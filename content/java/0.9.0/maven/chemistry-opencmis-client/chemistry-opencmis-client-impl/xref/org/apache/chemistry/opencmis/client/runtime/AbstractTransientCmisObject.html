91      protected void initialize(Session session, CmisObject object) {
92          this.session = session;
93          this.object = object;
94  
95          ObjectFactory of = getObjectFactory();
96  
97          // --- create snapshot ---
98  
99          // properties (modifiable)
100         properties = new LinkedHashMap<String, Property<?>>();
101         for (Property<?> property : object.getProperties()) {
102             properties.put(property.getId(),
103                     of.createProperty(property.getDefinition(), new ArrayList(property.getValues())));
104         }
105         isPropertyUpdateRequired = false;
106 
107         // allowable actions (unmodifiable)
108         allowableActions = object.getAllowableActions();
109 
110         // policies (modifiable)
111         policies = new ArrayList<Policy>();
112         if (object.getPolicies() != null) {
113             policies.addAll(object.getPolicies());
114         }
115         addPolicies = new HashSet<String>();
116         removePolicies = new HashSet<String>();
117 
118         // ACL (unmodifiable)
119         acl = object.getAcl();
120         addAces = new HashMap<AclPropagation, List<AceChangeHolder>>();
121         removeAces = new HashMap<AclPropagation, List<AceChangeHolder>>();
122 
123         // relationships (unmodifiable)
124         relationships = object.getRelationships();
125 
126         // renditions (unmodifiable)
127         renditions = object.getRenditions();
128 
129         // input extensions (unmodifiable)
130         inputExtensions = new HashMap<ExtensionLevel, List<CmisExtensionElement>>();
131         for (ExtensionLevel level : ExtensionLevel.values()) {
132             List<CmisExtensionElement> extension = object.getExtensions(level);
133             if (extension != null) {
134                 inputExtensions.put(level, extension);
135             }
136         }
137 
138         // output extensions (modifiable)
139         ouputExtensions = new HashMap<ExtensionLevel, List<CmisExtensionElement>>();
140 
141         isModified = false;
142         deleteAllVersions = true;
143         isMarkedForDelete = false;
144     }

399     public ObjectId save() {
400         if (!isModified()) {
401             // nothing has change, so there is nothing to do
402             return getObjectId();
403         }
404 
405         String objectId = getId();
406 
407         if (saveDelete(objectId)) {
408             // object has been deleted, there is nothing else to do
409             // ... and there is no object id anymore
410             return null;
411         }
412         String newObjectId = saveProperties(getId(), getChangeToken());
413         saveACL(newObjectId);
414         savePolicies(newObjectId);
415 
416         return getSession().createObjectId(newObjectId);
417     }

419     /**
420      * Fetches the latest change token of this object from the repository.
421      */
422     protected String getLatestChangeToken(String objectId) {
423         // determine the object id query name
424         PropertyDefinition<?> objectIdPropDef = getCmisObject().getType().getPropertyDefinitions()
425                 .get(PropertyIds.OBJECT_ID);
426         if (objectIdPropDef == null) {
427             return null;
428         }
429 
430         String objectIdQueryName = objectIdPropDef.getQueryName();
431         if (objectIdQueryName == null) {
432             return null;
433         }
434 
435         // determine the change token query name
436         PropertyDefinition<?> changeTokenPropDef = getCmisObject().getType().getPropertyDefinitions()
437                 .get(PropertyIds.CHANGE_TOKEN);
438         if (changeTokenPropDef == null) {
439             return null;
440         }
441 
442         String changeTokenQueryName = changeTokenPropDef.getQueryName();
443         if (changeTokenQueryName == null) {
444             return null;
445         }
446 
447         // get the change token property
448         Properties properties = getBinding().getObjectService().getProperties(getRepositoryId(), objectId,
449                 objectIdQueryName + "," + changeTokenQueryName, null);
450 
451         // if a change token is set, return it
452         PropertyData<?> changeToken = properties.getProperties().get(PropertyIds.CHANGE_TOKEN);
453 
454         if ((changeToken == null) || (changeToken.getFirstValue() == null)) {
455             return null;
456         }
457 
458         return changeToken.getFirstValue().toString();
459     }

470     protected Properties prepareProperties() {
471         Set<Updatability> updatebility = EnumSet.noneOf(Updatability.class);
472         updatebility.add(Updatability.READWRITE);
473 
474         // check if checked out
475         Boolean isCheckedOut = getPropertyValue(PropertyIds.IS_VERSION_SERIES_CHECKED_OUT);
476         if ((isCheckedOut != null) && isCheckedOut.booleanValue()) {
477             updatebility.add(Updatability.WHENCHECKEDOUT);
478         }
479 
480         // convert properties
481         Properties result = getObjectFactory().convertProperties(properties, getType(), getSecondaryTypes(),
482                 updatebility);
483 
484         // extensions
485         List<CmisExtensionElement> extensions = ouputExtensions.get(ExtensionLevel.PROPERTIES);
486         if (extensions != null) {
487             result.setExtensions(extensions);
488         }
489 
490         return result;
491     }

493     protected String saveProperties(String objectId, String changeToken) {
494         if (isPropertyUpdateRequired) {
495             Holder<String> objectIdHolder = new Holder<String>(objectId);
496             Holder<String> changeTokenHolder = new Holder<String>(changeToken);
497 
498             // convert properties
499             Properties props = prepareProperties();
500 
501             // it's time to update
502             getBinding().getObjectService().updateProperties(getRepositoryId(), objectIdHolder, changeTokenHolder,
503                     props, null);
504 
505             if (objectIdHolder.getValue() != null) {
506                 return objectIdHolder.getValue();
507             }
508         }
509 
510         return objectId;
511     }

525     protected Acl prepareAcl(List<AceChangeHolder> achList) {
526         if ((achList == null) || (achList.isEmpty())) {
527             return null;
528         }
529 
530         ObjectFactory of = getObjectFactory();
531 
532         List<Ace> aces = new ArrayList<Ace>();
533         for (AceChangeHolder ach : achList) {
534             aces.add(of.createAce(ach.getPrincipalId(), ach.getPermissions()));

540     protected void saveACL(String objectId) {
541         for (AclPropagation ap : AclPropagation.values()) {
542             if (!addAces.containsKey(ap) && !removeAces.containsKey(ap)) {
543                 continue;
544             }
545 
546             getBinding().getAclService().applyAcl(getRepositoryId(), objectId, prepareAcl(addAces.get(ap)),
547                     prepareAcl(removeAces.get(ap)), ap, null);
548         }
549 
550         if (addAces.containsKey(null) || removeAces.containsKey(null)) {
551             getBinding().getAclService().applyAcl(getRepositoryId(), objectId, prepareAcl(addAces.get(null)),
552                     prepareAcl(removeAces.get(null)), null, null);
553         }
554     }

558     /**
559      * Checks if a value matches a property definition.
560      * <p>
561      * Returns a list of values.
562      */
563     @SuppressWarnings("unchecked")
564     private static <T> List<T> checkProperty(PropertyDefinition<T> propertyDefinition, Object value) {
565 
566         // null values are ok for updates
567         if (value == null) {
568             return null;
569         }
570 
571         // single and multi value check
572         List<T> values = null;
573         if (value instanceof List<?>) {
574             if (propertyDefinition.getCardinality() != Cardinality.MULTI) {
575                 throw new IllegalArgumentException("Property '" + propertyDefinition.getId()
576                         + "' is not a multi value property!");
577             }
578 
579             values = (List<T>) value;
580             if (values.isEmpty()) {
581                 return values;
582             }
583         } else {
584             if (propertyDefinition.getCardinality() != Cardinality.SINGLE) {
585                 throw new IllegalArgumentException("Property '" + propertyDefinition.getId()
586                         + "' is not a single value property!");
587             }
588 
589             values = Collections.singletonList((T) value);
590         }
591 
592         // check if list contains null values
593         for (Object o : values) {
594             if (o == null) {
595                 throw new IllegalArgumentException("Property '" + propertyDefinition.getId()
596                         + "' contains null values!");
597             }
598         }
599 
600         // take a sample and test the data type
601         boolean typeMatch = false;
602         Object firstValue = values.get(0);
603 
604         switch (propertyDefinition.getPropertyType()) {
605         case STRING:
606         case ID:
607         case URI:
608         case HTML:
609             typeMatch = (firstValue instanceof String);
610             break;
611         case INTEGER:
612             typeMatch = (firstValue instanceof BigInteger) || (firstValue instanceof Byte)
613                     || (firstValue instanceof Short) || (firstValue instanceof Integer) || (firstValue instanceof Long);
614             break;
615         case DECIMAL:
616             typeMatch = (firstValue instanceof BigDecimal);
617             break;
618         case BOOLEAN:
619             typeMatch = (firstValue instanceof Boolean);
620             break;
621         case DATETIME:
622             typeMatch = (firstValue instanceof GregorianCalendar);
623             break;
624         }
625 
626         if (!typeMatch) {
627             throw new IllegalArgumentException("Value of property '" + propertyDefinition.getId()
628                     + "' does not match property type!");
629         }
630 
631         return values;
632     }
