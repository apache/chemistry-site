37  /**
38   * A parser implementation using a lower level interface to get more control about the 
39   * EXIF tags than Tika provides
40   * 
41   * @author Jens
42   *
43   */
44  public class MetadataParserExif extends AbstractMetadataParser  {
45  
46      private static final Logger LOG = LoggerFactory.getLogger(MetadataParserExif.class.getName());
47      
48      public void extractMetadata(File f, TypeDefinition td, Session session) throws MapperException {
49          
50          reset();
51          
52          // see  http://code.google.com/p/metadata-extractor/wiki/GettingStarted
53          try {
54              com.drew.metadata.Metadata metadata = ImageMetadataReader.readMetadata(f);
55              Iterator<?> it = metadata.getDirectoryIterator();
56              while (it.hasNext()) {
57                  Directory directory = (com.drew.metadata.Directory) it.next();
58                  Iterator<?> tagIt = directory.getTagIterator();
59                  while (tagIt.hasNext()) {
60                      Tag tag = (Tag) tagIt.next();
61                      Object o = directory.getObject(tag.getTagType());
62                      LOG.debug("Tag: " + tag + ", value: " + o + ", class: " + o.getClass() + 
63                              ", tag type: " + tag.getTagType() + ", hex-value: " + tag.getTagTypeHex());
64                      if (null != cmisProperties) {
65                          ((PropertyMapperExif)mapper).mapTagAndConvert(directory, tag, td);
66                      }
67                  }
68              }
69              Map<String, Object> props = ((PropertyMapperExif)mapper).getMappedProperties();
70              cmisProperties.putAll(props);
71          } catch (ImageProcessingException e) {            
72              LOG.error(e.toString(), e);
73          }
74  
75      }
76  
77  }
