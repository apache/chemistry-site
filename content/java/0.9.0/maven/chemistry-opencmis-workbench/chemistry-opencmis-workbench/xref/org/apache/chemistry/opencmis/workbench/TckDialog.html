291             for (CmisTestGroup group : runner.getGroups()) {
292                 group.setEnabled(select);

302         private JComboBox addComboBox(JPanel panel, String title, String rootTypeId, String defaultTypeId,
303                 boolean creatableOnly) {
304             final JLabel label = new JLabel(title);
305             label.setAlignmentX(Component.LEFT_ALIGNMENT);
306             panel.add(label);
307 
308             List<ObjectType> types = model.getTypesAsList(rootTypeId, creatableOnly);
309             String[] typeIds = new String[types.size()];
310 
311             int i = 0;
312             for (ObjectType type : types) {
313                 typeIds[i++] = type.getId();
314             }
315 
316             final JComboBox comboBox = new JComboBox(typeIds);
317             comboBox.setSelectedItem(defaultTypeId);
318             comboBox.setAlignmentX(Component.LEFT_ALIGNMENT);
319             comboBox.setMaximumSize(new Dimension(Short.MAX_VALUE, 10));
320             comboBox.setEnabled(typeIds.length > 0);
321             panel.add(comboBox);
322 
323             return comboBox;
324         }

327     private class TestTreeNode extends DefaultMutableTreeNode {
328         private static final long serialVersionUID = 1L;
329 
330         private final JTree tree;
331         private final CmisTestGroup group;
332         private final CmisTest test;
333 
334         public TestTreeNode(JTree tree, CmisTestGroup group) {
335             this.tree = tree;
336             this.group = group;
337             this.test = null;
338         }
339 
340         public TestTreeNode(JTree tree, CmisTest test) {
341             this.tree = tree;
342             this.test = test;
343             this.group = null;
344         }
345 
346         public CmisTestGroup getGroup() {
347             return group;
348         }
349 
350         public String getName() {
351             if (group != null) {
352                 return group.getName();
353             }
354 
355             return test.getName();
356         }
357 
358         public String getDescription() {
359             if (group != null) {
360                 return group.getDescription();
361             }
362 
363             return test.getDescription();
364         }
365 
366         public boolean isEnabled() {
367             if (group != null) {
368                 return group.isEnabled();
369             }
370 
371             return test.isEnabled();
372         }
373 
374         public void setEnabled(boolean enabled) {
375             DefaultTreeModel model = ((DefaultTreeModel) tree.getModel());
376 
377             if (group != null) {
378                 group.setEnabled(enabled);
379 
380                 for (int i = 0; i < getChildCount(); i++) {
381                     TestTreeNode node = (TestTreeNode) getChildAt(i);
382                     node.setEnabled(enabled);
383                     model.nodeChanged(node);
384                 }
385 
386                 return;
387             }
388 
389             test.setEnabled(enabled);
390 
391             if (enabled) {
392                 TestTreeNode node = (TestTreeNode) getParent();
393                 node.getGroup().setEnabled(true);
394                 model.nodeChanged(node);
395             }
396         }

584     private class DialogProgressMonitor implements CmisTestProgressMonitor {
585 
586         public DialogProgressMonitor(int numberOfGroups) {
587             groupsProgressBar.setStringPainted(true);
588             groupsProgressBar.setMinimum(0);
589             groupsProgressBar.setMaximum(numberOfGroups);
590             groupsProgressBar.setValue(0);
591         }
592 
593         public void startGroup(CmisTestGroup group) {
594             groupsProgressBar.setString(group.getName());
595 
596             testsProgressBar.setStringPainted(true);
597             testsProgressBar.setMinimum(0);
598             testsProgressBar.setMaximum(group.getTests().size());
599             testsProgressBar.setValue(0);
600         }
601 
602         public void endGroup(CmisTestGroup group) {
603             groupsProgressBar.setString("");
604             groupsProgressBar.setValue(groupsProgressBar.getValue() + 1);
605         }
606 
607         public void startTest(CmisTest test) {
608             testsProgressBar.setString(test.getName());
609         }
610 
611         public void endTest(CmisTest test) {
612             testsProgressBar.setString("");
613             testsProgressBar.setValue(testsProgressBar.getValue() + 1);
614 
615             for (CmisTestResult tr : test.getResults()) {
616                 int x = status.get(tr.getStatus());
617                 status.put(tr.getStatus(), x + 1);
618             }
619 
620             StringBuilder sb = new StringBuilder("<html>");
621             int x;
622 
623             x = status.get(CmisTestResultStatus.INFO);
624             if (x > 0) {
625                 sb.append("<font color='#000000'>[Info: " + x + "]</font>  ");
626             }
627 
628             x = status.get(CmisTestResultStatus.SKIPPED);
629             if (x > 0) {
630                 sb.append("<font color='#444444'>[Skipped: " + x + "]</font>  ");
631             }
632 
633             x = status.get(CmisTestResultStatus.OK);
634             if (x > 0) {
635                 sb.append("<font color='#009900'>[Ok: " + x + "]</font>  ");
636             }
637 
638             x = status.get(CmisTestResultStatus.WARNING);
639             if (x > 0) {
640                 sb.append("<font color='#999900'>[Warning: " + x + "]</font>  ");
641             }
642 
643             x = status.get(CmisTestResultStatus.FAILURE);
644             if (x > 0) {
645                 sb.append("<font color='#996000'>[Failure: " + x + "]</font>  ");
646             }
647 
648             x = status.get(CmisTestResultStatus.UNEXPECTED_EXCEPTION);
649             if (x > 0) {
650                 sb.append("<font color='#990000'>[Exception: " + x + "]</font>  ");
651             }
652 
653             sb.append("</html>");
654 
655             statusLabel.setText(sb.toString());
656         }

659         }

662     class TckTask extends SwingWorker<Void, Void> {
663         private final JDialog dialog;
664         private final TckDialogRunner runner;
665 
666         public TckTask(JDialog dialog, TckDialogRunner runner) {
667             this.dialog = dialog;
668             this.runner = runner;
669         }
670 
671         @Override
672         public Void doInBackground() {
673             try {
674                 runner.run(new DialogProgressMonitor(runner.getGroups().size()));
675             } catch (InterruptedException ie) {
676                 runner.cancel();
677             } catch (Exception e) {
678                 JOptionPane
679                         .showMessageDialog(owner, "Error: " + e.getMessage(), "TCK Error", JOptionPane.ERROR_MESSAGE);
680             }
681 
682             return null;
683         }
684 
685         @Override
686         public void done() {
