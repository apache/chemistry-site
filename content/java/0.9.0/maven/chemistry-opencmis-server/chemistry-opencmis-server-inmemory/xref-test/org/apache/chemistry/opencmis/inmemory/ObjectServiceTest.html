334         Holder<String> tokenHolder = new Holder<String>(changeToken);
335         fObjSvc.deleteContentStream(fRepositoryId, idHolder, tokenHolder, null);
336         
337         try {
338             props = fObjSvc.getProperties(fRepositoryId, id, PropertyIds.CHANGE_TOKEN, null);
339             changeToken = (String) props.getProperties().get(PropertyIds.CHANGE_TOKEN).getFirstValue();
340             tokenHolder = new Holder<String>(changeToken);

342                     .valueOf(-1) /* length */, null);
343             fail("getContentStream with non existing content should raise a CmisConstraintException");
344         } catch (Exception e) {
345             assertTrue(e instanceof CmisConstraintException);
346         }
347         
348         // create content again in a second call
349         ContentStream contentStream = createContent();
350         fObjSvc.setContentStream(fRepositoryId, idHolder, true, tokenHolder, contentStream, null);
351         sd = fObjSvc.getContentStream(fRepositoryId, id, null, BigInteger.valueOf(-1) /* offset */, BigInteger
352                 .valueOf(-1) /* length */, null);
353         verifyContentResult(sd);
354 
355         // update content and do not set overwrite flag, expect failure
356         try {
357             props = fObjSvc.getProperties(fRepositoryId, id, PropertyIds.CHANGE_TOKEN, null);
358             changeToken = (String) props.getProperties().get(PropertyIds.CHANGE_TOKEN).getFirstValue();
359             tokenHolder = new Holder<String>(changeToken);
360             fObjSvc.setContentStream(fRepositoryId, idHolder, false, tokenHolder, contentStream, null);
361             fail("setContentStream with existing content and no overWriteFlag should fail");
362         } catch (Exception e) {
363             assertTrue(e instanceof CmisContentAlreadyExistsException);
364         }
365 
366         // cleanup
367         fObjSvc.deleteObject(fRepositoryId, id, true, null);
368 
369         log.info("... testCreateDocumentWithContent() finished.");
370     }
