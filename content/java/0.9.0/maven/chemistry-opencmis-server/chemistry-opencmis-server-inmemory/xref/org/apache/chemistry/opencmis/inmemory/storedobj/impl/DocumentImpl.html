64      public ContentStream getContent(long offset, long length) {
65          if (null == fContent) {
66              return null;
67          } else if (offset <= 0 && length < 0) {
68              return fContent;
69          } else {
70              return fContent.getCloneWithLimits(offset, length);
71          }
72      }

74      public void setContent(ContentStream content, boolean mustPersist) {
75          if (null == content) {
76              fContent = null;
77          } else {
78              fContent = new ContentStreamDataImpl(MAX_CONTENT_SIZE_KB == null ? 0 : MAX_CONTENT_SIZE_KB);
79              String fileName = content.getFileName();
80              if (null == fileName || fileName.length() <= 0) {
81                  fileName = getName(); // use name of document as fallback
82              }
83              fContent.setFileName(fileName);
84              String mimeType = content.getMimeType();
85              if (null == mimeType || mimeType.length() <= 0) {
86                  mimeType = "application/octet-stream"; // use as fallback
87              }
88              fContent.setMimeType(mimeType);
89              fContent.setLastModified(getModifiedAt());
90              try {
91                  fContent.setContent(content.getStream());
92              } catch (IOException e) {
93                  e.printStackTrace();
94                  throw new RuntimeException("Failed to get content from InputStream", e);
95              }
96          }
97      }
98      
99      public void appendContent(ContentStream content) {
100         if (null == content) {
101             return;
102         } if (null == fContent) {
103             setContent(content, true);
104         } else {
105             try {
106                 fContent.appendContent(content.getStream());
107             } catch (IOException e) {
108                 e.printStackTrace();
109                 throw new CmisStorageException("Failed to append content: IO Exception", e);
110             }
111         }
112     }

115     @Override
116     public void fillProperties(Map<String, PropertyData<?>> properties, BindingsObjectFactory objFactory,
117             List<String> requestedIds) {
118 
119         super.fillProperties(properties, objFactory, requestedIds);
120 
121         // fill the version related properties (versions should override this
122         // but the spec requires some
123         // properties always to be set
124 
125         if (FilterParser.isContainedInFilter(PropertyIds.IS_IMMUTABLE, requestedIds)) {
126             properties.put(PropertyIds.IS_IMMUTABLE, objFactory.createPropertyBooleanData(PropertyIds.IS_IMMUTABLE,
127                     false));
128         }
129 
130         // Set the content related properties
131         if (FilterParser.isContainedInFilter(PropertyIds.CONTENT_STREAM_FILE_NAME, requestedIds)) {
132             properties.put(PropertyIds.CONTENT_STREAM_FILE_NAME, objFactory.createPropertyStringData(
133                     PropertyIds.CONTENT_STREAM_FILE_NAME, null != fContent ? fContent.getFileName() : (String)null) );
134         }
135         if (FilterParser.isContainedInFilter(PropertyIds.CONTENT_STREAM_ID, requestedIds)) {
136             properties.put(PropertyIds.CONTENT_STREAM_ID, objFactory.createPropertyStringData(
137                     PropertyIds.CONTENT_STREAM_ID, (String) null));
138         }
139         if (FilterParser.isContainedInFilter(PropertyIds.CONTENT_STREAM_LENGTH, requestedIds)) {
140             properties.put(PropertyIds.CONTENT_STREAM_LENGTH, objFactory.createPropertyIntegerData(
141                     PropertyIds.CONTENT_STREAM_LENGTH, null != fContent ? fContent.getBigLength() : null));
142         }
143         if (FilterParser.isContainedInFilter(PropertyIds.CONTENT_STREAM_MIME_TYPE, requestedIds)) {
144             properties.put(PropertyIds.CONTENT_STREAM_MIME_TYPE, objFactory.createPropertyStringData(
145                     PropertyIds.CONTENT_STREAM_MIME_TYPE, null != fContent ? fContent.getMimeType() : (String)null) );
146         }
147         
148         // Spec requires versioning properties even for unversioned documents
149         // overwrite the version related properties
150         if (FilterParser.isContainedInFilter(PropertyIds.VERSION_SERIES_ID, requestedIds)) {
151             properties.put(PropertyIds.VERSION_SERIES_ID, objFactory.createPropertyIdData(
152                     PropertyIds.VERSION_SERIES_ID, getId()));
153         }
154         if (FilterParser.isContainedInFilter(PropertyIds.IS_VERSION_SERIES_CHECKED_OUT, requestedIds)) {
155             properties.put(PropertyIds.IS_VERSION_SERIES_CHECKED_OUT, objFactory.createPropertyBooleanData(
156                     PropertyIds.IS_VERSION_SERIES_CHECKED_OUT, false));
157         }
158         if (FilterParser.isContainedInFilter(PropertyIds.VERSION_SERIES_CHECKED_OUT_BY, requestedIds)) {
159             properties.put(PropertyIds.VERSION_SERIES_CHECKED_OUT_BY, objFactory.createPropertyStringData(
160                     PropertyIds.VERSION_SERIES_CHECKED_OUT_BY, (String)null));
161         }
162         if (FilterParser.isContainedInFilter(PropertyIds.VERSION_SERIES_CHECKED_OUT_ID, requestedIds)) {
163             properties.put(PropertyIds.VERSION_SERIES_CHECKED_OUT_ID, objFactory.createPropertyIdData(
164                     PropertyIds.VERSION_SERIES_CHECKED_OUT_ID, (String)null));
165         }
166         if (FilterParser.isContainedInFilter(PropertyIds.IS_LATEST_VERSION, requestedIds)) {
167             properties.put(PropertyIds.IS_LATEST_VERSION, objFactory.createPropertyBooleanData(
168                     PropertyIds.IS_LATEST_VERSION, true));
169         }
170         if (FilterParser.isContainedInFilter(PropertyIds.IS_MAJOR_VERSION, requestedIds)) {
171             properties.put(PropertyIds.IS_MAJOR_VERSION, objFactory.createPropertyBooleanData(
172                     PropertyIds.IS_MAJOR_VERSION, true));
173         }
174         if (FilterParser.isContainedInFilter(PropertyIds.IS_LATEST_MAJOR_VERSION, requestedIds)) {
175             properties.put(PropertyIds.IS_LATEST_MAJOR_VERSION, objFactory.createPropertyBooleanData(
176                     PropertyIds.IS_LATEST_MAJOR_VERSION, true));
177         }
178         if (FilterParser.isContainedInFilter(PropertyIds.CHECKIN_COMMENT, requestedIds)) {
179             properties.put(PropertyIds.CHECKIN_COMMENT, objFactory.createPropertyStringData(
180                     PropertyIds.CHECKIN_COMMENT, (String )null));
181         }
182         if (FilterParser.isContainedInFilter(PropertyIds.VERSION_LABEL, requestedIds)) {
183             properties.put(PropertyIds.VERSION_LABEL, objFactory.createPropertyStringData(PropertyIds.VERSION_LABEL,
184                     (String) null));

187         // CMIS 1.1

189             properties.put(PropertyIds.IS_PRIVATE_WORKING_COPY,
190                     objFactory.createPropertyBooleanData(PropertyIds.IS_PRIVATE_WORKING_COPY, false));
191         }

199     @Override
200     public List<RenditionData> getRenditions(String renditionFilter,
201             long maxItems, long skipCount) {
202 
203         String tokenizer = "[\\s;]";
204         if (null==renditionFilter)
205             return null;
206         String[] formats = renditionFilter.split(tokenizer);
207         boolean isImageRendition = testRenditionFilterForImage(formats);
208 
209         if (isImageRendition && fContent != null && hasRendition(null)) {
210             List<RenditionData> renditions = new ArrayList<RenditionData>(1);
211             String mimeType = fContent.getMimeType();
212             RenditionDataImpl rendition = new RenditionDataImpl();
213             if (mimeType.equals("image/jpeg")) {
214                 rendition.setBigHeight(BigInteger.valueOf(THUMBNAIL_SIZE));
215                 rendition.setBigWidth(BigInteger.valueOf(THUMBNAIL_SIZE));
216                 rendition.setMimeType(RENDITION_MIME_TYPE_JPEG);
217             } else {
218                 rendition.setBigHeight(BigInteger.valueOf(ICON_SIZE));
219                 rendition.setBigWidth(BigInteger.valueOf(ICON_SIZE));
220                 rendition.setMimeType(RENDITION_MIME_TYPE_PNG);
221             }
222             rendition.setKind("cmis:thumbnail");
223             rendition.setRenditionDocumentId(getId());
224             rendition.setStreamId(getId() + RENDITION_SUFFIX);
225             rendition.setBigLength(BigInteger.valueOf(-1L));
226             rendition.setTitle(getName());
227             renditions.add(rendition);
228             return renditions;
229         } else {
230             return null;
231         }
232     }

234     @Override
235     public ContentStream getRenditionContent(String streamId, long offset, long length) {     
236         if (null == fContent)
237             return null;
238         
239         String mimeType = fContent.getMimeType();
240         
241         try {
242             if (isImage(mimeType)) {
243                 ImageThumbnailGenerator generator = new ImageThumbnailGenerator(getContent(0L, -1L).getStream());
244                 return generator.getRendition(THUMBNAIL_SIZE, 0);
245             } else if (isAudio(mimeType)) {
246                 return getIconFromResourceDir("/audio-x-generic.png");
247             } else if (isVideo(mimeType)) {
248                 return getIconFromResourceDir("/video-x-generic.png");
249             } else if (isPDF(mimeType)) {
250                 return getIconFromResourceDir("/application-pdf.png");
251             } else if (isWord(mimeType)) {
252                 return getIconFromResourceDir("/application-msword.png");
253             } else if (isPowerpoint(mimeType)) {
254                 return getIconFromResourceDir("/application-vnd.ms-powerpoint.png");
255             } else if (isExcel(mimeType)) {
256                 return getIconFromResourceDir("/application-vnd.ms-excel.png");
257             } else if (isHtml(mimeType)) {
258                 return getIconFromResourceDir("/text-html.png");
259             } else if (isPlainText(mimeType)) {
260                 return getIconFromResourceDir("/text-x-generic.png");
261             } else
262                 return null;
263         } catch (IOException e) {
264             LOG.error("Failed to generate rendition: ", e);
265             throw new CmisRuntimeException("Failed to generate rendition: " + e);
266         }
267     }

274         String mimeType = fContent.getMimeType();
275         
276         if (isImage(mimeType))

281             return true;
282         else if (isPDF(mimeType))
283             return true;
284         else if (isPowerpoint(mimeType))
285             return true;
286         else if (isExcel(mimeType))
287             return true;
288         else if (isWord(mimeType))
289             return true;
290         else if (isHtml(mimeType))
291             return true;
292         else if (isPlainText(mimeType))
293             return true;
294         else
295             return false;
296     }
