82      private void buildIdMap(String dirKey, Map<String, String> propMap, Properties properties) {
83          Set<String> keys = properties.stringPropertyNames();
84          String prefix = propPrefix + "." + dirKey + ".id.";
85          for (String key : keys) {
86              if (key.startsWith(prefix)) {
87                  String id = key.substring(prefix.length());
88                  String cmisPropId = properties.getProperty(key);
89                  if (null == cmisPropId) {
90                      throw new MapperException("Configuration key " + key + " must have a value assigned");
91                  }
92                  cmisPropId = cmisPropId.trim();
93                  LOG.debug("Found mapping for type " + dirKey + " with " + id + " to " + cmisPropId);
94                  propMap.put(id, cmisPropId);
95              }
96          }
97      }

111     private String getHexString(int tagType) {
112         StringBuffer hexStr = new StringBuffer();
113         hexStr.append(Integer.toHexString(tagType));
114         while (hexStr.length() < 4) {
115             hexStr.insert(0, "0");
116         }
117         hexStr.insert(0, "0x");
118         return hexStr.toString();
119     }

130     public void mapTagAndConvert(Directory dir, Tag tag, TypeDefinition td) {
131         String propId = null;
132         String hexStr = getHexString(tag.getTagType());
133 
134         if (GpsDirectory.class.equals(dir.getClass())) {
135             propId = propMapGps.get(hexStr);
136         } else if (ExifDirectory.class.equals(dir.getClass())) {
137             propId = propMapExif.get(hexStr);
138         } else if (JpegDirectory.class.equals(dir.getClass())) {
139             propId = propMapJpeg.get(hexStr);
140         } else {
141             propId = null;
142         }
143 
144         if (null != propId) {
145             if (null != td) {
146                 PropertyDefinition<?> pd = td.getPropertyDefinitions().get(propId);
147                 if (null == pd) {
148                     throw new MapperException("Unknown property id " + propId + " in type definition " + td.getId());
149                 }
150                 PropertyType pt = pd.getPropertyType();
151                 Object convValue = convertValue(dir, tag, pt);
152                 propMap.put(propId, convValue);
153             } else {
154                 // omit conversion if no type definition is available
155                 propMap.put(propId, dir.getObject(tag.getTagType()));
156             }
157         }
158     }

160     public Object convertValue(Directory dir, Tag tag, PropertyType propType) {
161 
162         Object res = null;
163         String hexStr = getHexString(tag.getTagType());
164 
165         // Handle all tags corresponding to their directory specifics
166         if (GpsDirectory.class.equals(dir.getClass())) {
167             // first check for those tags that need special consideration:
168             if (GpsDirectory.TAG_GPS_LONGITUDE == tag.getTagType()) {
169                 Object ref = dir.getObject(GpsDirectory.TAG_GPS_LONGITUDE_REF);
170                 boolean mustInv = ref != null && ref.equals("W");
171                 return convertGps(tag, dir, mustInv);
172             } else if (GpsDirectory.TAG_GPS_LATITUDE == tag.getTagType()) {
173                 Object ref = dir.getObject(GpsDirectory.TAG_GPS_LONGITUDE_REF);
174                 boolean mustInv = ref != null && ref.equals("S");
175                 return convertGps(tag, dir, mustInv);
176             } else {
177                 String propId = propMapGps.get(hexStr);
178                 LOG.debug("Found GPS tag '" + tag + "\', property mapped is: " + propId);
179                 if (null == propId) {
180                     LOG.info("Ignoring EXIF tag '" + tag + "\' no property mapped to this tag.");
181                 } else if (propType == null) {
182                     // should not happen and is a configuration error: we have a
183                     // property id but no type
184                     LOG.error("Ignoring EXIF tag '" + tag + "\' no property type mapped to this tag.");
185                 }
186                 Object src = dir.getObject(tag.getTagType());
187                 Class<?> clazz = src.getClass();
188                 if (clazz.equals(Rational.class)) {
189                     // expect a CMIS decimal property
190                     if (propType != PropertyType.DECIMAL) {
191                         throw new MapperException("Tag value has type Rational and expected CMIS Decimal, but found: "
192                                 + propType + " for tag: " + tag);
193                     }
194                     double d = ((Rational) src).doubleValue();
195                     res = d;
196                 } else if (clazz.equals(String.class)) {
197                     if (propType != PropertyType.STRING && propType != PropertyType.ID && propType != PropertyType.URI
198                             && propType != PropertyType.HTML && propType != PropertyType.DATETIME) {
199                         throw new MapperException("Tag value has type String and expected CMIS String, but found: "
200                                 + propType + " for tag: " + tag);
201                     }
202                     String s = ((String) src);
203                     res = s;
204                 } else {
205                     res = null;
206                 }
207             }
208         } else if (ExifDirectory.class.equals(dir.getClass())) {
209             // is there a property mapped to this tag?
210             String propId = propMapExif.get(hexStr);
211             LOG.debug("Found EXIF tag '" + tag + "\', property mapped is: " + propId);
212 
213             if (null == propId) {
214                 LOG.debug("Ignoring EXIF tag '" + tag + "\' no property mapped to this tag.");
215             } else if (propType == null) {
216                 // should not happen and is a configuration error: we have a
217                 // property id but no type
218                 LOG.error("Ignoring EXIF tag '" + tag + "\' no property type mapped to this tag.");
219             } else {
220                 Object src = dir.getObject(tag.getTagType());
221                 Class<?> clazz = src.getClass();
222                 // handle arrays and map them to multi-value properties
223                 if (clazz.isArray()) {
224                     LOG.error("Found a multi-value tag " + tag + ": multi value not implemented");
225                     return null;
226                 }
227                 if (clazz.equals(Rational.class)) {
228                     // expect a CMIS decimal property
229                     if (propType != PropertyType.DECIMAL) {
230                         throw new MapperException("Tag value has type Rational and expected CMIS Decimal, but found: "
231                                 + propType + " for tag: " + tag);
232                     }
233 
234                     if (tag.getTagType() == ExifDirectory.TAG_SHUTTER_SPEED) {
235                         // requires special handling, see Tika impl.
236                         double apexValue = ((Rational) src).doubleValue();
237                         res = PhotographicConversions.shutterSpeedToExposureTime(apexValue);
238                     } else if (tag.getTagType() == (ExifDirectory.TAG_APERTURE)) {
239                         double aperture = ((Rational) src).doubleValue();
240                         double fStop = PhotographicConversions.apertureToFStop(aperture);
241                         res = fStop;
242                     } else {
243                         // convert to a double
244                         double d = ((Rational) src).doubleValue();
245                         res = d;
246                     }
247                 } else if (clazz.equals(Integer.class)) {
248                     if (propType != PropertyType.INTEGER) {
249                         throw new MapperException("Tag value has type Integer and expected CMIS Integer, but found: "
250                                 + propType + " for tag: " + tag);
251                     }
252                     // convert to a long
253                     long l = ((Integer) src).longValue();
254                     res = l;
255                 } else if (clazz.equals(String.class)) {
256                     if (propType != PropertyType.STRING && propType != PropertyType.ID && propType != PropertyType.URI
257                             && propType != PropertyType.HTML && propType != PropertyType.DATETIME) {
258                         throw new MapperException("Tag value has type String and expected CMIS String, but found: "
259                                 + propType + " for tag: " + tag);
260                     }
261 
262                     // convert to a String
263                     if (propType == PropertyType.DATETIME) {
264                         // parse format: 2012:02:25 16:23:16
265                         DateFormat formatter = new SimpleDateFormat(dateFormat);
266                         GregorianCalendar cal;
267                         try {
268                             Date date = formatter.parse((String) src);
269                             // convert date to GreogorianCalendar as CMIS
270                             // expects
271                             cal = new GregorianCalendar();
272                             cal.setTime(date);
273                         } catch (ParseException e) {
274                             LOG.error(e.toString(), e);
275                             throw new MapperException("Unrecognized date format in EXIF date tag: " + src
276                                     + " for tag: " + tag + " expected: yyyy:MM:dd HH:mm:ss", e);
277                         }
278                         res = cal;
279                     } else {
280                         String s = ((String) src);
281                         res = s;
282                     }
283                 } else if (clazz.equals(Date.class)) {
284                     if (propType != PropertyType.DATETIME) {
285                         throw new MapperException("Tag value has type Date and expected CMIS DateTime, but found: "
286                                 + propType + " for tag: " + tag);
287                     }
288                     // convert to a String
289                     Date date = ((Date) src);
290                     // convert date to GreogorianCalendar as CMIS expects
291                     GregorianCalendar cal = new GregorianCalendar();
292                     cal.setTime(date);
293                     res = cal;
294                 } else if (clazz.equals(Boolean.class)) {
295                     if (propType != PropertyType.BOOLEAN) {
296                         throw new MapperException("Tag value has type Boolean and expected CMIS Boolean, but found: "
297                                 + propType + " for tag: " + tag);
298                     }
299                     // convert to a String
300                     Boolean b = ((Boolean) src);
301                     res = b;
302                 } else {
303                     LOG.debug("Tag value has unsupported type: " + clazz.getName() + " for EXIF tag: " + tag);
304                 }
305             }
306         } else if (JpegDirectory.class.equals(dir.getClass())) {
307             // is there a property mapped to this tag?
308             String propId = propMapJpeg.get(hexStr);
309             LOG.debug("Found JPEG tag '" + tag + "\', property mapped is: " + propId);
310 
311             if (null == propId) {
312                 LOG.info("Ignoring JPEG tag '" + tag + "\' no property mapped to this tag.");
313             } else if (propType == null) {
314                 // should not happen and is a configuration error: we have a
315                 // property id but no type
316                 LOG.error("Ignoring JPEG tag '" + tag + "\' no property type mapped to this tag.");
317             } else {
318                 Object src = dir.getObject(tag.getTagType());
319                 Class<?> clazz = src.getClass();
320 
321                 if (clazz.equals(Integer.class)) {
322                     if (propType != PropertyType.INTEGER) {
323                         throw new MapperException("Tag value has type Integer and expected CMIS Integer, but found: "
324                                 + propType + " for tag: " + tag);
325                     }
326                     // convert to a long
327                     long l = ((Integer) src).longValue();
328                     res = l;
329                 } else {
330                     LOG.debug("Tag value has unsupported type: " + clazz.getName() + " for JPEG tag: " + tag);
331                 }
332             }
333         }
334 
335         return res;
336     }

338     private static Object convertGps(Tag tag, Directory dir, boolean mustInvert) {
339         Double res = null;
340         Object src = dir.getObject(tag.getTagType());
341 
342         Class<?> stringArrayClass = src.getClass();
343         Class<?> stringArrayComponentType = stringArrayClass.getComponentType();
344         if (!stringArrayClass.isArray() || null == stringArrayComponentType || Array.getLength(src) != 3) {
345             throw new MapperException("GPS coordinate \"" + tag + "\" has unknown type.");
346         }
347         if (!stringArrayComponentType.equals(Rational.class)) {
348             throw new MapperException("GPS coordinate \"" + tag
349                     + "\" has unknown component type (expected Rational, found: " + stringArrayComponentType.getName()
350                     + ")");
351         }
352         // do conversion
353         Rational[] components;
354         components = (Rational[]) src;
355         int deg = components[0].intValue();
356         double min = components[1].doubleValue();
357         double sec = components[2].doubleValue();
358         Double d = (deg + min / SECS_PER_MINUTE + sec / SECS_PER_HOUR);
359         if (d > 0.0 && mustInvert) {
360             d = -d;
361         }
362         res = d;
363         return res;
364     }
