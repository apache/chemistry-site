87      public MultipartParser(HttpServletRequest request, TempStoreOutputStreamFactory streamFactory) throws IOException {
88          this.request = request;
89          this.streamFactory = streamFactory;
90          this.requestStream = request.getInputStream();
91  
92          extractBoundary();
93  
94          buffer = new byte[BUFFER_SIZE + boundary.length];
95          buffer2 = new byte[buffer.length];
96          bufferPosition = 0;
97          bufferCount = 0;
98          eof = false;
99  
100         hasContent = false;
101         fieldBytes = 0;
102 
103         fields = new HashMap<String, String[]>();
104         rawFields = new HashMap<String, byte[][]>();
105 
106         skipPreamble();
107     }

109     private void addField(String name, String value) {
110         String[] values = fields.get(name);
111 
112         if (values == null) {
113             fields.put(name, new String[] { value });
114         } else {
115             String[] newValues = new String[values.length + 1];
116             System.arraycopy(values, 0, newValues, 0, values.length);
117             newValues[newValues.length - 1] = value;
118             fields.put(name, newValues);
119         }
120     }

122     private void addRawField(String name, byte[] value) {
123         byte[][] values = rawFields.get(name);
124 
125         if (values == null) {
126             byte[][] newValue = new byte[][] { value };
127             rawFields.put(name, newValue);
128         } else {
129             byte[][] newValues = new byte[values.length + 1][];
130             System.arraycopy(values, 0, newValues, 0, values.length);
131             newValues[newValues.length - 1] = value;
132             rawFields.put(name, newValues);
133         }
134     }

136     private void extractBoundary() {
137         String requestContentType = request.getContentType();
138 
139         // parse content type and extract boundary
140         byte[] extractedBoundary = MimeHelper.getBoundaryFromMultiPart(requestContentType);
141         if (extractedBoundary == null) {
142             throw new CmisInvalidArgumentException("Invalid multipart request!");
143         }
144 
145         boundary = new byte[BOUNDARY_PREFIX.length + extractedBoundary.length];
146         System.arraycopy(BOUNDARY_PREFIX, 0, boundary, 0, BOUNDARY_PREFIX.length);
147         System.arraycopy(extractedBoundary, 0, boundary, BOUNDARY_PREFIX.length, extractedBoundary.length);
148 
149         // prepare boundary search
150 
151         int m = boundary.length;
152 
153         badCharacters = new int[256];
154         Arrays.fill(badCharacters, -1);
155 
156         for (int j = 0; j < m; j++) {
157             badCharacters[boundary[j] & 0xff] = j;
158         }
159 
160         int[] f = new int[m + 1];
161         goodSuffixes = new int[m + 1];
162         int i = m;
163         int j = m + 1;
164         f[i] = j;
165         while (i > 0) {
166             while (j <= m && boundary[i - 1] != boundary[j - 1]) {
167                 if (goodSuffixes[j] == 0) {
168                     goodSuffixes[j] = j - i;
169                 }
170                 j = f[j];
171             }
172             i--;
173             j--;
174             f[i] = j;
175         }
176 
177         j = f[0];
178         for (i = 0; i <= m; i++) {
179             if (goodSuffixes[i] == 0) {
180                 goodSuffixes[i] = j;
181             }
182 
183             if (i == j) {
184                 j = f[j];
185             }
186         }
187     }

189     private int findBoundary() {
190         if (bufferCount < boundary.length) {
191             if (eof) {
192                 throw new CmisInvalidArgumentException("Unexpected end of stream!");
193             } else {
194                 return -1;
195             }
196         }
197 
198         int m = boundary.length;
199 
200         int i = 0;
201         while (i <= bufferCount - m) {
202             int j = m - 1;
203             while (j >= 0 && boundary[j] == buffer[i + j]) {
204                 j--;
205             }
206 
207             if (j < 0) {
208                 return i;
209             } else {
210                 i += Math.max(goodSuffixes[j + 1], j - badCharacters[buffer[i + j] & 0xff]);
211             }
212         }
213 
214         return -1;
215     }

217     private void readBuffer() throws IOException {
218         if (bufferPosition < bufferCount) {
219             System.arraycopy(buffer, bufferPosition, buffer2, 0, bufferCount - bufferPosition);
220             bufferCount = bufferCount - bufferPosition;
221 
222             byte[] tmpBuffer = buffer2;
223             buffer2 = buffer;
224             buffer = tmpBuffer;
225         } else {
226             bufferCount = 0;
227         }
228 
229         bufferPosition = 0;
230 
231         if (eof) {
232             return;
233         }
234 
235         while (true) {
236             int r = requestStream.read(buffer, bufferCount, buffer.length - bufferCount);
237             if (r == -1) {
238                 eof = true;
239                 break;
240             }
241 
242             bufferCount += r;
243             if (buffer.length == bufferCount) {
244                 break;
245             }
246         }
247     }

249     private int nextByte() throws IOException {
250         if (bufferCount == 0) {
251             if (eof) {
252                 return -1;
253             } else {
254                 readBuffer();
255                 return nextByte();
256             }
257         }
258 
259         if (bufferCount > bufferPosition) {
260             return buffer[bufferPosition++] & 0xff;
261         }
262 
263         readBuffer();
264         return nextByte();
265     }

267     private String readLine() throws IOException {
268         StringBuilder sb = new StringBuilder();
269 
270         int r;
271         while ((r = nextByte()) > -1) {
272             if (r == CR) {
273                 if (nextByte() != LF) {
274                     throw new CmisInvalidArgumentException("Invalid multipart request!");
275                 }
276                 break;
277             }
278 
279             sb.append((char) r);
280         }
281 
282         return sb.toString();
283     }

285     private void readHeaders() throws IOException {
286         int b = nextByte();
287         if (b == -1) {
288             throw new CmisInvalidArgumentException("Unexpected end of stream!");
289         }
290 
291         if (b == DASH) {
292             b = nextByte();
293             if (b == DASH) {
294                 // expected end of stream
295                 headers = null;
296                 return;
297             }
298         } else if (b == CR) {
299             b = nextByte();
300             if (b == LF) {
301                 parseHeaders();
302                 return;
303             }
304         }
305 
306         throw new CmisInvalidArgumentException("Invalid multipart request!");
307     }

309     private void parseHeaders() throws IOException {
310         headers = new HashMap<String, String>();
311 
312         while (true) {
313             String line = readLine();
314             if (line.length() == 0) {
315                 // empty line -> end of headers
316                 break;
317             }
318 
319             int x = line.indexOf(':');
320             if (x > 0) {
321                 headers.put(line.substring(0, x).toLowerCase(Locale.ENGLISH).trim(), line.substring(x + 1).trim());
322             }
323         }
324     }

326     private byte[] readBodyBytes() throws IOException {
327         readBuffer();
328 
329         int boundaryPosition = findBoundary();
330 
331         if (boundaryPosition > -1) {
332             // the body bytes are completely in the buffer
333             int len = boundaryPosition - bufferPosition;
334             addFieldBytes(len);
335 
336             byte[] body = new byte[len];
337             System.arraycopy(buffer, bufferPosition, body, 0, len);
338             bufferPosition = boundaryPosition + boundary.length;
339             return body;
340         }
341 
342         // the body bytes are not completely in the buffer
343         // read all available bytes
344         int len = Math.min(BUFFER_SIZE, bufferCount) - bufferPosition;
345         addFieldBytes(len);
346 
347         byte[] bodyBytes = new byte[len + BUFFER_SIZE];
348         int bodyBytesPos = len;
349 
350         System.arraycopy(buffer, bufferPosition, bodyBytes, 0, len);
351         bufferPosition = bufferPosition + len;
352 
353         // read next chunk
354         while (true) {
355             readBuffer();
356 
357             boundaryPosition = findBoundary();
358 
359             if (boundaryPosition > -1) {
360                 // last chunk
361                 len = boundaryPosition - bufferPosition;
362                 addFieldBytes(len);
363 
364                 if (bodyBytesPos + len >= bodyBytes.length) {
365                     byte[] newBodyBytes = new byte[bodyBytesPos + len];
366                     System.arraycopy(bodyBytes, 0, newBodyBytes, 0, bodyBytesPos);
367                     bodyBytes = newBodyBytes;
368                 }
369                 System.arraycopy(buffer, bufferPosition, bodyBytes, bodyBytesPos, len);
370                 bodyBytesPos += len;
371 
372                 bufferPosition = boundaryPosition + boundary.length;
373                 break;
374             } else {
375                 // not the last chunk
376                 len = Math.min(BUFFER_SIZE, bufferCount) - bufferPosition;
377                 addFieldBytes(len);
378 
379                 if (bodyBytesPos + len >= bodyBytes.length) {
380                     int newSize = bodyBytes.length << 1;
381                     if (newSize < 0 || newSize > MAX_FIELD_BYTES) {
382                         newSize = MAX_FIELD_BYTES;
383                     }
384                     if (newSize < bodyBytesPos + len) {
385                         newSize = bodyBytesPos + BUFFER_SIZE;
386                     }
387 
388                     byte[] newBodyBytes = new byte[newSize];
389                     System.arraycopy(bodyBytes, 0, newBodyBytes, 0, bodyBytesPos);
390                     bodyBytes = newBodyBytes;
391                 }
392                 System.arraycopy(buffer, bufferPosition, bodyBytes, bodyBytesPos, len);
393                 bodyBytesPos += len;
394 
395                 bufferPosition = bufferPosition + len;
396             }
397         }
398 
399         if (bodyBytes.length == bodyBytesPos) {
400             return bodyBytes;
401         }
402 
403         byte[] returnBytes = new byte[bodyBytesPos];
404         System.arraycopy(bodyBytes, 0, returnBytes, 0, returnBytes.length);
405 
406         return returnBytes;
407     }

409     private void addFieldBytes(int len) {
410         fieldBytes += len;
411         if (fieldBytes > MAX_FIELD_BYTES) {
412             throw new CmisInvalidArgumentException("Limit exceeded!");
413         }
414     }

416     private void readBodyAsStream(String contentType, String filename) throws IOException {
417         TempStoreOutputStream stream = streamFactory.newOutputStream();
418         stream.setMimeType(contentType);
419         stream.setFileName(filename);
420 
421         try {
422             while (true) {
423                 readBuffer();
424 
425                 int boundaryPosition = findBoundary();
426 
427                 if (boundaryPosition > -1) {
428                     stream.write(buffer, bufferPosition, boundaryPosition - bufferPosition);
429                     bufferPosition = boundaryPosition + boundary.length;
430                     break;
431                 } else {
432                     int len = Math.min(BUFFER_SIZE, bufferCount) - bufferPosition;
433                     stream.write(buffer, bufferPosition, len);
434                     bufferPosition = bufferPosition + len;
435                 }
436             }
437 
438             stream.close();
439 
440             contentSize = BigInteger.valueOf(stream.getLength());
441             contentStream = stream.getInputStream();
442         } catch (IOException e) {
443             // if something went wrong, make sure the temp file will
444             // be deleted
445             stream.destroy(e);
446             throw e;
447         }
448     }

450     private void readBody() throws IOException {
451         String contentDisposition = headers.get("content-disposition");
452 
453         if (contentDisposition == null) {
454             throw new CmisInvalidArgumentException("Invalid multipart request!");
455         }
456 
457         Map<String, String> params = new HashMap<String, String>();
458         MimeHelper.decodeContentDisposition(contentDisposition, params);
459         boolean isContent = params.containsKey(MimeHelper.DISPOSITION_FILENAME);
460 
461         if (isContent) {
462             if (hasContent) {
463                 throw new CmisInvalidArgumentException("Only one content expected!");
464             }
465 
466             hasContent = true;
467 
468             filename = params.get(MimeHelper.DISPOSITION_FILENAME);
469 
470             if (filename != null) {
471                 // if the browser sent the full path,
472                 // extract the filename segment
473                 int pathsep = filename.lastIndexOf('/');
474                 if (pathsep > -1) {
475                     filename = filename.substring(pathsep + 1);
476                 }
477                 pathsep = filename.lastIndexOf('\\');
478                 if (pathsep > -1) {
479                     filename = filename.substring(pathsep + 1);
480                 }
481 
482                 filename = filename.trim();
483             }
484 
485             contentType = headers.get("content-type");
486             if (contentType == null) {
487                 contentType = Constants.MEDIATYPE_OCTETSTREAM;
488             }
489 
490             readBodyAsStream(contentType, filename);
491         } else {
492             String name = params.get(MimeHelper.DISPOSITION_NAME);
493             byte[] rawValue = readBodyBytes();
494 
495             if (CHARSET_FIELD.equalsIgnoreCase(name)) {
496                 charset = new String(rawValue, IOUtils.ISO_8859_1);
497                 return;
498             }
499 
500             String fieldContentType = headers.get("content-type");
501             if (fieldContentType != null) {
502                 String fieldCharset = MimeHelper.getCharsetFromContentType(fieldContentType);
503                 if (fieldCharset != null) {
504                     addField(name, new String(rawValue, fieldCharset));
505                     return;
506                 }
507             }
508 
509             addRawField(name, rawValue);
510         }
511     }

513     private void skipPreamble() throws IOException {
514         readBuffer();
515 
516         if (bufferCount < boundary.length - 2) {
517             throw new CmisInvalidArgumentException("Invalid multipart request!");
518         }
519 
520         for (int i = 2; i < boundary.length; i++) {
521             if (boundary[i] != buffer[i - 2]) {
522                 break;
523             }
524 
525             if (i == boundary.length - 1) {
526                 bufferPosition = boundary.length - 2;
527                 readBuffer();
528                 return;
529             }
530         }
531 
532         while (true) {
533             int boundaryPosition = findBoundary();
534 
535             if (boundaryPosition > -1) {
536                 bufferPosition = boundaryPosition + boundary.length;
537                 readBuffer();
538                 break;
539             }
540 
541             bufferPosition = BUFFER_SIZE + 1;
542             readBuffer();
543         }
544     }

546     private void skipEpilogue() {
547         try {
548             // read to the end of stream, but max 1 MB
549             int count = 0;
550             byte[] tmpBuf = new byte[4096];
551             int b;
552             while ((b = requestStream.read(tmpBuf)) > -1) {
553                 count += b;
554                 if (count >= 1024 * 1024) {
555                     break;
556                 }
557             }
558         } catch (IOException e) {
559             // ignore
560         }
561     }

563     private boolean readNext() throws IOException {
564         try {
565             readHeaders();
566 
567             // no headers -> end of request
568             if (headers == null) {
569                 skipEpilogue();
570                 return false;
571             }
572 
573             readBody();
574 
575             return true;
576         } catch (IOException e) {
577             IOUtils.closeQuietly(contentStream);
578 
579             skipEpilogue();
580 
581             throw e;
582         }
583     }

585     public void parse() throws IOException {
586         try {
587 
588             while (readNext()) {
589                 // nothing to do here, just read
590             }
591 
592             // apply charset
593             for (Map.Entry<String, byte[][]> e : rawFields.entrySet()) {
594 
595                 String[] otherValues = fields.get(e.getKey());
596                 int index = (otherValues != null ? otherValues.length : 0);
597 
598                 String[] values = new String[e.getValue().length + index];
599 
600                 if (otherValues != null) {
601                     System.arraycopy(otherValues, 0, values, 0, otherValues.length);
602                 }
603 
604                 for (byte[] rawValue : e.getValue()) {
605                     values[index++] = new String(rawValue, charset);
606                 }
607 
608                 fields.put(e.getKey(), values);
609             }
610         } catch (Exception e) {
611             if (contentStream != null) {
612                 IOUtils.closeQuietly(contentStream);
613             }
614 
615             skipEpilogue();
616 
617             fields = null;
618 
619             if (e instanceof UnsupportedEncodingException) {
620                 throw new CmisInvalidArgumentException("Encoding not supported!", e);
621             } else if (e instanceof CmisBaseException) {
622                 throw (CmisBaseException) e;
623             } else if (e instanceof IOException) {
624                 throw (IOException) e;
625             } else {
626                 throw new CmisRuntimeException(e.getMessage(), e);
627             }
628         } finally {
629             rawFields = null;
630         }
631     }

657     /**

659      */
660     public static final boolean isMultipartContent(HttpServletRequest request) {
661         String contentType = request.getContentType();
662 
663         if (contentType != null && contentType.toLowerCase(Locale.ENGLISH).startsWith(MULTIPART)) {
664             return true;
