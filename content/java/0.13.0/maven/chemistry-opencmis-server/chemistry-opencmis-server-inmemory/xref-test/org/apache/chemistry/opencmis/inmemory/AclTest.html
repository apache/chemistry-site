153         final InMemoryAce aceNew = new InMemoryAce("Hugo", Permission.WRITE); // will
154                                                                               // be
155                                                                               // added
156         final InMemoryAce aceRCopy = new InMemoryAce(BERTA, Permission.READ); // is

175         assertEquals(Permission.WRITE, acl1.getPermission("Hugo"));

255     @Test
256     public void testConvertToCmisAcl() {
257         Ace ace = aceN.toCommonsAce();
258         assertEquals(ANDREAS, ace.getPrincipalId());
259         assertEquals(1, ace.getPermissions().size());
260         assertEquals("", ace.getPermissions().get(0));
261 
262         ace = aceR.toCommonsAce();
263         assertEquals(BERTA, ace.getPrincipalId());
264         assertEquals(1, ace.getPermissions().size());
265         assertEquals(EnumBasicPermissions.CMIS_READ.value(), ace.getPermissions().get(0));
266 
267         ace = aceW.toCommonsAce();
268         assertEquals(CHRISTIAN, ace.getPrincipalId());
269         assertEquals(1, ace.getPermissions().size());
270         assertEquals(EnumBasicPermissions.CMIS_WRITE.value(), ace.getPermissions().get(0));
271 
272         ace = aceA.toCommonsAce();
273         assertEquals(DOROTHEE, ace.getPrincipalId());
274         assertEquals(1, ace.getPermissions().size());
275         assertEquals(EnumBasicPermissions.CMIS_ALL.value(), ace.getPermissions().get(0));
276 
277         InMemoryAcl acl = createDefaultAcl();
278         Acl commonsAcl = acl.toCommonsAcl();
279         assertEquals(4, commonsAcl.getAces().size());
280         assertTrue(hasCommonsAce(commonsAcl, ANDREAS, ""));
281         assertFalse(hasCommonsAce(commonsAcl, ANDREAS, EnumBasicPermissions.CMIS_READ.value()));
282         assertFalse(hasCommonsAce(commonsAcl, ANDREAS, EnumBasicPermissions.CMIS_WRITE.value()));
283         assertFalse(hasCommonsAce(commonsAcl, ANDREAS, EnumBasicPermissions.CMIS_ALL.value()));
284         assertTrue(hasCommonsAce(commonsAcl, BERTA, EnumBasicPermissions.CMIS_READ.value()));
285         assertTrue(hasCommonsAce(commonsAcl, CHRISTIAN, EnumBasicPermissions.CMIS_WRITE.value()));
286         assertTrue(hasCommonsAce(commonsAcl, DOROTHEE, EnumBasicPermissions.CMIS_ALL.value()));
287     }

330     private Ace createAce(String principalId, String permission) {
331         AccessControlEntryImpl ace = new AccessControlEntryImpl(new AccessControlPrincipalDataImpl(principalId),
332                 Arrays.asList(new String[] { permission }));
333         return ace;
334     }

336     private boolean hasCommonsAce(Acl acl, String principalId, String permission) {
337         for (Ace ace : acl.getAces()) {
338             if (ace.getPrincipalId().equals(principalId) && ace.getPermissions().get(0).equals(permission)) {
339                 return true;
340             }
341         }
342         return false;
