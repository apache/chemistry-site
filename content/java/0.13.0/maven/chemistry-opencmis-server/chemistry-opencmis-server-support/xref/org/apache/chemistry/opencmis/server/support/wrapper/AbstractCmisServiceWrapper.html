52  /**
53   * An abstract CMIS service wrapper.
54   * <p>
55   * All service wrappers managed by {@link CmisServiceWrapperManager} must be
56   * derived from this class and must provide a constructor that takes a
57   * {@link CmisService} object as the sole parameter.
58   */
59  public abstract class AbstractCmisServiceWrapper implements CallContextAwareCmisService, ProgressControlCmisService {
60  
61      private CmisService service;
62      private CallContext context;
63  
64      public AbstractCmisServiceWrapper(CmisService service) {
65          if (service == null) {
66              throw new IllegalArgumentException("Service must be set!");
67          }
68  
69          this.service = service;
70      }
71  
72      /**
73       * Initializes the wrapper with a set of parameters.
74       */
75      public void initialize(Object[] params) {
76      }
77  
78      /**
79       * Returns the wrapped service or the next service wrapper.
80       * 
81       * @return the wrapped service
82       */
83      public CmisService getWrappedService() {
84          return service;
85      }
86  
87      /**
88       * Sets the call context and propagates it down to the next service wrapper
89       * or service if it implements the {@link CallContextAwareCmisService}
90       * interface.
91       */
92      public void setCallContext(CallContext callContext) {
93          this.context = callContext;
94  
95          if (service instanceof CallContextAwareCmisService) {
96              ((CallContextAwareCmisService) service).setCallContext(callContext);
97          }
98      }
99  
100     /**
101      * Gets the current call context.
102      */
103     public CallContext getCallContext() {
104         return context;
105     }

110         if (service instanceof ProgressControlCmisService) {
111             return ((ProgressControlCmisService) service).beforeServiceCall();
112         }

114         return ProgressControlCmisService.Progress.CONTINUE;
115     }

117     public ProgressControlCmisService.Progress afterServiceCall() {
118         if (service instanceof ProgressControlCmisService) {
119             return ((ProgressControlCmisService) service).afterServiceCall();
120         }
121 
122         return ProgressControlCmisService.Progress.CONTINUE;
123     }
