47  /**
48   * OAuth 2.0 Authentication Provider.
49   * <p>
50   * This authentication provider implements OAuth 2.0 (RFC 6749) Bearer Tokens
51   * (RFC 6750).
52   * <p>
53   * The provider can be either configured with an authorization code or with an
54   * existing bearer token. Token endpoint and client ID are always required. If a
55   * client secret is required depends on the authorization server.
56   * <p>
57   * Configuration with authorization code:
58   * 
59   * <pre>
60   * SessionFactory factory = ...
61   * 
62   * Map&lt;String, String> parameter = new HashMap&lt;String, String>();
63   * 
64   * parameter.put(SessionParameter.ATOMPUB_URL, "http://localhost/cmis/atom");
65   * parameter.put(SessionParameter.BINDING_TYPE, BindingType.ATOMPUB.value());
66   * parameter.put(SessionParameter.REPOSITORY_ID, "myRepository");
67   * 
68   * parameter.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS, "org.apache.chemistry.opencmis.client.bindings.spi.OAuthAuthenticationProvider");
69   * 
70   * parameter.put(SessionParameter.OAUTH_TOKEN_ENDPOINT, "https://example.com/auth/oauth/token");
71   * parameter.put(SessionParameter.OAUTH_CLIENT_ID, "s6BhdRkqt3");
72   * parameter.put(SessionParameter.OAUTH_CLIENT_SECRET, "7Fjfp0ZBr1KtDRbnfVdmIw");
73   * 
74   * parameter.put(SessionParameter.OAUTH_CODE, "abc");
75   * 
76   * ...
77   * Session session = factory.createSession(parameter);
78   * </pre>
79   * 
80   * <p>
81   * Configuration with existing bearer token:
82   * 
83   * <pre>
84   * SessionFactory factory = ...
85   * 
86   * Map&lt;String, String> parameter = new HashMap&lt;String, String>();
87   * 
88   * parameter.put(SessionParameter.ATOMPUB_URL, "http://localhost/cmis/atom");
89   * parameter.put(SessionParameter.BINDING_TYPE, BindingType.ATOMPUB.value());
90   * parameter.put(SessionParameter.REPOSITORY_ID, "myRepository");
91   * 
92   * parameter.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS, "org.apache.chemistry.opencmis.client.bindings.spi.OAuthAuthenticationProvider");
93   *  
94   * parameter.put(SessionParameter.OAUTH_TOKEN_ENDPOINT, "https://example.com/auth/oauth/token");
95   * parameter.put(SessionParameter.OAUTH_CLIENT_ID, "s6BhdRkqt3");
96   * parameter.put(SessionParameter.OAUTH_CLIENT_SECRET, "7Fjfp0ZBr1KtDRbnfVdmIw");
97   * 
98   * parameter.put(SessionParameter.OAUTH_ACCESS_TOKEN, "2YotnFZFEjr1zCsicMWpAA");
99   * parameter.put(SessionParameter.OAUTH_REFRESH_TOKEN, "tGzv3JOkF0XG5Qx2TlKWIA");
100  * parameter.put(SessionParameter.OAUTH_EXPIRATION_TIMESTAMP, "1388237075127");
101  * 
102  * ...
103  * Session session = factory.createSession(parameter);
104  * </pre>
105  * 
106  * <p>
107  * Getting tokens at runtime:
108  * 
109  * <pre>
110  * OAuthAuthenticationProvider authProvider = (OAuthAuthenticationProvider) session.getBinding()
111  *         .getAuthenticationProvider();
112  * 
113  * // get the current token
114  * Token token = authProvider.getToken();
115  * 
116  * // listen for token refreshes
117  * authProvider.addTokenListener(new OAuthAuthenticationProvider.TokenListener() {
118  *     public void tokenRefreshed(Token token) {
119  *         // do something with the new token
120  *     }
121  * });
122  * </pre>
123  * 
124  * <p>
125  * OAuth errors can be handled like this:
126  * 
127  * <pre>
128  *  try {
129  *      ...
130  *      // CMIS calls
131  *      ...
132  *  } catch (CmisConnectionException connEx) {
133  *      if (connEx.getCause() instanceof CmisOAuthException) {
134  *          CmisOAuthException oauthEx = (CmisOAuthException) connEx.getCause();
135  * 
136  *          if (CmisOAuthException.ERROR_INVALID_GRANT.equals(oauthEx.getError()) ||
137  *              CmisOAuthException.ERROR_INVALID_TOKEN.equals(oauthEx.getError())) {
138  *              // ask the user to authenticate again
139  *          } else {
140  *             // a configuration or server problem
141  *          }
142  *    }
143  * }
144  * </pre>
145  */

154     private Token token = null;
155     private long defaultTokenLifetime = 3600;

158     @Override
159     public void setSession(BindingSession session) {
160         super.setSession(session);
161 
162         if (token == null) {
163             // get predefined access token
164             String accessToken = null;
165             if (session.get(SessionParameter.OAUTH_ACCESS_TOKEN) instanceof String) {
166                 accessToken = (String) session.get(SessionParameter.OAUTH_ACCESS_TOKEN);
167             }
168 
169             // get predefined refresh token
170             String refreshToken = null;
171             if (session.get(SessionParameter.OAUTH_REFRESH_TOKEN) instanceof String) {
172                 refreshToken = (String) session.get(SessionParameter.OAUTH_REFRESH_TOKEN);
173             }
174 
175             // get predefined expiration timestamp
176             long expirationTimestamp = 0;
177             if (session.get(SessionParameter.OAUTH_EXPIRATION_TIMESTAMP) instanceof String) {
178                 try {
179                     expirationTimestamp = Long.parseLong((String) session
180                             .get(SessionParameter.OAUTH_EXPIRATION_TIMESTAMP));
181                 } catch (NumberFormatException nfe) {
182                     // ignore
183                 }
184             } else if (session.get(SessionParameter.OAUTH_EXPIRATION_TIMESTAMP) instanceof Number) {
185                 expirationTimestamp = ((Number) session.get(SessionParameter.OAUTH_EXPIRATION_TIMESTAMP)).longValue();
186             }
187 
188             // get default token lifetime
189             if (session.get(SessionParameter.OAUTH_DEFAULT_TOKEN_LIFETIME) instanceof String) {
190                 try {
191                     defaultTokenLifetime = Long.parseLong((String) session
192                             .get(SessionParameter.OAUTH_DEFAULT_TOKEN_LIFETIME));
193                 } catch (NumberFormatException nfe) {
194                     // ignore
195                 }
196             } else if (session.get(SessionParameter.OAUTH_DEFAULT_TOKEN_LIFETIME) instanceof Number) {
197                 defaultTokenLifetime = ((Number) session.get(SessionParameter.OAUTH_DEFAULT_TOKEN_LIFETIME))
198                         .longValue();
199             }
200 
201             token = new Token(accessToken, refreshToken, expirationTimestamp);
202             fireTokenListner(token);
203         }
204     }

238     public void addTokenListener(TokenListener listner) {
239         if (listner == null) {
240             return;
241         }
242 
243         lock.writeLock().lock();
244         try {
245             if (tokenListeners == null) {
246                 tokenListeners = new ArrayList<OAuthAuthenticationProvider.TokenListener>();
247             }
248 
249             tokenListeners.add(listner);
250         } finally {
251             lock.writeLock().unlock();
252         }
253     }

258      * @param listner

262         if (listner == null) {
263             return;
264         }
265 
266         lock.writeLock().lock();
267         try {
268             if (tokenListeners != null) {
269                 tokenListeners.remove(listner);
270             }
271         } finally {
272             lock.writeLock().unlock();
273         }

276     /**
277      * Lets all token listeners know that there is a new token.
278      */
279     protected void fireTokenListner(Token token) {
280         if (tokenListeners == null) {
281             return;
282         }
283 
284         for (TokenListener listner : tokenListeners) {
285             listner.tokenRefreshed(token);
286         }

295     /**
296      * Gets the access token. If no access token is present or the access token
297      * is expired, a new token is requested.
298      * 
299      * @return the access token
300      */
301     protected String getAccessToken() {
302         lock.writeLock().lock();
303         try {
304             if (token.getAccessToken() == null) {
305                 if (token.getRefreshToken() == null) {
306                     requestToken();
307                 } else {
308                     refreshToken();
309                 }
310             } else if (token.isExpired()) {
311                 refreshToken();
312             }
313 
314             return token.getAccessToken();
315         } catch (CmisConnectionException ce) {
316             throw ce;
317         } catch (Exception e) {
318             throw new CmisConnectionException("Cannot get OAuth access token: " + e.getMessage(), e);
319         } finally {
320             lock.writeLock().unlock();
321         }
322     }

324     private void requestToken() throws IOException {
325         if (LOG.isDebugEnabled()) {
326             LOG.debug("Requesting new OAuth access token.");

337         if (LOG.isDebugEnabled()) {

342

348     private void makeRequest(boolean isRefresh) throws IOException {
349         Object tokenEndpoint = getSession().get(SessionParameter.OAUTH_TOKEN_ENDPOINT);
350         if (!(tokenEndpoint instanceof String)) {
351             throw new CmisConnectionException("Token endpoint not set!");
352         }
353 
354         if (isRefresh && token.getRefreshToken() == null) {
355             throw new CmisConnectionException("No refresh token!");
356         }
357 
358         // request token
359         HttpURLConnection conn = (HttpURLConnection) (new URL(tokenEndpoint.toString())).openConnection();
360         conn.setRequestMethod("POST");
361         conn.setDoInput(true);
362         conn.setDoOutput(true);
363         conn.setAllowUserInteraction(false);
364         conn.setUseCaches(false);
365         conn.setRequestProperty("User-Agent", ClientVersion.OPENCMIS_CLIENT);
366         conn.setRequestProperty("Content-Type", "application/x-www-form-urlencoded; charset=UTF-8");
367 
368         // compile request
369         Writer writer = new OutputStreamWriter(conn.getOutputStream(), IOUtils.UTF8);
370 
371         if (isRefresh) {
372             writer.write("grant_type=refresh_token");
373 
374             writer.write("&refresh_token=");
375             writer.write(IOUtils.encodeURL(token.getRefreshToken()));
376         } else {
377             writer.write("grant_type=authorization_code");
378 
379             Object code = getSession().get(SessionParameter.OAUTH_CODE);
380             if (code != null) {
381                 writer.write("&code=");
382                 writer.write(IOUtils.encodeURL(code.toString()));
383             }
384 
385             Object redirectUri = getSession().get(SessionParameter.OAUTH_REDIRECT_URI);
386             if (redirectUri != null) {
387                 writer.write("&redirect_uri=");
388                 writer.write(IOUtils.encodeURL(redirectUri.toString()));
389             }
390         }
391 
392         Object clientId = getSession().get(SessionParameter.OAUTH_CLIENT_ID);
393         if (clientId != null) {
394             writer.write("&client_id=");
395             writer.write(IOUtils.encodeURL(clientId.toString()));
396         }
397 
398         Object clientSecret = getSession().get(SessionParameter.OAUTH_CLIENT_SECRET);
399         if (clientSecret != null) {
400             writer.write("&client_secret=");
401             writer.write(IOUtils.encodeURL(clientSecret.toString()));
402         }
403 
404         writer.flush();
405 
406         // connect
407         conn.connect();
408 
409         // check success
410         if (conn.getResponseCode() != 200) {
411             JSONObject jsonResponse = parseResponse(conn);
412 
413             Object error = jsonResponse.get("error");
414             String errorStr = (error == null ? null : error.toString());
415 
416             Object description = jsonResponse.get("error_description");
417             String descriptionStr = (description == null ? null : description.toString());
418 
419             Object uri = jsonResponse.get("error_uri");
420             String uriStr = (uri == null ? null : uri.toString());
421 
422             if (LOG.isDebugEnabled()) {
423                 LOG.debug("OAuth token request failed: {}", jsonResponse.toJSONString());
424             }
425 
426             throw new CmisOAuthException("OAuth token request failed" + (errorStr == null ? "" : ": " + errorStr)
427                     + (descriptionStr == null ? "" : ": " + descriptionStr), errorStr, descriptionStr, uriStr);
428         }
429 
430         // parse response
431         JSONObject jsonResponse = parseResponse(conn);
432 
433         Object tokenType = jsonResponse.get("token_type");
434         if (!(tokenType instanceof String) || !"bearer".equalsIgnoreCase((String) tokenType)) {
435             throw new CmisOAuthException("Unsupported OAuth token type: " + tokenType);
436         }
437 
438         Object jsonAccessToken = jsonResponse.get("access_token");
439         if (!(jsonAccessToken instanceof String)) {
440             throw new CmisOAuthException("Invalid OAuth access_token!");
441         }
442 
443         Object jsonRefreshToken = jsonResponse.get("refresh_token");
444         if (jsonRefreshToken != null && !(jsonRefreshToken instanceof String)) {
445             throw new CmisOAuthException("Invalid OAuth refresh_token!");
446         }
447 
448         long expiresIn = defaultTokenLifetime;
449         Object jsonExpiresIn = jsonResponse.get("expires_in");
450         if (jsonExpiresIn != null) {
451             if (jsonExpiresIn instanceof Number) {
452                 expiresIn = ((Number) jsonExpiresIn).longValue();
453             } else if (jsonExpiresIn instanceof String) {
454                 try {
455                     expiresIn = Long.parseLong((String) jsonExpiresIn);
456                 } catch (NumberFormatException nfe) {
457                     throw new CmisOAuthException("Invalid OAuth expires_in value!");
458                 }
459             } else {
460                 throw new CmisOAuthException("Invalid OAuth expires_in value!");
461             }
462 
463             if (expiresIn <= 0) {
464                 expiresIn = defaultTokenLifetime;
465             }
466         }
467 
468         token = new Token(jsonAccessToken.toString(), (jsonRefreshToken == null ? null : jsonRefreshToken.toString()),
469                 expiresIn * 1000 + System.currentTimeMillis());
470         fireTokenListner(token);
471     }

473     private JSONObject parseResponse(HttpURLConnection conn) {
474         Reader reader = null;
475         try {
476             InputStream stream = null;
477 
478             int respCode = conn.getResponseCode();
479             if (respCode == 401) {
480                 Map<String, Map<String, String>> challenges = MimeHelper.getChallengesFromAuthenticateHeader(conn
481                         .getHeaderField("WWW-Authenticate"));
482 
483                 if (challenges != null && challenges.containsKey("bearer")) {
484                     Map<String, String> params = challenges.get("bearer");
485 
486                     String errorStr = params.get("error");
487                     String descriptionStr = params.get("error_description");
488                     String uriStr = params.get("error_uri");
489 
490                     if (LOG.isDebugEnabled()) {
491                         LOG.debug("Invalid OAuth token: {}", params.toString());
492                     }
493 
494                     throw new CmisOAuthException("Unauthorized" + (errorStr == null ? "" : ": " + errorStr)
495                             + (descriptionStr == null ? "" : ": " + descriptionStr), errorStr, descriptionStr, uriStr);
496                 }
497 
498                 throw new CmisOAuthException("Unauthorized!");
499             }
500 
501             if (respCode >= 200 && respCode < 300) {
502                 stream = conn.getInputStream();
503             } else {
504                 stream = conn.getErrorStream();
505             }
506             if (stream == null) {
507                 throw new CmisOAuthException("Invalid OAuth token response!");
508             }
509 
510             reader = new InputStreamReader(stream, extractCharset(conn));
511             JSONParser parser = new JSONParser();
512             Object response = parser.parse(reader);
513 
514             if (!(response instanceof JSONObject)) {
515                 throw new CmisOAuthException("Invalid OAuth token response!");
516             }
517 
518             return (JSONObject) response;
519         } catch (CmisConnectionException ce) {
520             throw ce;
521         } catch (Exception pe) {
522             throw new CmisOAuthException("Parsing the OAuth token response failed: " + pe.getMessage(), pe);
523         } finally {
524             IOUtils.consumeAndClose(reader);
525         }
526     }

528     private String extractCharset(HttpURLConnection conn) {
529         String charset = IOUtils.UTF8;
530 
531         String contentType = conn.getContentType();
532         if (contentType != null) {
533             String[] parts = contentType.split(";");
534             for (int i = 1; i < parts.length; i++) {
535                 String part = parts[i].trim().toLowerCase(Locale.ENGLISH);
536                 if (part.startsWith("charset")) {
537                     int x = part.indexOf('=');
538                     charset = part.substring(x + 1).trim();
539                     break;
540                 }
541             }
542         }
543 
544         return charset;
545     }

547     /**
548      * Token holder class.

550     public static class Token {
551         private String accessToken;
552         private String refreshToken;
553         private long expirationTimestamp;
554 
555         public Token(String accessToken, String refreshToken, long expirationTimestamp) {
556             this.accessToken = accessToken;
557             this.refreshToken = refreshToken;
558             this.expirationTimestamp = expirationTimestamp;
559         }
560 
561         /**
562          * Returns the access token.
563          * 
564          * @return the access token
565          */
566         public String getAccessToken() {
567             return accessToken;
568         }
569 
570         /**
571          * Returns the refresh token.
572          * 
573          * @return the refresh token
574          */
575         public String getRefreshToken() {
576             return refreshToken;
577         }
578 
579         /**
580          * Returns the timestamp when the access expires.
581          * 
582          * @return the timestamp in milliseconds since midnight, January 1, 1970
583          *         UTC.
584          */
585         public long getExpirationTimestamp() {
586             return expirationTimestamp;
587         }
588 
589         /**
590          * Returns whether the access token is expired or not.
591          * 
592          * @return {@code true} if the access token is expired, {@code false}
593          *         otherwise
594          */
595         public boolean isExpired() {
596             return System.currentTimeMillis() >= expirationTimestamp;
597         }
598 
599         @Override
600         public String toString() {
601             return "Access token: " + accessToken + " / Refresh token: " + refreshToken + " / Expires : "
602                     + expirationTimestamp;
603         }
