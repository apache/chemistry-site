168     @Override
169     public Map<String, String> getSessionParameters() {
170         int row = authTable.getSelectedRow();
171         if (row < 0) {
172             return Collections.emptyMap();
173         }
174 
175         // compile session parameters
176         CmisAuthentication auth = (CmisAuthentication) authTable.getValueAt(row,
177                 authTable.convertColumnIndexToView(DETAILS_COLUMN));
178         Map<String, String> parameters = reader.pepareSessionParameters(auth);
179 
180         // add other required parameters
181         if (CmisAuthentication.AUTH_BASIC.equals(auth.getType())
182                 || CmisAuthentication.AUTH_USERNAME_TOKEN.equals(auth.getType())
183                 || CmisAuthentication.AUTH_NTLM.equals(auth.getType())) {
184             // these authentication methods need a user and password
185             parameters.put(SessionParameter.USER, "");
186             parameters.put(SessionParameter.PASSWORD, "");
187         } else if (CmisAuthentication.AUTH_OAUTH.equals(auth.getType())) {
188             // OAuth need some extra parameters
189             parameters.put(SessionParameter.OAUTH_TOKEN_ENDPOINT, "");
190             parameters.put(SessionParameter.OAUTH_CLIENT_ID, "");
191             parameters.put(SessionParameter.OAUTH_CLIENT_SECRET, "");
192             parameters.put(SessionParameter.OAUTH_CODE, "");
193             parameters.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS, OAuthAuthenticationProvider.class.getName());
194         } else if (CmisAuthentication.AUTH_CERT.equals(auth.getType())) {
195             // client cert parameters
196             parameters.put(SessionParameter.CLIENT_CERT_KEYFILE, "");
197             parameters.put(SessionParameter.CLIENT_CERT_PASSPHRASE, "");
198             parameters.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS,
199                     ClientCertificateAuthenticationProvider.class.getName());
200         } else if (!CmisAuthentication.AUTH_NONE.equals(auth.getType())
201                 && !parameters.containsKey(SessionParameter.AUTHENTICATION_PROVIDER_CLASS)) {
202             // a custom authentication provider is required here
203             parameters.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS, "");
204         }
205 
206         return parameters;
207     }

341     static class CmisAuthenticationModel extends AbstractTableModel {
342 
343         private static final long serialVersionUID = 1L;
344 
345         private CmisEndpointsDocument endpointDocument;
346         private List<CmisAuthentication> authentications;
347 
348         public CmisAuthenticationModel() {
349             this.endpointDocument = new CmisEndpointsDocumentImpl();
350             this.authentications = Collections.emptyList();
351         }
352 
353         public CmisEndpointsDocument getCmisEndpointsDocument() {
354             return endpointDocument;
355         }
356 
357         public void setCmisEndpointDocument(CmisEndpointsDocument endpointDocument) {
358             this.endpointDocument = endpointDocument;
359             this.authentications = endpointDocument.getAuthenticationsSortedByPreference();
360             fireTableDataChanged();
361         }
362 
363         @Override
364         public String getColumnName(int columnIndex) {
365             return COLUMN_NAMES[columnIndex];
366         }
367 
368         @Override
369         public int getColumnCount() {
370             return COLUMN_NAMES.length;
371         }
372 
373         @Override
374         public int getRowCount() {
375             return authentications.size();
376         }
377 
378         @Override
379         public Object getValueAt(int rowIndex, int columnIndex) {
380             CmisAuthentication auth = authentications.get(rowIndex);
381 
382             switch (columnIndex) {
383             case 0:
384                 return auth.getPreference();
385             case 1:
386                 return auth.getEndpoint().getCmisVersion() == null ? "?" : auth.getEndpoint().getCmisVersion();
387             case 2:
388                 return auth.getEndpoint().getBinding() == null ? "?" : auth.getEndpoint().getBinding();
389             case 3:
390                 return auth.getType() == null ? "?" : auth.getType();
391             case 4:
392                 return auth;
393             default:
394                 return "";
395             }
396         }
397 
398         @Override
399         public Class<?> getColumnClass(int columnIndex) {
400             switch (columnIndex) {
401             case 0:
402                 return Integer.class;
403             case 4:
404                 return CmisAuthentication.class;
405             default:
406                 return String.class;
407             }
408         }

425             if (height > (getFontMetrics(getFont()).getHeight() + getInsets().bottom + getInsets().top)) {
426                 if (table.getRowHeight(row) != height) {

434         @Override

436             StringBuilder text = new StringBuilder(128);

438             if (value instanceof CmisAuthentication) {
439                 CmisAuthentication auth = (CmisAuthentication) value;

454                     text.append(auth.getEndpoint().getDisplayName());
455                 } else {
456                     text.append("???");
457                 }
458 
459                 if (auth.getEndpoint().getUrl() != null) {
460                     text.append("<br>Endpoint URL: ");
461                     text.append(auth.getEndpoint().getUrl());
462                 } else if (auth.getEndpoint().getRepositoryServiceWdsl() != null) {

465                 }
