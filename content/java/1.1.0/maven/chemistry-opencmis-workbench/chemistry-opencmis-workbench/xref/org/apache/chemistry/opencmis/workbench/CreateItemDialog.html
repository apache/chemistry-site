47  public class CreateItemDialog extends CreateDialog {
48  
49      private static final long serialVersionUID = 1L;
50  
51      private JRadioButton unfiledButton;
52      private JRadioButton currentPathButton;
53      private JTextField nameField;
54      private JComboBox<ObjectTypeItem> typeBox;
55  
56      public CreateItemDialog(Frame owner, ClientModel model) {
57          super(owner, "Create Item", model);
58          createGUI();
59      }
60  
61      private void createGUI() {
62          final CreateItemDialog thisDialog = this;
63  
64          boolean hasCurrentFolder = getClientModel().getCurrentFolder() != null;
65  
66          unfiledButton = new JRadioButton("create unfiled");
67          unfiledButton.setSelected(!hasCurrentFolder);
68  
69          currentPathButton = new JRadioButton("create in the current folder: "
70                  + (hasCurrentFolder ? getClientModel().getCurrentFolder().getPath() : ""));
71          currentPathButton.setSelected(hasCurrentFolder);
72          currentPathButton.setEnabled(hasCurrentFolder);
73  
74          ButtonGroup filedGroup = new ButtonGroup();
75          filedGroup.add(unfiledButton);
76          filedGroup.add(currentPathButton);
77  
78          JPanel buttonPanel = new JPanel(new FlowLayout(FlowLayout.LEFT));
79          buttonPanel.add(unfiledButton);
80          buttonPanel.add(currentPathButton);
81  
82          createRow("", buttonPanel, 0);
83  
84          nameField = new JTextField(60);
85          createRow("Name:", nameField, 1);
86  
87          ObjectTypeItem[] types = getTypes(BaseTypeId.CMIS_ITEM.value());
88          if (types.length == 0) {
89              JOptionPane.showMessageDialog(this, "No creatable type!", "Creatable Types", JOptionPane.ERROR_MESSAGE);
90              thisDialog.dispose();
91              return;
92          }
93  
94          typeBox = new JComboBox<ObjectTypeItem>(types);
95          typeBox.setSelectedIndex(0);
96          typeBox.addItemListener(new ItemListener() {
97              @Override
98              public void itemStateChanged(ItemEvent e) {
99                  TypeDefinition type = ((ObjectTypeItem) typeBox.getSelectedItem()).getObjectType();
100                 updateMandatoryOrOnCreateFields(type);
101             }
102         });
103 
104         ObjectTypeItem type = (ObjectTypeItem) typeBox.getSelectedItem();
105         updateMandatoryOrOnCreateFields(type.getObjectType());
106 
107         createRow("Type:", typeBox, 2);
108 
109         JButton createButton = new JButton("Create Item", new NewItemIcon(ClientHelper.BUTTON_ICON_SIZE,
110                 ClientHelper.BUTTON_ICON_SIZE));
111         createButton.addActionListener(new ActionListener() {
112             @Override
113             public void actionPerformed(ActionEvent event) {
114                 String name = nameField.getText();
115                 ObjectType type = ((ObjectTypeItem) typeBox.getSelectedItem()).getObjectType();
116 
117                 try {
118                     setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
119 
120                     ObjectId objectId = getClientModel().createItem(name, type.getId(),
121                             getMandatoryOrOnCreatePropertyValues(type), unfiledButton.isSelected());
122 
123                     if (objectId != null) {
124                         LoadObjectWorker.loadObject(getOwner(), getClientModel(), objectId.getId());
125                     }
126 
127                     thisDialog.setVisible(false);
128                     thisDialog.dispose();
129                 } catch (Exception e) {
130                     ClientHelper.showError(null, e);
131                 } finally {
132                     setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
133                     LoadFolderWorker.reloadFolder(getOwner(), getClientModel());
134                 }
135             }
136         });
137         createActionRow("", createButton, 3);
138 
139         getRootPane().setDefaultButton(createButton);
140 
141         showDialog();
142     }
