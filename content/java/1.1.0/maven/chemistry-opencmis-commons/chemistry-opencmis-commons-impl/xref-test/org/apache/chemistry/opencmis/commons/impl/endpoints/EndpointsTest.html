155     @Test
156     public void testWriteRead() throws Exception {
157         CmisEndpointImpl epA1 = new CmisEndpointImpl(CmisEndpoint.VERSION_1_0, CmisEndpoint.BINDING_ATOMPUB);
158         epA1.put(CmisEndpoint.KEY_DISPLAY_NAME, "My Endpoint 1");
159         epA1.put(CmisEndpoint.KEY_URL, "http://server1/cmis1");
160         epA1.put(CmisEndpoint.KEY_COOKIES, CmisEndpoint.COOKIES_REQUIRED);
161         epA1.put(CmisEndpoint.KEY_COMPRESSION, CmisEndpoint.COMPRESSION_BOTH);
162         epA1.put(CmisEndpoint.KEY_CSRF_HEADER, "X-CSRF");
163 
164         CmisEndpointImpl epA2 = new CmisEndpointImpl(CmisEndpoint.VERSION_1_1, CmisEndpoint.BINDING_BROWSER);
165         epA2.put(CmisEndpoint.KEY_DISPLAY_NAME, "My Endpoint 2");
166         epA2.put(CmisEndpoint.KEY_URL, "http://server2/cmis2");
167 
168         CmisEndpointImpl epA3 = new CmisEndpointImpl(CmisEndpoint.VERSION_1_1, CmisEndpoint.BINDING_WEBSERVICES);
169         epA3.put(CmisEndpoint.KEY_SOAP_VERSION, CmisEndpoint.SOAP_VERSION_1_1);
170 
171         List<CmisEndpoint> endpoints = new ArrayList<CmisEndpoint>();
172         endpoints.add(epA1);
173         endpoints.add(epA2);
174         endpoints.add(epA3);
175 
176         CmisEndpointsDocumentImpl docA = new CmisEndpointsDocumentImpl(endpoints);
177         assertEquals(3, docA.getEndpoints().size());
178 
179         // -- write --
180         String json = CmisEndpointsDocumentHelper.write(docA);
181         assertNotNull(json);
182 
183         // -- read --
184         CmisEndpointsDocument docB = CmisEndpointsDocumentHelper.read(json);
185         assertNotNull(docB);
186 
187         // -- check --
188         assertEqualsDocument(docA, docB);
189     }
