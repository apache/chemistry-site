34      @Test
35      public void testXmlDateTimeParser() {
36          GregorianCalendar cal1 = DateTimeHelper.parseXmlDateTime("2012-12-24T09:15:06.123Z");
37          assertEquals(2012, cal1.get(Calendar.YEAR));
38          assertEquals(11, cal1.get(Calendar.MONTH));
39          assertEquals(24, cal1.get(Calendar.DAY_OF_MONTH));
40          assertEquals(9, cal1.get(Calendar.HOUR_OF_DAY));
41          assertEquals(15, cal1.get(Calendar.MINUTE));
42          assertEquals(6, cal1.get(Calendar.SECOND));
43          assertEquals(123, cal1.get(Calendar.MILLISECOND));
44          assertEquals(0, cal1.getTimeZone().getRawOffset());
45  
46          GregorianCalendar cal2 = DateTimeHelper.parseXmlDateTime("2013-02-04T23:45:55.9876543");
47          assertEquals(2013, cal2.get(Calendar.YEAR));
48          assertEquals(1, cal2.get(Calendar.MONTH));
49          assertEquals(4, cal2.get(Calendar.DAY_OF_MONTH));
50          assertEquals(23, cal2.get(Calendar.HOUR_OF_DAY));
51          assertEquals(45, cal2.get(Calendar.MINUTE));
52          assertEquals(55, cal2.get(Calendar.SECOND));
53          assertEquals(987, cal2.get(Calendar.MILLISECOND));
54          assertEquals(0, cal2.getTimeZone().getRawOffset());
55  
56          GregorianCalendar cal3 = DateTimeHelper.parseXmlDateTime("2013-01-02T03:04:05.678+05:00");
57          assertEquals(2013, cal3.get(Calendar.YEAR));
58          assertEquals(0, cal3.get(Calendar.MONTH));
59          assertEquals(2, cal3.get(Calendar.DAY_OF_MONTH));
60          assertEquals(3, cal3.get(Calendar.HOUR_OF_DAY));
61          assertEquals(4, cal3.get(Calendar.MINUTE));
62          assertEquals(5, cal3.get(Calendar.SECOND));
63          assertEquals(678, cal3.get(Calendar.MILLISECOND));
64          assertEquals(5 * 60 * 60 * 1000, cal3.getTimeZone().getRawOffset());
65      }

152     @Test
153     public void testHttpDateTimeParser() {
154         GregorianCalendar cal1 = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
155         cal1.set(2012, 11, 24, 9, 10, 11);
156         cal1.set(Calendar.MILLISECOND, 0);
157 
158         String dateStr1 = DateTimeHelper.formatHttpDateTime(cal1.getTime());
159         assertEquals("Mon, 24 Dec 2012 09:10:11 GMT", dateStr1);
160 
161         Date date1 = DateTimeHelper.parseHttpDateTime(dateStr1);
162 
163         assertEquals(cal1.getTime(), date1);
164 
165         GregorianCalendar cal2 = new GregorianCalendar(TimeZone.getTimeZone("GMT+05:00"));
166         cal2.set(2013, 0, 1, 2, 3, 4);
167         cal2.set(Calendar.MILLISECOND, 0);
168 
169         String dateStr2 = "  '" + DateTimeHelper.formatHttpDateTime(cal2.getTimeInMillis()) + "' ";
170         Date date2 = DateTimeHelper.parseHttpDateTime(dateStr2);
171 
172         assertEquals(cal2.getTime(), date2);
173 
174         GregorianCalendar cal3 = new GregorianCalendar(TimeZone.getTimeZone("GMT+05:00"));
175         cal3.set(2012, 11, 24, 9, 10, 11);
176         cal3.set(Calendar.MILLISECOND, 0);

181         assertEquals(cal3.getTime(), date3);
