70      static <T> PropertyValidator<T> createPropertyValidator(PropertyDefinition<?> propDef) {
71          PropertyValidator<T> result = null;
72          if (propDef instanceof PropertyIntegerDefinition) {
73              result = (PropertyValidator<T>) new PropertyValidatorInteger();
74          } else if (propDef instanceof PropertyDecimalDefinition) {
75              result = (PropertyValidator<T>) new PropertyValidatorDecimal();
76          } else if (propDef instanceof PropertyStringDefinition) {
77              result = (PropertyValidator<T>) new PropertyValidatorString();
78          } else {
79              result = new PropertyValidator<T>();
80          }
81          return result;
82      }

84      /*
85       * property validations: not readonly, all required are given, all are known
86       * in type cardinality: no multi values for single value, def min max check
87       * for Integer and Decimal, choices and in list Strings, max length set
88       * default value for omitted properties

90      static class PropertyValidator<T> {
91  
92          public void validate(PropertyDefinition<T> propDef, PropertyData<T> prop) {
93  
94              // check general constraints for all property types
95              if (propDef.getCardinality() == Cardinality.SINGLE) {
96                  if (prop.getValues() != null && prop.getValues().size() > 1) {
97                      throw new CmisConstraintException("The property with id " + propDef.getId()
98                              + " is single valued, but multiple values are passed " + prop.getValues());
99                  }
100             }
101 
102             if (isNotEmpty(propDef.getChoices())) {
103                 validateChoices(propDef, prop);
104             }
105         }
106 
107         private void validateChoices(PropertyDefinition<T> propDef, PropertyData<T> prop) {
108             boolean isAllowedValue = true;
109             boolean hasMultiValueChoiceLists = false;
110             for (Choice<?> allowedValue : propDef.getChoices()) {
111                 if (allowedValue.getValue() != null && allowedValue.getValue().size() > 1) {
112                     hasMultiValueChoiceLists = true;
113                 }
114             }
115 
116             if (propDef.isOpenChoice() != null && propDef.isOpenChoice()) {
117             	isAllowedValue = true;
118             } else if (hasMultiValueChoiceLists) {
119                 // check if value is in list
120 
121             	// do a complex check if this combination of actual values is
122                 // allowed check if value is in list
123                 isAllowedValue = false;
124                 List<?> actualValues = prop.getValues();
125                 for (Choice<?> allowedValue : propDef.getChoices()) {
126                     if (allowedValue.getValue().size() == actualValues.size()) {
127                         boolean listValuesAreEqual = true;
128                         Iterator<?> it = allowedValue.getValue().iterator();
129                         for (Object actualValue : actualValues) {
130                             if (!actualValue.equals(it.next())) {
131                                 listValuesAreEqual = false;
132                                 break;
133                             }
134                         }
135                         if (listValuesAreEqual) {
136                             isAllowedValue = true;
137                         }
138                     }
139 
140                     if (isAllowedValue) {
141                         break;
142                     }
143                 }
144 
145             } else {
146                 List<T> allowedValues = getAllowedValues(propDef.getChoices());
147                 // do a simpler check if all values are choice elements
148 
149                 for (Object actualValue : prop.getValues()) {
150                     if (!allowedValues.contains(actualValue)) {
151                         isAllowedValue = false;
152                         break;
153                     }
154                 }
155             }
156 
157             if (!isAllowedValue) {
158                 throw new CmisConstraintException("The property with id " + propDef.getId()
159                         + " has a fixed set of values. Value(s) " + prop.getValues() + " are not listed.");
160             }
161         }
162 
163         /**
164          * Calculate the list of allowed values for this property definition by
165          * recursively collecting all choice values from property definition
166          * 
167          * @param propDef
168          *            property definition
169          * @return list of possible values in complete hierarchy
170          */
171         private List<T> getAllowedValues(List<Choice<T>> choices) {
172             List<T> allowedValues = new ArrayList<T>(choices.size());
173             for (Choice<T> choice : choices) {
174                 if (isNotEmpty(choice.getValue())) {
175                     allowedValues.add(choice.getValue().get(0));
176                 }
177                 if (isNotEmpty(choice.getChoice())) {
178                     List<Choice<T>> x = choice.getChoice();
179                     allowedValues.addAll(getAllowedValues(x));
180                 }
181             }
182             return allowedValues;
183         }

186     static class PropertyValidatorInteger extends PropertyValidator<BigInteger> {
187 
188         @Override
189         public void validate(PropertyDefinition<BigInteger> propDef, PropertyData<BigInteger> property) {
190 
191             super.validate(propDef, property);
192 
193             BigInteger propVal = property.getFirstValue();
194             BigInteger minVal = ((PropertyIntegerDefinition) propDef).getMinValue();
195             BigInteger maxVal = ((PropertyIntegerDefinition) propDef).getMaxValue();
196 
197             // check min and max
198             if (minVal != null && propVal != null && propVal.compareTo(minVal) == -1) {
199                 throw new CmisConstraintException("For property with id " + propDef.getId() + " the value " + propVal
200                         + " is less than the minimum value " + minVal);
201             }
202             if (maxVal != null && propVal != null && propVal.compareTo(maxVal) == 1) {
203                 throw new CmisConstraintException("For property with id " + propDef.getId() + " the value " + propVal
204                         + " is bigger than the maximum value " + maxVal);
205             }
206         }
207     }

245                 throw new CmisConstraintException("For property with id " + propDef.getId() + " the length of " + len

255     @SuppressWarnings("unchecked")
256     public static <T> void validateProperties(TypeDefinition typeDef, Properties properties, boolean checkMandatory,
257             boolean cmis11) {
258         List<String> propDefsRequired = getMandatoryPropDefs(typeDef.getPropertyDefinitions());
259 
260         if (properties != null) {
261             for (PropertyData<?> prop : properties.getProperties().values()) {
262                 String propertyId = prop.getId();
263                 if (null == propertyId) {
264                     throw new CmisInvalidArgumentException("Property id cannot be null");
265                 }
266                 BaseTypeId baseTypeId = typeDef.getBaseTypeId();
267 
268                 // check that all mandatory attributes are present
269                 if (checkMandatory && propDefsRequired.contains(propertyId)) {
270                     propDefsRequired.remove(propertyId);
271                 }
272 
273                 if (isSystemProperty(baseTypeId, propertyId, cmis11)) {
274                     continue; // ignore system properties for validation
275                 }
276 
277                 // Check if all properties are known in the type
278                 if (typeContainsProperty(typeDef, propertyId)) {
279                     // check all type specific constraints:
280                     PropertyDefinition<T> propDef = getPropertyDefinition(typeDef, propertyId);
281                     PropertyValidator<T> validator = createPropertyValidator(propDef);
282                     validator.validate(propDef, (PropertyData<T>) prop);
283                 } else {
284                     throw new CmisConstraintException("Unknown property " + propertyId + " in type " + typeDef.getId());
285                 }
286             }
287         }
288 
289         if (checkMandatory && !propDefsRequired.isEmpty()) {
290             throw new CmisConstraintException("The following mandatory properties are missing: " + propDefsRequired);
291         }
292     }

294     @SuppressWarnings("unchecked")
295     public static <T> void validateProperties(List<TypeDefinition> typeDefs, Properties properties,
296             boolean checkMandatory) {
297         if (properties == null) {
298             return;
299         }
300 
301         Map<String, Boolean> checkedProperties = new HashMap<String, Boolean>();
302         for (String propId : properties.getProperties().keySet()) {
303             checkedProperties.put(propId, false);
304         }
305 
306         for (TypeDefinition typeDef : typeDefs) {
307 
308             List<String> propDefsRequired = getMandatoryPropDefs(typeDef.getPropertyDefinitions());
309 
310             for (PropertyData<?> prop : properties.getProperties().values()) {
311                 String propertyId = prop.getId();
312                 if (null == propertyId) {
313                     throw new CmisInvalidArgumentException("Property id cannot be null");
314                 }
315                 BaseTypeId baseTypeId = typeDef.getBaseTypeId();
316 
317                 // check that all mandatory attributes are present
318                 if (checkMandatory && propDefsRequired.contains(propertyId)) {
319                     propDefsRequired.remove(propertyId);
320                 }
321 
322                 if (isSystemProperty(baseTypeId, propertyId, true)) {
323                     checkedProperties.put(prop.getId(), true); // ignore system
324                                                                // properties for
325                                                                // validation
326                 } else if (typeContainsProperty(typeDef, propertyId)) {
327                     // Check if all properties are known in the type
328                     // marked the property as found in a type of primary or
329                     // secondary types
330                     checkedProperties.put(prop.getId(), true);
331 
332                     // check all type specific constraints:
333                     PropertyDefinition<T> propDef = getPropertyDefinition(typeDef, propertyId);
334                     PropertyValidator<T> validator = createPropertyValidator(propDef);
335                     validator.validate(propDef, (PropertyData<T>) prop);
336                 }
337             }
338 
339             if (checkMandatory && !propDefsRequired.isEmpty()) {
340                 throw new CmisConstraintException("The following mandatory properties are missing: " + propDefsRequired);
341             }
342         }
343 
344         // check if all properties are known in a type definition
345         List<String> unknownProperties = new ArrayList<String>();
346         for (String propId : properties.getProperties().keySet()) {
347             if (!checkedProperties.get(propId)) {
348                 unknownProperties.add(propId);
349             }
350         }
351         if (!unknownProperties.isEmpty()) {
352             throw new CmisConstraintException(
353                     "The following properties are not known in any of the types of this object: " + unknownProperties);
354         }
355     }

381     protected static void validateAllowedTypes(TypeDefinition typeDef, List<String> allowedTypes, String description) {
382         if (isNullOrEmpty(allowedTypes)) {
383             return; // all types are allowed
384         }
385 
386         for (String allowedType : allowedTypes) {
387             if (allowedType.equals(typeDef.getId())) {
388                 return;
389             }
390         }
391         throw new CmisConstraintException("The requested type " + typeDef.getId() + " is not allowed " + description);
392     }

401     public static void validateContentAllowed(DocumentTypeDefinition typeDef, boolean hasContent) {
402         ContentStreamAllowed contentAllowed = typeDef.getContentStreamAllowed();
403         if (ContentStreamAllowed.REQUIRED == contentAllowed && !hasContent) {
404             throw new CmisConstraintException("Type " + typeDef.getId()
405                     + " requires content but document has no content.");
406         } else if (ContentStreamAllowed.NOTALLOWED == contentAllowed && hasContent) {
407             throw new CmisConstraintException("Type " + typeDef.getId()
408                     + " does not allow content but document has content.");
409         }
410     }

412     private static List<String> getMandatoryPropDefs(Map<String, PropertyDefinition<?>> propDefs) {
413         List<String> res = new ArrayList<String>();
414         if (null != propDefs) {
415             for (PropertyDefinition<?> propDef : propDefs.values()) {
416                 if (propDef.isRequired() && !isMandatorySystemProperty(propDef.getId())) {
417                     res.add(propDef.getId());
418                 }
419             }
420         }
421         return res;
422     }

469     private static boolean isSystemProperty(BaseTypeId baseTypeId, String propertyId, boolean cmis11) {
470 
471         if (propertyId.equals(PropertyIds.NAME)) {
472             return true;
473         } else if (propertyId.equals(PropertyIds.OBJECT_ID)) {
474             return true;
475         } else if (propertyId.equals(PropertyIds.OBJECT_TYPE_ID)) {
476             return true;
477         } else if (propertyId.equals(PropertyIds.BASE_TYPE_ID)) {
478             return true;
479         } else if (propertyId.equals(PropertyIds.CREATED_BY)) {
480             return true;
481         } else if (propertyId.equals(PropertyIds.CREATION_DATE)) {
482             return true;
483         } else if (propertyId.equals(PropertyIds.LAST_MODIFIED_BY)) {
484             return true;
485         } else if (propertyId.equals(PropertyIds.LAST_MODIFICATION_DATE)) {
486             return true;
487         } else if (propertyId.equals(PropertyIds.CHANGE_TOKEN)) {
488             return true;
489         } else if (cmis11 && propertyId.equals(PropertyIds.DESCRIPTION)) {
490             return true;
491         } else if (cmis11 && propertyId.equals(PropertyIds.SECONDARY_OBJECT_TYPE_IDS)) {
492             return true;
493         }
494 
495         if (baseTypeId.equals(BaseTypeId.CMIS_DOCUMENT)) {
496             if (propertyId.equals(PropertyIds.IS_IMMUTABLE)) {
497                 return true;
498             } else if (propertyId.equals(PropertyIds.IS_LATEST_VERSION)) {
499                 return true;
500             } else if (propertyId.equals(PropertyIds.IS_MAJOR_VERSION)) {
501                 return true;
502             } else if (propertyId.equals(PropertyIds.VERSION_SERIES_ID)) {
503                 return true;
504             } else if (propertyId.equals(PropertyIds.IS_LATEST_MAJOR_VERSION)) {
505                 return true;
506             } else if (propertyId.equals(PropertyIds.VERSION_LABEL)) {
507                 return true;
508             } else if (propertyId.equals(PropertyIds.IS_VERSION_SERIES_CHECKED_OUT)) {
509                 return true;
510             } else if (propertyId.equals(PropertyIds.VERSION_SERIES_CHECKED_OUT_BY)) {
511                 return true;
512             } else if (propertyId.equals(PropertyIds.VERSION_SERIES_CHECKED_OUT_ID)) {
513                 return true;
514             } else if (propertyId.equals(PropertyIds.CHECKIN_COMMENT)) {
515                 return true;
516             } else if (propertyId.equals(PropertyIds.CONTENT_STREAM_LENGTH)) {
517                 return true;
518             } else if (propertyId.equals(PropertyIds.CONTENT_STREAM_MIME_TYPE)) {
519                 return true;
520             } else if (propertyId.equals(PropertyIds.CONTENT_STREAM_FILE_NAME)) {
521                 return true;
522             } else if (propertyId.equals(PropertyIds.CONTENT_STREAM_ID)) {
523                 return true;
524             }
525 
526             return false;
527         } else if (baseTypeId.equals(BaseTypeId.CMIS_FOLDER)) {
528             if (propertyId.equals(PropertyIds.PARENT_ID)) {
529                 return true;
530             } else if (propertyId.equals(PropertyIds.ALLOWED_CHILD_OBJECT_TYPE_IDS)) {
531                 return true;
532             } else if (propertyId.equals(PropertyIds.PATH)) {
533                 return true;
534             }
535 
536             return false;
537         } else if (baseTypeId.equals(BaseTypeId.CMIS_POLICY)) {
538             if (propertyId.equals(PropertyIds.SOURCE_ID)) {
539                 return true;
540             } else if (propertyId.equals(PropertyIds.TARGET_ID)) {
541                 return true;
542             }
543             return false;
544         } else { // relationship
545             if (propertyId.equals(PropertyIds.POLICY_TEXT)) {
546                 return true;
547             }
548 
549             return false;
550         }
