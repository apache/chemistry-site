406     public ObjectData create(CallContext context, Properties properties, String folderId, ContentStream contentStream,
407             VersioningState versioningState, ObjectInfoHandler objectInfos) {
408         debug("create");
409         boolean userReadOnly = checkUser(context, true);
410 
411         String typeId = FileShareUtils.getObjectTypeId(properties);
412         TypeDefinition type = typeManager.getInternalTypeDefinition(typeId);
413         if (type == null) {
414             throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
415         }
416 
417         String objectId = null;
418         if (type.getBaseTypeId() == BaseTypeId.CMIS_DOCUMENT) {
419             objectId = createDocument(context, properties, folderId, contentStream, versioningState);
420         } else if (type.getBaseTypeId() == BaseTypeId.CMIS_FOLDER) {
421             if (contentStream != null || versioningState != null) {
422                 throw new CmisInvalidArgumentException("Cannot create a folder with content or a versioning state!");
423             }
424 
425             objectId = createFolder(context, properties, folderId);
426         } else {
427             throw new CmisObjectNotFoundException("Cannot create object of type '" + typeId + "'!");
428         }
429 
430         return compileObjectData(context, getFile(objectId), null, false, false, userReadOnly, objectInfos);
431     }

433     /**
434      * CMIS createDocument.
435      */
436     public String createDocument(CallContext context, Properties properties, String folderId,
437             ContentStream contentStream, VersioningState versioningState) {
438         debug("createDocument");
439         checkUser(context, true);
440 
441         // check properties
442         if (properties == null || properties.getProperties() == null) {
443             throw new CmisInvalidArgumentException("Properties must be set!");
444         }
445 
446         // check versioning state
447         if (!(VersioningState.NONE == versioningState || versioningState == null)) {
448             throw new CmisConstraintException("Versioning not supported!");
449         }
450 
451         // check type
452         String typeId = FileShareUtils.getObjectTypeId(properties);
453         TypeDefinition type = typeManager.getInternalTypeDefinition(typeId);
454         if (type == null) {
455             throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
456         }
457         if (type.getBaseTypeId() != BaseTypeId.CMIS_DOCUMENT) {
458             throw new CmisInvalidArgumentException("Type must be a document type!");
459         }
460 
461         // compile the properties
462         PropertiesImpl props = compileWriteProperties(typeId, context.getUsername(), context.getUsername(), properties);
463 
464         // check the name
465         String name = FileShareUtils.getStringProperty(properties, PropertyIds.NAME);
466         if (!isValidName(name)) {
467             throw new CmisNameConstraintViolationException("Name is not valid!");
468         }
469 
470         // get parent File
471         File parent = getFile(folderId);
472         if (!parent.isDirectory()) {
473             throw new CmisObjectNotFoundException("Parent is not a folder!");
474         }
475 
476         // check the file
477         File newFile = new File(parent, name);
478         if (newFile.exists()) {
479             throw new CmisNameConstraintViolationException("Document already exists!");
480         }
481 
482         // create the file
483         try {
484             newFile.createNewFile();
485         } catch (IOException e) {
486             throw new CmisStorageException("Could not create file: " + e.getMessage(), e);
487         }
488 
489         // write content, if available
490         if (contentStream != null && contentStream.getStream() != null) {
491             writeContent(newFile, contentStream.getStream());
492         }
493 
494         // set creation date
495         addPropertyDateTime(props, typeId, null, PropertyIds.CREATION_DATE,
496                 FileShareUtils.millisToCalendar(newFile.lastModified()));
497 
498         // write properties
499         writePropertiesFile(newFile, props);
500 
501         return getId(newFile);
502     }

504     /**
505      * CMIS createDocumentFromSource.
506      */
507     public String createDocumentFromSource(CallContext context, String sourceId, Properties properties,
508             String folderId, VersioningState versioningState) {
509         debug("createDocumentFromSource");
510         checkUser(context, true);
511 
512         // check versioning state
513         if (!(VersioningState.NONE == versioningState || versioningState == null)) {
514             throw new CmisConstraintException("Versioning not supported!");
515         }
516 
517         // get parent File
518         File parent = getFile(folderId);
519         if (!parent.isDirectory()) {
520             throw new CmisObjectNotFoundException("Parent is not a folder!");
521         }
522 
523         // get source File
524         File source = getFile(sourceId);
525         if (!source.isFile()) {
526             throw new CmisObjectNotFoundException("Source is not a document!");
527         }
528 
529         // file name
530         String name = source.getName();
531 
532         // get properties
533         PropertiesImpl sourceProperties = new PropertiesImpl();
534         readCustomProperties(source, sourceProperties, null, new ObjectInfoImpl());
535 
536         // get the type id
537         String typeId = FileShareUtils.getIdProperty(sourceProperties, PropertyIds.OBJECT_TYPE_ID);
538         if (typeId == null) {
539             typeId = BaseTypeId.CMIS_DOCUMENT.value();
540         }
541 
542         // copy properties
543         PropertiesImpl newProperties = new PropertiesImpl();
544         for (PropertyData<?> prop : sourceProperties.getProperties().values()) {
545             if (prop.getId().equals(PropertyIds.OBJECT_TYPE_ID) || prop.getId().equals(PropertyIds.CREATED_BY)
546                     || prop.getId().equals(PropertyIds.CREATION_DATE)
547                     || prop.getId().equals(PropertyIds.LAST_MODIFIED_BY)) {
548                 continue;
549             }
550 
551             newProperties.addProperty(prop);
552         }
553 
554         // replace properties
555         if (properties != null) {
556             // find new name
557             String newName = FileShareUtils.getStringProperty(properties, PropertyIds.NAME);
558             if (newName != null) {
559                 if (!isValidName(newName)) {
560                     throw new CmisNameConstraintViolationException("Name is not valid!");
561                 }
562                 name = newName;
563             }
564 
565             // get the property definitions
566             TypeDefinition type = typeManager.getInternalTypeDefinition(typeId);
567             if (type == null) {
568                 throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
569             }
570             if (type.getBaseTypeId() != BaseTypeId.CMIS_DOCUMENT) {
571                 throw new CmisInvalidArgumentException("Type must be a document type!");
572             }
573 
574             // replace with new values
575             for (PropertyData<?> prop : properties.getProperties().values()) {
576                 PropertyDefinition<?> propType = type.getPropertyDefinitions().get(prop.getId());
577 
578                 // do we know that property?
579                 if (propType == null) {
580                     throw new CmisConstraintException("Property '" + prop.getId() + "' is unknown!");
581                 }

585                     throw new CmisConstraintException("Property '" + prop.getId() + "' cannot be updated!");
586                 }
587 
588                 // empty properties are invalid
589                 if (isEmptyProperty(prop)) {
590                     throw new CmisConstraintException("Property '" + prop.getId() + "' must not be empty!");
591                 }
592 
593                 newProperties.addProperty(prop);
594             }
595         }
596 
597         addPropertyId(newProperties, typeId, null, PropertyIds.OBJECT_TYPE_ID, typeId);
598         addPropertyString(newProperties, typeId, null, PropertyIds.CREATED_BY, context.getUsername());
599         addPropertyDateTime(newProperties, typeId, null, PropertyIds.CREATION_DATE,
600                 FileShareUtils.millisToCalendar(System.currentTimeMillis()));

604         File newFile = new File(parent, name);
605         if (newFile.exists()) {
606             throw new CmisNameConstraintViolationException("Document already exists.");
607         }
608

610         try {

612         } catch (IOException e) {
613             throw new CmisStorageException("Could not create file: " + e.getMessage(), e);
614         }

619         } catch (IOException e) {
620             throw new CmisStorageException("Could not roead or write content: " + e.getMessage(), e);
621         }

950      * Checks and updates a property set that can be written to disc.
951      */
952     private Properties updateProperties(String typeId, String creator, GregorianCalendar creationDate, String modifier,
953             Properties oldProperties, Properties properties) {
954         PropertiesImpl result = new PropertiesImpl();
955 
956         if (properties == null) {
957             throw new CmisConstraintException("No properties!");
958         }
959 
960         // get the property definitions
961         TypeDefinition type = typeManager.getInternalTypeDefinition(typeId);
962         if (type == null) {
963             throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
964         }
965 
966         // copy old properties
967         for (PropertyData<?> prop : oldProperties.getProperties().values()) {
968             PropertyDefinition<?> propType = type.getPropertyDefinitions().get(prop.getId());
969 
970             // do we know that property?
971             if (propType == null) {
972                 throw new CmisConstraintException("Property '" + prop.getId() + "' is unknown!");
973             }
974 
975             // only add read/write properties
976             if (propType.getUpdatability() != Updatability.READWRITE) {
977                 continue;
978             }
979 
980             result.addProperty(prop);
981         }
982 
983         // update properties
984         for (PropertyData<?> prop : properties.getProperties().values()) {
985             PropertyDefinition<?> propType = type.getPropertyDefinitions().get(prop.getId());
986 
987             // do we know that property?
988             if (propType == null) {
989                 throw new CmisConstraintException("Property '" + prop.getId() + "' is unknown!");
990             }
991 
992             // can it be set?
993             if (propType.getUpdatability() == Updatability.READONLY) {
994                 throw new CmisConstraintException("Property '" + prop.getId() + "' is readonly!");
995             }
996 
997             if (propType.getUpdatability() == Updatability.ONCREATE) {
998                 throw new CmisConstraintException("Property '" + prop.getId() + "' can only be set on create!");
999             }
1000 
1001             // default or value
1002             if (isEmptyProperty(prop)) {
1003                 addPropertyDefault(result, propType);
1004             } else {
1005                 result.addProperty(prop);
1006             }
1007         }
1008 
1009         addPropertyId(result, typeId, null, PropertyIds.OBJECT_TYPE_ID, typeId);
1010         addPropertyString(result, typeId, null, PropertyIds.CREATED_BY, creator);
1011         addPropertyDateTime(result, typeId, null, PropertyIds.CREATION_DATE, creationDate);
1012         addPropertyString(result, typeId, null, PropertyIds.LAST_MODIFIED_BY, modifier);

1014         return result;

1172

1175         if (skip < 0) {

1177         }
1178 
1179         int max = maxItems == null ? Integer.MAX_VALUE : maxItems.intValue();

1182         }

1185         File folder = getFile(folderId);
1186         if (!folder.isDirectory()) {
1187             throw new CmisObjectNotFoundException("Not a folder!");
1188         }
1189 
1190         // get the children
1191         List<File> children = new ArrayList<File>();
1192         for (File child : folder.listFiles()) {
1193             // skip hidden and shadow files
1194             if (child.isHidden() || child.getName().equals(SHADOW_FOLDER) || child.getPath().endsWith(SHADOW_EXT)) {
1195                 continue;
1196             }
1197 
1198             children.add(child);
1199         }
1200 
1201         // very basic sorting
1202         if (orderBy != null) {
1203             boolean desc = false;
1204             String queryName = orderBy;
1205 
1206             int commaIdx = orderBy.indexOf(',');
1207             if (commaIdx > -1) {
1208                 queryName = orderBy.substring(0, commaIdx);
1209             }
1210 
1211             queryName = queryName.trim();
1212             if (queryName.toLowerCase(Locale.ENGLISH).endsWith(" desc")) {
1213                 desc = true;
1214                 queryName = queryName.substring(0, queryName.length() - 5).trim();
1215             }
1216 
1217             Comparator<File> comparator = null;
1218 
1219             if ("cmis:name".equals(queryName)) {
1220                 comparator = new Comparator<File>() {
1221                     @Override
1222                     public int compare(File f1, File f2) {
1223                         return f1.getName().toLowerCase(Locale.ENGLISH)
1224                                 .compareTo(f2.getName().toLowerCase(Locale.ENGLISH));
1225                     }
1226                 };
1227             } else if ("cmis:creationDate".equals(queryName) || "cmis:lastModificationDate".equals(queryName)) {
1228                 comparator = new Comparator<File>() {
1229                     @Override
1230                     public int compare(File f1, File f2) {
1231                         return Long.compare(f1.lastModified(), f2.lastModified());
1232                     }
1233                 };
1234             } else if ("cmis:contentStreamLength".equals(queryName)) {
1235                 comparator = new Comparator<File>() {
1236                     @Override
1237                     public int compare(File f1, File f2) {
1238                         return Long.compare(f1.length(), f2.length());
1239                     }
1240                 };
1241             } else if ("cmis:objectId".equals(queryName)) {
1242                 comparator = new Comparator<File>() {
1243                     @Override
1244                     public int compare(File f1, File f2) {
1245                         try {
1246                             return fileToId(f1).compareTo(fileToId(f2));
1247                         } catch (IOException e) {
1248                             return 0;

1251                 };
1252             } else if ("cmis:baseTypeId".equals(queryName)) {
1253                 comparator = new Comparator<File>() {
1254                     @Override
1255                     public int compare(File f1, File f2) {
1256                         if (f1.isDirectory() == f2.isDirectory()) {
1257                             return 0;
1258                         }
1259                         return f1.isDirectory() ? -1 : 1;
1260                     }
1261                 };
1262             } else if ("cmis:createdBy".equals(queryName) || "cmis:lastModifiedBy".equals(queryName)) {
1263                 // do nothing
1264             } else {
1265                 throw new CmisInvalidArgumentException("Cannot sort by " + queryName + ".");
1266             }

1269                 Collections.sort(children, comparator);
1270                 if (desc) {
1271                     Collections.reverse(children);
1272                 }
1273             }

1275

1277         if (context.isObjectInfoRequired()) {
1278             compileObjectData(context, folder, null, false, false, userReadOnly, objectInfos);
1279         }
1280 
1281         // prepare result
1282         ObjectInFolderListImpl result = new ObjectInFolderListImpl();
1283         result.setObjects(new ArrayList<ObjectInFolderData>());
1284         result.setHasMoreItems(false);
1285         int count = 0;
1286 
1287         // iterate through children
1288         for (File child : children) {
1289             count++;
1290 
1291             if (skip > 0) {
1292                 skip--;
1293                 continue;
1294             }
1295 
1296             if (result.getObjects().size() >= max) {

1692     /**
1693      * Reads and adds properties.
1694      */
1695     private void readCustomProperties(File file, PropertiesImpl properties, Set<String> filter,
1696             ObjectInfoImpl objectInfo) {
1697         File propFile = getPropertiesFile(file);
1698 
1699         // if it doesn't exists, ignore it
1700         if (!propFile.exists()) {
1701             return;
1702         }
1703 
1704         // parse it
1705         ObjectData obj = null;
1706         InputStream stream = null;
1707         try {
1708             stream = new BufferedInputStream(new FileInputStream(propFile), 64 * 1024);
1709             XMLStreamReader parser = XMLUtils.createParser(stream);
1710             XMLUtils.findNextStartElemenet(parser);
1711             obj = XMLConverter.convertObject(parser);
1712             parser.close();
1713         } catch (Exception e) {
1714             LOG.warn("Unvalid CMIS properties: {}", propFile.getAbsolutePath(), e);
1715         } finally {
1716             IOUtils.closeQuietly(stream);
1717         }
1718 
1719         if (obj == null || obj.getProperties() == null) {
1720             return;
1721         }
1722 
1723         // add it to properties
1724         for (PropertyData<?> prop : obj.getProperties().getPropertyList()) {
1725             // overwrite object info
1726             if (prop instanceof PropertyString) {
1727                 String firstValueStr = ((PropertyString) prop).getFirstValue();
1728                 if (PropertyIds.NAME.equals(prop.getId())) {
1729                     objectInfo.setName(firstValueStr);
1730                 } else if (PropertyIds.OBJECT_TYPE_ID.equals(prop.getId())) {
1731                     objectInfo.setTypeId(firstValueStr);
1732                 } else if (PropertyIds.CREATED_BY.equals(prop.getId())) {
1733                     objectInfo.setCreatedBy(firstValueStr);
1734                 } else if (PropertyIds.CONTENT_STREAM_MIME_TYPE.equals(prop.getId())) {
1735                     objectInfo.setContentType(firstValueStr);
1736                 } else if (PropertyIds.CONTENT_STREAM_FILE_NAME.equals(prop.getId())) {
1737                     objectInfo.setFileName(firstValueStr);
1738                 }
1739             }
1740 
1741             if (prop instanceof PropertyDateTime) {
1742                 GregorianCalendar firstValueCal = ((PropertyDateTime) prop).getFirstValue();
1743                 if (PropertyIds.CREATION_DATE.equals(prop.getId())) {
1744                     objectInfo.setCreationDate(firstValueCal);
1745                 } else if (PropertyIds.LAST_MODIFICATION_DATE.equals(prop.getId())) {
1746                     objectInfo.setLastModificationDate(firstValueCal);
1747                 }
1748             }
1749 
1750             // check filter
1751             if (filter != null) {
1752                 if (!filter.contains(prop.getQueryName())) {
1753                     continue;
1754                 } else {
1755                     filter.remove(prop.getQueryName());
1756                 }
1757             }
1758 
1759             // don't overwrite id
1760             if (PropertyIds.OBJECT_ID.equals(prop.getId())) {
1761                 continue;
1762             }
1763 
1764             // don't overwrite base type
1765             if (PropertyIds.BASE_TYPE_ID.equals(prop.getId())) {
1766                 continue;
1767             }
1768 
1769             // add it
1770             properties.replaceProperty(prop);
1771         }
1772     }

1774     /**
1775      * Checks and compiles a property set that can be written to disc.
1776      */
1777     private PropertiesImpl compileWriteProperties(String typeId, String creator, String modifier, Properties properties) {
1778         PropertiesImpl result = new PropertiesImpl();
1779         Set<String> addedProps = new HashSet<String>();
1780 
1781         if (properties == null || properties.getProperties() == null) {
1782             throw new CmisConstraintException("No properties!");
1783         }
1784 
1785         // get the property definitions
1786         TypeDefinition type = typeManager.getInternalTypeDefinition(typeId);
1787         if (type == null) {
1788             throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
1789         }
1790 
1791         // check if all required properties are there
1792         for (PropertyData<?> prop : properties.getProperties().values()) {
1793             PropertyDefinition<?> propType = type.getPropertyDefinitions().get(prop.getId());
1794 
1795             // do we know that property?
1796             if (propType == null) {
1797                 throw new CmisConstraintException("Property '" + prop.getId() + "' is unknown!");
1798             }
1799 
1800             // can it be set?
1801             if (propType.getUpdatability() == Updatability.READONLY) {
1802                 throw new CmisConstraintException("Property '" + prop.getId() + "' is readonly!");
1803             }
1804 
1805             // empty properties are invalid
1806             // TODO: check
1807             // if (isEmptyProperty(prop)) {
1808             // throw new CmisConstraintException("Property '" + prop.getId() +
1809             // "' must not be empty!");
1810             // }
1811 
1812             // add it
1813             result.addProperty(prop);
1814             addedProps.add(prop.getId());
1815         }
1816 
1817         // check if required properties are missing
1818         for (PropertyDefinition<?> propDef : type.getPropertyDefinitions().values()) {
1819             if (!addedProps.contains(propDef.getId()) && propDef.getUpdatability() != Updatability.READONLY) {
1820                 if (!addPropertyDefault(result, propDef) && propDef.isRequired()) {
1821                     throw new CmisConstraintException("Property '" + propDef.getId() + "' is required!");
1822                 }
1823             }
1824         }
1825 
1826         addPropertyId(result, typeId, null, PropertyIds.OBJECT_TYPE_ID, typeId);
1827         addPropertyString(result, typeId, null, PropertyIds.CREATED_BY, creator);
1828         addPropertyString(result, typeId, null, PropertyIds.LAST_MODIFIED_BY, modifier);
1829 
1830         return result;
1831     }
