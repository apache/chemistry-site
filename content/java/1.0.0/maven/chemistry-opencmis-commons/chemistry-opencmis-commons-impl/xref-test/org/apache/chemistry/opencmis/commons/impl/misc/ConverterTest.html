176     /**
177      * Asserts OpenCMIS data objects.
178      */
179     protected void assertDataObjectsEquals(String name, Object expected, Object actual) throws Exception {
180 
181         if ((expected == null) && (actual == null)) {
182             return;
183         }
184 
185         if ((expected == null) && (actual instanceof Collection<?>)) {
186             assertTrue(((Collection<?>) actual).isEmpty());
187             return;
188         } else if ((expected instanceof Collection<?>) && (actual == null)) {
189             assertTrue(((Collection<?>) expected).isEmpty());
190             return;
191         } else if ((expected == null) || (actual == null)) {
192             fail("Data object is null! name: " + name + " / expected: " + expected + " / actual: " + actual);
193         }
194 
195         // handle simple types
196         if ((expected instanceof String) || (expected instanceof Boolean) || (expected instanceof BigInteger)
197                 || (expected instanceof BigDecimal) || (expected instanceof Enum<?>)) {
198             assertEquals(expected, actual);
199 
200             return;
201         } else if (expected instanceof List<?>) {
202             List<?> expectedList = (List<?>) expected;
203             List<?> actualList = (List<?>) actual;
204 
205             assertEquals(expectedList.size(), actualList.size());
206 
207             for (int i = 0; i < expectedList.size(); i++) {
208                 assertDataObjectsEquals(name + "[" + i + "]", expectedList.get(i), actualList.get(i));
209             }
210 
211             return;
212         } else if (expected instanceof Map<?, ?>) {
213             Map<?, ?> expectedMap = (Map<?, ?>) expected;
214             Map<?, ?> actualMap = (Map<?, ?>) actual;
215 
216             assertEquals(expectedMap.size(), actualMap.size());
217 
218             for (Map.Entry<?, ?> entry : expectedMap.entrySet()) {
219                 assertTrue(actualMap.containsKey(entry.getKey()));
220                 assertDataObjectsEquals(name + "[" + entry.getKey() + "]", entry.getValue(),
221                         actualMap.get(entry.getKey()));
222             }
223 
224             return;
225         }
226 
227         for (Method m : expected.getClass().getMethods()) {
228             if (!m.getName().startsWith("get") && !m.getName().startsWith("supports")) {
229                 continue;
230             }
231 
232             if (m.getName().equals("getClass")) {
233                 continue;
234             }
235 
236             if (m.getParameterTypes().length != 0) {
237                 continue;
238             }
239 
240             Object expectedValue = m.invoke(expected, new Object[0]);
241             Object actualValue = m.invoke(actual, new Object[0]);
242 
243             assertDataObjectsEquals(name + "." + m.getName(), expectedValue, actualValue);
244         }
245     }
