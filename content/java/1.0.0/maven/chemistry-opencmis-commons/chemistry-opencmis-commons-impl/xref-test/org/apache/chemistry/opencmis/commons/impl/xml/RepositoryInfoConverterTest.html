218     protected void assertXmlRepositoryInfo10(RepositoryInfo repInfo, boolean validate) throws Exception {
219         ByteArrayOutputStream out = new ByteArrayOutputStream();
220 
221         XMLStreamWriter writer = createWriter(out);
222         XMLConverter.writeRepositoryInfo(writer, CmisVersion.CMIS_1_0, TEST_NAMESPACE, repInfo);
223         closeWriter(writer);
224 
225         byte[] xml = out.toByteArray();
226 
227         if (validate) {
228             validate(xml, CmisVersion.CMIS_1_0);
229         }
230 
231         XMLStreamReader parser = createParser(xml);
232         RepositoryInfo result = XMLConverter.convertRepositoryInfo(parser);
233         closeParser(parser);
234 
235         assertNotNull(result);
236         assertDataObjectsEquals("RepositoryInfo", repInfo, result, cmis10ignoreMethods);
237         assertNull(result.getExtensions());
238     }
