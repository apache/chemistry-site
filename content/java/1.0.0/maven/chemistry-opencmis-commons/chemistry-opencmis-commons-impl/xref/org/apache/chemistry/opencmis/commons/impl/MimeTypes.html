221     /**
222      * Returns the MIME type for file extension.
223      */
224     public static String getMIMEType(String ext) {
225         if (ext == null) {
226             return OCTET_STREAM;
227         }
228 
229         int x = ext.lastIndexOf('.');
230         if (x > -1) {
231             ext = ext.substring(x + 1);
232         }
233 
234         return TYPE_MAP.getContentType("x." + ext.toLowerCase(Locale.ENGLISH));
235     }
236 
237     /**
238      * Returns the MIME type for a file.
239      */
240     public static String getMIMEType(File file) {
241         if (file == null) {
242             return OCTET_STREAM;
243         }
244 
245         return TYPE_MAP.getContentType(file);
246     }
247 
248     /**
249      * Guesses a extension from a MIME type.
250      */
251     public static String getExtension(String mimeType) {
252         if (mimeType == null) {
253             return "";
254         }
255 
256         int x = mimeType.indexOf(';');
257         if (x > -1) {
258             mimeType = mimeType.substring(0, x);
259         }
260         mimeType = mimeType.trim().toLowerCase(Locale.ENGLISH);
261 
262         String extension = MIME2EXT.get(mimeType);
263         return (extension == null || extension.length() == 0) ? "" : "." + extension;
264     }
