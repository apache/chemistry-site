89      public AclEditorFrame(final ClientModel model, final CmisObject object) {
90          super();
91  
92          this.model = model;
93          this.object = object;
94  
95          // get users
96          List<String> princiaplList = new ArrayList<String>();
97          try {
98              princiaplList.add("");
99              princiaplList.add("cmis:user");
100 
101             String user = model.getClientSession().getSessionParameters().get(SessionParameter.USER);
102             if (user != null && user.length() > 0) {
103                 princiaplList.add(user);
104             }
105 
106             String anonymous = model.getRepositoryInfo().getPrincipalIdAnonymous();
107             if (anonymous != null && anonymous.length() > 0) {
108                 princiaplList.add(anonymous);
109             }
110 
111             String anyone = model.getRepositoryInfo().getPrincipalIdAnyone();
112             if (anyone != null && anyone.length() > 0) {
113                 princiaplList.add(anyone);
114             }
115 
116             if (object.getAcl() != null && object.getAcl().getAces() != null) {
117                 List<String> aclPrinciaplList = new ArrayList<String>();
118 
119                 for (Ace ace : object.getAcl().getAces()) {
120                     String pid = ace.getPrincipalId();
121                     if (!princiaplList.contains(pid) && !aclPrinciaplList.contains(pid)) {
122                         aclPrinciaplList.add(pid);
123                     }
124                 }
125 
126                 Collections.sort(aclPrinciaplList);
127 
128                 princiaplList.addAll(aclPrinciaplList);
129             }
130         } catch (Exception ex) {
131             princiaplList = new ArrayList<String>();
132             princiaplList.add("");
133             princiaplList.add("cmis:user");
134         }
135 
136         // get permissions
137         List<String> permissionsList = new ArrayList<String>();
138         try {
139             permissionsList.add("");
140             for (PermissionDefinition pd : model.getRepositoryInfo().getAclCapabilities().getPermissions()) {
141                 permissionsList.add(pd.getId());
142             }
143         } catch (Exception ex) {
144             permissionsList = new ArrayList<String>();
145             permissionsList.add("");
146             permissionsList.add("cmis:read");
147             permissionsList.add("cmis:write");
148             permissionsList.add("cmis:all");
149         }
150 
151         addAceList = new AceList(princiaplList, permissionsList);
152         removeAceList = new AceList(princiaplList, permissionsList);
153 
154         createGUI();
155     }

277     /**
278      * Applies the ACEs.
279      */
280     private boolean doApply() {
281         try {
282             setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
283 
284             List<Ace> adds = addAceList.getAces();
285             List<Ace> removes = removeAceList.getAces();
286 
287             if (adds != null || removes != null) {
288                 AclPropagation aclPropagation = AclPropagation.REPOSITORYDETERMINED;
289                 if (propagationObjectOnlyButton.isSelected()) {
290                     aclPropagation = AclPropagation.OBJECTONLY;
291                 }
292 
293                 if (propagationPropagteButton.isSelected()) {
294                     aclPropagation = AclPropagation.PROPAGATE;
295                 }
296 
297                 object.applyAcl(adds, removes, aclPropagation);
298                 LoadObjectWorker.reloadObject(this, model);
299             }
300 
301             return true;
302         } catch (Exception ex) {
303             ClientHelper.showError(this, ex);
304             return false;
305         } finally {
306             setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
307         }

321         public AceList(final List<String> principals, final List<String> permissions) {
322             super();
323 
324             panels = new ArrayList<AceInputPanel>();
325             this.principals = principals;
326             this.permissions = permissions;
327 
328             setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
329         }
330 
331         public synchronized void addNewAce() {
332             AceInputPanel acePanel = new AceInputPanel(this, principals, permissions, panels.size());
333             panels.add(acePanel);
334 
335             add(acePanel);
336         }
337 
338         public synchronized void removeAce(int position) {
339             panels.remove(position);
340             for (int i = position; i < panels.size(); i++) {
341                 panels.get(i).updatePosition(i);
342             }
343 
344             removeAll();
345             for (AceInputPanel p : panels) {
346                 add(p);
347             }
348 
349             revalidate();
350         }
351 
352         public synchronized List<Ace> getAces() {
353             List<Ace> result = new ArrayList<Ace>();
354 
355             for (AceInputPanel p : panels) {
356                 result.add(p.getAce());
357             }
358 
359             return result.isEmpty() ? null : result;
360         }

489         private void updatePermissionsPanel(boolean focus) {
490             boolean changed = false;
491 
492             if (!permissionBoxes.isEmpty()) {
493                 int i = 0;
494                 while (i < permissionBoxes.size() - 1) {
495                     if (permissionBoxes.get(i).getSelectedItem().toString().trim().length() == 0) {
496                         permissionBoxes.remove(i);
497                         changed = true;
498                     } else {
499                         i++;
500                     }
501                 }
502 
503                 if (permissionBoxes.get(permissionBoxes.size() - 1).getSelectedItem().toString().trim().length() > 0) {
504                     permissionBoxes.add(createPermissionBox());
505                     changed = true;
506                 }
507             } else {
508                 permissionBoxes.add(createPermissionBox());
509                 changed = true;
510             }
511 
512             if (changed) {
513                 permissionsPanel.removeAll();
514 
515                 for (JComboBox<String> box : permissionBoxes) {
516                     permissionsPanel.add(box);
517                 }
518 
519                 revalidate();
520 
521                 if (focus) {
522                     permissionBoxes.get(permissionBoxes.size() - 1).requestFocusInWindow();
523                 }
524             }
525         }
