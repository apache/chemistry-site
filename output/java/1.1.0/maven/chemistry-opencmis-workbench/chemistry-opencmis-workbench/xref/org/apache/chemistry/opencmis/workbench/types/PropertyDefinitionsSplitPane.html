117         public PropertyDefinitionsTable(final PropertyDefinitionInfoPanel propertyInfoPanel) {

121             setModel(new PropertyDefinitionTableModel(this));

164                 @Override
165                 public void valueChanged(ListSelectionEvent e) {
166                     if (e.getValueIsAdjusting()) {
167                         return;
168                     }
169 
170                     int row = getSelectedRow();
171                     if (row > -1) {
172                         String propId = getModel().getValueAt(getRowSorter().convertRowIndexToModel(row), ID_COLUMN)
173                                 .toString();
174                         propertyInfoPanel.setPropertyDefinition(type.getPropertyDefinitions().get(propId));
175                     }
176                 }
177             });
178 
179             setFillsViewportHeight(true);
180         }

182         public void setType(ObjectType type) {
183             this.type = type;
184 
185             if (type != null && type.getPropertyDefinitions() != null) {
186                 propertyDefintions = new ArrayList<PropertyDefinition<?>>();
187                 for (PropertyDefinition<?> propDef : type.getPropertyDefinitions().values()) {
188                     propertyDefintions.add(propDef);
189                 }
190 
191                 Collections.sort(propertyDefintions, new Comparator<PropertyDefinition<?>>() {
192                     @Override
193                     public int compare(PropertyDefinition<?> pd1, PropertyDefinition<?> pd2) {
194                         return pd1.getId().compareTo(pd2.getId());
195                     }
196                 });
197             } else {
198                 propertyDefintions = null;
199             }
200 
201             ((AbstractTableModel) getModel()).fireTableDataChanged();
202 
203             if (type != null && type.getPropertyDefinitions() != null && !type.getPropertyDefinitions().isEmpty()) {
204                 getSelectionModel().setSelectionInterval(0, 0);
205             } else {
206                 propertyInfoPanel.setPropertyDefinition(null);
207             }
208         }
209 
210         public ObjectType getType() {

212         }

218         static class PropertyDefinitionTableModel extends AbstractTableModel {
219 
220             private static final long serialVersionUID = 1L;
221 
222             private final PropertyDefinitionsTable table;
223 
224             public PropertyDefinitionTableModel(PropertyDefinitionsTable table) {
225                 this.table = table;
226             }
227 
228             @Override
229             public String getColumnName(int columnIndex) {
230                 return COLUMN_NAMES[columnIndex];
231             }
232 
233             @Override
234             public int getColumnCount() {
235                 return COLUMN_NAMES.length;
236             }
237 
238             @Override
239             public int getRowCount() {
240                 if (table.getPropertyDefinitions() == null) {
241                     return 0;
242                 }
243 
244                 return table.getPropertyDefinitions().size();
245             }
246 
247             @Override
248             public Object getValueAt(int rowIndex, int columnIndex) {
249                 PropertyDefinition<?> propDef = table.getPropertyDefinitions().get(rowIndex);
250 
251                 switch (columnIndex) {
252                 case 0:
253                     return propDef.getDisplayName();
254                 case 1:
255                     return propDef.getId();
256                 case 2:
257                     return propDef.getDescription();
258                 case 3:
259                     return propDef.getLocalNamespace();
260                 case 4:
261                     return propDef.getLocalName();
262                 case 5:
263                     return propDef.getQueryName();
264                 case 6:
265                     return propDef.getPropertyType();
266                 case 7:
267                     return propDef.getCardinality();
268                 case 8:
269                     return propDef.getUpdatability();
270                 case 9:
271                     return propDef.isQueryable();
272                 case 10:
273                     return propDef.isOrderable();
274                 case 11:
275                     return propDef.isRequired();
276                 case 12:
277                     return propDef.isInherited();
278                 case 13:
279                     return propDef.getDefaultValue();
280                 case 14:
281                     return propDef.isOpenChoice();
282                 case 15:
283                     return propDef.getChoices();
284                 default:
285                 }
286 
287                 return null;
288             }
289 
290             @Override
291             public Class<?> getColumnClass(int columnIndex) {
292                 if ((columnIndex == 13) || (columnIndex == 15)) {
293                     return Collection.class;
294                 }
295 
296                 return super.getColumnClass(columnIndex);
297             }
298         }

334         @SuppressWarnings({ "unchecked", "rawtypes" })
335         public void setPropertyDefinition(PropertyDefinition<?> propDef) {
336             if (propDef != null) {
337                 nameField.setText(propDef.getDisplayName());
338                 descriptionField.setText(propDef.getDescription());
339                 idField.setText(propDef.getId());
340                 localNamespaceField.setText(propDef.getLocalNamespace());
341                 localNameField.setText(propDef.getLocalName());
342                 queryNameField.setText(propDef.getQueryName());
343 
344                 propertyTypeField
345                         .setText(propDef.getPropertyType() == null ? "???" : propDef.getPropertyType().value());
346                 cardinalityField.setText(propDef.getCardinality() == null ? "???" : propDef.getCardinality().value());
347                 updatabilityField
348                         .setText(propDef.getUpdatability() == null ? "???" : propDef.getUpdatability().value());
349                 queryableLabel.setValue(is(propDef.isQueryable()));
350                 orderableLabel.setValue(is(propDef.isOrderable()));
351                 requiredLabel.setValue(is(propDef.isRequired()));
352                 inheritedLabel.setValue(is(propDef.isInherited()));
353                 defaultValueField.setList(propDef.getDefaultValue());
354                 openChoiceLabel.setValue(is(propDef.isOpenChoice()));
355 
356                 List choices = (List) propDef.getChoices();
357                 choicesTree.setData(choices == null || choices.isEmpty() ? null : choices);
358 
359                 if (propDef instanceof PropertyIntegerDefinition) {
360                     maxLengthField.setText("");
361                     BigInteger min = ((PropertyIntegerDefinition) propDef).getMinValue();
362                     minField.setText(min == null ? "" : min.toString());
363                     BigInteger max = ((PropertyIntegerDefinition) propDef).getMaxValue();
364                     maxField.setText(max == null ? "" : max.toString());
365                     precisionField.setText("");
366                     dateTimeResolutionField.setText("");
367                 } else if (propDef instanceof PropertyDecimalDefinition) {
368                     maxLengthField.setText("");
369                     BigDecimal min = ((PropertyDecimalDefinition) propDef).getMinValue();
370                     minField.setText(min == null ? "" : min.toPlainString());
371                     BigDecimal max = ((PropertyDecimalDefinition) propDef).getMaxValue();
372                     maxField.setText(max == null ? "" : max.toPlainString());
373                     DecimalPrecision precision = ((PropertyDecimalDefinition) propDef).getPrecision();
374                     precisionField.setText(precision == null ? "" : precision.value().toString() + "-bit precision");
375                     dateTimeResolutionField.setText("");
376                 } else if (propDef instanceof PropertyStringDefinition) {
377                     BigInteger maxLen = ((PropertyStringDefinition) propDef).getMaxLength();
378                     maxLengthField.setText(maxLen == null ? "" : maxLen.toString());
379                     minField.setText("");
380                     maxField.setText("");
381                     precisionField.setText("");
382                     dateTimeResolutionField.setText("");
383                 } else if (propDef instanceof PropertyDateTimeDefinition) {
384                     maxLengthField.setText("");
385                     minField.setText("");
386                     maxField.setText("");
387                     precisionField.setText("");
388                     DateTimeResolution dateTimeResolution = ((PropertyDateTimeDefinition) propDef)
389                             .getDateTimeResolution();
390                     dateTimeResolutionField.setText(dateTimeResolution == null ? "" : dateTimeResolution.value());
391                 } else {
392                     minField.setText("");
393                     maxField.setText("");
394                     maxLengthField.setText("");
395                     precisionField.setText("");
396                     dateTimeResolutionField.setText("");
397                 }
398 
399                 extensionsTree.setData(propDef.getExtensions());
400             } else {
401                 nameField.setText("");
402                 descriptionField.setText("");
403                 idField.setText("");
404                 localNamespaceField.setText("");
405                 localNameField.setText("");
406                 queryNameField.setText("");
407                 propertyTypeField.setText("");
408                 cardinalityField.setText("");
409                 updatabilityField.setText("");
410                 queryableLabel.setValue(false);
411                 orderableLabel.setValue(false);
412                 requiredLabel.setValue(false);
413                 inheritedLabel.setValue(false);
414                 defaultValueField.removeAll();
415                 openChoiceLabel.setValue(false);
416                 choicesTree.setData(null);
417                 maxLengthField.setText("");
418                 minField.setText("");
419                 maxField.setText("");
420                 precisionField.setText("");
421                 dateTimeResolutionField.setText("");
422                 extensionsTree.setData(null);
423             }
424 
425             regenerateGUI();
426         }
