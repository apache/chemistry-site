271     @Override
272     public Map<String, String> getSessionParameters() {
273         String url = urlField.getText().trim();
274 
275         BindingType binding = BindingType.ATOMPUB;
276         if (bindingWebServicesButton.isSelected()) {
277             binding = BindingType.WEBSERVICES;
278         } else if (bindingBrowserButton.isSelected()) {
279             binding = BindingType.BROWSER;
280         }
281 
282         String username = usernameField.getText();
283         String password = new String(passwordField.getPassword());
284 
285         ClientSession.Authentication authentication = ClientSession.Authentication.NONE;
286         if (authenticationStandardButton.isSelected()) {
287             authentication = ClientSession.Authentication.STANDARD;
288         } else if (authenticationNTLMButton.isSelected()) {
289             authentication = ClientSession.Authentication.NTLM;
290         } else if (authenticationOAuthButton.isSelected()) {
291             authentication = ClientSession.Authentication.OAUTH_BEARER;
292         } else if (authenticationCertButton.isSelected()) {
293             authentication = ClientSession.Authentication.CLIENT_CERT;
294         }
295 
296         Locale locale = null;
297         String language = languageField.getText().trim();
298         if (language.length() > 1 && language.charAt(0) != '-' && language.charAt(0) != '_') {
299             int sep1 = language.indexOf('-');
300             int sep2 = language.indexOf('_');
301 
302             if (sep1 > 0) {
303                 locale = new Locale(language.substring(0, sep1), language.substring(sep1 + 1));
304             } else if (sep2 > 0) {
305                 locale = new Locale(language.substring(0, sep2), language.substring(sep2 + 1));
306             } else {
307                 locale = new Locale(language);
308             }
309         }
310 
311         long connectTimeout = 0;
312         if (connectTimeoutField.getValue() instanceof Number) {
313             connectTimeout = ((Number) connectTimeoutField.getValue()).longValue() * 1000;
314             if (connectTimeout < 0) {
315                 connectTimeoutField.setValue(0);
316                 connectTimeout = 0;
317             }
318         }
319 
320         long readTimeout = 0;
321         if (readTimeoutField.getValue() instanceof Number) {
322             readTimeout = ((Number) readTimeoutField.getValue()).longValue() * 1000;
323             if (readTimeout < 0) {
324                 readTimeoutField.setValue(0);
325                 readTimeout = 0;
326             }
327         }
328 
329         return ClientSession.createSessionParameters(url, binding, username, password, authentication,
330                 compressionOnButton.isSelected(), clientCompressionOnButton.isSelected(), cookiesOnButton.isSelected(),
331                 csrfHeaderField.getText(), locale, connectTimeout, readTimeout);
332     }
