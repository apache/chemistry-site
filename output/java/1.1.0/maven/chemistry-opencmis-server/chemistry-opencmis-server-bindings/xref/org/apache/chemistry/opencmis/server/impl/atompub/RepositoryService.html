52      public static class GetRepositories extends AbstractAtomPubServiceCall {
53          @Override
54          public void serve(CallContext context, CmisService service, String repositoryId, HttpServletRequest request,
55                  HttpServletResponse response) throws Exception {
56              assert context != null;
57              assert service != null;
58              assert request != null;
59              assert response != null;
60  
61              // get parameters
62              repositoryId = getStringParameter(request, Constants.PARAM_REPOSITORY_ID);
63  
64              // execute
65              List<RepositoryInfo> infoDataList = null;
66  
67              if (stopBeforeService(service)) {
68                  return;
69              }
70  
71              if (repositoryId == null) {
72                  infoDataList = service.getRepositoryInfos(null);
73              } else {
74                  infoDataList = Collections.singletonList(service.getRepositoryInfo(repositoryId, null));
75                  if (context instanceof CallContextImpl) {
76                      ((CallContextImpl) context).put(CallContext.REPOSITORY_ID, repositoryId);
77                  }
78              }
79  
80              if (stopAfterService(service)) {
81                  return;
82              }
83  
84              // set headers
85              response.setStatus(HttpServletResponse.SC_OK);
86              response.setContentType(Constants.MEDIATYPE_SERVICE);
87  
88              // write XML
89              ServiceDocument serviceDoc = new ServiceDocument();
90  
91              serviceDoc.startDocument(response.getOutputStream(), getNamespaces(service));
92              serviceDoc.startServiceDocument();
93  
94              if (infoDataList != null) {
95                  CmisVersion cmisVersion = context.getCmisVersion();
96                  for (RepositoryInfo infoData : infoDataList) {
97                      if (infoData == null) {
98                          continue;
99                      }
100 
101                     String repId = infoData.getId();
102                     UrlBuilder baseUrl = compileBaseUrl(request, repId);
103 
104                     boolean supportsQuery = false;
105                     boolean supportsUnFiling = false;
106                     boolean supportsMultifiling = false;
107                     boolean supportsFolderTree = false;
108                     boolean supportsRootDescendants = false;
109                     boolean supportsChanges = false;
110 
111                     if (infoData.getCapabilities() != null) {
112                         RepositoryCapabilities cap = infoData.getCapabilities();
113 
114                         if (cap.getQueryCapability() != null) {
115                             supportsQuery = (cap.getQueryCapability() != CapabilityQuery.NONE);
116                         }
117 
118                         if (cap.isUnfilingSupported() != null) {
119                             supportsUnFiling = cap.isUnfilingSupported();
120                         }
121 
122                         if (cap.isMultifilingSupported() != null) {
123                             supportsMultifiling = cap.isMultifilingSupported();
124                         }
125 
126                         if (cap.isGetFolderTreeSupported() != null) {
127                             supportsFolderTree = cap.isGetFolderTreeSupported();
128                         }
129 
130                         if (cap.isGetDescendantsSupported() != null) {
131                             supportsRootDescendants = cap.isGetDescendantsSupported();
132                         }
133 
134                         if (cap.getChangesCapability() != null) {
135                             supportsChanges = (cap.getChangesCapability() != CapabilityChanges.NONE);
136                         }
137                     }
138 
139                     serviceDoc.startWorkspace(infoData.getId());
140 
141                     // add collections
142 
143                     // - root collection
144                     serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_CHILDREN, infoData.getRootFolderId()),
145                             Constants.COLLECTION_ROOT, "Root Collection", Constants.MEDIATYPE_ENTRY,
146                             Constants.MEDIATYPE_CMISATOM);
147 
148                     // - types collection
149                     serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_TYPES, null), Constants.COLLECTION_TYPES,
150                             "Types Collection", "");
151 
152                     // - query collection
153                     if (supportsQuery) {
154                         serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_QUERY, null),
155                                 Constants.COLLECTION_QUERY, "Query Collection", Constants.MEDIATYPE_QUERY);
156                     }
157 
158                     // - checked out collection collection
159                     serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_CHECKEDOUT, null),
160                             Constants.COLLECTION_CHECKEDOUT, "Checked Out Collection", Constants.MEDIATYPE_CMISATOM);
161 
162                     // - unfiled collection collection
163                     if (supportsUnFiling || supportsMultifiling) {
164                         serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_UNFILED, null),
165                                 Constants.COLLECTION_UNFILED, "Unfiled Collection", Constants.MEDIATYPE_CMISATOM);
166                     }
167 
168                     // - bulk update collection
169                     serviceDoc.writeCollection(compileUrl(baseUrl, RESOURCE_BULK_UPDATE, null),
170                             Constants.COLLECTION_BULK_UPDATE, "Bulk Update Collection", Constants.MEDIATYPE_CMISATOM);
171 
172                     // add repository info
173                     serviceDoc.writeRepositoryInfo(infoData, cmisVersion);
174 
175                     // add links
176 
177                     // - types descendants
178                     serviceDoc.writeLink(Constants.REP_REL_TYPEDESC, compileUrl(baseUrl, RESOURCE_TYPESDESC, null),
179                             Constants.MEDIATYPE_FEED, null);
180 
181                     // - folder tree
182                     if (supportsFolderTree) {
183                         serviceDoc.writeLink(Constants.REP_REL_FOLDERTREE,
184                                 compileUrl(baseUrl, RESOURCE_FOLDERTREE, infoData.getRootFolderId()),
185                                 Constants.MEDIATYPE_DESCENDANTS, null);
186                     }
187 
188                     // - root descendants
189                     if (supportsRootDescendants) {
190                         serviceDoc.writeLink(Constants.REP_REL_ROOTDESC,
191                                 compileUrl(baseUrl, RESOURCE_DESCENDANTS, infoData.getRootFolderId()),
192                                 Constants.MEDIATYPE_DESCENDANTS, infoData.getRootFolderId());
193                     }
194 
195                     // - changes
196                     if (supportsChanges) {
197                         serviceDoc.writeLink(Constants.REP_REL_CHANGES, compileUrl(baseUrl, RESOURCE_CHANGES, null),
198                                 Constants.MEDIATYPE_FEED, null);
199                     }
200 
201                     // add URI templates
202 
203                     // - object by id
204                     String url = compileUrl(baseUrl, RESOURCE_OBJECTBYID, null)
205                             + "?id={id}&filter={filter}&includeAllowableActions={includeAllowableActions}"
206                             + "&includeACL={includeACL}&includePolicyIds={includePolicyIds}"
207                             + "&includeRelationships={includeRelationships}&renditionFilter={renditionFilter}";
208                     serviceDoc.writeUriTemplate(url, Constants.TEMPLATE_OBJECT_BY_ID, Constants.MEDIATYPE_ENTRY);
209 
210                     // - object by path
211                     url = compileUrl(baseUrl, RESOURCE_OBJECTBYPATH, null)
212                             + "?path={path}&filter={filter}&includeAllowableActions={includeAllowableActions}"
213                             + "&includeACL={includeACL}&includePolicyIds={includePolicyIds}"
214                             + "&includeRelationships={includeRelationships}&renditionFilter={renditionFilter}";
215                     serviceDoc.writeUriTemplate(url, Constants.TEMPLATE_OBJECT_BY_PATH, Constants.MEDIATYPE_ENTRY);
216 
217                     // - type by id
218                     url = compileUrl(baseUrl, RESOURCE_TYPE, null) + "?id={id}";
219                     serviceDoc.writeUriTemplate(url, Constants.TEMPLATE_TYPE_BY_ID, Constants.MEDIATYPE_ENTRY);
220 
221                     // - query
222                     if (supportsQuery) {
223                         url = compileUrl(baseUrl, RESOURCE_QUERY, null)
224                                 + "?q={q}&searchAllVersions={searchAllVersions}&includeAllowableActions={includeAllowableActions}"
225                                 + "&includeRelationships={includeRelationships}&maxItems={maxItems}&skipCount={skipCount}";
226                         serviceDoc.writeUriTemplate(url, Constants.TEMPLATE_QUERY, Constants.MEDIATYPE_FEED);
227                     }
228 
229                     serviceDoc.endWorkspace();
230                 }
231             }
232 
233             serviceDoc.endServiceDocument();
234             serviceDoc.endDocument();
235         }
236     }

239      * Renders a type children collection.

241     public static class GetTypeChildren extends AbstractAtomPubServiceCall {
242         @Override
243         public void serve(CallContext context, CmisService service, String repositoryId, HttpServletRequest request,
244                 HttpServletResponse response) throws Exception {
245             assert context != null;
246             assert service != null;
247             assert repositoryId != null;
248             assert request != null;
249             assert response != null;
250 
251             // get parameters
252             String typeId = getStringParameter(request, Constants.PARAM_TYPE_ID);
253             boolean includePropertyDefinitions = getBooleanParameter(request, Constants.PARAM_PROPERTY_DEFINITIONS,
254                     false);
255             BigInteger maxItems = getBigIntegerParameter(request, Constants.PARAM_MAX_ITEMS);
256             BigInteger skipCount = getBigIntegerParameter(request, Constants.PARAM_SKIP_COUNT);
257 
258             // execute
259             if (stopBeforeService(service)) {
260                 return;
261             }
262 
263             TypeDefinitionList typeList = service.getTypeChildren(repositoryId, typeId, includePropertyDefinitions,
264                     maxItems, skipCount, null);
265 
266             if (stopAfterService(service)) {
267                 return;
268             }
269 
270             BigInteger numItems = (typeList == null ? null : typeList.getNumItems());
271             Boolean hasMoreItems = (typeList == null ? null : typeList.hasMoreItems());
272 
273             String parentTypeId = null;
274             String typeName = "Type Children";
275 
276             // in order to get the parent type, we need the type definition of
277             // this type as well
278             if (typeId != null) {
279                 TypeDefinition typeDefinition = service.getTypeDefinition(repositoryId, typeId, null);
280 
281                 parentTypeId = (typeDefinition == null ? null : typeDefinition.getParentTypeId());
282                 typeName = (typeDefinition == null ? typeId : typeDefinition.getDisplayName());
283             }
284 
285             // write XML
286             response.setStatus(HttpServletResponse.SC_OK);
287             response.setContentType(Constants.MEDIATYPE_FEED);
288 
289             AtomFeed feed = new AtomFeed();
290             feed.startDocument(response.getOutputStream(), getNamespaces(service));
291             feed.startFeed(true);
292 
293             // write basic Atom feed elements
294             feed.writeFeedElements(typeId, null, TYPE_AUTHOR, typeName, new GregorianCalendar(), null, numItems);
295 
296             // write links
297             UrlBuilder baseUrl = compileBaseUrl(request, repositoryId);
298 
299             feed.writeServiceLink(baseUrl.toString(), repositoryId);
300 
301             UrlBuilder selfLink = compileUrlBuilder(baseUrl, RESOURCE_TYPES, null);
302             selfLink.addParameter(Constants.PARAM_TYPE_ID, typeId);
303             selfLink.addParameter(Constants.PARAM_PROPERTY_DEFINITIONS, includePropertyDefinitions);
304             selfLink.addParameter(Constants.PARAM_MAX_ITEMS, maxItems);
305             selfLink.addParameter(Constants.PARAM_SKIP_COUNT, skipCount);
306             feed.writeSelfLink(selfLink.toString(), typeId);
307 
308             feed.writeViaLink(compileUrl(baseUrl, RESOURCE_TYPE, typeId));
309 
310             UrlBuilder downLink = compileUrlBuilder(baseUrl, RESOURCE_TYPESDESC, null);
311             downLink.addParameter(Constants.PARAM_TYPE_ID, typeId);
312             feed.writeDownLink(downLink.toString(), Constants.MEDIATYPE_DESCENDANTS);
313 
314             if (parentTypeId != null) {
315                 feed.writeUpLink(compileUrl(baseUrl, RESOURCE_TYPE, parentTypeId), Constants.MEDIATYPE_ENTRY);
316             }
317 
318             // write paging links
319             UrlBuilder pagingUrl = compileUrlBuilder(baseUrl, RESOURCE_TYPES, null);
320             pagingUrl.addParameter(Constants.PARAM_TYPE_ID, typeId);
321             pagingUrl.addParameter(Constants.PARAM_PROPERTY_DEFINITIONS, includePropertyDefinitions);
322             feed.writePagingLinks(pagingUrl, maxItems, skipCount, numItems, hasMoreItems, PAGE_SIZE);
323 
324             // write collection
325             UrlBuilder collectionUrl = compileUrlBuilder(baseUrl, RESOURCE_TYPES, null);
326             collectionUrl.addParameter(Constants.PARAM_TYPE_ID, typeId);
327             feed.writeCollection(collectionUrl.toString(), null, "Types Collection", "");
328 
329             // write type entries
330             if (typeList != null && typeList.getList() != null) {
331                 AtomEntry entry = new AtomEntry(feed.getWriter());
332                 for (TypeDefinition type : typeList.getList()) {
333                     writeTypeEntry(entry, type, null, repositoryId, baseUrl, false, context.getCmisVersion());
334                 }
335             }
336 
337             // write extensions
338             feed.writeExtensions(typeList);
339 
340             // we are done
341             feed.endFeed();
342             feed.endDocument();
343         }
