95      private void reset() {
96          rect = new ComplexRectangle(-1.6, -1.2, -0.1, 0.1);
97          juliaPoint = null; // new ComplexPoint();
98          maxIterations = DEFAULT_MAX_ITERATIONS;
99  
100         SecureRandom ran = new SecureRandom();
101         color = colorSchemes[ran.nextInt(colorSchemes.length)];
102         parts = ran.nextInt(13) + 3;
103         LOG.debug("Parts: " + parts);
104         maxIterations = DEFAULT_MAX_ITERATIONS;
105         LOG.debug("Original rect " + ": (" + rect.getRMin() + "r," + rect.getRMax() + "r, " + rect.getIMin() + "i, "
106                 + rect.getIMax() + "i)");
107         randomizeRect(rect);
108     }

110     public ByteArrayOutputStream generateFractal() throws IOException {
111         ByteArrayOutputStream bos = null;
112 
113         if (stepInBatch == ZOOM_STEPS_PER_BATCH) {
114             stepInBatch = 0;
115             reset();
116         }
117 
118         ++stepInBatch;
119         LOG.debug("Generating rect no " + stepInBatch + ": (" + rect.getRMin() + "r," + rect.getRMax() + "r, "
120                 + rect.getIMin() + "i, " + rect.getIMax() + "i)");
121         LOG.debug("   width: " + rect.getWidth() + " height: " + rect.getHeight());
122         bos = genFractal(rect, juliaPoint);
123 
124         double r1New = rect.getWidth() * newColTile / parts + rect.getRMin();
125         double r2New = rect.getWidth() * (newColTile + 1) / parts + rect.getRMin();
126         double i1New = rect.getIMax() - (rect.getHeight() * newRowTile / parts);
127         double i2New = rect.getIMax() - (rect.getHeight() * (newRowTile + 1) / parts);
128         rect.set(r1New, r2New, i1New, i2New);
129         randomizeRect(rect);
130         LOG.debug("Done generating fractals.");
131 
132         return bos;
133     }

135     private void randomizeRect(ComplexRectangle rect) {
136         SecureRandom rnd = new SecureRandom();
137         
138         double jitterFactor = 0.15; // +/- 15%
139         double ran = rnd.nextDouble() * jitterFactor + (1.0 - jitterFactor);
140         double width = rect.getWidth() * ran;
141         ran = rnd.nextDouble() * jitterFactor + (1.0 - jitterFactor);
142         double height = rect.getHeight() * ran;
143         ran = rnd.nextDouble() * jitterFactor + (1.0 - jitterFactor);
144         double r1 = (rect.getWidth() - width) * ran + rect.getRMin();
145         ran = rnd.nextDouble() * jitterFactor + (1.0 - jitterFactor);
146         double i1 = (rect.getHeight() - height) * ran + rect.getIMin();
147         rect.set(r1, r1 + width, i1, i1 + height);
148     }

150     /**
151      * Creates a fractal image as JPEG in memory and return it.
152      * 
153      * @param rect
154      *            rectangle of mandelbrot or julia set
155      * @param juliaPoint
156      *            point in Julia set or null
157      * @return byte array with JPEG stream
158      */
159     public ByteArrayOutputStream genFractal(ComplexRectangle rect, ComplexPoint juliaPoint) throws IOException {
160 
161         boolean isJulia = null != juliaPoint;
162         expandRectToFitImage(rect);
163         initializeColors();
164 
165         maxIterations = maybeGuessMaxIterations(maxIterations, rect, isJulia);
166         LOG.debug("using " + maxIterations + " iterations.");
167         detectDeepZoom(rect);
168 
169         calculator = new FractalCalculator(rect, maxIterations, IMAGE_WIDTH, IMAGE_HEIGHT, getCurrentColorMap(),
170                 juliaPoint);
171         int[][] iterations = calculator.calcFractal();
172         BufferedImage image = calculator.mapItersToColors(iterations);
173         findNewRect(image, iterations);
174 
175         // create image in memory
176         ByteArrayOutputStream bos = new ByteArrayOutputStream(200 * 1024);
177         ImageOutputStream ios = ImageIO.createImageOutputStream(bos);
178         Iterator<ImageWriter> writers = ImageIO.getImageWritersByFormatName("jpg");
179         ImageWriter imageWriter = writers.next();
180 
181         JPEGImageWriteParam params = new JPEGImageWriteParam(Locale.getDefault());
182         params.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
183         params.setCompressionQuality(0.9f);
184 
185         imageWriter.setOutput(ios);
186         imageWriter.write(null, new IIOImage(image, null, null), params);
187         ios.close();
188 
189         return bos;
190     }

200     protected void expandRectToFitImage(ComplexRectangle complexRect) {
201         // The complex rectangle must be scaled to fit the pixel image view.
202         // Method: compare the width/height ratios of the two rectangles.
203         double imageWHRatio = 1.0;
204         double complexWHRatio = 1.0;
205         double iMin = complexRect.getIMin();
206         double iMax = complexRect.getIMax();
207         double rMin = complexRect.getRMin();
208         double rMax = complexRect.getRMax();
209         double complexWidth = rMax - rMin;
210         double complexHeight = iMax - iMin;
211 
212         if ((IMAGE_WIDTH > 0) && (IMAGE_HEIGHT > 0)) {
213             imageWHRatio = ((double) IMAGE_WIDTH / (double) IMAGE_HEIGHT);
214         }
215 
216         if ((complexWidth > 0) && (complexHeight > 0)) {
217             complexWHRatio = complexWidth / complexHeight;
218         } else {
219             return;
220         }
221 
222         if (Double.compare(imageWHRatio, complexWHRatio) == 0) {
223             return;
224         }
225 
226         if (imageWHRatio < complexWHRatio) {
227             // Expand vertically
228             double newHeight = complexWidth / imageWHRatio;
229             double heightDifference = Math.abs(newHeight - complexHeight);
230             iMin = iMin - heightDifference / 2;
231             iMax = iMax + heightDifference / 2;
232         } else {
233             // Expand horizontally
234             double newWidth = complexHeight * imageWHRatio;
235             double widthDifference = Math.abs(newWidth - complexWidth);
236             rMin = rMin - widthDifference / 2;
237             rMax = rMax + widthDifference / 2;
238         }
239         complexRect.set(rMin, rMax, iMin, iMax);
240     }

242     private int guessNewMaxIterations(ComplexRectangle cr, boolean isJulia) {
243         // The higher the zoom factor, the more iterations that are needed to
244         // see
245         // the detail. Guess at a number to produce a cool looking fractal:
246         double zoom = INITIAL_RECT.getWidth() / cr.getWidth();
247         if (zoom < 1.0) {
248             zoom = 1.0; // forces logZoom >= 0
249         }
250         double logZoom = Math.log(zoom);
251         double magnitude = (logZoom / 2.3) - 2.0; // just a guess.
252         if (magnitude < 1.0) {
253             magnitude = 1.0;
254         }
255         double iterations = INITIAL_ITERATIONS * (magnitude * logZoom + 1.0);
256         if (isJulia) {
257             iterations *= 2.0; // Julia sets tend to need more iterations.
258         }
259         return (int) iterations;
260     }

262     private int maybeGuessMaxIterations(int maxIterations, ComplexRectangle cr, boolean isJulia) {
263         // If the user did not change the number of iterations, make a guess...
264         if (previousIterations == maxIterations) {
265             maxIterations = guessNewMaxIterations(cr, isJulia);
266         }
267         previousIterations = maxIterations;
268         return maxIterations;
269     }

271     private boolean detectDeepZoom(ComplexRectangle cr) {
272         // "Deep Zoom" occurs when the precision provided by the Java type
273         // double
274         // runs out of resolution. The use of BigDecimal is required to fix
275         // this.
276         double deltaDiv2 = cr.getWidth() / ((IMAGE_WIDTH) * 2.0);
277         String min = "" + (cr.getRMin());
278         String minPlus = "" + (cr.getRMin() + deltaDiv2);
279 
280         if (Double.valueOf(min).doubleValue() == Double.valueOf(minPlus).doubleValue()) {
281             LOG.warn("Deep Zoom...  Drawing resolution will be degraded ;-(");
282             return true;
283         }
284         return false;
285     }

428     private void findNewRect(BufferedImage image, int[][] iterations) {
429 
430         int newWidth = image.getWidth() / parts;
431         int newHeight = image.getHeight() / parts;
432         int i = 0, j = 0;
433         int noTiles = (image.getWidth() / newWidth) * (image.getHeight() / newHeight); // equals
434                                                                                        // parts
435                                                                                        // but
436                                                                                        // be
437                                                                                        // aware
438                                                                                        // of
439                                                                                        // rounding
440                                                                                        // errors!
441         double[] stdDev = new double[noTiles];
442 
443         for (int y = 0; y + newHeight <= image.getHeight(); y += newHeight) {
444             for (int x = 0; x + newWidth <= image.getWidth(); x += newWidth) {
445                 Rectangle subRect = new Rectangle(x, y, newWidth, newHeight);
446                 stdDev[i * parts + j] = calcStdDev(iterations, subRect);
447                 ++j;
448             }
449             ++i;
450             j = 0;
451         }
452 
453         // find tile with greatest std deviation:
454         double max = 0;
455         int index = 0;
456         for (i = 0; i < noTiles; i++) {
457             if (stdDev[i] > max) {
458                 index = i;
459                 max = stdDev[i];
460             }
461         }
462         newRowTile = index / parts;
463         newColTile = index % parts;
464     }

466     private double calcStdDev(int[][] iterations, Rectangle rect) {
467 
468         int sum = 0;
469         long sumSquare = 0;
470 
471         for (int x = rect.x; x < rect.x + rect.width; x += 1) {
472             for (int y = rect.y; y < rect.y + rect.height; y += 1) {
473                 int iters = iterations[x][y];
474                 sum += iters;
475                 sumSquare += iters * iters;
476             }
477         }
478         int count = rect.width * rect.height;
479         double mean = 0.0;
480 
481         mean = (double) sum / count;
482         return Math.sqrt(sumSquare / (count - (mean * mean)));
483     }
