69      public InMemoryService(StoreManager sm, CallContext ctx) {
70          storeManager = sm;
71          callContext = ctx;
72          fRepSvc = new InMemoryRepositoryServiceImpl(storeManager);
73          fNavSvc = new InMemoryNavigationServiceImpl(storeManager);
74          fObjSvc = new InMemoryObjectServiceImpl(storeManager);
75          fVerSvc = new InMemoryVersioningServiceImpl(storeManager, fObjSvc);
76          fDisSvc = new InMemoryDiscoveryServiceImpl(storeManager);
77          fMultiSvc = new InMemoryMultiFilingServiceImpl(storeManager);
78          fRelSvc = new InMemoryRelationshipServiceImpl(storeManager, fRepSvc);
79          fPolSvc = new InMemoryPolicyServiceImpl(storeManager);
80          fAclSvc = new InMemoryAclService(storeManager);
81      }
