19  package org.apache.chemistry.opencmis.server.support;
20  
21  import java.util.Collection;
22  import java.util.List;
23  
24  import org.apache.chemistry.opencmis.commons.definitions.TypeDefinition;
25  import org.apache.chemistry.opencmis.commons.definitions.TypeDefinitionContainer;
26  
27  public interface TypeManager {
28  
29      /**
30       * return a type definition from the type definition id
31       * 
32       * @param typeId
33       *            id of the type definition
34       * @return type definition for this id
35       */
36      TypeDefinitionContainer getTypeById(String typeId);
37  
38      /**
39       * return a type definition from the type query name or null if not found
40       * 
41       * @param typeQueryName
42       *            query name of the type definition
43       * @return type definition for this query name
44       */
45      TypeDefinition getTypeByQueryName(String typeQueryName);
46  
47      /**
48       * return a list of all types known in this repository
49       * Note: This method is not needed for the query parser.
50       * 
51       * @return
52       *      list of type definitions
53       */
54      Collection<TypeDefinitionContainer> getTypeDefinitionList();
55  
56      /**
57       * return a list of the root types as defined in the CMIS spec (for
58       * document, folder, policy and relationship
59       * Note: This method is not needed for the query parser.
60       * 
61       * @return
62       *      list of type definitions
63       */
64      List<TypeDefinitionContainer> getRootTypes();
65  
66      /**
67       * retrieve the property id from a type for a given property query name 
68       * 
69       * @param typeDefinition
70       *      type definition containing query name
71       * @param propQueryName
72       *      query name of property
73       * @return
74       *      property id of property or null if not found
75       */
76      String getPropertyIdForQueryName(TypeDefinition typeDefinition, String propQueryName);
77  
78      /**
79       * Add a type to the type system. Add all properties from inherited types,
80       * add type to children of parent types.
81       * Note: This method is not needed for the query parser.
82       * 
83       * @param typeDefinition
84       *            new type to add
85       * @param addInheritedProperties
86       *            add properties from supertype to type definition
87       */
88      void addTypeDefinition(TypeDefinition typeDefinition, boolean addInheritedProperties);
89  
90      /**
91       * Modify an existing type definition.
92       * Note: This method is not needed for the query parser.
93       * 
94       * @param typeDefinition
95       *            type to be modified
96       */
97      void updateTypeDefinition(TypeDefinition typeDefinition);
98  
99      /**
100      * Delete a type from the type system. Delete will succeed only if type is
101      * not in use. Otherwise an exception is thrown.
102      * Note: This method is not needed for the query parser.
103      * 
104      * @param typeId
105      *            id of type to be deleted
106      */
107     void deleteTypeDefinition(String typeId);
108 }
