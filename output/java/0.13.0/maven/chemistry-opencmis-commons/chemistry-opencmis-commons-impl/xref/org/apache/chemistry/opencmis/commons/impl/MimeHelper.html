50      // RFC 2045
51      private static final String MIME_SPECIALS = "()<>@,;:\\\"/[]?=" + "\t ";

57      private static final char[] HEX_DIGITS = "0123456789ABCDEF".toCharArray();

59      private static final byte[] HEX_DECODE = new byte[0x80];
60      static {
61          for (int i = 0; i < HEX_DIGITS.length; i++) {
62              HEX_DECODE[HEX_DIGITS[i]] = (byte) i;
63              HEX_DECODE[Character.toLowerCase(HEX_DIGITS[i])] = (byte) i;
64          }

70      /**
71       * Encodes a value per RFC 2231.
72       * <p>
73       * This is used to pass non-ASCII parameters to MIME parameter lists.
74       * <p>
75       * This implementation always uses UTF-8 and no language.
76       * <p>
77       * See <a href="http://tools.ietf.org/html/rfc2231">RFC 2231</a> for
78       * details.
79       * 
80       * @param value
81       *            the value to encode
82       * @param buf
83       *            the buffer to fill
84       * @return {@code true} if an encoding was needed, or {@code false} if no
85       *         encoding was actually needed
86       */
87      protected static boolean encodeRFC2231value(String value, StringBuilder buf) {
88          assert value != null;
89          assert buf != null;
90  
91          String charset = IOUtils.UTF8;
92          buf.append(charset);
93          buf.append("''"); // no language
94          byte[] bytes = IOUtils.toUTF8Bytes(value);
95  
96          boolean encoded = false;
97          for (int i = 0; i < bytes.length; i++) {
98              int ch = bytes[i] & 0xff;
99              if (ch <= 32 || ch >= 127 || RFC2231_SPECIALS.indexOf(ch) != -1) {
100                 buf.append('%');
101                 buf.append(HEX_DIGITS[ch >> 4]);
102                 buf.append(HEX_DIGITS[ch & 0xf]);
103                 encoded = true;
104             } else {
105                 buf.append((char) ch);
106             }
107         }
108         return encoded;
109     }

111     /**
112      * Encodes a MIME parameter per RFC 2231.
113      * <p>
114      * This implementation always uses UTF-8 and no language.
115      * <p>
116      * See <a href="http://tools.ietf.org/html/rfc2231">RFC 2231</a> for
117      * details.
118      * 
119      * @param value
120      *            the string to encode
121      * @return the encoded string
122      */
123     protected static String encodeRFC2231(String key, String value) {
124         StringBuilder buf = new StringBuilder();
125         boolean encoded = encodeRFC2231value(value, buf);
126         if (encoded) {
127             return "; " + key + "*=" + buf.toString();
128         } else {
129             return "; " + key + "=" + value;
130         }

171      * Decodes the Content-Disposition header value according to RFC 2183 and
172      * RFC 2231.
173      * <p>
174      * Does not deal with continuation lines.
175      * <p>
176      * See <a href="http://tools.ietf.org/html/rfc2231">RFC 2231</a> for
177      * details.
178      * 
179      * @param value
180      *            the header value to decode
181      * @param params
182      *            the map of parameters to fill
183      * @return the disposition
184      * 
185      */
186     public static String decodeContentDisposition(String value, Map<String, String> params) {
187         try {
188             HeaderTokenizer tokenizer = new HeaderTokenizer(value);
189             // get the first token, which must be an ATOM
190             Token token = tokenizer.next();
191             if (token.getType() != Token.ATOM) {
192                 return null;
193             }
194             String disposition = token.getValue();
195             // value ignored in this method
196 
197             // the remainder is the parameters
198             String remainder = tokenizer.getRemainder();
199             if (remainder != null) {
200                 getParameters(remainder, params);
201             }
202             return disposition;
203         } catch (ParseException e) {
204             return null;
205         }

241     /**
242      * Parses a WWW-Authenticate header value.
243      * 
244      * @param value
245      *            the header value to parse
246      * 
247      * @return a map with the (lower case) challenge name as key and as the
248      *         value a sub-map with parameters of the challenge
249      */
250     public static Map<String, Map<String, String>> getChallengesFromAuthenticateHeader(String value) {
251         if (value == null || value.length() == 0) {
252             return null;
253         }
254 
255         final String trimValue = value.trim();
256 
257         Map<String, Map<String, String>> result = new HashMap<String, Map<String, String>>();
258 
259         boolean inQuotes = false;
260         boolean inName = true;
261         String challenge = null;
262         String paramName = "";
263         StringBuilder sb = new StringBuilder();
264         for (int i = 0; i < trimValue.length(); i++) {
265             char c = trimValue.charAt(i);
266 
267             if (c == '\\') {
268                 if (!inQuotes) {
269                     return null;
270                 }
271                 if (trimValue.length() > i && trimValue.charAt(i + 1) == '\\') {
272                     sb.append('\\');
273                     i++;
274                 } else if (trimValue.length() > i && trimValue.charAt(i + 1) == '"') {
275                     sb.append('"');
276                     i++;
277                 } else {
278                     return null;
279                 }
280             } else if (c == '"') {
281                 if (inName) {
282                     return null;
283                 }
284                 if (inQuotes) {
285                     Map<String, String> authMap = result.get(challenge);
286                     if (authMap == null) {
287                         return null;
288                     }
289                     authMap.put(paramName, sb.toString());
290                 }
291                 sb = new StringBuilder();
292                 inQuotes = !inQuotes;
293             } else if (c == '=') {
294                 if (inName) {
295                     paramName = sb.toString().trim();
296 
297                     int spcIdx = paramName.indexOf(' ');
298                     if (spcIdx > -1) {
299                         challenge = paramName.substring(0, spcIdx).toLowerCase(Locale.ENGLISH);
300                         result.put(challenge, new HashMap<String, String>());
301                         paramName = paramName.substring(spcIdx).trim();
302                     }
303 
304                     sb = new StringBuilder();
305                     inName = false;
306                 } else if (!inQuotes) {
307                     return null;
308                 }
309             } else if (c == ',') {
310                 if (inName) {
311                     challenge = sb.toString().trim().toLowerCase(Locale.ENGLISH);
312                     result.put(challenge, new HashMap<String, String>());
313                     sb = new StringBuilder();
314                 } else {
315                     if (inQuotes) {
316                         sb.append(c);
317                     } else {
318                         Map<String, String> authMap = result.get(challenge);
319                         if (authMap == null) {
320                             return null;
321                         }
322                         if (!authMap.containsKey(paramName)) {
323                             authMap.put(paramName, sb.toString().trim());
324                         }
325                         sb = new StringBuilder();
326                         inName = true;
327                     }
328                 }
329             } else {
330                 sb.append(c);
331             }
332         }
333         if (inQuotes) {
334             return null;
335         }
336         if (inName) {
337             challenge = sb.toString().trim().toLowerCase(Locale.ENGLISH);
338             result.put(challenge, new HashMap<String, String>());
339         } else {
340             Map<String, String> authMap = result.get(challenge);
341             if (authMap == null) {
342                 return null;
343             }
344             if (!authMap.containsKey(paramName)) {
345                 authMap.put(paramName, sb.toString().trim());
346             }
347         }
348 
349         return result;
350     }

353      * Gets the boundary from a <code>multipart/formdata</code> content type
354      * header.
355      * 
356      * @param value
357      *            the header value to decode
358      * @return the boundary as a byte array or <code>null</code> if no valid
359      *         boundary available
360      */
361     public static byte[] getBoundaryFromMultiPart(String value) {
362         try {
363             HeaderTokenizer tokenizer = new HeaderTokenizer(value, ";", true);
364 
365             // get the first token, which must be an ATOM
366             Token token = tokenizer.next();
367             if (token.getType() != Token.ATOM) {
368                 return null;
369             }
370 
371             // check content type
372             String multipartContentType = token.getValue();
373             if (multipartContentType == null
374                     || !(multipartContentType.equalsIgnoreCase("multipart/form-data") || multipartContentType
375                             .equalsIgnoreCase("multipart/related"))) {
376                 return null;
377             }
378 
379             // the remainder is the parameters
380             String remainder = tokenizer.getRemainder();
381             if (remainder != null) {
382                 Map<String, String> params = new HashMap<String, String>();
383                 getParameters(remainder, params);
384 
385                 String boundaryStr = params.get("boundary");
386                 if (boundaryStr != null && boundaryStr.length() > 0) {
387                     try {
388                         return boundaryStr.getBytes(IOUtils.ISO_8859_1);
389                     } catch (UnsupportedEncodingException e) {
390                         // shouldn't happen...
391                         throw new CmisRuntimeException("Unsupported encoding 'ISO-8859-1'", e);
392                     }
393                 }
394             }
395         } catch (ParseException e) {
396             return null;
397         }
398 
399         return null;
400     }

448     protected static class HeaderTokenizer {
449 
450         private static final Token EOF = new Token(Token.EOF, null);
451 
452         private final String header;
453 
454         private final String delimiters;
455 
456         private final boolean skipComments;
457 
458         private int pos;
459 
460         public HeaderTokenizer(String header) {
461             this(header, MIME_SPECIALS, true);
462         }

464         protected HeaderTokenizer(String header, String delimiters, boolean skipComments) {
465             this.header = header;
466             this.delimiters = delimiters;
467             this.skipComments = skipComments;
468         }
469 
470         public String getRemainder() {
471             return header.substring(pos);
472         }
473 
474         public Token next() throws ParseException {
475             return readToken();
476         }

478         /**
479          * Read an ATOM token from the parsed header.
480          * 
481          * @return A token containing the value of the atom token.
482          */
483         private Token readAtomicToken() {
484             // skip to next delimiter
485             int start = pos;
486             while (++pos < header.length()) {
487                 // break on the first non-atom character.
488                 char ch = header.charAt(pos);
489                 if (delimiters.indexOf(header.charAt(pos)) != -1 || ch < 32 || ch >= 127) {
490                     break;
491                 }
492             }
493             return new Token(Token.ATOM, header.substring(start, pos));
494         }
495 
496         /**
497          * Read the next token from the header.
498          * 
499          * @return The next token from the header. White space is skipped, and
500          *         comment tokens are also skipped if indicated.
501          */
502         private Token readToken() throws ParseException {
503             if (pos >= header.length()) {
504                 return EOF;
505             } else {
506                 char c = header.charAt(pos);
507                 // comment token...read and skip over this
508                 if (c == '(') {
509                     Token comment = readComment();
510                     if (skipComments) {
511                         return readToken();
512                     } else {
513                         return comment;
514                     }
515                     // quoted literal
516                 } else if (c == '\"') {
517                     return readQuotedString();
518                     // white space, eat this and find a real token.
519                 } else if (WHITE.indexOf(c) != -1) {
520                     eatWhiteSpace();
521                     return readToken();
522                     // either a CTL or special. These characters have a
523                     // self-defining token type.
524                 } else if (c < 32 || c >= 127 || delimiters.indexOf(c) != -1) {
525                     pos++;
526                     return new Token((int) c, String.valueOf(c));
527                 } else {
528                     // start of an atom, parse it off.
529                     return readAtomicToken();
530                 }
531             }
532         }

534         /**
535          * Extract a substring from the header string and apply any
536          * escaping/folding rules to the string.
537          * 
538          * @param start
539          *            The starting offset in the header.
540          * @param end
541          *            The header end offset + 1.
542          * @return The processed string value.
543          */
544         private String getEscapedValue(int start, int end) throws ParseException {
545             StringBuilder value = new StringBuilder();
546             for (int i = start; i < end; i++) {
547                 char ch = header.charAt(i);
548                 // is this an escape character?
549                 if (ch == '\\') {
550                     i++;
551                     if (i == end) {
552                         throw new ParseException("Invalid escape character");
553                     }
554                     value.append(header.charAt(i));
555                 } else if (ch == '\r') {
556                     // line breaks are ignored, except for naked '\n'
557                     // characters, which are consider parts of linear
558                     // whitespace.
559                     // see if this is a CRLF sequence, and skip the second if it
560                     // is.
561                     if (i < end - 1 && header.charAt(i + 1) == '\n') {
562                         i++;
563                     }
564                 } else {
565                     // just append the ch value.
566                     value.append(ch);
567                 }
568             }
569             return value.toString();
570         }
571 
572         /**
573          * Read a comment from the header, applying nesting and escape rules to
574          * the content.
575          * 
576          * @return A comment token with the token value.
577          */
578         private Token readComment() throws ParseException {
579             int start = pos + 1;
580             int nesting = 1;
581             boolean requiresEscaping = false;
582             // skip to end of comment/string
583             while (++pos < header.length()) {
584                 char ch = header.charAt(pos);
585                 if (ch == ')') {
586                     nesting--;
587                     if (nesting == 0) {
588                         break;
589                     }
590                 } else if (ch == '(') {
591                     nesting++;
592                 } else if (ch == '\\') {
593                     pos++;
594                     requiresEscaping = true;
595                 } else if (ch == '\r') {
596                     // we need to process line breaks also
597                     requiresEscaping = true;
598                 }
599             }
600             if (nesting != 0) {
601                 throw new ParseException("Unbalanced comments");
602             }
603             String value;
604             if (requiresEscaping) {
605                 value = getEscapedValue(start, pos);
606             } else {
607                 value = header.substring(start, pos++);
608             }
609             return new Token(Token.COMMENT, value);
610         }
611 
612         /**
613          * Parse out a quoted string from the header, applying escaping rules to
614          * the value.
615          * 
616          * @return The QUOTEDSTRING token with the value.
617          * @exception ParseException
618          */
619         private Token readQuotedString() throws ParseException {
620             int start = pos + 1;
621             boolean requiresEscaping = false;
622             // skip to end of comment/string
623             while (++pos < header.length()) {
624                 char ch = header.charAt(pos);
625                 if (ch == '"') {
626                     String value;
627                     if (requiresEscaping) {
628                         value = getEscapedValue(start, pos++);
629                     } else {
630                         value = header.substring(start, pos++);
631                     }
632                     return new Token(Token.QUOTEDSTRING, value);
633                 } else if (ch == '\\') {
634                     pos++;
635                     requiresEscaping = true;
636                 } else if (ch == '\r') {
637                     // we need to process line breaks also
638                     requiresEscaping = true;
639                 }
640             }
641             throw new ParseException("Missing '\"'");
642         }

644         /**
645          * Skip white space in the token string.
646          */
647         private void eatWhiteSpace() {
648             // skip to end of whitespace
649             while (++pos < header.length() && WHITE.indexOf(header.charAt(pos)) != -1) {
650                 // just read
651             }
652         }

655     /*
656      * Tweaked from geronimo-javamail_1.4_spec-1.7.1. ParameterList
657      */
658     protected static Map<String, String> getParameters(String list, Map<String, String> params) throws ParseException {
659         HeaderTokenizer tokenizer = new HeaderTokenizer(list);
660         while (true) {
661             Token token = tokenizer.next();
662             switch (token.getType()) {
663             case Token.EOF:
664                 // the EOF token terminates parsing.
665                 return params;
666 
667             case ';':
668                 // each new parameter is separated by a semicolon, including
669                 // the first, which separates
670                 // the parameters from the main part of the header.
671                 // the next token needs to be a parameter name
672                 token = tokenizer.next();
673                 // allow a trailing semicolon on the parameters.
674                 if (token.getType() == Token.EOF) {
675                     return params;
676                 }
677 
678                 if (token.getType() != Token.ATOM) {
679                     throw new ParseException("Invalid parameter name: " + token.getValue());
680                 }
681 
682                 // get the parameter name as a lower case version for better
683                 // mapping.
684                 String name = token.getValue().toLowerCase(Locale.ENGLISH);
685 
686                 token = tokenizer.next();
687 
688                 // parameters are name=value, so we must have the "=" here.
689                 if (token.getType() != '=') {
690                     throw new ParseException("Missing '='");
691                 }
692 
693                 // now the value, which may be an atom or a literal
694                 token = tokenizer.next();
695 
696                 if (token.getType() != Token.ATOM && token.getType() != Token.QUOTEDSTRING) {
697                     throw new ParseException("Invalid parameter value: " + token.getValue());
698                 }
699 
700                 String value = token.getValue();
701 
702                 // we might have to do some additional decoding. A name that
703                 // ends with "*" is marked as being encoded, so if requested, we
704                 // decode the value.
705                 if (name.endsWith("*")) {
706                     name = name.substring(0, name.length() - 1);
707                     value = decodeRFC2231value(value);
708                 }
709                 params.put(name, value);
710                 break;
711             default:
712                 throw new ParseException("Missing ';'");
713             }
714         }
715     }

717     protected static String decodeRFC2231value(String value) {
718         int q1 = value.indexOf('\'');
719         if (q1 == -1) {
720             // missing charset
721             return value;
722         }
723         String mimeCharset = value.substring(0, q1);
724         int q2 = value.indexOf('\'', q1 + 1);
725         if (q2 == -1) {
726             // missing language
727             return value;
728         }
729         byte[] bytes = fromHex(value.substring(q2 + 1));
730         try {
731             return new String(bytes, getJavaCharset(mimeCharset));
732         } catch (UnsupportedEncodingException e) {
733             // incorrect encoding
734             return value;
735         }
736     }

738     protected static byte[] fromHex(String data) {
739         ByteArrayOutputStream out = new ByteArrayOutputStream(data.length());
740         for (int i = 0; i < data.length();) {
741             char c = data.charAt(i++);
742             if (c == '%') {
743                 if (i > data.length() - 2) {
744                     break; // unterminated sequence
745                 }
746                 byte b1 = HEX_DECODE[data.charAt(i++) & 0x7f];
747                 byte b2 = HEX_DECODE[data.charAt(i++) & 0x7f];
748                 out.write((b1 << 4) | b2);
749             } else {
750                 out.write((byte) c);
751             }
752         }
753         return out.toByteArray();
754     }
