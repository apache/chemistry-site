85       * {@inheritDoc}

87       * <p>
88       * <b>Implementation Hints:</b>
89       * <ul>

91       * <li>Implementation is required. Convenience implementation is present.</li>

94      public RepositoryInfo getRepositoryInfo(String repositoryId, ExtensionsData extension) {
95          RepositoryInfo result = null;
96  
97          List<RepositoryInfo> repositories = getRepositoryInfos(extension);
98          if (repositories != null) {
99              for (RepositoryInfo ri : repositories) {
100                 if (ri.getId().equals(repositoryId)) {
101                     result = ri;
102                     break;
103                 }
104             }
105         }
106 
107         if (result == null) {
108             throw new CmisObjectNotFoundException("Repository '" + repositoryId + "' does not exist!");
109         }
110 
111         return result;
112     }

143      * <b>Implementation Hints:</b>
144      * <ul>

147      * </ul>

149     public List<TypeDefinitionContainer> getTypeDescendants(String repositoryId, String typeId, BigInteger depth,
150             Boolean includePropertyDefinitions, ExtensionsData extension) {
151         // check depth
152         int d = (depth == null ? -1 : depth.intValue());
153         if (d == 0) {
154             throw new CmisInvalidArgumentException("Depth must not be 0!");
155         }
156         if (typeId == null) {
157             d = -1;
158         }
159 
160         List<TypeDefinitionContainer> result = new ArrayList<TypeDefinitionContainer>();
161 
162         TypeDefinitionList children = getTypeChildren(repositoryId, typeId, includePropertyDefinitions,
163                 BigInteger.valueOf(Integer.MAX_VALUE), BigInteger.ZERO, null);
164 
165         if (children != null && isNotEmpty(children.getList())) {
166             for (TypeDefinition td : children.getList()) {
167                 TypeDefinitionContainerImpl tdc = new TypeDefinitionContainerImpl(td);
168                 addTypeChildren(repositoryId, includePropertyDefinitions, (d > 0 ? d - 1 : -1), tdc);
169                 result.add(tdc);
170             }
171         }
172 
173         return result;
174     }
175 
176     /**
177      * Helper method for
178      * {@link #getTypeDescendants(String, String, BigInteger, Boolean, ExtensionsData)}
179      * .
180      */
181     private void addTypeChildren(String repositoryId, Boolean includePropertyDefinitions, int depth,
182             TypeDefinitionContainerImpl container) {
183 
184         if (depth == 0) {
185             return;
186         }
187 
188         TypeDefinitionList children = getTypeChildren(repositoryId, container.getTypeDefinition().getId(),
189                 includePropertyDefinitions, BigInteger.valueOf(Integer.MAX_VALUE), BigInteger.ZERO, null);
190 
191         if (children != null && isNotEmpty(children.getList())) {
192             List<TypeDefinitionContainer> list = new ArrayList<TypeDefinitionContainer>();
193             container.setChildren(list);
194 
195             for (TypeDefinition td : children.getList()) {
196                 TypeDefinitionContainerImpl tdc = new TypeDefinitionContainerImpl(td);
197                 addTypeChildren(repositoryId, includePropertyDefinitions, (depth > 0 ? depth - 1 : -1), tdc);
198                 list.add(tdc);
199             }
200         }
201     }

360     /**
361      * {@inheritDoc}
362      * 
363      * <p>
364      * <b>Implementation Hints:</b>
365      * <ul>
366      * <li>Bindings: AtomPub</li>
367      * <li>Implementation is optional. Convenience implementation is present.</li>
368      * <li>Object infos should contain the newly created object.</li>
369      * </ul>
370      */
371     public String create(String repositoryId, Properties properties, String folderId, ContentStream contentStream,
372             VersioningState versioningState, List<String> policies, ExtensionsData extension) {
373         // check properties
374         if (properties == null || properties.getProperties() == null) {
375             throw new CmisInvalidArgumentException("Properties must be set!");
376         }
377 
378         // check object type id
379         PropertyData<?> obbjectTypeIdProperty = properties.getProperties().get(PropertyIds.OBJECT_TYPE_ID);
380         if (obbjectTypeIdProperty == null || !(obbjectTypeIdProperty.getFirstValue() instanceof String)) {
381             throw new CmisInvalidArgumentException("Property '" + PropertyIds.OBJECT_TYPE_ID + "' must be set!");
382         }
383 
384         // get the type
385         String objectTypeId = obbjectTypeIdProperty.getFirstValue().toString();
386         TypeDefinition type = getTypeDefinition(repositoryId, objectTypeId, null);
387 
388         // create object
389         String newId;
390         switch (type.getBaseTypeId()) {
391         case CMIS_DOCUMENT:
392             newId = createDocument(repositoryId, properties, folderId, contentStream, versioningState, policies, null,
393                     null, extension);
394             break;
395         case CMIS_FOLDER:
396             newId = createFolder(repositoryId, properties, folderId, policies, null, null, extension);
397             break;
398         case CMIS_POLICY:
399             newId = createPolicy(repositoryId, properties, folderId, policies, null, null, extension);
400             break;
401         case CMIS_ITEM:
402             newId = createItem(repositoryId, properties, folderId, policies, null, null, extension);
403             break;
404         default:
405             newId = null;
406         }
407 
408         // check new object id
409         if (newId == null) {
410             throw new CmisRuntimeException("Creation failed!");
411         }
412 
413         // return the new object id
414         return newId;
415     }

1026      * Returns the object info map.
1027      */
1028     private Map<String, ObjectInfo> getObjectInfoMap() {

1030             objectInfoMap = new HashMap<String, ObjectInfo>();
1031         }

1036     /**
1037      * {@inheritDoc}
1038      * 
1039      * <p>
1040      * <b>Implementation Hints:</b>
1041      * <ul>
1042      * <li>Bindings: AtomPub</li>
1043      * <li>If the object info is not found, the object info will be assembled.
1044      * To do that the repository info, the object, the object parent, the object
1045      * history and the base type definitions will be fetched. If you want to
1046      * change this behavior, override this method.</li>
1047      * </ul>
1048      */
1049     public ObjectInfo getObjectInfo(String repositoryId, String objectId) {
1050         Map<String, ObjectInfo> oim = getObjectInfoMap();
1051         ObjectInfo info = oim.get(objectId);
1052         if (info == null) {
1053             // object info has not been found -> create one
1054             try {
1055                 // switch off object info collection to avoid side effects
1056                 addObjectInfos = false;
1057 
1058                 // get the object and its info
1059                 ObjectData object = getObject(repositoryId, objectId, null, Boolean.TRUE, IncludeRelationships.BOTH,
1060                         "*", Boolean.TRUE, Boolean.FALSE, null);
1061                 info = getObjectInfoIntern(repositoryId, object);
1062 
1063                 // switch on object info collection
1064                 addObjectInfos = true;
1065 
1066                 // add object info
1067                 addObjectInfo(info);
1068             } catch (Exception e) {
1069                 info = null;
1070 
1071                 if (LOG.isTraceEnabled()) {
1072                     LOG.trace("Getting the object info for object {} in repository {}  failed: {}", objectId,
1073                             repositoryId, e.toString(), e);
1074                 }
1075             } finally {
1076                 addObjectInfos = true;
1077             }
1078         }
1079         return info;
1080     }

1082     /**
1083      * Collects the {@link ObjectInfo} about an object.
1084      * 
1085      * @param repositoryId
1086      *            the repository id
1087      * @param object
1088      *            the object
1089      * @return the collected object info
1090      */
1091     protected ObjectInfo getObjectInfoIntern(String repositoryId, ObjectData object) {
1092         // if the object has no properties, stop here
1093         if (object.getProperties() == null || object.getProperties().getProperties() == null) {
1094             throw new CmisRuntimeException("No properties!");
1095         }
1096 
1097         ObjectInfoImpl info = new ObjectInfoImpl();
1098 
1099         // get the repository info
1100         RepositoryInfo repositoryInfo = null;
1101         try {
1102             repositoryInfo = getRepositoryInfo(repositoryId, null);
1103         } catch (CmisRuntimeException e) {
1104             LOG.error("getRepositoryInfo returned an error while compiling object info for object {}.", object.getId(),
1105                     e);
1106             throw e;
1107         }
1108 
1109         // general properties
1110         info.setObject(object);
1111         info.setId(object.getId());
1112         info.setName(getStringProperty(object, PropertyIds.NAME));
1113         info.setCreatedBy(getStringProperty(object, PropertyIds.CREATED_BY));
1114         info.setCreationDate(getDateTimeProperty(object, PropertyIds.CREATED_BY));
1115         info.setLastModificationDate(getDateTimeProperty(object, PropertyIds.LAST_MODIFICATION_DATE));
1116         info.setTypeId(getIdProperty(object, PropertyIds.OBJECT_TYPE_ID));
1117         info.setBaseType(object.getBaseTypeId());
1118 
1119         // versioning
1120         info.setIsCurrentVersion(object.getBaseTypeId() == BaseTypeId.CMIS_DOCUMENT);
1121         info.setWorkingCopyId(null);
1122         info.setWorkingCopyOriginalId(null);
1123 
1124         info.setVersionSeriesId(getIdProperty(object, PropertyIds.VERSION_SERIES_ID));
1125         if (info.getVersionSeriesId() != null) {
1126             Boolean isLatest = getBooleanProperty(object, PropertyIds.IS_LATEST_VERSION);
1127             info.setIsCurrentVersion(isLatest == null ? true : isLatest.booleanValue());
1128 
1129             Boolean isCheckedOut = getBooleanProperty(object, PropertyIds.IS_VERSION_SERIES_CHECKED_OUT);
1130             if (isCheckedOut != null && isCheckedOut.booleanValue()) {
1131                 info.setWorkingCopyId(getIdProperty(object, PropertyIds.VERSION_SERIES_CHECKED_OUT_ID));
1132 
1133                 // get latest version
1134                 try {
1135                     List<ObjectData> versions = getAllVersions(repositoryId, object.getId(), info.getVersionSeriesId(),
1136                             null, Boolean.FALSE, null);
1137                     if (isNotEmpty(versions)) {
1138                         info.setWorkingCopyOriginalId(versions.get(0).getId());
1139                     }
1140                 } catch (CmisNotSupportedException nse) {
1141                     if (LOG.isWarnEnabled()) {
1142                         LOG.warn("getAllVersions is not implemented! Object info for object {} might be incorrect.",
1143                                 info.getId(), nse);
1144                     }
1145                 }
1146             }
1147         }
1148 
1149         // content
1150         String fileName = getStringProperty(object, PropertyIds.CONTENT_STREAM_FILE_NAME);
1151         String mimeType = getStringProperty(object, PropertyIds.CONTENT_STREAM_MIME_TYPE);
1152         String streamId = getIdProperty(object, PropertyIds.CONTENT_STREAM_ID);
1153         BigInteger length = getIntegerProperty(object, PropertyIds.CONTENT_STREAM_LENGTH);
1154         boolean hasContent = fileName != null || mimeType != null || streamId != null || length != null;
1155         if (hasContent) {
1156             info.setHasContent(hasContent);
1157             info.setContentType(mimeType);
1158             info.setFileName(fileName);
1159         } else {
1160             info.setHasContent(false);
1161             info.setContentType(null);
1162             info.setFileName(null);
1163         }
1164 
1165         // parent
1166         if (object.getBaseTypeId() == BaseTypeId.CMIS_RELATIONSHIP) {
1167             info.setHasParent(false);
1168         } else if (object.getBaseTypeId() == BaseTypeId.CMIS_FOLDER) {
1169             info.setHasParent(!object.getId().equals(repositoryInfo.getRootFolderId()));
1170         } else {
1171             try {
1172                 List<ObjectParentData> parents = getObjectParents(repositoryId, object.getId(), null, Boolean.FALSE,
1173                         IncludeRelationships.NONE, "cmis:none", Boolean.FALSE, null);
1174                 info.setHasParent(isNotEmpty(parents));
1175             } catch (CmisInvalidArgumentException e) {
1176                 info.setHasParent(false);
1177             } catch (CmisNotSupportedException nse) {
1178                 if (LOG.isWarnEnabled()) {
1179                     LOG.warn("getObjectParents is not implemented! Object info for object {} might be incorrect.",
1180                             info.getId(), nse);
1181                 }
1182             }
1183         }
1184 
1185         // policies and relationships
1186         info.setSupportsRelationships(false);
1187         info.setSupportsPolicies(false);
1188 
1189         try {
1190             TypeDefinitionList baseTypesList = getTypeChildren(repositoryId, null, Boolean.FALSE,
1191                     BigInteger.valueOf(6), BigInteger.ZERO, null);
1192             for (TypeDefinition type : baseTypesList.getList()) {
1193                 if (BaseTypeId.CMIS_RELATIONSHIP.value().equals(type.getId())) {
1194                     info.setSupportsRelationships(true);
1195                 } else if (BaseTypeId.CMIS_POLICY.value().equals(type.getId())) {
1196                     info.setSupportsPolicies(true);
1197                 }
1198             }
1199         } catch (CmisNotSupportedException nse) {
1200             if (LOG.isWarnEnabled()) {
1201                 LOG.warn("getTypeChildren is not implemented! Object info for object {} might be incorrect.",
1202                         info.getId(), nse);
1203             }
1204         }
1205 
1206         // renditions
1207         info.setRenditionInfos(null);
1208         List<RenditionData> renditions = object.getRenditions();
1209         if (isNotEmpty(renditions)) {
1210             List<RenditionInfo> renditionInfos = new ArrayList<RenditionInfo>();
1211             for (RenditionData rendition : renditions) {
1212                 RenditionInfoImpl renditionInfo = new RenditionInfoImpl();
1213                 renditionInfo.setId(rendition.getStreamId());
1214                 renditionInfo.setKind(rendition.getKind());
1215                 renditionInfo.setContentType(rendition.getMimeType());
1216                 renditionInfo.setTitle(rendition.getTitle());
1217                 renditionInfo.setLength(rendition.getBigLength());
1218                 renditionInfos.add(renditionInfo);
1219             }
1220             info.setRenditionInfos(renditionInfos);
1221         }
1222 
1223         // relationships
1224         info.setRelationshipSourceIds(null);
1225         info.setRelationshipTargetIds(null);
1226         List<ObjectData> relationships = object.getRelationships();
1227         if (isNotEmpty(relationships)) {
1228             List<String> sourceIds = new ArrayList<String>();
1229             List<String> targetIds = new ArrayList<String>();
1230             for (ObjectData relationship : relationships) {
1231                 String sourceId = getIdProperty(relationship, PropertyIds.SOURCE_ID);
1232                 String targetId = getIdProperty(relationship, PropertyIds.TARGET_ID);
1233                 if (object.getId().equals(sourceId)) {
1234                     sourceIds.add(relationship.getId());
1235                 }
1236                 if (object.getId().equals(targetId)) {
1237                     targetIds.add(relationship.getId());
1238                 }
1239             }
1240             if (isNotEmpty(sourceIds)) {
1241                 info.setRelationshipSourceIds(sourceIds);
1242             }
1243             if (isNotEmpty(targetIds)) {
1244                 info.setRelationshipTargetIds(targetIds);
1245             }
1246         }
1247 
1248         // global settings
1249         info.setHasAcl(false);
1250         info.setSupportsDescendants(false);
1251         info.setSupportsFolderTree(false);
1252 
1253         RepositoryCapabilities capabilities = repositoryInfo.getCapabilities();
1254         if (capabilities != null) {
1255             info.setHasAcl(capabilities.getAclCapability() == CapabilityAcl.DISCOVER
1256                     || capabilities.getAclCapability() == CapabilityAcl.MANAGE);
1257             if (object.getBaseTypeId() == BaseTypeId.CMIS_FOLDER) {
1258                 info.setSupportsDescendants(Boolean.TRUE.equals(capabilities.isGetDescendantsSupported()));
1259                 info.setSupportsFolderTree(Boolean.TRUE.equals(capabilities.isGetFolderTreeSupported()));
1260             }
1261         }
1262 
1263         return info;
1264     }
