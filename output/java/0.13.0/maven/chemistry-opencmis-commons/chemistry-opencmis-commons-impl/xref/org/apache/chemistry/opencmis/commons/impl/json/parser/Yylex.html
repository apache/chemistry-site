74      private static int zzUnpackAction(String packed, int offset, int[] result) {
75          int i = 0; /* index in packed string */
76          int j = offset; /* index in unpacked array */
77          int l = packed.length();
78          while (i < l) {
79              int count = packed.charAt(i++);
80              int value = packed.charAt(i++);
81              do {
82                  result[j++] = value;
83              } while (--count > 0);
84          }
85          return j;
86      }

258     /**
259      * Refills the input buffer.
260      * 
261      * @return <code>false</code>, iff there was new input.
262      * 
263      * @exception java.io.IOException
264      *                if any I/O-Error occurs
265      */
266     private boolean zzRefill() throws java.io.IOException {
267 
268         /* first: make room (if you can) */
269         if (zzStartRead > 0) {
270             System.arraycopy(zzBuffer, zzStartRead, zzBuffer, 0, zzEndRead - zzStartRead);
271 
272             /* translate stored positions */
273             zzEndRead -= zzStartRead;
274             zzCurrentPos -= zzStartRead;
275             zzMarkedPos -= zzStartRead;
276             zzStartRead = 0;
277         }
278 
279         /* is the buffer big enough? */
280         if (zzCurrentPos >= zzBuffer.length) {
281             /* if not: blow it up */
282             char newBuffer[] = new char[zzCurrentPos * 2];
283             System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
284             zzBuffer = newBuffer;
285         }
286 
287         /* finally: fill the buffer with new input */
288         int numRead = zzReader.read(zzBuffer, zzEndRead, zzBuffer.length - zzEndRead);
289 
290         if (numRead > 0) {
291             zzEndRead += numRead;
292             return false;
293         }
294         // unlikely but not impossible: read 0 characters, but not at end of
295         // stream
296         if (numRead == 0) {
297             int c = zzReader.read();
298             if (c == -1) {
299                 return true;
300             } else {
301                 zzBuffer[zzEndRead++] = (char) c;
302                 return false;
303             }
304         }
305 
306         // numRead < 0
307         return true;
308     }

390     /**
391      * Reports an error that occured while scanning.
392      * 
393      * In a wellformed scanner (no or only correct usage of yypushback(int) and
394      * a match-all fallback rule) this method will only be called with things
395      * that "Can't Possibly Happen". If this method is called, something is
396      * seriously wrong (e.g. a JFlex bug producing a faulty scanner etc.).
397      * 
398      * Usual syntax/scanner level error handling should be done in error
399      * fallback rules.
400      * 
401      * @param errorCode
402      *            the code of the errormessage to display
403      */
404     private void zzScanError(int errorCode) {
405         String message;
406         try {
407             message = ZZ_ERROR_MSG[errorCode];
408         } catch (ArrayIndexOutOfBoundsException e) {
409             message = ZZ_ERROR_MSG[ZZ_UNKNOWN_ERROR];
410         }
411 
412         throw new Error(message);
413     }

415     /**
416      * Pushes the specified amount of characters back into the input stream.
417      * 
418      * They will be read again by then next call of the scanning method
419      * 
420      * @param number
421      *            the number of characters to be read again. This number must
422      *            not be greater than yylength()!
423      */
424     public void yypushback(int number) {
425         if (number > yylength()) {
426             zzScanError(ZZ_PUSHBACK_2BIG);
427         }
428 
429         zzMarkedPos -= number;
430     }

432     /**
433      * Resumes scanning until the next regular expression is matched, the end of
434      * input is encountered or an I/O-Error occurs.
435      * 
436      * @return the next token
437      * @exception java.io.IOException
438      *                if any I/O-Error occurs
439      */
440     public Yytoken yylex() throws java.io.IOException, JSONParseException {
441         int zzInput;
442         int zzAction;
443 
444         // cached fields:
445         int zzCurrentPosL;
446         int zzMarkedPosL;
447         int zzEndReadL = zzEndRead;
448         char[] zzBufferL = zzBuffer;
449         char[] zzCMapL = ZZ_CMAP;
450 
451         int[] zzTransL = ZZ_TRANS;
452         int[] zzRowMapL = ZZ_ROWMAP;
453         int[] zzAttrL = ZZ_ATTRIBUTE;
454 
455         while (true) {
456             zzMarkedPosL = zzMarkedPos;
457 
458             yychar += zzMarkedPosL - zzStartRead;
459 
460             zzAction = -1;
461 
462             zzCurrentPosL = zzMarkedPosL;
463             zzCurrentPos = zzMarkedPosL;
464             zzStartRead = zzMarkedPosL;
465 
466             zzState = ZZ_LEXSTATE[zzLexicalState];
467 
468             zzForAction: {
469                 while (true) {
470 
471                     if (zzCurrentPosL < zzEndReadL) {
472                         zzInput = zzBufferL[zzCurrentPosL++];
473                     } else if (zzAtEOF) {
474                         zzInput = YYEOF;
475                         break zzForAction;
476                     } else {
477                         // store back cached positions
478                         zzCurrentPos = zzCurrentPosL;
479                         zzMarkedPos = zzMarkedPosL;
480                         boolean eof = zzRefill();
481                         // get translated positions and possibly new buffer
482                         zzCurrentPosL = zzCurrentPos;
483                         zzMarkedPosL = zzMarkedPos;
484                         zzBufferL = zzBuffer;
485                         zzEndReadL = zzEndRead;
486                         if (eof) {
487                             zzInput = YYEOF;
488                             break zzForAction;
489                         } else {
490                             zzInput = zzBufferL[zzCurrentPosL++];
491                         }
492                     }
493                     int zzNext = zzTransL[zzRowMapL[zzState] + zzCMapL[zzInput]];
494                     if (zzNext == -1) {
495                         break zzForAction;
496                     }
497                     zzState = zzNext;
498 
499                     int zzAttributes = zzAttrL[zzState];
500                     if ((zzAttributes & 1) == 1) {
501                         zzAction = zzState;
502                         zzMarkedPosL = zzCurrentPosL;
503                         if ((zzAttributes & 8) == 8) {
504                             break zzForAction;
505                         }
506                     }
507 
508                 }
509             }
510 
511             // store back cached position
512             zzMarkedPos = zzMarkedPosL;
513 
514             switch (zzAction < 0 ? zzAction : ZZ_ACTION[zzAction]) {
515             case 11: {
516                 sb.append(yytext());
517                 break;
518             }
519             case 25:
520                 break;
521             case 4: {
522                 sb.delete(0, sb.length());
523                 yybegin(STRING_BEGIN);
524                 break;
525             }
526             case 26:
527                 break;
528             case 16: {
529                 sb.append('\b');
530                 break;
531             }
532             case 27:
533                 break;
534             case 6: {
535                 return new Yytoken(Yytoken.TYPE_RIGHT_BRACE, null);
536             }
537             case 28:
538                 break;
539             case 23: {
540                 Boolean val = Boolean.valueOf(yytext());
541                 return new Yytoken(Yytoken.TYPE_VALUE, val);
542             }
543             case 29:
544                 break;
545             case 22: {
546                 return new Yytoken(Yytoken.TYPE_VALUE, null);
547             }
548             case 30:
549                 break;
550             case 13: {
551                 yybegin(YYINITIAL);
