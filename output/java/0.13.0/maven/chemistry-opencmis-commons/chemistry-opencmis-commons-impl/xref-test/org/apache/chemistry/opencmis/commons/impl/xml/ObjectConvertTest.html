161     protected ObjectDataImpl createObjectData(boolean addRelationships, CmisVersion cmisVersion, boolean withChanges,
162             boolean withExtensions) {
163         ObjectDataImpl result = new ObjectDataImpl();
164 
165         // properties
166         PropertiesImpl properties = new PropertiesImpl();
167         properties.addProperty(createIdPropertyData(PropertyIds.BASE_TYPE_ID, BaseTypeId.CMIS_DOCUMENT.value()));
168         properties.addProperty(createIdPropertyData(PropertyIds.OBJECT_ID, randomString()));
169         properties.addProperty(createIdPropertyData(PropertyIds.OBJECT_TYPE_ID, randomString()));
170         properties.addProperty(createPropertyData(PropertyType.BOOLEAN, 0));
171         properties.addProperty(createPropertyData(PropertyType.BOOLEAN, 1));
172         properties.addProperty(createPropertyData(PropertyType.BOOLEAN, randomInt(8) + 2));
173         properties.addProperty(createPropertyData(PropertyType.DATETIME, 0));
174         properties.addProperty(createPropertyData(PropertyType.DATETIME, 1));
175         properties.addProperty(createPropertyData(PropertyType.DATETIME, randomInt(8) + 2));
176         properties.addProperty(createPropertyData(PropertyType.DECIMAL, 0));
177         properties.addProperty(createPropertyData(PropertyType.DECIMAL, 1));
178         properties.addProperty(createPropertyData(PropertyType.DECIMAL, randomInt(8) + 2));
179         properties.addProperty(createPropertyData(PropertyType.HTML, 0));
180         properties.addProperty(createPropertyData(PropertyType.HTML, 1));
181         properties.addProperty(createPropertyData(PropertyType.HTML, randomInt(8) + 2));
182         properties.addProperty(createPropertyData(PropertyType.ID, 0));
183         properties.addProperty(createPropertyData(PropertyType.ID, 1));
184         properties.addProperty(createPropertyData(PropertyType.ID, randomInt(8) + 2));
185         properties.addProperty(createPropertyData(PropertyType.INTEGER, 0));
186         properties.addProperty(createPropertyData(PropertyType.INTEGER, 1));
187         properties.addProperty(createPropertyData(PropertyType.INTEGER, randomInt(8) + 2));
188         properties.addProperty(createPropertyData(PropertyType.STRING, 0));
189         properties.addProperty(createPropertyData(PropertyType.STRING, 1));
190         properties.addProperty(createPropertyData(PropertyType.STRING, randomInt(8) + 2));
191         properties.addProperty(createPropertyData(PropertyType.URI, 0));
192         properties.addProperty(createPropertyData(PropertyType.URI, 1));
193         properties.addProperty(createPropertyData(PropertyType.URI, randomInt(8) + 2));
194         if (withExtensions) {
195             properties.setExtensions(createExtensions(3));
196         }
197         result.setProperties(properties);
198 
199         // allowable actions
200         AllowableActionsImpl allowableActions = new AllowableActionsImpl();
201         Set<Action> actions = new HashSet<Action>();
202         for (Action action : Action.values()) {
203             if (action == Action.CAN_CREATE_ITEM && cmisVersion == CmisVersion.CMIS_1_0) {
204                 continue;
205             }
206             actions.add(action);
207         }
208         allowableActions.setAllowableActions(actions);
209         result.setAllowableActions(allowableActions);
210 
211         // relationships
212         if (addRelationships) {
213             List<ObjectData> relationships = new ArrayList<ObjectData>();
214             for (int i = 0; i < randomInt(4) + 1; i++) {
215                 relationships.add(createObjectData(false, cmisVersion, withChanges, withExtensions));
216             }
217             result.setRelationships(relationships);
218         }
219 
220         // change event info
221         if (withChanges) {
222             ChangeEventInfoDataImpl changeEventInfo = new ChangeEventInfoDataImpl();
223             changeEventInfo.setChangeTime(randomDateTime());
224             changeEventInfo.setChangeType(randomEnum(ChangeType.class));
225             result.setChangeEventInfo(changeEventInfo);
226         }
227 
228         // ACL
229         AccessControlListImpl acl = new AccessControlListImpl();
230         List<Ace> aces = new ArrayList<Ace>();
231         for (int i = 0; i < randomInt(9) + 1; i++) {
232             AccessControlEntryImpl ace = new AccessControlEntryImpl();
233             List<String> permissions = new ArrayList<String>();
234             for (int j = 0; j < randomInt(4) + 1; j++) {
235                 permissions.add(randomString());
236             }
237             ace.setPermissions(permissions);
238             ace.setDirect(randomBoolean());
239             ace.setPrincipal(new AccessControlPrincipalDataImpl(randomString()));
240             aces.add(ace);
241         }
242         acl.setAces(aces);
243         result.setAcl(acl);
244 
245         result.setIsExactAcl(randomBoolean());
246 
247         // policy ids
248         List<String> pIds = new ArrayList<String>();
249         for (int i = 0; i < randomInt(9) + 1; i++) {
250             pIds.add(randomString());
251         }
252         PolicyIdListImpl policyList = new PolicyIdListImpl();
253         policyList.setPolicyIds(pIds);
254         result.setPolicyIds(policyList);
255 
256         // renditions
257         List<RenditionData> renditions = new ArrayList<RenditionData>();
258         for (int i = 0; i < randomInt(4) + 1; i++) {
259             RenditionDataImpl rendition = new RenditionDataImpl();
260             rendition.setKind(randomString());
261             rendition.setMimeType(randomString());
262             rendition.setRenditionDocumentId(randomString());
263             rendition.setStreamId(randomString());
264             rendition.setTitle(randomString());
265             rendition.setBigLength(randomInteger());
266             rendition.setBigHeight(randomInteger());
267             rendition.setBigWidth(randomInteger());
268 
269             renditions.add(rendition);
270         }
271         result.setRenditions(renditions);
272 
273         // extensions
274         if (withExtensions) {
275             result.setExtensions(createExtensions(5));
276         }
277 
278         return result;
279     }
