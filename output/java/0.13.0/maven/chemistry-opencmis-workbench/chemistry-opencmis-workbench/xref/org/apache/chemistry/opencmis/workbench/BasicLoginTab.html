99      private void createBindingButtons(Container pane) {
100         JPanel bindingContainer = new JPanel();
101         bindingContainer.setLayout(new BoxLayout(bindingContainer, BoxLayout.LINE_AXIS));
102         char bc = System.getProperty(SYSPROP_BINDING, "atom").toLowerCase(Locale.ENGLISH).charAt(0);
103         boolean atom = (bc == 'a');
104         boolean ws = (bc == 'w');
105         boolean browser = (bc == 'b');
106         bindingAtomButton = new JRadioButton("AtomPub", atom);
107         bindingWebServicesButton = new JRadioButton("Web Services", ws);

213     @Override
214     public Map<String, String> getSessionParameters() {
215         String url = urlField.getText().trim();
216 
217         BindingType binding = BindingType.ATOMPUB;
218         if (bindingWebServicesButton.isSelected()) {
219             binding = BindingType.WEBSERVICES;
220         } else if (bindingBrowserButton.isSelected()) {
221             binding = BindingType.BROWSER;
222         }
223 
224         String username = usernameField.getText();
225         String password = new String(passwordField.getPassword());
226 
227         ClientSession.Authentication authentication = ClientSession.Authentication.NONE;
228         if (authenticationStandardButton.isSelected()) {
229             authentication = ClientSession.Authentication.STANDARD;
230         } else if (authenticationNTLMButton.isSelected()) {
231             authentication = ClientSession.Authentication.NTLM;
232         } else if (authenticationOAuthButton.isSelected()) {
233             authentication = ClientSession.Authentication.OAUTH_BEARER;
234         }
235         return ClientSession.createSessionParameters(url, binding, username, password, authentication,
236                 compressionOnButton.isSelected(), clientCompressionOnButton.isSelected(), cookiesOnButton.isSelected());
237     }
