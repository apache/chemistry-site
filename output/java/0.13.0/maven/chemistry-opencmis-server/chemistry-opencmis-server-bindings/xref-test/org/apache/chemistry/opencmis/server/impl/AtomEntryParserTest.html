220     @Test
221     public void testBigStream() throws Exception {
222         byte[] begin = IOUtils
223                 .toUTF8Bytes("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>"
224                         + "<entry xmlns=\"http://www.w3.org/2005/Atom\" xmlns:app=\"http://www.w3.org/2007/app\" xmlns:cmisra=\"http://docs.oasis-open.org/ns/cmis/restatom/200908/\">"
225                         + "<cmisra:content><cmisra:mediatype>video/raw</cmisra:mediatype><cmisra:base64>");
226         byte[] end = IOUtils
227                 .toUTF8Bytes("</cmisra:base64></cmisra:content><cmisra:object xmlns:cmis=\"http://docs.oasis-open.org/ns/cmis/core/200908/\">"
228                         + "<cmis:properties></cmis:properties></cmisra:object><title>some.file</title></entry>");
229 
230         int contenSize = 50 * 1024 * 1024;
231         int base64Size = 4 * (contenSize + 2 - ((contenSize + 2) % 3)) / 3;
232 
233         byte[] entry = new byte[begin.length + base64Size + end.length];
234 
235         System.arraycopy(begin, 0, entry, 0, begin.length);
236         System.arraycopy(end, 0, entry, entry.length - end.length, end.length);
237         Arrays.fill(entry, begin.length, entry.length - end.length, (byte) 'a');
238 
239         TempStoreOutputStreamFactory streamFactory = TempStoreOutputStreamFactory.newInstance(null, THRESHOLD,
240                 MAX_SIZE, false);
241         AtomEntryParser aep = new AtomEntryParser(new ByteArrayInputStream(entry), streamFactory);
242         ContentStream contentStream = aep.getContentStream();
243 
244         assertNotNull(contentStream);
245         assertNotNull(contentStream.getStream());
246 
247         contentStream.getStream().close();
248 
249         aep.release();
250         assertNull(aep.getContentStream());
251     }

253     private static byte[] parse(byte[] entry, String mimeType) throws Exception {
254         TempStoreOutputStreamFactory streamFactory = TempStoreOutputStreamFactory.newInstance(null, THRESHOLD,
255                 MAX_SIZE, false);
256         AtomEntryParser aep = new AtomEntryParser(new ByteArrayInputStream(entry), streamFactory);
257         ContentStream contentStream = aep.getContentStream();
258 
259         assertNotNull(contentStream);
260         assertNotNull(contentStream.getStream());
261 
262         assertEquals(mimeType, contentStream.getMimeType());
263         if (contentStream.getStream() instanceof ThresholdInputStream) {
264             assertEquals(mimeType, ((ThresholdInputStream) contentStream.getStream()).getMimeType());
265         }
266 
267         ByteArrayOutputStream baos = new ByteArrayOutputStream();
268 
269         IOUtils.copy(contentStream.getStream(), baos);
270 
271         contentStream.getStream().close();
272 
273         aep.release();
274         assertNull(aep.getContentStream());
275 
276         return baos.toByteArray();
277     }
