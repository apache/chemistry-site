83      /**
84       * Compiles the base URL for links, collections and templates.
85       */
86      public UrlBuilder compileBaseUrl(HttpServletRequest request, String repositoryId) {
87          String baseUrl = (String) request.getAttribute(Dispatcher.BASE_URL_ATTRIBUTE);
88          if (baseUrl != null) {
89              int repIdPos = baseUrl.indexOf(REPOSITORY_PLACEHOLDER);
90              if (repIdPos < 0) {
91                  return new UrlBuilder(baseUrl);
92              } else {
93                  return new UrlBuilder(baseUrl.substring(0, repIdPos) + repositoryId
94                          + baseUrl.substring(repIdPos + REPOSITORY_PLACEHOLDER.length()));
95              }
96          }
97  
98          UrlBuilder url = new UrlBuilder(request.getScheme(), request.getServerName(), request.getServerPort(), null);
99  
100         url.addPath(request.getContextPath());
101         url.addPath(request.getServletPath());
102         url.addPathSegment(repositoryId);
103 
104         return url;
105     }

123     /**
124      * Writes JSON to the servlet response and adds a callback wrapper if
125      * requested.
126      */
127     public void writeJSON(JSONStreamAware json, HttpServletRequest request, HttpServletResponse response)
128             throws IOException {
129         String token = getStringParameter(request, Constants.PARAM_TOKEN);
130 
131         if (token != null && "POST".equals(request.getMethod())) {
132             response.setContentType(HTML_MIME_TYPE);
133             response.setContentLength(0);
134         } else {
135             response.setContentType(JSON_MIME_TYPE);
136             response.setCharacterEncoding(IOUtils.UTF8);
137 
138             PrintWriter pw = response.getWriter();
139 
140             String callback = getStringParameter(request, Constants.PARAM_CALLBACK);
141             if (callback != null) {
142                 if (!callback.matches("[A-Za-z0-9._\\[\\]]*")) {
143                     throw new CmisInvalidArgumentException("Invalid callback name!");
144                 }
145                 pw.print(callback + "(");
146             }
147 
148             json.writeJSONString(pw);
149 
150             if (callback != null) {
151                 pw.print(");");
152             }
153 
154             pw.flush();
155         }
156     }

207     /**
208      * Sets a transaction cookie.
209      */
210     public void setCookie(HttpServletRequest request, HttpServletResponse response, String repositoryId, String token,
211             String value, int expiry) {
212         if (token != null && token.length() > 0) {
213             String cookieValue = IOUtils.encodeURL(value);
214 
215             Cookie transactionCookie = new Cookie(getCookieName(token), cookieValue);
216             transactionCookie.setMaxAge(expiry);
217             transactionCookie.setPath(request.getContextPath() + request.getServletPath() + "/" + repositoryId);
218             transactionCookie.setSecure(request.isSecure());
219 
220             response.addCookie(transactionCookie);
221         }
222     }

235     public Properties createNewProperties(ControlParser controlParser, TypeCache typeCache) {
236         Map<String, List<String>> properties = controlParser.getProperties();
237         if (properties == null) {
238             return null;
239         }
240 
241         // load primary type
242         List<String> objectTypeIdsValues = properties.get(PropertyIds.OBJECT_TYPE_ID);
243         if (isNotEmpty(objectTypeIdsValues)) {
244             TypeDefinition typeDef = typeCache.getTypeDefinition(objectTypeIdsValues.get(0));
245             if (typeDef == null) {
246                 throw new CmisInvalidArgumentException("Invalid type: " + objectTypeIdsValues.get(0));
247             }
248         }
249 
250         // load secondary types
251         List<String> secondaryObjectTypeIdsValues = properties.get(PropertyIds.SECONDARY_OBJECT_TYPE_IDS);
252         if (isNotEmpty(secondaryObjectTypeIdsValues)) {
253             for (String secTypeId : secondaryObjectTypeIdsValues) {
254                 TypeDefinition typeDef = typeCache.getTypeDefinition(secTypeId);
255                 if (typeDef == null) {
256                     throw new CmisInvalidArgumentException("Invalid type: " + secTypeId);
257                 }
258             }
259         }
260 
261         // create properties
262         PropertiesImpl result = new PropertiesImpl();
263         for (Map.Entry<String, List<String>> property : properties.entrySet()) {
264             PropertyDefinition<?> propDef = typeCache.getPropertyDefinition(property.getKey());
265             if (propDef == null) {
266                 throw new CmisInvalidArgumentException(property.getKey() + " is unknown!");
267             }
268 
269             result.addProperty(createPropertyData(propDef, property.getValue()));
270         }
271 
272         return result;
273     }

275     public Properties createUpdateProperties(ControlParser controlParser, String typeId, List<String> secondaryTypeIds,
276             List<String> objectIds, TypeCache typeCache) {
277         Map<String, List<String>> properties = controlParser.getProperties();
278         if (properties == null) {
279             return null;
280         }
281 
282         // load primary type
283         if (typeId != null) {
284             TypeDefinition typeDef = typeCache.getTypeDefinition(typeId);
285             if (typeDef == null) {
286                 throw new CmisInvalidArgumentException("Invalid type: " + typeId);
287             }
288         }
289 
290         // load secondary types
291         List<String> secondaryObjectTypeIdsValues = properties.get(PropertyIds.SECONDARY_OBJECT_TYPE_IDS);
292         if (isNotEmpty(secondaryObjectTypeIdsValues)) {
293             for (String secTypeId : secondaryObjectTypeIdsValues) {
294                 TypeDefinition typeDef = typeCache.getTypeDefinition(secTypeId);
295                 if (typeDef == null) {
296                     throw new CmisInvalidArgumentException("Invalid type: " + secTypeId);
297                 }
298             }
299         }
300 
301         if (secondaryTypeIds != null) {
302             for (String secTypeId : secondaryTypeIds) {
303                 TypeDefinition typeDef = typeCache.getTypeDefinition(secTypeId);
304                 if (typeDef == null) {
305                     throw new CmisInvalidArgumentException("Invalid secondary type: " + secTypeId);
306                 }
307             }
308         }
309 
310         // create properties
311         PropertiesImpl result = new PropertiesImpl();
312         for (Map.Entry<String, List<String>> property : properties.entrySet()) {
313             PropertyDefinition<?> propDef = typeCache.getPropertyDefinition(property.getKey());
314             if (propDef == null && objectIds != null) {
315                 for (String objectId : objectIds) {
316                     typeCache.getTypeDefinitionForObject(objectId);
317                     propDef = typeCache.getPropertyDefinition(property.getKey());
318                     if (propDef != null) {
319                         break;
320                     }
321                 }
322             }
323             if (propDef == null) {
324                 throw new CmisInvalidArgumentException(property.getKey() + " is unknown!");
325             }
326 
327             result.addProperty(createPropertyData(propDef, property.getValue()));
328         }
329 
330         return result;
331     }

333     @SuppressWarnings("unchecked")
334     private PropertyData<?> createPropertyData(PropertyDefinition<?> propDef, Object value) {
335 
336         List<String> strValues;
337         if (value == null) {
338             strValues = Collections.emptyList();
339         } else if (value instanceof String) {
340             strValues = new ArrayList<String>();
341             strValues.add((String) value);
342         } else {
343             strValues = (List<String>) value;
344         }
345 
346         PropertyData<?> propertyData = null;
347         switch (propDef.getPropertyType()) {
348         case STRING:
349             propertyData = new PropertyStringImpl(propDef.getId(), strValues);
350             break;
351         case ID:
352             propertyData = new PropertyIdImpl(propDef.getId(), strValues);
353             break;
354         case BOOLEAN:
355             List<Boolean> boolValues = new ArrayList<Boolean>(strValues.size());
356             for (String s : strValues) {
357                 boolValues.add(Boolean.valueOf(s));
358             }
359             propertyData = new PropertyBooleanImpl(propDef.getId(), boolValues);
360             break;
361         case INTEGER:
362             List<BigInteger> intValues = new ArrayList<BigInteger>(strValues.size());
363             try {
364                 for (String s : strValues) {
365                     intValues.add(new BigInteger(s));
366                 }
367             } catch (NumberFormatException e) {
368                 throw new CmisInvalidArgumentException(propDef.getId() + " value is not an integer value!", e);
369             }
370             propertyData = new PropertyIntegerImpl(propDef.getId(), intValues);
371             break;
372         case DECIMAL:
373             List<BigDecimal> decValues = new ArrayList<BigDecimal>(strValues.size());
374             try {
375                 for (String s : strValues) {
376                     decValues.add(new BigDecimal(s));
377                 }
378             } catch (NumberFormatException e) {
379                 throw new CmisInvalidArgumentException(propDef.getId() + " value is not an integer value!", e);
380             }
381             propertyData = new PropertyDecimalImpl(propDef.getId(), decValues);
382             break;
383         case DATETIME:
384             List<GregorianCalendar> calValues = new ArrayList<GregorianCalendar>(strValues.size());
385             for (String s : strValues) {
386                 GregorianCalendar cal;
387                 try {
388                     long timestamp = Long.parseLong(s);
389                     cal = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
390                     cal.setTimeInMillis(timestamp);
391                 } catch (NumberFormatException e) {
392                     cal = DateTimeHelper.parseXmlDateTime(s);
393                 }
394 
395                 if (cal == null) {
396                     throw new CmisInvalidArgumentException(propDef.getId() + " value is not an datetime value!");
397                 }
398 
399                 calValues.add(cal);
400             }
401 
402             propertyData = new PropertyDateTimeImpl(propDef.getId(), calValues);
403             break;
404         case HTML:
405             propertyData = new PropertyHtmlImpl(propDef.getId(), strValues);
406             break;
407         case URI:
408             propertyData = new PropertyUriImpl(propDef.getId(), strValues);
409             break;
410         default:
411             assert false;
412         }
413 
414         return propertyData;
415     }
