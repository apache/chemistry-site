61      /*
62       * In the public interface of this class we return the singleton containing
63       * the required types for testing.
64       * @see org.apache.chemistry.opencmis.inmemory.TypeCreator#createTypesList()
65       */
66      @Override

68          return SINGLETON_TYPES;

71      public static List<TypeDefinition> getTypesList() {
72          return SINGLETON_TYPES;
73      }
74  
75      public static TypeDefinition getTypeById(String typeId) {
76          for (TypeDefinition typeDef : SINGLETON_TYPES) {
77              if (typeDef.getId().equals(typeId)) {
78                  return typeDef;
79              }
80          }
81          return null;
82      }
