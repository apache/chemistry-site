73      /**
74       * Check if repository is known and that object exists. To avoid later calls
75       * to again retrieve the object from the id return the retrieved object for
76       * later use.
77       * 
78       * @param repositoryId
79       *            repository id
80       * @param objectId
81       *            object id
82       * @return object for objectId
83       */
84      protected StoredObject checkStandardParameters(String repositoryId, String objectId) {
85          if (null == repositoryId) {
86              throw new CmisInvalidArgumentException(REPOSITORY_ID_CANNOT_BE_NULL);
87          }
88  
89          if (null == objectId) {
90              throw new CmisInvalidArgumentException(OBJECT_ID_CANNOT_BE_NULL);
91          }
92  
93          ObjectStore objStore = fStoreManager.getObjectStore(repositoryId);
94  
95          if (objStore == null) {
96              throw new CmisObjectNotFoundException(UNKNOWN_REPOSITORY_ID + repositoryId);
97          }
98  
99          StoredObject so = objStore.getObjectById(objectId);
100 
101         if (so == null) {
102             throw new CmisObjectNotFoundException(UNKNOWN_OBJECT_ID + objectId);
103         }
104 
105         return so;
106     }

194     protected void checkPolicies(String repositoryId, List<String> policyIds) {
195         if (policyIds != null && policyIds.size() > 0) {
196             for (String policyId : policyIds) {
197                 TypeDefinitionContainer tdc = fStoreManager.getTypeById(repositoryId, policyId);
198                 if (tdc == null) {
199                     throw new CmisInvalidArgumentException("Unknown policy type: " + policyId);
200                 }
201                 if (tdc.getTypeDefinition().getBaseTypeId() != BaseTypeId.CMIS_POLICY) {
202                     throw new CmisInvalidArgumentException(policyId + " is not a policy type");
203                 }
204             }
205         }

208     protected void checkCreatablePropertyTypes(String repositoryId,
209             Collection<PropertyDefinition<?>> propertyDefinitions )
210     {
211         RepositoryInfo repositoryInfo = fStoreManager.getRepositoryInfo(repositoryId);
212         RepositoryCapabilities repositoryCapabilities = repositoryInfo.getCapabilities();
213         CreatablePropertyTypes creatablePropertyTypes = repositoryCapabilities.getCreatablePropertyTypes();
214         
215         Set<PropertyType> creatablePropertyTypeSet = creatablePropertyTypes.canCreate();
216         for (PropertyDefinition<?> propertyDefinition : propertyDefinitions)
217         {
218             if (!creatablePropertyTypeSet.contains(propertyDefinition.getPropertyType()))
219                 throw new CmisConstraintException("propertyDefinition " + propertyDefinition.getId() +
220                         "is of not creatable type " + propertyDefinition.getPropertyType());
221             
222             // mandatory properties must have a default value
223             if (propertyDefinition.isRequired() && 
224                     (propertyDefinition.getDefaultValue() == null))
225             {
226                 throw new CmisConstraintException("property: " + propertyDefinition.getId() + 
227                         "required properties must have a default value");
228             }
229         }
230     }

232     protected void checkSettableAttributes(String repositoryId, TypeDefinition oldTypeDefinition,
233             TypeDefinition newTypeDefinition )
234     {
235         RepositoryInfo repositoryInfo = fStoreManager.getRepositoryInfo(repositoryId);
236         RepositoryCapabilities repositoryCapabilities = repositoryInfo.getCapabilities();
237         NewTypeSettableAttributes newTypeSettableAttributes = repositoryCapabilities.getNewTypeSettableAttributes();
238         
239         if (null == newTypeSettableAttributes)
240             return; // no restrictions defined
241         if (newTypeSettableAttributes.canSetControllableAcl() &&
242                 newTypeSettableAttributes.canSetControllablePolicy() &&
243                 newTypeSettableAttributes.canSetCreatable() &&
244                 newTypeSettableAttributes.canSetDescription() &&
245                 newTypeSettableAttributes.canSetDisplayName() &&
246                 newTypeSettableAttributes.canSetFileable() &&
247                 newTypeSettableAttributes.canSetFulltextIndexed() &&
248                 newTypeSettableAttributes.canSetId() &&
249                 newTypeSettableAttributes.canSetIncludedInSupertypeQuery() &&
250                 newTypeSettableAttributes.canSetLocalName() &&
251                 newTypeSettableAttributes.canSetLocalNamespace() &&
252                 newTypeSettableAttributes.canSetQueryable() &&
253                 newTypeSettableAttributes.canSetQueryName())
254             return;  // all is allowed
255         if (!newTypeSettableAttributes.canSetControllableAcl() && 
256                 oldTypeDefinition.isControllableAcl() != newTypeDefinition.isControllableAcl())
257             throw new CmisConstraintException("controllableAcl is not settable in repository " + repositoryId + ", but " +
258                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in controllableAcl");
259         if (!newTypeSettableAttributes.canSetControllablePolicy() && 
260                 oldTypeDefinition.isControllablePolicy() != newTypeDefinition.isControllablePolicy())
261             throw new CmisConstraintException("controllablePolicy is not settable in repository " + repositoryId + ", but " +
262                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in controllablePolicy");
263         if (!newTypeSettableAttributes.canSetCreatable() && 
264                 oldTypeDefinition.isCreatable() != newTypeDefinition.isCreatable())
265             throw new CmisConstraintException("isCreatable is not settable in repository " + repositoryId + ", but " +
266                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in isCreatable");
267         if (!newTypeSettableAttributes.canSetDescription() && 
268                 oldTypeDefinition.getDescription() != newTypeDefinition.getDescription())
269             throw new CmisConstraintException("description is not settable in repository " + repositoryId + ", but " +
270                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their description");
271         if (!newTypeSettableAttributes.canSetDisplayName() && 
272                 oldTypeDefinition.getDisplayName() != newTypeDefinition.getDisplayName())
273             throw new CmisConstraintException("displayName is not settable in repository " + repositoryId + ", but " +
274                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their displayName");
275         if (!newTypeSettableAttributes.canSetFileable() && 
276                 oldTypeDefinition.isFileable() != newTypeDefinition.isFileable())
277             throw new CmisConstraintException("fileable is not settable in repository " + repositoryId + ", but " +
278                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in isFileable");
279         if (!newTypeSettableAttributes.canSetFulltextIndexed() && 
280                 oldTypeDefinition.isFulltextIndexed() != newTypeDefinition.isFulltextIndexed())
281             throw new CmisConstraintException("fulltextIndexed is not settable in repository " + repositoryId + ", but " +
282                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in isFulltextIndexed");
283         // TODO  how can the ids differ?
284         if (!newTypeSettableAttributes.canSetId() && 
285                 oldTypeDefinition.getId() != newTypeDefinition.getId())
286             throw new CmisConstraintException("id is not settable in repository " + repositoryId + ", but " +
287                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their id");
288         if (!newTypeSettableAttributes.canSetIncludedInSupertypeQuery() && 
289                 oldTypeDefinition.isIncludedInSupertypeQuery() != newTypeDefinition.isIncludedInSupertypeQuery())
290             throw new CmisConstraintException("includedInSupertypeQuery is not settable in repository " + repositoryId + ", but " +
291                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their isIncludedInSupertypeQuery");
292         if (!newTypeSettableAttributes.canSetLocalName() && 
293                 oldTypeDefinition.getLocalName() != newTypeDefinition.getLocalName())
294             throw new CmisConstraintException("localName is not settable in repository " + repositoryId + ", but " +
295                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their localName");
296         if (!newTypeSettableAttributes.canSetLocalNamespace() && 
297                 oldTypeDefinition.getLocalNamespace() != newTypeDefinition.getLocalNamespace())
298             throw new CmisConstraintException("localNamespace is not settable in repository " + repositoryId + ", but " +
299                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their localNamespace");
300         if (!newTypeSettableAttributes.canSetQueryable() && 
301                 oldTypeDefinition.isQueryable() != newTypeDefinition.isQueryable())
302             throw new CmisConstraintException("queryable is not settable in repository " + repositoryId + ", but " +
303                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their isQueryable");
304         if (!newTypeSettableAttributes.canSetQueryName() && 
305                 oldTypeDefinition.getQueryName() != newTypeDefinition.getQueryName())
306             throw new CmisConstraintException("queryName is not settable in repository " + repositoryId + ", but " +
307                      oldTypeDefinition.getId() + " and " + newTypeDefinition.getId() + " differ in their queryName");
308     }

310     protected void checkUpdatePropertyDefinitions(Map<String,PropertyDefinition<?>> oldPropertyDefinitions,
311             Map<String,PropertyDefinition<?>> newPropertyDefinitions)
312     {
313         for(PropertyDefinition<?> newPropertyDefinition : newPropertyDefinitions.values())
314         {                   
315             PropertyDefinition<?> oldPropertyDefinition = oldPropertyDefinitions.get(newPropertyDefinition.getId());
316         
317             if (oldPropertyDefinition.isInherited())
318                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
319                         " update of inherited properties is not allowed");
320             if (!(oldPropertyDefinition.isRequired()) && newPropertyDefinition.isRequired())
321                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
322                         " optional properties must not be changed to required");    
323             if (oldPropertyDefinition.getPropertyType() != newPropertyDefinition.getPropertyType())
324                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
325                         " cannot update the propertyType (" + oldPropertyDefinition.getPropertyType() + ")");   
326             if (oldPropertyDefinition.getCardinality() != newPropertyDefinition.getCardinality())
327                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
328                         " cannot update the cardinality (" + oldPropertyDefinition.getCardinality() + ")"); 
329             
330             if (oldPropertyDefinition.isOpenChoice() && !newPropertyDefinition.isOpenChoice() )
331                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
332                         " open choice cannot change from true to false");
333             
334             // check choices
335             if (!oldPropertyDefinition.isOpenChoice())
336             {
337                 List<?> oldChoices = oldPropertyDefinition.getChoices();
338                 if (null == oldChoices)
339                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
340                             " there should be any choices when it's no open choice");
341                 List<?> newChoices = newPropertyDefinition.getChoices();
342                 if (null == newChoices)
343                 throw new CmisConstraintException("property: " + newPropertyDefinition.getId() + 
344                         " there should be any choices when it's no open choice");
345                 ListIterator<?> newChoicesIterator = newChoices.listIterator();
346                 for (Object oldChoiceObject : oldChoices)
347                 {
348                     Object newChoiceObject = newChoicesIterator.next();
349                     if (!(oldChoiceObject instanceof Choice))
350                         throw new CmisConstraintException("property: " + newPropertyDefinition.getId() + 
351                                 " old choice object is not of class Choice: " + oldChoiceObject.toString());
352                     if (!(newChoiceObject instanceof Choice))
353                         throw new CmisConstraintException("property: " + newPropertyDefinition.getId() + 
354                                 " new choice object is not of class Choice: " + newChoiceObject.toString());
355                     Choice<?> oldChoice = (Choice<?>) oldChoiceObject;
356                     Choice<?> newChoice = (Choice<?>) newChoiceObject;
357                     List<?> oldValues = oldChoice.getValue();
358                     List<?> newValues = newChoice.getValue();
359                     for (Object oldValue : oldValues)
360                     {                       
361                         if (! newValues.contains(oldValue))
362                             throw new CmisConstraintException("property: " + newPropertyDefinition.getId() + 
363                                     " value: " + oldValue.toString() + " is not in new values of the new choice");
364                     }       
365                 }   
366             }
367         
368             // check restrictions
369             if (oldPropertyDefinition instanceof PropertyDecimalDefinition)
370             {
371                 PropertyDecimalDefinition oldPropertyDecimalDefinition = (PropertyDecimalDefinition) oldPropertyDefinition;
372                 PropertyDecimalDefinition newPropertyDecimalDefinition = (PropertyDecimalDefinition) newPropertyDefinition;
373                 
374                 BigDecimal oldMinValue = oldPropertyDecimalDefinition.getMinValue();
375                 BigDecimal newMinValue = newPropertyDecimalDefinition.getMinValue();
376                 if (null != newMinValue &&
377                         (oldMinValue == null || (newMinValue.compareTo( oldMinValue) > 0)))
378                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
379                             " minValue " + oldMinValue + " cannot be further restricted to " + newMinValue);
380                 
381                 BigDecimal oldMaxValue = oldPropertyDecimalDefinition.getMaxValue();
382                 BigDecimal newMaxValue = newPropertyDecimalDefinition.getMaxValue();
383                 if (null != newMaxValue &&
384                         (oldMaxValue == null || (newMaxValue.compareTo( oldMaxValue) < 0)))
385                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
386                             " maxValue " + oldMaxValue + " cannot be further restricted to " + newMaxValue);
387             }
388             if (oldPropertyDefinition instanceof PropertyIntegerDefinition)
389             {
390                 PropertyIntegerDefinition oldPropertyIntegerDefinition = (PropertyIntegerDefinition) oldPropertyDefinition;
391                 PropertyIntegerDefinition newPropertyIntegerDefinition = (PropertyIntegerDefinition) newPropertyDefinition;
392                 
393                 BigInteger oldMinValue = oldPropertyIntegerDefinition.getMinValue();
394                 BigInteger newMinValue = newPropertyIntegerDefinition.getMinValue();
395                 if (null != newMinValue &&
396                         (oldMinValue == null || (newMinValue.compareTo( oldMinValue) > 0)))
397                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
398                             " minValue " + oldMinValue + " cannot be further restricted to " + newMinValue);
399                 
400                 BigInteger oldMaxValue = oldPropertyIntegerDefinition.getMaxValue();
401                 BigInteger newMaxValue = newPropertyIntegerDefinition.getMaxValue();
402                 if (null != newMaxValue &&
403                         (oldMaxValue == null || (newMaxValue.compareTo( oldMaxValue) < 0)))
404                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
405                             " maxValue " + oldMaxValue + " cannot be further restricted to " + newMaxValue);
406             }
407             if (oldPropertyDefinition instanceof PropertyStringDefinition)
408             {
409                 PropertyStringDefinition oldPropertyStringDefinition = (PropertyStringDefinition) oldPropertyDefinition;
410                 PropertyStringDefinition newPropertyStringDefinition = (PropertyStringDefinition) newPropertyDefinition;
411                 
412                 BigInteger oldMaxValue = oldPropertyStringDefinition.getMaxLength();
413                 BigInteger newMaxValue = newPropertyStringDefinition.getMaxLength();
414                 if (null != newMaxValue &&
415                         (oldMaxValue == null || (newMaxValue.compareTo( oldMaxValue) < 0)))
416                     throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
417                             " maxValue " + oldMaxValue + " cannot be further restricted to " + newMaxValue);
418             }
419         }
420         
421         // check for removed properties
422         for(PropertyDefinition<?> oldPropertyDefinition : oldPropertyDefinitions.values())
423         {
424             PropertyDefinition<?> newPropertyDefinition = newPropertyDefinitions.get(oldPropertyDefinition.getId());
425             if (null == newPropertyDefinition)
426             {
427                 throw new CmisConstraintException("property: " + oldPropertyDefinition.getId() + 
428                         " cannot remove that property");    
429             }
430         }
431     }

433     protected void checkUpdateType (TypeDefinition updateType, TypeDefinition type)
434     {
435         if (updateType.getId() != type.getId())
436             throw new CmisConstraintException("type to update must be of the same id: " + updateType.getId() + ", " + type.getId());
437         if (updateType.getBaseTypeId() != type.getBaseTypeId())
438             throw new CmisConstraintException("base type to update must be the same: " + updateType.getBaseTypeId() + ", " + type.getBaseTypeId());
439         // anything else should be ignored          
440     }
