143     @Override
144     public void setModifiedAtNow() {
145     	GregorianCalendar now = getNow();
146     	// ensure a larger time for modification date and change token:
147     	while (now.getTimeInMillis() == fModifiedAt.getTimeInMillis()) {
148     		try {
149     			Thread.sleep(1);
150     		} catch (InterruptedException ex) {    			
151     		}
152     		now = getNow();
153     	}
154         this.fModifiedAt = now;
155     }

256     @Override
257     public void fillProperties(Map<String, PropertyData<?>> properties, BindingsObjectFactory objFactory,
258             List<String> requestedIds) {
259 
260         if (FilterParser.isContainedInFilter(PropertyIds.NAME, requestedIds)) {
261             properties.put(PropertyIds.NAME, objFactory.createPropertyStringData(PropertyIds.NAME, getName()));
262         }
263         if (FilterParser.isContainedInFilter(PropertyIds.OBJECT_ID, requestedIds)) {
264             properties.put(PropertyIds.OBJECT_ID, objFactory.createPropertyIdData(PropertyIds.OBJECT_ID, getId()));
265         }
266         if (FilterParser.isContainedInFilter(PropertyIds.OBJECT_TYPE_ID, requestedIds)) {
267             properties.put(PropertyIds.OBJECT_TYPE_ID,
268                     objFactory.createPropertyIdData(PropertyIds.OBJECT_TYPE_ID, getTypeId()));
269         }
270         // set the base type id PropertyIds.CMIS_BASE_TYPE_ID outside because it
271         // requires the type definition
272         if (FilterParser.isContainedInFilter(PropertyIds.CREATED_BY, requestedIds)) {
273             properties.put(PropertyIds.CREATED_BY,
274                     objFactory.createPropertyStringData(PropertyIds.CREATED_BY, getCreatedBy()));
275         }
276         if (FilterParser.isContainedInFilter(PropertyIds.CREATION_DATE, requestedIds)) {
277             properties.put(PropertyIds.CREATION_DATE,
278                     objFactory.createPropertyDateTimeData(PropertyIds.CREATION_DATE, getCreatedAt()));
279         }
280         if (FilterParser.isContainedInFilter(PropertyIds.LAST_MODIFIED_BY, requestedIds)) {
281             properties.put(PropertyIds.LAST_MODIFIED_BY,
282                     objFactory.createPropertyStringData(PropertyIds.LAST_MODIFIED_BY, getModifiedBy()));
283         }
284         if (FilterParser.isContainedInFilter(PropertyIds.LAST_MODIFICATION_DATE, requestedIds)) {
285             properties.put(PropertyIds.LAST_MODIFICATION_DATE,
286                     objFactory.createPropertyDateTimeData(PropertyIds.LAST_MODIFICATION_DATE, getModifiedAt()));
287         }
288         if (FilterParser.isContainedInFilter(PropertyIds.CHANGE_TOKEN, requestedIds)) {
289             String token = getChangeToken();
290             properties.put(PropertyIds.CHANGE_TOKEN,
291                     objFactory.createPropertyStringData(PropertyIds.CHANGE_TOKEN, token));
292         }
293 
294         // CMIS 1.1 properties:
295         if (FilterParser.isContainedInFilter(PropertyIds.DESCRIPTION, requestedIds)) {
296             properties.put(PropertyIds.DESCRIPTION,
297                     objFactory.createPropertyStringData(PropertyIds.DESCRIPTION, description));
298         }
299         if (FilterParser.isContainedInFilter(PropertyIds.SECONDARY_OBJECT_TYPE_IDS, requestedIds)) {
300             properties.put(PropertyIds.SECONDARY_OBJECT_TYPE_IDS,
301                     objFactory.createPropertyIdData(PropertyIds.SECONDARY_OBJECT_TYPE_IDS, secondaryTypeIds));
302         }
303 
304         // add custom properties of type definition to the collection
305         if (null != fProperties) {
306             for (Entry<String, PropertyData<?>> prop : fProperties.entrySet()) {
307                 if (FilterParser.isContainedInFilter(prop.getKey(), requestedIds)) {
308                     properties.put(prop.getKey(), prop.getValue());
309                 }
310             }
311         }
312

331     /*
332      * Add CMIS_CREATED_BY, CMIS_CREATION_DATE, CMIS_LAST_MODIFIED_BY,
333      * CMIS_LAST_MODIFICATION_DATE, CMIS_CHANGE_TOKEN system properties to the
334      * list of properties with current values.
335      */
336     @SuppressWarnings("unchecked")
337     private void addSystemBaseProperties(Map<String, PropertyData<?>> properties, String user, boolean isCreated) {
338         if (user == null) {
339             user = UNKNOWN_USER;
340         }
341 
342         // Note that initial creation and modification date is set in
343         // constructor.
344         setModifiedBy(user);
345         if (null != properties) {
346             if (null != properties.get(PropertyIds.DESCRIPTION)) {
347                 setDescription((String) properties.get(PropertyIds.DESCRIPTION).getFirstValue());
348             }
349 
350             if (properties.containsKey(PropertyIds.SECONDARY_OBJECT_TYPE_IDS)) {
351                 secondaryTypeIds.clear();
352             }
353             PropertyData<?> secondaryTypeProp = properties.get(PropertyIds.SECONDARY_OBJECT_TYPE_IDS);
354             if (null != secondaryTypeProp) {
355                 secondaryTypeIds.addAll((List<String>) secondaryTypeProp.getValues());
356             }
357         }
358         if (isCreated) {
359             setCreatedBy(user);
360             setName((String) properties.get(PropertyIds.NAME).getFirstValue());
361             setTypeId((String) properties.get(PropertyIds.OBJECT_TYPE_ID).getFirstValue());
362         } else {
363             setModifiedAtNow();
364         }
365     }

516     protected ContentStream getIconFromResourceDir(String name) throws IOException {
517 
518         InputStream imageStream = StoredObjectImpl.class.getResourceAsStream(name);
519         ByteArrayOutputStream ba = new ByteArrayOutputStream();
520         byte[] buffer = new byte[BUFFER_SIZE];
521         int noBytesRead = 0;
522 
523         try {
524             while ((noBytesRead = imageStream.read(buffer)) >= 0) {
525                 ba.write(buffer, 0, noBytesRead);
526             }
527         } finally {
528             IOUtils.closeQuietly(ba);
529             IOUtils.closeQuietly(imageStream);
530         }
531 
532         ContentStreamDataImpl content = new ContentStreamDataImpl(0);
533         content.setFileName(name);
534         content.setMimeType("image/png");
535         content.setContent(new ByteArrayInputStream(ba.toByteArray()));
536         return content;
537     }

539     protected boolean testRenditionFilterForImage(String[] formats) {
540         if (formats.length == 1 && null != formats[0] && formats[0].equals("cmis:none")) {
541             return false;
542         } else {
543             return arrayContainsString(formats, "*") || arrayContainsString(formats, "image/*")
544                     || arrayContainsString(formats, "image/jpeg");
545         }
546     }

548     private boolean arrayContainsString(String[] formats, String val) {
549         for (String s : formats) {
550             if (val.equals(s)) {
551                 return true;
552             }
553         }
554         return false;
555     }
