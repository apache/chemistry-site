68  /**
69   * The object store is the central core of the in-memory repository. It is based
70   * on huge HashMap map mapping ids to objects in memory. To allow access from
71   * multiple threads a Java concurrent HashMap is used that allows parallel
72   * access methods.
73   * <p>
74   * Certain methods in the in-memory repository must guarantee constraints. For
75   * example a folder enforces that each child has a unique name. Therefore
76   * certain operations must occur in an atomic manner. In the example it must be
77   * guaranteed that no write access occurs to the map between acquiring the
78   * iterator to find the children and finishing the add operation when no name
79   * conflicts can occur. For this purpose this class has methods to lock an
80   * unlock the state of the repository. It is very important that the caller
81   * acquiring the lock enforces an unlock under all circumstances. Typical code
82   * is:
83   * <p>
84   * 
85   * <pre>
86   * ObjectStoreImpl os = ... ;
87   * try {
88   *     os.lock();
89   * } finally {
90   *     os.unlock();
91   * }
92   * </pre>
93   * 
94   * The locking is very coarse-grained. Productive implementations would probably
95   * implement finer grained locks on a folder or document rather than the
96   * complete repository.
97   */
98  public class ObjectStoreImpl implements ObjectStore {

165     private Fileable findObjectWithPathInDescendents(String path, String user, String prefix, Fileable fo) {
166         if (path.equals(prefix)) {
167             return fo;
168         } else if (fo instanceof Folder) {
169             List<Fileable> children = getChildren((Folder) fo);
170             for (Fileable child : children) {
171                 String foundPath = prefix.length() == 1 ? prefix + child.getName() : prefix + Filing.PATH_SEPARATOR
172                         + child.getName();
173                 if (path.startsWith(foundPath)) {
174                     Fileable found = findObjectWithPathInDescendents(path, user, foundPath, child);
175                     if (null != found) {
176                         return found; // note that there can be multiple folders
177                                       // with the same prefix like folder1,
178                                       // folder10
179                     }
180                 }
181             }
182         }
183         return null;
184     }

193     @Override
194     public void deleteObject(String objectId, Boolean allVersions, String user) {
195         StoredObject obj = fStoredObjectMap.get(objectId);
196 
197         if (null == obj) {
198             throw new CmisObjectNotFoundException("Cannot delete object with id  " + objectId
199                     + ". Object does not exist.");
200         }
201 
202         if (obj instanceof FolderImpl) {
203             deleteFolder(objectId, user);
204         } else if (obj instanceof DocumentVersion) {
205             DocumentVersion vers = (DocumentVersion) obj;
206             VersionedDocument parentDoc = vers.getParentDocument();
207             boolean otherVersionsExists;
208             if (allVersions != null && allVersions) {
209                 otherVersionsExists = false;
210                 List<DocumentVersion> allVers = parentDoc.getAllVersions();
211                 for (DocumentVersion ver : allVers) {
212                     fStoredObjectMap.remove(ver.getId());
213                 }
214             } else {
215                 fStoredObjectMap.remove(objectId);
216                 otherVersionsExists = parentDoc.deleteVersion(vers);
217             }
218 
219             if (!otherVersionsExists) {
220                 fStoredObjectMap.remove(parentDoc.getId());
221             }
222         } else {
223             fStoredObjectMap.remove(objectId);
224         }
225     }

228         String id = so.getId();

230         if (null == id) {

233         fStoredObjectMap.put(id, so);
234         return id;
235     }

284     @Override
285     public Document createDocument(Map<String, PropertyData<?>> propMap, String user, Folder folder,
286             ContentStream contentStream, List<String> policies, Acl addACEs, Acl removeACEs) {
287         String name = (String) propMap.get(PropertyIds.NAME).getFirstValue();
288         DocumentImpl doc = new DocumentImpl();
289         doc.createSystemBasePropertiesWhenCreated(propMap, user);
290         doc.setCustomProperties(propMap);
291         doc.setRepositoryId(fRepositoryId);
292         doc.setName(name);
293         if (null != folder) {
294             if (hasChild(folder, name)) {
295                 throw new CmisNameConstraintViolationException("Cannot create document an object with name " + name
296                         + " already exists in folder " + getFolderPath(folder.getId()));
297             }
298             doc.addParentId(folder.getId());
299         }
300         ContentStream content = setContent(doc, contentStream);
301         doc.setContent(content);
302         int aclId = getAclId(((FolderImpl) folder), addACEs, removeACEs);
303         doc.setAclId(aclId);
304         if (null != policies) {
305             doc.setAppliedPolicies(policies);
306         }
307         String id = storeObject(doc);
308         doc.setId(id);
309         applyAcl(doc, addACEs, removeACEs);
310         return doc;
311     }

459     @Override
460     public void updateObject(StoredObject so, Map<String, PropertyData<?>> newProperties, String user) {
461         // nothing to do
462         Map<String, PropertyData<?>> properties = so.getProperties();
463         for (String key : newProperties.keySet()) {
464             PropertyData<?> value = newProperties.get(key);
465 
466             if (key.equals(PropertyIds.SECONDARY_OBJECT_TYPE_IDS)) {
467                 properties.put(key, value); // preserve it even if it is empty!
468             } else  if (null == value || value.getValues() == null || value.getFirstValue() == null) {
469                 // delete property
470                 properties.remove(key);
471             } else {
472                 properties.put(key, value);
473             }
474         }
475         // update system properties and secondary object type ids
476         so.updateSystemBasePropertiesWhenModified(properties, user);
477         properties.remove(PropertyIds.SECONDARY_OBJECT_TYPE_IDS);
478     }

497     @Override
498     public List<StoredObject> getRelationships(String objectId, List<String> typeIds, RelationshipDirection direction) {
499 
500         List<StoredObject> res = new ArrayList<StoredObject>();
501 
502         if (typeIds != null && typeIds.size() > 0) {
503             for (String typeId : typeIds) {
504                 for (StoredObject so : fStoredObjectMap.values()) {
505                     if (so instanceof Relationship && so.getTypeId().equals(typeId)) {
506                         Relationship ro = (Relationship) so;
507                         if (ro.getSourceObjectId().equals(objectId)
508                                 && (RelationshipDirection.EITHER == direction 
509                                     || RelationshipDirection.SOURCE == direction)) {
510                             res.add(so);
511                         } else if (ro.getTargetObjectId().equals(objectId)
512                                 && (RelationshipDirection.EITHER == direction 
513                                     || RelationshipDirection.TARGET == direction)) {
514                             res.add(so);
515                         }
516                     }
517                 }
518             }
519         } else {
520             res = getAllRelationships(objectId, direction);
521         }
522         return res;
523     }

580     public int getAclId(StoredObjectImpl so, Acl addACEs, Acl removeACEs) {
581         InMemoryAcl newAcl;
582         boolean removeDefaultAcl = false;
583         int aclId = 0;
584 
585         if (so == null) {
586             newAcl = new InMemoryAcl();
587         } else {
588             aclId = so.getAclId();
589             newAcl = getInMemoryAcl(aclId);
590             if (null == newAcl) {
591                 newAcl = new InMemoryAcl();
592             } else {
593                 // copy list so that we can safely change it without effecting
594                 // the original
595                 newAcl = new InMemoryAcl(newAcl.getAces());
596             }
597         }
598 
599         if (newAcl.size() == 0 && addACEs == null && removeACEs == null) {
600             return 0;
601         }
602 
603         if (null != removeACEs) {
604             for (Ace ace : removeACEs.getAces()) {
605                 InMemoryAce inMemAce = new InMemoryAce(ace);
606                 if (inMemAce.equals(InMemoryAce.getDefaultAce())) {
607                     removeDefaultAcl = true;
608                 }
609             }
610         }
611 
612         if (so != null && 0 == aclId && !removeDefaultAcl) {
613             return 0; // if object grants full access to everyone and it will
614                       // not be removed we do nothing
615         }
616 
617         // add ACEs
618         if (null != addACEs) {
619             for (Ace ace : addACEs.getAces()) {
620                 InMemoryAce inMemAce = new InMemoryAce(ace);
621                 if (inMemAce.equals(InMemoryAce.getDefaultAce())) {
622                     return 0; // if everyone has full access there is no need to
623                 }
624                 // add additional ACLs.
625                 newAcl.addAce(inMemAce);
626             }
627         }
628 
629         // remove ACEs
630         if (null != removeACEs) {
631             for (Ace ace : removeACEs.getAces()) {
632                 InMemoryAce inMemAce = new InMemoryAce(ace);
633                 newAcl.removeAce(inMemAce);
634             }
635         }
636 
637         if (newAcl.size() > 0) {
638             return addAcl(newAcl);
639         } else {
640             return 0;
641         }
642     }

644     private void deleteFolder(String folderId, String user) {
645         StoredObject folder = fStoredObjectMap.get(folderId);
646         if (folder == null) {
647             throw new CmisInvalidArgumentException("Unknown object with id:  " + folderId);
648         }
649 
650         if (!(folder instanceof FolderImpl)) {
651             throw new CmisInvalidArgumentException("Cannot delete folder with id:  " + folderId
