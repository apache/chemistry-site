1057     private static boolean aclEquals(Acl acl1, Acl acl2) {
1058         if (acl1 == acl2) {
1059             return true;
1060         }
1061         if (acl1 == null || acl2 == null) {
1062             return false;
1063         }
1064         if (acl1.getClass() != acl2.getClass()) {
1065             return false;
1066         }
1067         if (acl1.getAces().size() != acl2.getAces().size()) {
1068             return false;
1069         }
1070         for (int i = 0; i < acl1.getAces().size(); i++) {
1071             aclHasAce(acl1.getAces(), acl2.getAces().get(i));
1072         }
1073         return true;
1074     }

1076     private static boolean aclHasAce(List<Ace> aces, Ace ace) {
1077         for (Ace ace2 : aces) {
1078             if (!ace.getPrincipalId().equals(ace2.getPrincipalId())) {
1079                 continue;
1080             }
1081             if (ace.getPermissions().size() != ace2.getPermissions().size()) {
1082                 continue;
