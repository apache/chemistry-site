114     @Test
115     public void testCheckOutBasic() {
116         String verId = createDocument(PROP_NAME, fRootFolderId, VersioningState.MAJOR);
117 
118         ObjectData version = fObjSvc.getObject(fRepositoryId, verId, "*", false, IncludeRelationships.NONE, null,
119                 false, false, null);
120         String docId = getVersionSeriesId(verId, version.getProperties().getProperties());
121         assertTrue(null != docId && docId.length() > 0);
122 
123         assertFalse(isCheckedOut(version.getProperties().getProperties()));
124 
125         Holder<Boolean> contentCopied = new Holder<Boolean>();
126         Holder<String> idHolder = new Holder<String>(verId); // or should this
127         // be version
128         // series?
129         fVerSvc.checkOut(fRepositoryId, idHolder, null, contentCopied);
130         String pwcId = idHolder.getValue();
131         // test that object is checked out and that all properties are set
132         // correctly
133         Properties props = fObjSvc.getProperties(fRepositoryId, pwcId, "*", null);
134         String changeToken = (String) props.getProperties().get(PropertyIds.CHANGE_TOKEN).getFirstValue();
135         checkVersionProperties(pwcId, VersioningState.CHECKEDOUT, props.getProperties(), null);
136 
137         // Test that a second checkout is not possible
138         try {
139             fVerSvc.checkOut(fRepositoryId, idHolder, null, contentCopied);
140             fail("Checking out a document that is already checked-out should fail.");
141         } catch (Exception e) {
142             assertTrue(e instanceof CmisUpdateConflictException);
143         }
144         // version and version series should be checked out now
145         // assertTrue(isCheckedOut(docId));
146         assertTrue(isCheckedOut(pwcId));
147 
148         // Set a new content and modify property
149         ContentStream altContent = fCreator.createAlternateContent();
150         idHolder = new Holder<String>(pwcId);
151         Holder<String> tokenHolder = new Holder<String>(changeToken);
152         fObjSvc.setContentStream(fRepositoryId, idHolder, true, tokenHolder, altContent, null);
153         fCreator.updateProperty(idHolder.getValue(), VersionTestTypeSystemCreator.PROPERTY_ID, PROP_VALUE_NEW);
154 
155         // Test that a check-in as same user is possible
156         String checkinComment = "Checkin without content and properties.";
157         fVerSvc.checkIn(fRepositoryId, idHolder, true, null, null, checkinComment, null, null, null, null);
158         // Neither the version nor the version series should be checked out any
159         // longer:
160         assertFalse(isCheckedOut(idHolder.getValue()));
161         // assertFalse(isCheckedOut(docId));
162         ContentStream retrievedContent = fObjSvc.getContentStream(fRepositoryId, idHolder.getValue(), null,
163                 BigInteger.valueOf(-1) /* offset */, BigInteger.valueOf(-1) /* length */, null);
164         assertTrue(fCreator.verifyContent(fCreator.createAlternateContent(), retrievedContent));
165         assertTrue(fCreator.verifyProperty(idHolder.getValue(), VersionTestTypeSystemCreator.PROPERTY_ID,
166                 PROP_VALUE_NEW));
167 
168         List<ObjectData> allVersions = fVerSvc.getAllVersions(fRepositoryId, docId, docId, "*", false, null);
169         assertEquals(2, allVersions.size());
170     }

507     private void checkVersionProperties(String docId, VersioningState versioningState,
508             Map<String, PropertyData<?>> props, String checkinComment) {
509         for (PropertyData<?> pd : props.values()) {
510             log.info("return property id: " + pd.getId() + ", value: " + pd.getValues());
511         }
512 
513         DocumentTypeDefinition typeDef = (DocumentTypeDefinition) fRepSvc.getTypeDefinition(fRepositoryId,
514                 VersionTestTypeSystemCreator.VERSION_TEST_DOCUMENT_TYPE_ID, null);
515         PropertyBoolean pdb = (PropertyBoolean) props.get(PropertyIds.IS_LATEST_VERSION);
516         assertNotNull(pdb);
517         boolean bVal = pdb.getFirstValue();
518         if (versioningState != VersioningState.CHECKEDOUT) {
519             assertTrue(bVal);
520         } else {
521             assertFalse(bVal);
522         }
523         pdb = (PropertyBoolean) props.get(PropertyIds.IS_MAJOR_VERSION);
524         assertNotNull(pdb);
525         bVal = pdb.getFirstValue();
