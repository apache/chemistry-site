147     public Document copy(ObjectId targetFolderId, Map<String, ?> properties, VersioningState versioningState,
148             List<Policy> policies, List<Ace> addAces, List<Ace> removeAces, OperationContext context) {
149 
150         ObjectId newId = null;
151         try {
152             newId = getSession().createDocumentFromSource(this, properties, targetFolderId, versioningState, policies,
153                     addAces, removeAces);
154         } catch (CmisNotSupportedException nse) {
155             newId = copyViaClient(targetFolderId, properties, versioningState, policies, addAces, removeAces);
156         }
157 
158         // if no context is provided the object will not be fetched
159         if (context == null || newId == null) {
160             return null;
161         }
162         // get the new object
163         CmisObject object = getSession().getObject(newId, context);
164         if (!(object instanceof Document)) {
165             throw new CmisRuntimeException("Newly created object is not a document! New id: " + newId);
166         }
167 
168         return (Document) object;
169     }

175     /**
176      * Copies the document manually. The content is streamed from the repository
177      * and back.
178      */
179     protected ObjectId copyViaClient(ObjectId targetFolderId, Map<String, ?> properties,
180             VersioningState versioningState, List<Policy> policies, List<Ace> addAces, List<Ace> removeAces) {
181         Map<String, Object> newProperties = new HashMap<String, Object>();
182 
183         OperationContext allPropsContext = getSession().createOperationContext();
184         allPropsContext.setFilterString("*");
185         allPropsContext.setIncludeAcls(false);
186         allPropsContext.setIncludeAllowableActions(false);
187         allPropsContext.setIncludePathSegments(false);
188         allPropsContext.setIncludePolicies(false);
189         allPropsContext.setIncludeRelationships(IncludeRelationships.NONE);
190         allPropsContext.setRenditionFilterString("cmis:none");
191 
192         Document allPropsDoc = (Document) getSession().getObject(this, allPropsContext);
193 
194         for (Property<?> prop : allPropsDoc.getProperties()) {
195             if (prop.getDefinition().getUpdatability() == Updatability.READWRITE
196                     || prop.getDefinition().getUpdatability() == Updatability.ONCREATE) {
197                 newProperties.put(prop.getId(), prop.getValue());
198             }
199         }
200 
201         if (properties != null) {
202             newProperties.putAll(properties);
203         }
204 
205         ContentStream contentStream = allPropsDoc.getContentStream();
206         try {
207             return getSession().createDocument(newProperties, targetFolderId, contentStream, versioningState, policies,
208                     addAces, removeAces);
209         } finally {
210             if (contentStream != null) {
211                 InputStream stream = contentStream.getStream();
212                 if (stream != null) {
213                     try {
214                         stream.close();
215                     } catch (IOException ioe) {
216                         throw new CmisRuntimeException("Cannot close source stream!", ioe);
217                     }
218                 }
219             }
220         }
221     }

229     public ObjectId checkOut() {
230         String newObjectId = null;
231 
232         readLock();
233         try {
234             String objectId = getObjectId();
235             Holder<String> objectIdHolder = new Holder<String>(objectId);
236 
237             getBinding().getVersioningService().checkOut(getRepositoryId(), objectIdHolder, null, null);
238             newObjectId = objectIdHolder.getValue();
239         } finally {
240             readUnlock();
241         }
242 
243         // remove original version from cache, the path and a few versioning
244         // properties are not valid anymore
245         getSession().removeObjectFromCache(this);
246 
247         if (newObjectId == null) {
248             return null;
249         }
250 
251         return getSession().createObjectId(newObjectId);
252     }

263     public ObjectId checkIn(boolean major, Map<String, ?> properties, ContentStream contentStream,
264             String checkinComment, List<Policy> policies, List<Ace> addAces, List<Ace> removeAces) {
265         String newObjectId = null;
266 
267         readLock();
268         try {
269             Holder<String> objectIdHolder = new Holder<String>(getObjectId());
270 
271             ObjectFactory of = getObjectFactory();
272 
273             Set<Updatability> updatebility = EnumSet.noneOf(Updatability.class);
274             updatebility.add(Updatability.READWRITE);
275             updatebility.add(Updatability.WHENCHECKEDOUT);
276 
277             getBinding().getVersioningService().checkIn(getRepositoryId(), objectIdHolder, major,
278                     of.convertProperties(properties, getType(), getSecondaryTypes(), updatebility),
279                     of.convertContentStream(contentStream), checkinComment, of.convertPolicies(policies),
280                     of.convertAces(addAces), of.convertAces(removeAces), null);
281 
282             newObjectId = objectIdHolder.getValue();
283         } finally {
284             readUnlock();
285         }
286 
287         // remove PWC from cache, it doesn't exist anymore
288         getSession().removeObjectFromCache(this);
289 
290         if (newObjectId == null) {
291             return null;
292         }
293 
294         return getSession().createObjectId(newObjectId);
295     }

301     public List<Document> getAllVersions(OperationContext context) {
302         String objectId;
303         String versionSeriesId;
304 
305         readLock();
306         try {
307             objectId = getObjectId();
308             versionSeriesId = getVersionSeriesId();
309         } finally {
310             readUnlock();
311         }
312 
313         List<ObjectData> versions = getBinding().getVersioningService().getAllVersions(getRepositoryId(), objectId,
314                 versionSeriesId, context.getFilterString(), context.isIncludeAllowableActions(), null);
315 
316         ObjectFactory objectFactory = getSession().getObjectFactory();
317 
318         List<Document> result = new ArrayList<Document>();
319         if (versions != null) {
320             for (ObjectData objectData : versions) {
321                 CmisObject doc = objectFactory.convertObject(objectData, context);
322                 if (!(doc instanceof Document)) {
323                     // should not happen...
324                     continue;
325                 }
326 
327                 result.add((Document) doc);
328             }
329         }
330 
331         return result;
332

357     public ContentStream getContentStream(String streamId, BigInteger offset, BigInteger length) {
358         // get the stream
359         ContentStream contentStream = getSession().getContentStream(this, streamId, offset, length);
360 
361         if (contentStream == null) {
362             return null;
363         }
364 
365         // the AtomPub binding doesn't return a file name
366         // -> get the file name from properties, if present
367         String filename = contentStream.getFileName();
368         if (filename == null) {
369             filename = getContentStreamFileName();
370         }
371 
372         long lengthLong = (contentStream.getBigLength() == null ? -1 : contentStream.getBigLength().longValue());
373 
374         // convert and return stream object
375         return getSession().getObjectFactory().createContentStream(filename, lengthLong, contentStream.getMimeType(),
376                 contentStream.getStream(), contentStream instanceof PartialContentStream);
377     }
