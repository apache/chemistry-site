38  public class RelationshipServiceImpl extends AbstractBrowserBindingService implements RelationshipService {

40      /**
41       * Constructor.
42       */
43      public RelationshipServiceImpl(BindingSession session) {
44          setSession(session);
45      }

47      public ObjectList getObjectRelationships(String repositoryId, String objectId, Boolean includeSubRelationshipTypes,
48              RelationshipDirection relationshipDirection, String typeId, String filter, Boolean includeAllowableActions,
49              BigInteger maxItems, BigInteger skipCount, ExtensionsData extension) {
50          // build URL
51          UrlBuilder url = getObjectUrl(repositoryId, objectId, Constants.SELECTOR_RELATIONSHIPS);
52          url.addParameter(Constants.PARAM_SUB_RELATIONSHIP_TYPES, includeSubRelationshipTypes);
53          url.addParameter(Constants.PARAM_RELATIONSHIP_DIRECTION, relationshipDirection);
54          url.addParameter(Constants.PARAM_TYPE_ID, typeId);
55          url.addParameter(Constants.PARAM_FILTER, filter);
56          url.addParameter(Constants.PARAM_ALLOWABLE_ACTIONS, includeAllowableActions);
57          url.addParameter(Constants.PARAM_MAX_ITEMS, maxItems);
58          url.addParameter(Constants.PARAM_SKIP_COUNT, skipCount);
59          url.addParameter(Constants.PARAM_SUCCINCT, getSuccinctParameter());
60          url.addParameter(Constants.PARAM_DATETIME_FORMAT, getDateTimeFormatParameter());
61  
62          // read and parse
63          Response resp = read(url);
64          Map<String, Object> json = parseObject(resp.getStream(), resp.getCharset());
65  
66          TypeCache typeCache = new ClientTypeCacheImpl(repositoryId, this);
67  
68          return JSONConverter.convertObjectList(json, typeCache, false);
69      }
70  }
