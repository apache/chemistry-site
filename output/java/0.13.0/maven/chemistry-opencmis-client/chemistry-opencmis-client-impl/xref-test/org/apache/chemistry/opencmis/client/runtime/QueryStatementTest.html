53      @Test
54      public void testWherePlacholder() {
55          Session session = new SessionImpl(new HashMap<String, String>(), null, null, null, null);
56          String query;
57          QueryStatement st;
58  
59          // strings
60          query = "SELECT * FROM cmis:document WHERE abc:string = ?";
61          st = new QueryStatementImpl(session, query);
62          st.setString(1, "test");
63          assertEquals("SELECT * FROM cmis:document WHERE abc:string = 'test'", st.toQueryString());
64  
65          query = "SELECT * FROM cmis:document WHERE abc:string = ?";
66          st = new QueryStatementImpl(session, query);
67          st.setString(1, "te'st");
68          assertEquals("SELECT * FROM cmis:document WHERE abc:string = 'te\\'st'", st.toQueryString());
69  
70          // likes
71          query = "SELECT * FROM cmis:document WHERE abc:string LIKE ?";
72          st = new QueryStatementImpl(session, query);
73          st.setStringLike(1, "%test%");
74          assertEquals("SELECT * FROM cmis:document WHERE abc:string LIKE '%test%'", st.toQueryString());
75  
76          query = "SELECT * FROM cmis:document WHERE abc:string LIKE ?";
77          st = new QueryStatementImpl(session, query);
78          st.setStringLike(1, "\\_test\\%blah\\\\blah");
79          assertEquals("SELECT * FROM cmis:document WHERE abc:string LIKE '\\_test\\%blah\\\\\\\\blah'",
80                  st.toQueryString());
81  
82          // contains
83  
84          // *, ? and - are treated as text search operators: 1st level escaping:
85          // none, 2nd level escaping: none
86          // \*, \? and \- are used as literals, 1st level escaping: none, 2nd
87          // level escaping: \\*, \\?, \\-
88          // ' and " are used as literals, 1st level escaping: \', \", 2nd level
89          // escaping: \\\', \\\",
90          // \ plus any other character, 1st level escaping \\ plus character, 2nd
91          // level: \\\\ plus character
92  
93          query = "SELECT * FROM cmis:document WHERE CONTAINS(?)";
94          st = new QueryStatementImpl(session, query);
95          st.setStringContains(1, "John's");
96          assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('John\\\\\\'s')", st.toQueryString());
97          st.setStringContains(1, "foo -bar");
98          assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo -bar')", st.toQueryString());
99          st.setStringContains(1, "foo*");
100         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo*')", st.toQueryString());
101         st.setStringContains(1, "foo?");
102         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo?')", st.toQueryString());
103         st.setStringContains(1, "foo\\-bar");
104         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo\\\\-bar')", st.toQueryString());
105         st.setStringContains(1, "foo\\*");
106         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo\\\\*')", st.toQueryString());
107         st.setStringContains(1, "foo\\?");
108         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('foo\\\\?')", st.toQueryString());
109         st.setStringContains(1, "\"Cool\"");
110         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('\\\\\\\"Cool\\\\\\\"')", st.toQueryString());
111         st.setStringContains(1, "c:\\MyDcuments");
112         assertEquals("SELECT * FROM cmis:document WHERE CONTAINS('c:\\\\MyDcuments')", st.toQueryString());
113 
114         // ids
115         query = "SELECT * FROM cmis:document WHERE abc:id = ?";
116         st = new QueryStatementImpl(session, query);
117         st.setId(1, new ObjectIdImpl("123"));
118         assertEquals("SELECT * FROM cmis:document WHERE abc:id = '123'", st.toQueryString());
119 
120         // booleans
121         query = "SELECT * FROM cmis:document WHERE abc:bool = ?";
122         st = new QueryStatementImpl(session, query);
123         st.setBoolean(1, true);
124         assertEquals("SELECT * FROM cmis:document WHERE abc:bool = TRUE", st.toQueryString());
125 
126         // numbers
127         query = "SELECT * FROM cmis:document WHERE abc:int = ? AND abc:int2 = 123";
128         st = new QueryStatementImpl(session, query);
129         st.setNumber(1, 42);
130         assertEquals("SELECT * FROM cmis:document WHERE abc:int = 42 AND abc:int2 = 123", st.toQueryString());
131 
132         // dateTime
133         query = "SELECT * FROM cmis:document WHERE abc:dateTime = TIMESTAMP ?";
134         GregorianCalendar cal = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
135         cal.clear();
136         cal.set(2012, 1, 2, 3, 4, 5);
137 
138         st = new QueryStatementImpl(session, query);
139         st.setDateTime(1, cal);
140         assertEquals("SELECT * FROM cmis:document WHERE abc:dateTime = TIMESTAMP '2012-02-02T03:04:05.000Z'",
141                 st.toQueryString());
142 
143         st = new QueryStatementImpl(session, query);
144         st.setDateTime(1, cal.getTimeInMillis());
145         assertEquals("SELECT * FROM cmis:document WHERE abc:dateTime = TIMESTAMP '2012-02-02T03:04:05.000Z'",
146                 st.toQueryString());
147 
148         // dateTime Timestamp
149         query = "SELECT * FROM cmis:document WHERE abc:dateTime = ?";
150 
151         st = new QueryStatementImpl(session, query);
152         st.setDateTimeTimestamp(1, cal);
153         assertEquals("SELECT * FROM cmis:document WHERE abc:dateTime = TIMESTAMP '2012-02-02T03:04:05.000Z'",
154                 st.toQueryString());
155 
156         st = new QueryStatementImpl(session, query);
157         st.setDateTimeTimestamp(1, cal.getTimeInMillis());
158         assertEquals("SELECT * FROM cmis:document WHERE abc:dateTime = TIMESTAMP '2012-02-02T03:04:05.000Z'",
159                 st.toQueryString());
160 
161         query = "SELECT * FROM cmis:document WHERE abc:dateTime IN (?)";
162 
163         st = new QueryStatementImpl(session, query);
164         st.setDateTimeTimestamp(1, cal.getTime(), cal.getTime());
165         assertEquals("SELECT * FROM cmis:document WHERE abc:dateTime "
166                 + "IN (TIMESTAMP '2012-02-02T03:04:05.000Z',TIMESTAMP '2012-02-02T03:04:05.000Z')", st.toQueryString());
167     }
