69      @Override
70      public void service(ServletContext servletContext, HttpServletRequest request, HttpServletResponse response) {
71  
72          String repositoryId = null; // TODO: determine repository id
73          String login = request.getParameter(PARAM_LOGIN);
74          try {
75              if (LOGIN_TOKEN.equals(login)) {
76                  sendTokens(servletContext, request, response);
77              } else if (LOGIN_SCRIPT.equals(login)) {
78                  sendJavaScript(servletContext, request, response, repositoryId);
79              } else if (LOGIN_CONTROLLER.equals(login)) {
80                  sendControllerContent(servletContext, request, response, repositoryId);
81              } else if (LOGIN_LOGIN.equals(login)) {
82                  login(servletContext, request, response, repositoryId);
83              } else if (LOGIN_LOGOUT.equals(login)) {
84                  logout(servletContext, request, response);
85              } else {
86                  throw new CmisObjectNotFoundException();
87              }
88          } catch (Exception e) {
89              if (e instanceof CmisBaseException) {
90                  throw (CmisBaseException) e;
91              } else {
92                  throw new CmisRuntimeException("Internal Error!", e);
93              }
94          }
95      }

98       * Sends a batch of new tokens.
99       */
100     protected void sendTokens(ServletContext servletContext, HttpServletRequest request, HttpServletResponse response)
101             throws IOException {
102         JSONArray result = new JSONArray();
103 
104         if (!SimpleTokenHandlerSessionHelper.checkApplicationKey(request)) {
105             // PARANOIA: remove the application key if the presented key is
106             // wrong, don't allow a second attempt
107             String appId = SimpleTokenHandlerSessionHelper.getApplicationIdFromKey(SimpleTokenHandlerSessionHelper
108                     .getKey(request));
109             if (appId != null) {
110                 SimpleTokenHandlerSessionHelper.removeApplicationKey(request, appId);
111             }
112         } else {
113             // generate a batch of tokens
114             String appId = SimpleTokenHandlerSessionHelper.getApplicationIdFromKey(SimpleTokenHandlerSessionHelper
115                     .getKey(request));
116 
117             for (int i = 0; i < TOKEN_BATCH_SIZE; i++) {
118                 // create token
119                 String token = SimpleTokenHandlerSessionHelper.generateKey(appId);
120 
121                 // put token into session and it to the response
122                 SimpleTokenHandlerSessionHelper.addToken(request, token);
123                 result.add(token);
124             }
125         }
126 
127         printJSON(response, result);
128     }

133     protected void sendJavaScript(ServletContext servletContext, HttpServletRequest request,
134             HttpServletResponse response, String repositoryId) throws IOException {
135 
136         UrlBuilder baseUrl = URL_SERVICE_CALL.compileBaseUrl(request, repositoryId);
137         URL url = new URL(baseUrl.toString());
138 
139         request.setAttribute(ATTR_PREFIX + "domain", encodeJavaScriptString(url.getProtocol() + "://" + url.getHost()
140                 + (url.getPort() > -1 ? ":" + url.getPort() : "")));
141         request.setAttribute(ATTR_PREFIX + "serviceUrl", encodeJavaScriptString(baseUrl.toString()));
142         request.setAttribute(ATTR_PREFIX + "iframeUrl",
143                 encodeJavaScriptString(baseUrl.addParameter(PARAM_LOGIN, LOGIN_CONTROLLER).toString()));
144 
145         response.setContentType("application/json; charset=UTF-8");
146 
147         RequestDispatcher dispatcher = servletContext.getRequestDispatcher(JSP_PATH + JSP_SCRIPT);
148         try {
149             dispatcher.include(request, response);
150         } catch (Exception e) {
151             throw new CmisRuntimeException("Internal error!", e);
152         }

211     protected void authenticate(ServletContext servletContext, HttpServletRequest request,
212             HttpServletResponse response, String repositoryId) throws IOException {
213         if (SimpleTokenHandlerSessionHelper.checkFormKey(request)) {
214             // login form returns
215 
216             String trustapp = request.getParameter(SimpleTokenHandlerSessionHelper.PARAM_TRUSTAPP);
217             String user = request.getParameter(SimpleTokenHandlerSessionHelper.PARAM_USER);
218             String password = request.getParameter(SimpleTokenHandlerSessionHelper.PARAM_PASSWORD);
219 
220             if (!"1".equals(trustapp)) {
221                 String error = "Please confirm that you trust the application!";
222                 showLoginForm(servletContext, request, response, error);
223                 return;
224             }
225 
226             if (!authenticate(servletContext, request, response, user, password)) {
227                 String error = "Invalid credentials!";
228                 showLoginForm(servletContext, request, response, error);
229                 return;
230             }
231 
232             // authentication successful -> create application key and
233             // forward to application
234             String appId = SimpleTokenHandlerSessionHelper.getApplicationIdFromKey(SimpleTokenHandlerSessionHelper
235                     .getKey(request));
236             String appKey = SimpleTokenHandlerSessionHelper.generateKey(appId);
237             SimpleTokenHandlerSessionHelper.setApplicationKey(request, appKey);
238             SimpleTokenHandlerSessionHelper.setUser(request, appId, user);
239             SimpleTokenHandlerSessionHelper.removeFormKey(request, appId);
240 
241             URL appURL = SimpleTokenHandlerSessionHelper.getApplicationURL(request, appId);
242 
243             response.sendRedirect(response.encodeRedirectURL(appURL.toString()));
244 
245             return;
246         }
247 
248         // check URL
249         String url = request.getParameter(SimpleTokenHandlerSessionHelper.PARAM_URL);
250 
251         if (url == null || url.trim().length() < 8 || !url.toLowerCase(Locale.ENGLISH).startsWith("http")) {
252             response.sendError(HttpServletResponse.SC_BAD_REQUEST);
253             return;
254         }
255 
256         URL appURL = null;
257         try {
258             appURL = new URL(url);
259         } catch (Exception e) {
260             response.sendError(HttpServletResponse.SC_BAD_REQUEST);
261             return;
262         }
263 
264         // check login key
265         String loginKey = request.getParameter(SimpleTokenHandlerSessionHelper.PARAM_KEY);
266 
267         if (loginKey == null || loginKey.trim().length() == 0) {
268             // no login key set -> generate one and return it
269 
270             String appId = SimpleTokenHandlerSessionHelper.generateAppId();
271             loginKey = SimpleTokenHandlerSessionHelper.generateKey(appId);
272             String formKey = SimpleTokenHandlerSessionHelper.generateKey(appId);
273 
274             SimpleTokenHandlerSessionHelper.setLoginKey(request, loginKey, formKey, appURL);
275 
276             String formURL = encodeJavaScriptString(URL_SERVICE_CALL.compileBaseUrl(request, repositoryId)
277                     .addParameter(PARAM_LOGIN, LOGIN_LOGIN)
278                     .addParameter(SimpleTokenHandlerSessionHelper.PARAM_KEY, formKey).toString());
279 
280             JSONObject result = new JSONObject();
281             result.put("ok", 0);
282             result.put("key", loginKey);
283             result.put("url", formURL);
284 
285             printJSON(response, result);
286 
287             return;
288         }
289 
290         String appId = SimpleTokenHandlerSessionHelper.getApplicationIdFromKey(SimpleTokenHandlerSessionHelper
291                 .getKey(request));
292         String appKey = SimpleTokenHandlerSessionHelper.getApplicationKey(request, appId);
293 
294         // check if login key fits
295         if (appKey == null || !SimpleTokenHandlerSessionHelper.checkLoginKey(request)) {
296             // PARANOIA: remove keys, don't allow a second attempt
297             SimpleTokenHandlerSessionHelper.removeLoginKey(request, appId);
298             SimpleTokenHandlerSessionHelper.removeApplicationKey(request, appId);
299             response.sendError(400);
300             return;
301         }
302 
303         // remove login key - it should not be reused
304         SimpleTokenHandlerSessionHelper.removeLoginKey(request, appId);
305 
306         // return application key
307         JSONObject result = new JSONObject();
308         result.put("ok", 1);
309         result.put("key", appKey);
310 
311         printJSON(response, result);
312     }

354     protected String encodeJavaScriptString(final String s) {
355         if (s == null) {
356             return null;
357         }
358 
359         StringBuilder sb = new StringBuilder(s.length() + 16);
360 
361         for (int i = 0; i < s.length(); i++) {
362             char c = s.charAt(i);
363             if (c == '\'') {
364                 sb.append("\\'");
365             } else {
366                 sb.append(c);
367             }
368         }
369 
370         return sb.toString();
371     }

373     protected String encodeHTMLString(final String s) {
374         if (s == null) {
375             return null;
376         }
377 
378         StringBuilder sb = new StringBuilder(s.length() + 64);
379 
380         for (int i = 0; i < s.length(); i++) {
381             char c = s.charAt(i);
382 
383             if (c == '\t' || c == '\n' || c == '\r' || (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
384                     || (c >= '0' && c <= '9') || c == ' ' || c == '.' || c == ',' || c == '-' || c == '_') {
385                 sb.append(c);
386             } else if (c <= 0x1f || (c >= 0x7f && c <= 0x9f)) {
387                 sb.append(' ');
388             } else if (c == '&') {
389                 sb.append("&amp;");
390             } else if (c == '"') {
391                 sb.append("&quot;");
392             } else if (c == '<') {
393                 sb.append("&lt;");
394             } else if (c == '>') {
395                 sb.append("&gt;");
396             } else {
397                 sb.append("&#x" + Integer.toHexString(c) + ";");
398             }
399         }
400 
401         return sb.toString();
402     }
