221      * Retrieves the index-th folder from given level of the hierarchy starting
222      * at rootId.
223      */
224     public String getFolderId(String rootId, int level, int index) {
225         String objectId = rootId;
226         final String requiredProperties = PropertyIds.OBJECT_ID + "," + PropertyIds.OBJECT_TYPE_ID + ","
227                 + PropertyIds.BASE_TYPE_ID;
228         // Note: This works because first folders are created then documents
229         for (int i = 0; i < level; i++) {
230             ObjectInFolderList result = fNavSvc.getChildren(fRepositoryId, objectId, requiredProperties,
231                     PropertyIds.OBJECT_TYPE_ID, false, IncludeRelationships.NONE, null, true, BigInteger.valueOf(-1),
232                     BigInteger.valueOf(-1), null);
233             List<ObjectInFolderData> children = result.getObjects();
234             ObjectData child = children.get(index).getObject();
235             objectId = (String) child.getProperties().getProperties().get(PropertyIds.OBJECT_ID).getFirstValue();
236         }
237         return objectId;
238     }

248     public String getDocumentId(String folderId, int index) {
249         String docId = null;
250         final String requiredProperties = PropertyIds.OBJECT_ID + "," + PropertyIds.OBJECT_TYPE_ID + ","
251                 + PropertyIds.BASE_TYPE_ID;
252         ObjectInFolderList result = fNavSvc.getChildren(fRepositoryId, folderId, requiredProperties,
253                 PropertyIds.OBJECT_TYPE_ID, false, IncludeRelationships.NONE, null, true, BigInteger.valueOf(-1),
254                 BigInteger.valueOf(-1), null);
255         List<ObjectInFolderData> children = result.getObjects();
256         int numDocsFound = 0;
257         for (int i = 0; i < children.size(); i++) {
258             ObjectData child = children.get(i).getObject();
259             docId = (String) child.getProperties().getProperties().get(PropertyIds.OBJECT_ID).getFirstValue();
260             if (child.getBaseTypeId().equals(BaseTypeId.CMIS_DOCUMENT)) {
261                 if (numDocsFound == index) {
262                     return docId;
263                 } else {
264                     numDocsFound++;
265                 }
266             }
267         }
268         return docId;
269     }

300     public String[] createDocuments(String folderId, int count) {
301         String[] result;
302 
303         for (int i = 0; i < count; i++) {
304             String id = createDocument(folderId, 0, 0);
305             fTopLevelDocsCreated.add(id);
306         }
307         if (fCleanup) {
308             deleteTree();
309             result = null;
310         } else {
311             result = new String[count];
312             for (int i = 0; i < fTopLevelDocsCreated.size(); i++) {
313                 result[i] = fTopLevelDocsCreated.get(i);
314             }
315         }

342     private void createFolderHierachy(String parentId, int level, int levels, int childrenPerLevel) {
343         String id = null;
344 
345         if (level >= levels) {
346             return;
347         }
348 
349         LOG.debug(" create folder for parent id: " + parentId + ", in level " + level + ", max levels " + levels);
350 
351         for (int i = 0; i < childrenPerLevel; i++) {
352             Properties props = createFolderProperties(i, level);
353             id = fObjSvc.createFolder(fRepositoryId, props, parentId, null, null, null, null);
354             if (level == 0) {
355                 fTopLevelFoldersCreated.add(id);
356             }
357 
358             if (id != null) {
359                 ++fFoldersInTotalCount;
360                 createFolderHierachy(id, level + 1, levels, childrenPerLevel);
361             }
362         }
363         for (int j = 0; j < fNoDocumentsToCreate; j++) {
364             id = createDocument(parentId, j, level);
365             if (level == 0) {
366                 fTopLevelDocsCreated.add(id);
367             }
368         }
369     }

371     private String createFolder(String parentId) {
372         Properties props = createFolderProperties(0, 0);
373         String id = null;
374         id = fObjSvc.createFolder(fRepositoryId, props, parentId, null, null, null, null);
375         fTopLevelFoldersCreated.add(id);
376         return id;
377     }

379     private String createDocument(String folderId, int no, int level) {
380         ContentStream contentStream = null;
381         VersioningState versioningState = VersioningState.NONE;
382         List<String> policies = null;
383         Acl addACEs = null;
384         Acl removeACEs = null;
385         ExtensionsData extension = null;
386 
387         LOG.debug("create document in folder " + folderId);
388         Properties props = createDocumentProperties(no, level);
389         String id = null;
390 
391         if (fContentSizeInK > 0) {
392             switch (fContentKind) {
393             case STATIC_TEXT:
394                 contentStream = createContentStaticText();
395                 break;
396             case LOREM_IPSUM_TEXT:
397                 contentStream = createContentLoremIpsumText();
398                 break;
399             case LOREM_IPSUM_HTML:
400                 contentStream = createContentLoremIpsumHtml();
401                 break;
402             case IMAGE_FRACTAL_JPEG:
403                 contentStream = createContentFractalimageJpeg();
404                 break;
405             }
406         }
407 
408         id = fObjSvc.createDocument(fRepositoryId, props, folderId, contentStream, versioningState, policies, addACEs,
409                 removeACEs, extension);
410 
411         if (null == id) {
412             LOG.error("createDocument failed.");
413         }
414         ++fDocumentsInTotalCount;
415         return id;
416     }

418     private void deleteTree() {
419 
420         // delete all documents from first level
421         for (String id : fTopLevelDocsCreated) {
422             deleteObject(id);
423         }
424 
425         // delete recursively all folders from first level
426         for (String id : fTopLevelFoldersCreated) {
427             fObjSvc.deleteTree(fRepositoryId, id, true, UnfileObject.DELETE, true, null);
428         }
429     }

433     }

460     public ContentStream createContentStaticText() {
461         ContentStreamImpl content = new ContentStreamImpl();
462         content.setFileName("data.txt");
463         content.setMimeType("text/plain");
464         int len = fContentSizeInK * 1024; // size of document in K
465         byte[] b = { 0x61, 0x62, 0x63, 0x64, 0x65, 0x66, 0x67, 0x68, 0x69, 0x6a, 0x6b, 0x6c, 0x6d, 0x6e, 0x0c, 0x0a,
466                 0x61, 0x62, 0x63, 0x64, 0x65, 0x66, 0x67, 0x68, 0x69, 0x6a, 0x6b, 0x6c, 0x6d, 0x6e, 0x0c, 0x0a }; // 32
467         // Bytes
468         ByteArrayOutputStream ba = new ByteArrayOutputStream(len);
469         try {
470             for (int j = 0; j < fContentSizeInK; j++) {
471                 // write 1K of data
472                 for (int i = 0; i < 32; i++) {
473                     ba.write(b);
474                 }
475             }
476         } catch (IOException e) {
477             throw new RuntimeException("Failed to fill content stream with data", e);
478         }
479         content.setStream(new ByteArrayInputStream(ba.toByteArray()));
480         return content;
481     }

561     public void dumpFolder(String folderId, String propertyFilter) {
562         LOG.debug("starting dumpFolder() id " + folderId + " ...");
563         boolean allRequiredPropertiesArePresent = propertyFilter != null && propertyFilter.equals("*"); // can
564         // be
565         // optimized
566         final String requiredProperties = allRequiredPropertiesArePresent ? propertyFilter : PropertyIds.OBJECT_ID
567                 + "," + PropertyIds.NAME + "," + PropertyIds.OBJECT_TYPE_ID + "," + PropertyIds.BASE_TYPE_ID;
568         // if all required properties are contained in the filter use we use the
569         // filter otherwise
570         // we use our own set and get those from the filter later in an extra
571         // call
572         String propertyFilterIntern = allRequiredPropertiesArePresent ? propertyFilter : requiredProperties;
573         dumpFolder(folderId, propertyFilterIntern, 0);
574     }

576     private void dumpFolder(String folderId, String propertyFilter, int depth) {
577         boolean allRequiredPropertiesArePresent = propertyFilter.equals("*"); // can
578         // be
579         // optimized
580         StringBuilder prefix = new StringBuilder();
581         for (int i = 0; i < depth; i++) {
582             prefix.append("   ");
583         }
584 
585         ObjectInFolderList result = fNavSvc.getChildren(fRepositoryId, folderId, propertyFilter, null, false,
586                 IncludeRelationships.NONE, null, true, BigInteger.valueOf(-1), BigInteger.valueOf(-1), null);
587         List<ObjectInFolderData> folders = result.getObjects();
588         if (null != folders) {
589             LOG.debug(prefix + "found " + folders.size() + " children in folder " + folderId);
590             int no = 0;
591             for (ObjectInFolderData folder : folders) {
592                 LOG.debug(prefix.toString() + ++no + ": found object with id: " + folder.getObject().getId()
593                         + " and path segment: " + folder.getPathSegment());
594                 dumpObjectProperties(folder.getObject(), depth, propertyFilter, !allRequiredPropertiesArePresent);
595                 String objectTypeBaseId = folder.getObject().getBaseTypeId().value();
596                 if (objectTypeBaseId.equals(BaseTypeId.CMIS_FOLDER.value())) {
597                     dumpFolder(folder.getObject().getId(), propertyFilter, depth + 1);
598                 } else if (objectTypeBaseId.equals(BaseTypeId.CMIS_DOCUMENT.value())) {
599                     dumpObjectProperties(folder.getObject(), depth + 1, propertyFilter,
600                             !allRequiredPropertiesArePresent);
601                 }
602             }
603         }
604         LOG.debug(""); // add empty line
605     }

607     private void dumpObjectProperties(ObjectData object, int depth, String propertyFilter, boolean mustFetchProperties) {
608         final SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssZ");
609         StringBuilder prefix = new StringBuilder();
610         for (int i = 0; i < depth; i++) {
611             prefix.append("   ");
612         }
613 
614         LOG.debug(prefix + "found object id " + object.getId());
615         Map<String, PropertyData<?>> propMap;
616         if (mustFetchProperties) {
617             String objId = (String) object.getProperties().getProperties().get(PropertyIds.OBJECT_ID).getFirstValue();
618             Properties props = fObjSvc.getProperties(fRepositoryId, objId, propertyFilter, null);
619             propMap = props.getProperties();
620         } else {
621             propMap = object.getProperties().getProperties();
622         }
623         StringBuilder valueStr = new StringBuilder("[");
624         for (Map.Entry<String, PropertyData<?>> entry : propMap.entrySet()) {
625             if (entry.getValue().getValues().size() > 1) {
626                 if (entry.getValue().getFirstValue() instanceof GregorianCalendar) {
627                     for (Object obj : entry.getValue().getValues()) {
628                         GregorianCalendar cal = (GregorianCalendar) obj;
629                         valueStr.append(df.format(cal.getTime())).append(", ");
630                     }
631                     valueStr.append("]");
632                 } else {
633                     valueStr = new StringBuilder(entry.getValue().getValues().toString());
634                 }
635             } else {
636                 Object value = entry.getValue().getFirstValue();
637                 if (null != value) {
638                     valueStr = new StringBuilder(value.toString());
639                     if (value instanceof GregorianCalendar) {
640                         valueStr = new StringBuilder(df.format(((GregorianCalendar) entry.getValue().getFirstValue())
641                                 .getTime()));
642                     }
643                 }
644             }
645             LOG.debug(prefix + entry.getKey() + ": " + valueStr);
646         }
647         LOG.debug(""); // add empty line
648     }
