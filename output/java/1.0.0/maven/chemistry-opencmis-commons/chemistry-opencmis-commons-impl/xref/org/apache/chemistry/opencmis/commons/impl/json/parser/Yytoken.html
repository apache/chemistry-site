19  package org.apache.chemistry.opencmis.commons.impl.json.parser;
20  
21  /**
22   * (Taken from JSON.simple &lt;http://code.google.com/p/json-simple/&gt; and
23   * modified for OpenCMIS.)
24   * 
25   * @author FangYidong&lt;fangyidong@yahoo.com.cn&gt;
26   */
27  public class Yytoken {
28      public static final int TYPE_VALUE = 0;// JSON primitive value:
29                                             // string,number,boolean,null
30      public static final int TYPE_LEFT_BRACE = 1;
31      public static final int TYPE_RIGHT_BRACE = 2;
32      public static final int TYPE_LEFT_SQUARE = 3;
33      public static final int TYPE_RIGHT_SQUARE = 4;
34      public static final int TYPE_COMMA = 5;
35      public static final int TYPE_COLON = 6;
36      public static final int TYPE_EOF = -1;// end of file
37  
38      public int type = 0;
39      public Object value = null;
40  
41      public Yytoken(int type, Object value) {
42          this.type = type;
43          this.value = value;
44      }
45  
46      @Override
47      public String toString() {
48          StringBuilder sb = new StringBuilder(64);
49          switch (type) {
50          case TYPE_VALUE:
51              sb.append("VALUE(").append(value).append(')');
52              break;
53          case TYPE_LEFT_BRACE:
54              sb.append("LEFT BRACE({)");
55              break;
56          case TYPE_RIGHT_BRACE:
57              sb.append("RIGHT BRACE(})");
58              break;
59          case TYPE_LEFT_SQUARE:
60              sb.append("LEFT SQUARE([)");
61              break;
62          case TYPE_RIGHT_SQUARE:
63              sb.append("RIGHT SQUARE(])");
64              break;
65          case TYPE_COMMA:
66              sb.append("COMMA(,)");
67              break;
68          case TYPE_COLON:
69              sb.append("COLON(:)");
70              break;
71          case TYPE_EOF:
72              sb.append("END OF FILE");
73              break;
74          default:
75              assert false;
76          }
77          return sb.toString();
78      }
79  }
