127     private void processCmdLine(String[] args) {
128 
129         OptionParser parser = new OptionParser();
130         fCmd = parser.accepts(CMD).withRequiredArg().describedAs("Command to perform (see below)");
131         fRepoId = parser.accepts(REPOSITORY_ID).withOptionalArg().describedAs("Repository used");
132         fDocType = parser.accepts(FILLER_DOCUMENT_TYPE_ID).withOptionalArg().defaultsTo(
133                 BaseTypeId.CMIS_DOCUMENT.value()).describedAs("Document type created");
134         fFolderType = parser.accepts(FILLER_FOLDER_TYPE_ID).withOptionalArg()
135                 .defaultsTo(BaseTypeId.CMIS_FOLDER.value()).describedAs("Folder type created");
136         fDocsPerFolder = parser.accepts(FILLER_DOCS_PER_FOLDER).withOptionalArg().ofType(Integer.class).describedAs(
137                 "Documents on each level").defaultsTo(1);
138         fFolderPerFolder = parser.accepts(FILLER_FOLDERS_PER_FOLDER).withOptionalArg().ofType(Integer.class)
139                 .describedAs(" Folders on each level").defaultsTo(0);
140         fDepth = parser.accepts(FILLER_DEPTH).withOptionalArg().ofType(Integer.class).describedAs("Levels of folders")
141                 .defaultsTo(1);
142         fContentSize = parser.accepts(FILLER_CONTENT_SIZE).withOptionalArg().ofType(Integer.class).describedAs(
143                 "Content size of each doc").defaultsTo(0);
144         fCount = parser.accepts(COUNT).withOptionalArg().ofType(Integer.class).defaultsTo(1).describedAs(
145                 "Repeat a command n times (partially implemented)");
146         fCleanup = parser.accepts(CLEANUP).withOptionalArg().ofType(Boolean.class).defaultsTo(false).describedAs(
147                 "Clean all created objects at the end");
148         fRootFolder = parser.accepts(ROOTFOLDER).withOptionalArg().ofType(String.class).describedAs(
149                 "folder id used as root to create objects (default repository root folder)");
150         fThreads = parser.accepts(THREADS).withOptionalArg().ofType(Integer.class).defaultsTo(1).describedAs(
151                 "Number of threads to start in parallel");
152 //        fFileName = parser.accepts(FILE).withRequiredArg().ofType(String.class).describedAs("Input File");
153         fContentKindStr = parser.accepts(CONTENT_KIND).withOptionalArg().ofType(String.class).defaultsTo("lorem/text")
154                 .describedAs("kind of content: static/text, lorem/text, lorem/html, fractal/jpeg");
155         fFileNamePattern = parser.accepts(FILE_NAME_PATTERN).withOptionalArg().ofType(String.class).defaultsTo("ContentData-%03d.bin")
156                 .describedAs("file name pattern to be used with CreateFiles action");
157         fLocalDir = parser.accepts(LOCAL_DIR).withOptionalArg().ofType(String.class).defaultsTo(".")
158                 .describedAs("name of a directory to be recursively copied to the repository");
159         fLocalFile = parser.accepts(LOCAL_FILE).withOptionalArg().ofType(String.class)
160                 .describedAs("file name of a file to be copied to the repository");
161         OptionSet options = parser.parse(args);
162 
163         if (options.valueOf(fCmd) == null || options.has("?")) {
164             usage(parser);
165         }
166 
167         String binding = getBinding();
168         
169         if (binding.equals(BINDING_WS)) {
170             fBindingType = BindingType.WEBSERVICES;
171         } else if (binding.equals(BINDING_ATOM)) {
172             fBindingType = BindingType.ATOMPUB;
173         } else if (binding.equals(BINDING_BROWSER)) {
174             fBindingType = BindingType.BROWSER;
175         } else {
176             System.out.println("Error: Unknown binding: " + binding + " allowed values: "
177                     + BINDING_WS + " or " + BINDING_ATOM + " or " + BINDING_BROWSER);
178             return;
179         }
180 
181         String kind = options.valueOf(fContentKindStr);
182         if (null == kind) {
183             if (options.valueOf(fContentSize) > 0)
184                 fContentKind = ObjectGenerator.CONTENT_KIND.StaticText;
185             else
186                 fContentKind = null;
187         } else if (kind.equals("static/text"))
188             fContentKind = ObjectGenerator.CONTENT_KIND.StaticText;
189         else if (kind.equals("lorem/text"))
190             fContentKind = ObjectGenerator.CONTENT_KIND.LoremIpsumText;
191         else if (kind.equals("lorem/html"))
192             fContentKind = ObjectGenerator.CONTENT_KIND.LoremIpsumHtml;
193         else if (kind.equals("fractal/jpeg"))
194             fContentKind = ObjectGenerator.CONTENT_KIND.ImageFractalJpeg;
195         else {
196             System.out.println("Unknown content kind: " + options.valueOf(fContentKindStr));
197             System.out.println("  must be one of static/text, lorem/text, lorem/html, fractal/jpeg");
198             usage(parser);
199         }
200 
201         if (null == options.valueOf(fCmd)) {
202             System.out.println("No command given.");
203             usage(parser);
204         } else if (options.valueOf(fCmd).equals("FillRepository")) {
205             fillRepository(options);
206         } else if (options.valueOf(fCmd).equals("CreateDocument")) {
207             createSingleDocument(options);
208         } else if (options.valueOf(fCmd).equals("CreateFolder")) {
209             createFolders(options);
210         } else if (options.valueOf(fCmd).equals("RepositoryInfo")) {
211             repositoryInfo(options);
212 //        } else if (options.valueOf(fCmd).equals("CreateTypes")) {
213 //            createTypes(options);
214         } else if (options.valueOf(fCmd).equals("CreateFiles")) {
215             createFiles(options);
216         } else if (options.valueOf(fCmd).equals("CopyFiles")) {
217             transferFiles(options);
218         } else if (options.valueOf(fCmd).equals("CopyFilesTest")) { // undocumented
219             transferFilesTest(options);
220         } else {
221             System.out.println("Unknown cmd: " + options.valueOf(fCmd));
222             usage(parser);
223         }
224     }

265     private void fillRepository(String repoId, int docsPerFolder, int foldersPerFolders, int depth,
266             String documentType, String folderType, int contentSizeInKB, String rootFolderId, boolean doCleanup) {
267 
268         MultiThreadedObjectGenerator.ObjectGeneratorRunner runner = MultiThreadedObjectGenerator.prepareForCreateTree(
269                 getClientBindings(), repoId, docsPerFolder, foldersPerFolders, depth, documentType, folderType,
270                 contentSizeInKB, rootFolderId, fContentKind, doCleanup);
271         ObjectGenerator gen = runner.getObjectGenerator();
272         runner.doCreateTree();
273 
274         System.out.println();
275         System.out.println("Result:");
276         System.out.println("Filling repository succeeded.");
277         System.out.println("Folder used as root for creation (null=rootFolderId): " + rootFolderId);
278         System.out.println("Number of documents created: " + gen.getDocumentsInTotal());
279         System.out.println("Number of folders created: " + gen.getFoldersInTotal());
280         gen.printTimings();
281     }

283     private void fillRepositoryMT(int noThreads, String repoId, int docsPerFolder, int foldersPerFolders, int depth,
284             String documentType, String folderType, int contentSizeInKB, String rootFolderId, boolean doCleanup) {
285 
286         // Step 1: create a root folder for each thread
287         MultiThreadedObjectGenerator.ObjectGeneratorRunner runner = MultiThreadedObjectGenerator
288                 .prepareForCreateFolder(getClientBindings(), repoId, folderType, rootFolderId, noThreads, doCleanup);
289         String[] folderIds = runner.doCreateFolder();
290 
291         // Step 2: fill each root folder with an object tree
292         MultiThreadedObjectGenerator.ObjectGeneratorRunner[] runners = MultiThreadedObjectGenerator
293                 .prepareForCreateTreeMT(getClientBindings(), repoId, docsPerFolder, foldersPerFolders, depth, documentType,
294                         folderType, contentSizeInKB, folderIds, fContentKind, doCleanup);
295 
296         MultiThreadedObjectGenerator.runMultiThreaded(runners);
297         System.out.println("Filling repository succeeded.");
298     }

484     private void createFiles(OptionSet options) {
485         ContentStream contentStream = null;
486         String fileNamePattern = options.valueOf(fFileNamePattern);
487         int count = options.valueOf(fCount);
488         int contentSize = options.valueOf(fContentSize);
489         
490         System.out.println("Creating local files with content: ");
491         System.out.println("Kind: " + options.valueOf(fDocsPerFolder));
492         System.out.println("Number of files: " + count);
493         System.out.println("File name pattern: " + fileNamePattern);
494         System.out.println("Kind of content: " + options.valueOf(fContentKindStr));
495         System.out.println("Size of content (text only): " + contentSize);
496         
497         ObjectGenerator objGen = new ObjectGenerator(null, null, null, null, null, fContentKind);
498         objGen.setContentSizeInKB(contentSize);
499         
500         InputStream is = null;
501         FileOutputStream os = null;
502         
503         try {
504             for (int i=0; i<count; i++) {
505                 String fileName = String.format(fileNamePattern, i);
506                 System.out.println("Generating file: " + fileName);
507                 if (contentSize > 0) {
508                     switch (fContentKind) {
509                     case StaticText:
510                         contentStream = objGen.createContentStaticText();
511                         break;
512                     case LoremIpsumText:
513                         contentStream =  objGen.createContentLoremIpsumText();
514                         break;
515                     case LoremIpsumHtml:
516                         contentStream =  objGen.createContentLoremIpsumHtml();
517                         break;
518                     case ImageFractalJpeg:
519                         contentStream =  objGen.createContentFractalimageJpeg();
