223      * Validates the given XML.
224      */
225     protected void validate(byte[] xmlDocument, CmisVersion cmisVersion) {
226         Validator validator = null;
227         if (cmisVersion == CmisVersion.CMIS_1_0) {
228             if (schema10 != null) {
229                 validator = schema10.newValidator();
230             } else {
231                 LOG.warn("CMIS 1.0 schema not loaded. Cannot validate XML.");
232                 return;
233             }
234         } else {
235             if (schema11 != null) {
236                 validator = schema11.newValidator();
237             } else {
238                 LOG.warn("CMIS 1.1 schema not loaded. Cannot validate XML.");
239                 return;
240             }
241         }
242 
243         Source source = new StreamSource(new ByteArrayInputStream(xmlDocument));
244 
245         try {
246             validator.validate(source);
247         } catch (Exception e) {
248             try {
249                 LOG.error("Schema validation failed:\n" + format(xmlDocument));
250                 System.out.println(format(xmlDocument));
251             } catch (TransformerException e1) {
252             }
253             fail("Schema " + cmisVersion.value() + " validation failed: " + e);
254         }
255     }

258      * Formats an XML document.
259      */
260     protected String format(byte[] xmlDocument) throws TransformerException {
261         TransformerFactory tf = TransformerFactory.newInstance();
262         tf.setAttribute("indent-number", new Integer(2));
263 
264         Transformer transformer = tf.newTransformer();
265         transformer.setOutputProperty(OutputKeys.INDENT, "yes");
266 
267         StreamResult result = new StreamResult(new StringWriter());
268         Source source = new StreamSource(new ByteArrayInputStream(xmlDocument));
269         transformer.transform(source, result);
270 
271         return result.getWriter().toString();
272     }

478     /**
479      * Compares two data objects.
480      */
481     protected void assertDataObjectsEquals(String name, Object expected, Object actual, Set<String> ignoreMethods) {
482 
483         LOG.debug(name + ": " + expected + " / " + actual);
484 
485         if ((expected == null) && (actual == null)) {
486             return;
487         }
488 
489         if ((expected == null) || (actual == null)) {
490             fail("Data object is null! name: " + name + " / expected: " + expected + " / actual: " + actual);
491         }
492 
493         // handle simple types
494         if ((expected instanceof String) || (expected instanceof Boolean) || (expected instanceof BigInteger)
495                 || (expected instanceof BigDecimal) || (expected instanceof Enum<?>)) {
496             assertEquals(name, expected, actual);
497 
498             return;
499         } else if (expected instanceof GregorianCalendar) {
500             assertEquals(name, ((GregorianCalendar) expected).getTimeInMillis(),
501                     ((GregorianCalendar) actual).getTimeInMillis());
502 
503             return;
504         } else if (expected instanceof List<?>) {
505             List<?> expectedList = (List<?>) expected;
506             List<?> actualList = (List<?>) actual;
507 
508             assertEquals(name + ".length", expectedList.size(), actualList.size());
509 
510             for (int i = 0; i < expectedList.size(); i++) {
511                 assertDataObjectsEquals(name + "[" + i + "]", expectedList.get(i), actualList.get(i), ignoreMethods);
512             }
513 
514             return;
515         } else if (expected instanceof Map<?, ?>) {
516             Map<?, ?> expectedMap = (Map<?, ?>) expected;
517             Map<?, ?> actualMap = (Map<?, ?>) actual;
518 
519             assertEquals(name + ".length", expectedMap.size(), actualMap.size());
520 
521             for (Map.Entry<?, ?> entry : expectedMap.entrySet()) {
522                 assertTrue(name + ".contains[" + entry.getKey() + "]", actualMap.containsKey(entry.getKey()));
523                 assertDataObjectsEquals(name + "[" + entry.getKey() + "]", entry.getValue(),
524                         actualMap.get(entry.getKey()), ignoreMethods);
525             }
526 
527             return;
528         } else if (expected instanceof Set<?>) {
529             Set<?> expectedSet = (Set<?>) expected;
530             Set<?> actualSet = (Set<?>) actual;
531 
532             assertEquals(name + ".length", expectedSet.size(), actualSet.size());
533 
534             for (Object item : expectedSet) {
535                 assertTrue(name + ".contains[" + item + "]", actualSet.contains(item));
536             }
537 
538             return;
539         }
540 
541         for (Method m : expected.getClass().getMethods()) {
542             if (!m.getName().startsWith("get") && !m.getName().startsWith("is") && !m.getName().startsWith("supports")) {
543                 continue;
544             }
545 
546             if (ignoreMethods != null && ignoreMethods.contains(m.getName())) {
547                 continue;
548             }
549 
550             if (m.getName().equals("getClass")) {
551                 continue;
552             }
553 
554             if (m.getParameterTypes().length != 0) {
555                 continue;
556             }
557 
558             try {
559                 Object expectedValue = m.invoke(expected, new Object[0]);
560                 Object actualValue = m.invoke(actual, new Object[0]);
561 
562                 assertDataObjectsEquals(name + "." + m.getName(), expectedValue, actualValue, ignoreMethods);
563             } catch (Exception e) {
564                 fail(e.toString());
565             }
566         }
567     }
