333     /**
334      * Creates a property definition object.
335      */
336     private static PropertyDefinition<?> createPropDef(String id, String displayName, String description,
337             PropertyType datatype, Cardinality cardinality, Updatability updateability, boolean inherited,
338             boolean required) {
339         AbstractPropertyDefinition<?> result = null;
340 
341         switch (datatype) {
342         case BOOLEAN:
343             result = new PropertyBooleanDefinitionImpl();
344             break;
345         case DATETIME:
346             result = new PropertyDateTimeDefinitionImpl();
347             break;
348         case DECIMAL:
349             result = new PropertyDecimalDefinitionImpl();
350             break;
351         case HTML:
352             result = new PropertyHtmlDefinitionImpl();
353             break;
354         case ID:
355             result = new PropertyIdDefinitionImpl();
356             break;
357         case INTEGER:
358             result = new PropertyIntegerDefinitionImpl();
359             break;
360         case STRING:
361             result = new PropertyStringDefinitionImpl();
362             break;
363         case URI:
364             result = new PropertyUriDefinitionImpl();
365             break;
366         default:
367             throw new RuntimeException("Unknown datatype! Spec change?");
368         }
369 
370         result.setId(id);
371         result.setLocalName(id);
372         result.setDisplayName(displayName);
373         result.setDescription(description);
374         result.setPropertyType(datatype);
375         result.setCardinality(cardinality);
376         result.setUpdatability(updateability);
377         result.setIsInherited(inherited);
378         result.setIsRequired(required);
379         result.setIsQueryable(false);
380         result.setIsOrderable(false);
381         result.setQueryName(id);
382 
383         return result;
384     }

386     /**
387      * Adds a type to collection with inheriting base type properties.
388      */
389     public boolean addType(TypeDefinition type) {
390         if (type == null) {
391             return false;
392         }
393 
394         if (type.getBaseTypeId() == null) {
395             return false;
396         }
397 
398         // find base type
399         TypeDefinition baseType = null;
400         if (type.getBaseTypeId() == BaseTypeId.CMIS_DOCUMENT) {
401             baseType = copyTypeDefintion(types.get(DOCUMENT_TYPE_ID).getTypeDefinition());
402         } else if (type.getBaseTypeId() == BaseTypeId.CMIS_FOLDER) {
403             baseType = copyTypeDefintion(types.get(FOLDER_TYPE_ID).getTypeDefinition());
404         } else if (type.getBaseTypeId() == BaseTypeId.CMIS_ITEM) {
405             baseType = copyTypeDefintion(types.get(ITEM_TYPE_ID).getTypeDefinition());
406         } else if (type.getBaseTypeId() == BaseTypeId.CMIS_RELATIONSHIP) {
407             baseType = copyTypeDefintion(types.get(RELATIONSHIP_TYPE_ID).getTypeDefinition());
408         } else if (type.getBaseTypeId() == BaseTypeId.CMIS_POLICY) {
409             baseType = copyTypeDefintion(types.get(POLICY_TYPE_ID).getTypeDefinition());
410         } else {
411             return false;
412         }
413 
414         AbstractTypeDefinition newType = (AbstractTypeDefinition) copyTypeDefintion(type);
415 
416         // copy property definition
417         for (PropertyDefinition<?> propDef : baseType.getPropertyDefinitions().values()) {
418             ((AbstractPropertyDefinition<?>) propDef).setIsInherited(true);
419             newType.addPropertyDefinition(propDef);
420         }
421 
422         // add it
423         addTypeInteral(newType);
424 
425         LOG.info("Added type '" + newType.getId() + "'.");
426 
427         return true;
428     }

430     /**
431      * Adds a type to collection.
432      */
433     private void addTypeInteral(AbstractTypeDefinition type) {
434         if (type == null) {
435             return;
436         }
437 
438         if (types.containsKey(type.getId())) {
439             // can't overwrite a type
440             return;
441         }
442 
443         TypeDefinitionContainerImpl tc = new TypeDefinitionContainerImpl();
444         tc.setTypeDefinition(type);
445 
446         // add to parent
447         if (type.getParentTypeId() != null) {
448             TypeDefinitionContainerImpl tdc = types.get(type.getParentTypeId());
449             if (tdc != null) {
450                 if (tdc.getChildren() == null) {
451                     tdc.setChildren(new ArrayList<TypeDefinitionContainer>());
452                 }
453                 tdc.getChildren().add(tc);
454             }
455         }
456 
457         types.put(type.getId(), tc);
458         typesList.add(tc);
459     }

461     /**
462      * CMIS getTypesChildren.
463      */
464     public TypeDefinitionList getTypesChildren(CallContext context, String typeId, boolean includePropertyDefinitions,
465             BigInteger maxItems, BigInteger skipCount) {
466         TypeDefinitionListImpl result = new TypeDefinitionListImpl(new ArrayList<TypeDefinition>());
467 
468         int skip = (skipCount == null ? 0 : skipCount.intValue());
469         if (skip < 0) {
470             skip = 0;
471         }
472 
473         int max = (maxItems == null ? Integer.MAX_VALUE : maxItems.intValue());
474         if (max < 1) {
475             return result;
476         }
477 
478         if (typeId == null) {
479             if (skip < 1) {
480                 result.getList().add(copyTypeDefintion(types.get(FOLDER_TYPE_ID).getTypeDefinition()));
481                 max--;
482             }
483             if ((skip < 2) && (max > 0)) {
484                 result.getList().add(copyTypeDefintion(types.get(DOCUMENT_TYPE_ID).getTypeDefinition()));
485                 max--;
486             }
487 
488             result.setHasMoreItems((result.getList().size() + skip) < 2);
489             result.setNumItems(BigInteger.valueOf(2));
490         } else {
491             TypeDefinitionContainer tc = types.get(typeId);
492             if ((tc == null) || (tc.getChildren() == null)) {
493                 return result;
494             }
495 
496             for (TypeDefinitionContainer child : tc.getChildren()) {
497                 if (skip > 0) {
498                     skip--;
499                     continue;
500                 }
501 
502                 result.getList().add(copyTypeDefintion(child.getTypeDefinition()));
503 
504                 max--;
505                 if (max == 0) {
506                     break;
507                 }
508             }
509 
510             result.setHasMoreItems((result.getList().size() + skip) < tc.getChildren().size());
511             result.setNumItems(BigInteger.valueOf(tc.getChildren().size()));
512         }
513 
514         if (!includePropertyDefinitions) {
515             for (TypeDefinition type : result.getList()) {
516                 type.getPropertyDefinitions().clear();
517             }
518         }
519 
520         return result;
521     }

523     /**
524      * CMIS getTypesDescendants.
525      */
526     public List<TypeDefinitionContainer> getTypesDescendants(CallContext context, String typeId, BigInteger depth,
527             Boolean includePropertyDefinitions) {
528         List<TypeDefinitionContainer> result = new ArrayList<TypeDefinitionContainer>();
529 
530         // check depth
531         int d = (depth == null ? -1 : depth.intValue());
532         if (d == 0) {
533             throw new CmisInvalidArgumentException("Depth must not be 0!");
534         }
535         if (typeId == null) {
536             d = -1;
537         }
538 
539         // set property definition flag to default value if not set
540         boolean ipd = (includePropertyDefinitions == null ? false : includePropertyDefinitions.booleanValue());
541 
542         if (typeId == null) {
543             result.add(getTypesDescendants(d, types.get(FOLDER_TYPE_ID), ipd));
544             result.add(getTypesDescendants(d, types.get(DOCUMENT_TYPE_ID), ipd));
545             // result.add(getTypesDescendants(depth,
546             // fTypes.get(RELATIONSHIP_TYPE_ID), includePropertyDefinitions));
547             // result.add(getTypesDescendants(depth, fTypes.get(POLICY_TYPE_ID),
548             // includePropertyDefinitions));
549         } else {
550             TypeDefinitionContainer tc = types.get(typeId);
551             if (tc != null) {
552                 result.add(getTypesDescendants(d, tc, ipd));
553             }
554         }
555 
556         return result;
557     }

559     /**
560      * Gathers the type descendants tree.
561      */
562     private TypeDefinitionContainer getTypesDescendants(int depth, TypeDefinitionContainer tc,
563             boolean includePropertyDefinitions) {
564         TypeDefinitionContainerImpl result = new TypeDefinitionContainerImpl();
565 
566         TypeDefinition type = copyTypeDefintion(tc.getTypeDefinition());
567         if (!includePropertyDefinitions) {
568             type.getPropertyDefinitions().clear();
569         }
570 
571         result.setTypeDefinition(type);
572 
573         if (depth != 0) {
574             if (tc.getChildren() != null) {
575                 result.setChildren(new ArrayList<TypeDefinitionContainer>());
576                 for (TypeDefinitionContainer tdc : tc.getChildren()) {
577                     result.getChildren().add(
578                             getTypesDescendants(depth < 0 ? -1 : depth - 1, tdc, includePropertyDefinitions));
579                 }
580             }
581         }
582 
583         return result;
584     }
