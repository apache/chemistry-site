49      /**
50       * Children Collection GET.
51       */
52      public static class GetChildren extends AbstractAtomPubServiceCall {
53          public void serve(CallContext context, CmisService service, String repositoryId, HttpServletRequest request,
54                  HttpServletResponse response) throws Exception {
55              // get parameters
56              String folderId = getStringParameter(request, Constants.PARAM_ID);
57              String filter = getStringParameter(request, Constants.PARAM_FILTER);
58              String orderBy = getStringParameter(request, Constants.PARAM_ORDER_BY);
59              Boolean includeAllowableActions = getBooleanParameter(request, Constants.PARAM_ALLOWABLE_ACTIONS);
60              IncludeRelationships includeRelationships = getEnumParameter(request, Constants.PARAM_RELATIONSHIPS,
61                      IncludeRelationships.class);
62              String renditionFilter = getStringParameter(request, Constants.PARAM_RENDITION_FILTER);
63              Boolean includePathSegment = getBooleanParameter(request, Constants.PARAM_PATH_SEGMENT);
64              BigInteger maxItems = getBigIntegerParameter(request, Constants.PARAM_MAX_ITEMS);
65              BigInteger skipCount = getBigIntegerParameter(request, Constants.PARAM_SKIP_COUNT);
66  
67              // execute
68              ObjectInFolderList children = service.getChildren(repositoryId, folderId, filter, orderBy,
69                      includeAllowableActions, includeRelationships, renditionFilter, includePathSegment, maxItems,
70                      skipCount, null);
71  
72              if (children == null) {
73                  throw new CmisRuntimeException("Children are null!");
74              }
75  
76              ObjectInfo folderInfo = service.getObjectInfo(repositoryId, folderId);
77              if (folderInfo == null) {
78                  throw new CmisRuntimeException("Folder Object Info is missing!");
79              }
80  
81              // set headers
82              response.setStatus(HttpServletResponse.SC_OK);
83              response.setContentType(Constants.MEDIATYPE_FEED);
84  
85              // write XML
86              AtomFeed feed = new AtomFeed();
87              feed.startDocument(response.getOutputStream(), getNamespaces(service));
88              feed.startFeed(true);
89  
90              // write basic Atom feed elements
91              feed.writeFeedElements(folderInfo.getId(), folderInfo.getAtomId(), folderInfo.getCreatedBy(),
92                      folderInfo.getName(), folderInfo.getLastModificationDate(), null, children.getNumItems());
93  
94              // write links
95              UrlBuilder baseUrl = compileBaseUrl(request, repositoryId);
96  
97              feed.writeServiceLink(baseUrl.toString(), repositoryId);
98  
99              UrlBuilder selfLink = compileUrlBuilder(baseUrl, RESOURCE_CHILDREN, folderInfo.getId());
100             selfLink.addParameter(Constants.PARAM_FILTER, filter);
101             selfLink.addParameter(Constants.PARAM_ORDER_BY, orderBy);
102             selfLink.addParameter(Constants.PARAM_ALLOWABLE_ACTIONS, includeAllowableActions);
103             selfLink.addParameter(Constants.PARAM_RELATIONSHIPS, includeRelationships);
104             selfLink.addParameter(Constants.PARAM_RENDITION_FILTER, renditionFilter);
105             selfLink.addParameter(Constants.PARAM_PATH_SEGMENT, includePathSegment);
106             selfLink.addParameter(Constants.PARAM_MAX_ITEMS, maxItems);
107             selfLink.addParameter(Constants.PARAM_SKIP_COUNT, skipCount);
108             feed.writeSelfLink(selfLink.toString(), null);
109 
110             feed.writeDescribedByLink(compileUrl(baseUrl, RESOURCE_TYPE, folderInfo.getTypeId()));
111 
112             feed.writeAllowableActionsLink(compileUrl(baseUrl, RESOURCE_ALLOWABLEACIONS, folderInfo.getId()));
113 
114             feed.writeDownLink(compileUrl(baseUrl, RESOURCE_CHILDREN, folderInfo.getId()), Constants.MEDIATYPE_FEED);
115 
116             if (folderInfo.supportsDescendants()) {
117                 feed.writeDownLink(compileUrl(baseUrl, RESOURCE_DESCENDANTS, folderInfo.getId()),
118                         Constants.MEDIATYPE_DESCENDANTS);
119             }
120 
121             if (folderInfo.supportsFolderTree()) {
122                 feed.writeFolderTreeLink(compileUrl(baseUrl, RESOURCE_FOLDERTREE, folderInfo.getId()));
123             }
124 
125             if (folderInfo.hasParent()) {
126                 feed.writeUpLink(compileUrl(baseUrl, RESOURCE_PARENTS, folderInfo.getId()), Constants.MEDIATYPE_FEED);
127             }
128 
129             if (folderInfo.getRenditionInfos() != null) {
130                 for (RenditionInfo ri : folderInfo.getRenditionInfos()) {
131                     feed.writeAlternateLink(compileUrl(baseUrl, RESOURCE_CONTENT, ri.getId()), ri.getContenType(),
132                             ri.getKind(), ri.getTitle(), ri.getLength());
133                 }
134             }
135 
136             if (folderInfo.hasAcl()) {
137                 feed.writeAclLink(compileUrl(baseUrl, RESOURCE_ACL, folderInfo.getId()));
138             }
139 
140             if (folderInfo.supportsPolicies()) {
141                 feed.writeAclLink(compileUrl(baseUrl, RESOURCE_POLICIES, folderInfo.getId()));
142             }
143 
144             if (folderInfo.supportsRelationships()) {
145                 feed.writeRelationshipsLink(compileUrl(baseUrl, RESOURCE_RELATIONSHIPS, folderInfo.getId()));
146             }
147 
148             UrlBuilder pagingUrl = new UrlBuilder(compileUrlBuilder(baseUrl, RESOURCE_CHILDREN, folderInfo.getId()));
149             pagingUrl.addParameter(Constants.PARAM_FILTER, filter);
150             pagingUrl.addParameter(Constants.PARAM_ORDER_BY, orderBy);
151             pagingUrl.addParameter(Constants.PARAM_ALLOWABLE_ACTIONS, includeAllowableActions);
152             pagingUrl.addParameter(Constants.PARAM_RELATIONSHIPS, includeRelationships);
153             pagingUrl.addParameter(Constants.PARAM_RENDITION_FILTER, renditionFilter);
154             pagingUrl.addParameter(Constants.PARAM_PATH_SEGMENT, includePathSegment);
155             feed.writePagingLinks(pagingUrl, maxItems, skipCount, children.getNumItems(), children.hasMoreItems(),
156                     PAGE_SIZE);
157 
158             // write collection
159             feed.writeCollection(compileUrl(baseUrl, RESOURCE_CHILDREN, folderInfo.getId()), null, "Folder collection",
160                     Constants.MEDIATYPE_CMISATOM);
161 
162             // write entries
163             if (children.getObjects() != null) {
164                 AtomEntry entry = new AtomEntry(feed.getWriter());
165                 for (ObjectInFolderData object : children.getObjects()) {
166                     if ((object == null) || (object.getObject() == null)) {
167                         continue;
168                     }
169                     writeObjectEntry(service, entry, object.getObject(), null, repositoryId, object.getPathSegment(),
170                             null, baseUrl, false, context.getCmisVersion());
171                 }
172             }
173 
174             // we are done
175             feed.endFeed();
176             feed.endDocument();
177         }
178     }

453                 }
454             } else {
455                 folderInfo = new ObjectInfoImpl();
456                 GregorianCalendar now = new GregorianCalendar();
457 
458                 ((ObjectInfoImpl) folderInfo).setId("uri:x-checkedout");
459                 ((ObjectInfoImpl) folderInfo).setName("Checked Out");
460                 ((ObjectInfoImpl) folderInfo).setCreatedBy("");
461                 ((ObjectInfoImpl) folderInfo).setCreationDate(now);
462                 ((ObjectInfoImpl) folderInfo).setLastModificationDate(now);
463                 ((ObjectInfoImpl) folderInfo).setHasParent(false);
464                 ((ObjectInfoImpl) folderInfo).setSupportsDescendants(false);
465                 ((ObjectInfoImpl) folderInfo).setSupportsFolderTree(false);
466             }
467 
468             // set headers
469             response.setStatus(HttpServletResponse.SC_OK);
470             response.setContentType(Constants.MEDIATYPE_FEED);
471 
472             // write XML
473             AtomFeed feed = new AtomFeed();
474             feed.startDocument(response.getOutputStream(), getNamespaces(service));
475             feed.startFeed(true);
476 
477             // write basic Atom feed elements
478             feed.writeFeedElements(folderInfo.getId(), folderInfo.getAtomId(), folderInfo.getCreatedBy(),
479                     folderInfo.getName(), folderInfo.getLastModificationDate(), null, checkedOut.getNumItems());
480 
481             // write links

483 
484             feed.writeServiceLink(baseUrl.toString(), repositoryId);
485 
486             UrlBuilder selfLink = compileUrlBuilder(baseUrl, RESOURCE_CHECKEDOUT, folderInfo.getId());

488             selfLink.addParameter(Constants.PARAM_ORDER_BY, orderBy);
489             selfLink.addParameter(Constants.PARAM_ALLOWABLE_ACTIONS, includeAllowableActions);
490             selfLink.addParameter(Constants.PARAM_RELATIONSHIPS, includeRelationships);
491             selfLink.addParameter(Constants.PARAM_MAX_ITEMS, maxItems);
492             selfLink.addParameter(Constants.PARAM_SKIP_COUNT, skipCount);
493             feed.writeSelfLink(selfLink.toString(), null);
494 
495             UrlBuilder pagingUrl = compileUrlBuilder(baseUrl, RESOURCE_CHECKEDOUT, folderInfo.getId());
496             pagingUrl.addParameter(Constants.PARAM_FILTER, filter);
497             pagingUrl.addParameter(Constants.PARAM_ORDER_BY, orderBy);
498             pagingUrl.addParameter(Constants.PARAM_ALLOWABLE_ACTIONS, includeAllowableActions);
499             pagingUrl.addParameter(Constants.PARAM_RELATIONSHIPS, includeRelationships);
500             pagingUrl.addParameter(Constants.PARAM_RENDITION_FILTER, renditionFilter);
501             feed.writePagingLinks(pagingUrl, maxItems, skipCount, checkedOut.getNumItems(), checkedOut.hasMoreItems(),

503 
504             // write entries

506                 AtomEntry entry = new AtomEntry(feed.getWriter());

508                     if (object == null) {
509                         continue;
510                     }
511                     writeObjectEntry(service, entry, object, null, repositoryId, null, null, baseUrl, false,
512                             context.getCmisVersion());
513                 }
514             }
515 
516             // we are done
517             feed.endFeed();
518             feed.endDocument();
519         }
