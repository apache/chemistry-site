204     /**
205      * See CMIS 1.0 section 2.2.4.1 createDocument
206      */
207     public String createDocument(Session session, Properties properties, String folderId, ContentStream contentStream,
208             VersioningState versioningState) {
209 
210         log.debug("createDocument");
211 
212         // check properties
213         if (properties == null || properties.getProperties() == null) {
214             throw new CmisInvalidArgumentException("Properties must be set!");
215         }
216 
217         // check type
218         String typeId = PropertyHelper.getTypeId(properties);
219         TypeDefinition type = typeManager.getType(typeId);
220         if (type == null) {
221             throw new CmisObjectNotFoundException("Type '" + typeId + "' is unknown!");
222         }
223 
224         boolean isVersionable = JcrTypeManager.isVersionable(type);
225         if (!isVersionable && versioningState != VersioningState.NONE) {
226             throw new CmisConstraintException("Versioning not supported for " + typeId);
227         }
228 
229         if (isVersionable && versioningState == VersioningState.NONE) {
230             throw new CmisConstraintException("Versioning required for " + typeId);
231         }
232 
233         // get the name
234         String name = PropertyHelper.getStringProperty(properties, PropertyIds.NAME);
235         // get parent Node and create child
236         JcrFolder parent = getJcrNode(session, folderId).asFolder();
237         JcrDocumentTypeHandler typeHandler = typeHandlerManager.getDocumentTypeHandler(typeId);
238         JcrNode jcrNode = typeHandler.createDocument(parent, JcrConverter.toJcrName(name), properties, contentStream, versioningState);
239         return jcrNode.getId();
240     }

243      * See CMIS 1.0 section 2.2.4.2 createDocumentFromSource
244      */
245     public String createDocumentFromSource(Session session, String sourceId, Properties properties, String folderId,
246             VersioningState versioningState) {
247 
248         log.debug("createDocumentFromSource");
249 
250         // get parent folder Node
251         JcrFolder parent = getJcrNode(session, folderId).asFolder();
252 
253         // get source document Node
254         JcrDocument source = getJcrNode(session, sourceId).asDocument();
255 
256         boolean isVersionable = source.isVersionable();
257         if (!isVersionable && versioningState != VersioningState.NONE) {
258             throw new CmisConstraintException("Versioning not supported for " + sourceId);
259         }
260 
261         if (isVersionable && versioningState == VersioningState.NONE) {
262             throw new CmisConstraintException("Versioning required for " + sourceId);
263         }
264 
265         // create child from source
266         JcrNode jcrNode = parent.addNodeFromSource(source, properties);
267         return jcrNode.getId();
268     }

439     /**
440      * See CMIS 1.0 section 2.2.3.1 getChildren
441      */
442     public ObjectInFolderList getChildren(Session session, String folderId, String filter,
443             Boolean includeAllowableActions, Boolean includePathSegment, BigInteger maxItems, BigInteger skipCount,
444             ObjectInfoHandler objectInfos, boolean requiresObjectInfo) {
445 
446         log.debug("getChildren");
447 
448         // skip and max
449         int skip = skipCount == null ? 0 : skipCount.intValue();
450         if (skip < 0) {
451             skip = 0;
452         }
453 
454         int max = maxItems == null ? Integer.MAX_VALUE : maxItems.intValue();
455         if (max < 0) {
456             max = Integer.MAX_VALUE;
457         }
458 
459         // get the folder
460         JcrFolder jcrFolder = getJcrNode(session, folderId).asFolder();
461 
462         // set object info of the the folder
463         if (requiresObjectInfo) {
464             jcrFolder.compileObjectType(null, false, objectInfos, requiresObjectInfo);
465         }
466 
467         // prepare result
468         ObjectInFolderListImpl result = new ObjectInFolderListImpl();
469         result.setObjects(new ArrayList<ObjectInFolderData>());
470         result.setHasMoreItems(false);
471         int count = 0;
472 
473         // iterate through children
474         Set<String> splitFilter = splitFilter(filter);
475         Iterator<JcrNode> childNodes = jcrFolder.getNodes();
476         while (childNodes.hasNext()) {
477             JcrNode child = childNodes.next();            
478             count++;
479 
480             if (skip > 0) {
481                 skip--;
482                 continue;
483             }
484 
485             if (result.getObjects().size() >= max) {
486                 result.setHasMoreItems(true);
487                 continue;
488             }
489 
490             // build and add child object
491             ObjectInFolderDataImpl objectInFolder = new ObjectInFolderDataImpl();
492             objectInFolder.setObject(child.compileObjectType(splitFilter, includeAllowableActions, objectInfos,
493                     requiresObjectInfo));
494 
495             if (Boolean.TRUE.equals(includePathSegment)) {
496                 objectInFolder.setPathSegment(child.getName());
497             }
498 
499             result.getObjects().add(objectInFolder);
500         }
501 
502         result.setNumItems(BigInteger.valueOf(count));
503         return result;
504     }

556     /**
557      * See CMIS 1.0 section 2.2.3.5 getObjectParents
558      */
559     public List<ObjectParentData> getObjectParents(Session session, String objectId, String filter,
560             Boolean includeAllowableActions, Boolean includeRelativePathSegment, ObjectInfoHandler objectInfos,
561             boolean requiresObjectInfo) {
562 
563         log.debug("getObjectParents");
564 
565         // get the file or folder
566         JcrNode jcrNode = getJcrNode(session, objectId);
567 
568         // don't climb above the root folder
569         if (jcrNode.isRoot()) {
570             return Collections.emptyList();
571         }
572 
573         // set object info of the the object
574         if (requiresObjectInfo) {
575             jcrNode.compileObjectType(null, false, objectInfos, requiresObjectInfo);
576         }
577 
578         // get parent
579         JcrNode parent = jcrNode.getParent();
580         ObjectData object = parent.compileObjectType(splitFilter(filter), includeAllowableActions, objectInfos,
581                 requiresObjectInfo);
582 
583         ObjectParentDataImpl result = new ObjectParentDataImpl();
584         result.setObject(object);
585         if (Boolean.TRUE.equals(includeRelativePathSegment)) {
586             result.setRelativePathSegment(parent.getName());
587         }
588 
589         return Collections.singletonList((ObjectParentData) result);
590     }

593      * See CMIS 1.0 section 2.2.4.9 getObjectByPath
594      */
595     public ObjectData getObjectByPath(Session session, String folderPath, String filter, boolean includeAllowableActions,
596             boolean includeACL, ObjectInfoHandler objectInfos, boolean requiresObjectInfo) {
597 
598         log.debug("getObjectByPath");
599 
600         // check path 
601         if (folderPath == null || !PathManager.isAbsolute(folderPath)) {
602             throw new CmisInvalidArgumentException("Invalid folder path!");
603         }
604 
605         JcrNode root = getJcrNode(session, PathManager.CMIS_ROOT_ID);
606         JcrNode jcrNode;
607         if (PathManager.isRoot(folderPath)) {
608             jcrNode = root;
609         }
610         else {
611             String path = PathManager.relativize(PathManager.CMIS_ROOT_PATH, folderPath);
612             jcrNode = root.getNode(path);
613         }
614 
615         return jcrNode.compileObjectType(splitFilter(filter), includeAllowableActions, objectInfos, requiresObjectInfo);
616     }

618     /**
619      * See CMIS 1.0 section 2.2.3.6 getCheckedOutDocs
620      */
621     public ObjectList getCheckedOutDocs(Session session, String folderId, String filter, String orderBy,
622             Boolean includeAllowableActions, BigInteger maxItems, BigInteger skipCount) {
623 
624         log.debug("getCheckedOutDocs");
625 
626         // skip and max
627         int skip = skipCount == null ? 0 : skipCount.intValue();
628         if (skip < 0) {
629             skip = 0;
630         }
631 
632         int max = maxItems == null ? Integer.MAX_VALUE : maxItems.intValue();
633         if (max < 0) {
634             max = Integer.MAX_VALUE;
635         }
636 
637         try {
638             // Build xpath query of the form
639             // '//path/to/folderId//*[jcr:isCheckedOut='true' and (not(@jcr:createdBy) or @jcr:createdBy='admin')]'
640             String xPath = "/*[jcr:isCheckedOut='true' " +
641                     "and (not(@jcr:createdBy) or @jcr:createdBy='" + session.getUserID() + "')]";
642             
643             if (folderId != null) {
644                 JcrFolder jcrFolder = getJcrNode(session, folderId).asFolder();
645                 String path = jcrFolder.getNode().getPath();
646                 if ("/".equals(path)) {
647                     path = "";
648                 }
649                 xPath = '/' + Util.escape(path) + xPath;
650             }
651             else {
652                 xPath = '/' + xPath;
653             }
654 
655             // Execute query
656             QueryManager queryManager = session.getWorkspace().getQueryManager();

783     public List<ObjectData> getAllVersions(Session session, String objectId, String filter,
784             Boolean includeAllowableActions, ObjectInfoHandler objectInfos, boolean requiresObjectInfo) {
785 
786         log.debug("getAllVersions");
787 
788         // check id
789         if (objectId == null) {
790             throw new CmisInvalidArgumentException("Object Id must be set.");
791         }
792 
793         Set<String> splitFilter = splitFilter(filter);
794 
795         // get the node
796         JcrNode jcrNode = getJcrNode(session, objectId);
797 
798         // Collect versions
799         if (jcrNode.isVersionable()) {
800             JcrVersionBase jcrVersion = jcrNode.asVersion();
801 
802             Iterator<JcrVersion> versions = jcrVersion.getVersions();
803             if (versions.hasNext()) {
804                 versions.next(); // skip root version
805             }
806 
807             List<ObjectData> allVersions = new ArrayList<ObjectData>();
808             while (versions.hasNext()) {
809                 JcrVersion version = versions.next();
810                 ObjectData objectData = version.compileObjectType(splitFilter, includeAllowableActions, objectInfos,
811                         requiresObjectInfo);
812                 allVersions.add(objectData);
813             }
814 
815             // Add pwc if checked out
816             if (jcrVersion.isDocumentCheckedOut()) {
817                 JcrPrivateWorkingCopy pwc = jcrVersion.getPwc();
818                 ObjectData objectData = pwc.compileObjectType(splitFilter, includeAllowableActions, objectInfos,
819                         requiresObjectInfo);
820 
821                 allVersions.add(objectData);
822             }
823 
824             // CMIS mandates descending order
825             Collections.reverse(allVersions);
826             return allVersions;
827         }
828         else {
829             // Single version
830             ObjectData objectData = jcrNode.compileObjectType(splitFilter, includeAllowableActions, objectInfos,
831                     requiresObjectInfo);

841     public ObjectList query(final Session session, String statement, Boolean searchAllVersions,
842             Boolean includeAllowableActions, BigInteger maxItems, BigInteger skipCount) {
843 
844         log.debug("query");
845 
846         if (searchAllVersions) {
847             throw new CmisNotSupportedException("Not supported: query for all versions");
848         }
849 
850         // skip and max
851         int skip = skipCount == null ? 0 : skipCount.intValue();  
852         if (skip < 0) {
853             skip = 0;
854         }
855 
856         int max = maxItems == null ? Integer.MAX_VALUE : maxItems.intValue();
857         if (max < 0) {
858             max = Integer.MAX_VALUE;
859         }
860 
861         QueryTranslator queryTranslator = new QueryTranslator(typeManager) {
862             @Override
863             protected String jcrPathFromId(String id) {
864                 try {
865                     JcrFolder folder = getJcrNode(session, id).asFolder();
866                     String path = folder.getNode().getPath();
867                     return Util.escape(path);                    
868                 }
869                 catch (RepositoryException e) {
870                     log.debug(e.getMessage(), e);
871                     throw new CmisRuntimeException(e.getMessage(), e);
872                 }
873             }
874 
875             @Override
876             protected String jcrPathFromCol(TypeDefinition fromType, String name) {
877                 return typeHandlerManager.getIdentifierMap(fromType.getId()).jcrPathFromCol(name);
878             }
879 
880             @Override
881             protected String jcrTypeName(TypeDefinition fromType) {
882                 return typeHandlerManager.getIdentifierMap(fromType.getId()).jcrTypeName();
883             }
884 
885             @Override
886             protected String jcrTypeCondition(TypeDefinition fromType) {
887                 return typeHandlerManager.getIdentifierMap(fromType.getId()).jcrTypeCondition();
888             }
889         };
890 
891         String xPath = queryTranslator.translateToXPath(statement);
892         try {  
893             // Execute query
894             QueryManager queryManager = session.getWorkspace().getQueryManager();
895             Query query = queryManager.createQuery(xPath, Query.XPATH);
896 
897             if (skip > 0) {
898                 query.setOffset(skip);
899             }
900             if (max < Integer.MAX_VALUE) {
901                 query.setLimit(max + 1);    // One more in order to detect whether there are more items
902             }
903 
904             QueryResult queryResult = query.execute();
905 
906             // prepare results
907             ObjectListImpl result = new ObjectListImpl();
908             result.setObjects(new ArrayList<ObjectData>());
909             result.setHasMoreItems(false);
910 
911             // iterate through children
912             int count = 0;
913             NodeIterator nodes = queryResult.getNodes();
914             while (nodes.hasNext() && result.getObjects().size() < max) {
915                 Node node = nodes.nextNode();
916                 JcrNode jcrNode = typeHandlerManager.create(node);
917                 count++;
918 
919                 // Get pwc if this node is versionable and checked out
920                 if (jcrNode.isVersionable() && jcrNode.asVersion().isCheckedOut()) {
921                     jcrNode = jcrNode.asVersion().getPwc();
922                 }
923 
924                 // build and add child object
925                 ObjectData objectData = jcrNode.compileObjectType(null, includeAllowableActions, null, false);
926                 result.getObjects().add(objectData);
927             }
928 
929             result.setHasMoreItems(nodes.hasNext());
930             result.setNumItems(BigInteger.valueOf(count));
931             return result;
932         }
933         catch (RepositoryException e) {
934             log.debug(e.getMessage(), e);
935             throw new CmisRuntimeException(e.getMessage(), e);
936         }
937     }

1053     /**
1054      * Transitively gather the children of a node down to a specific depth
1055      */
1056     private static void gatherDescendants(JcrFolder jcrFolder, List<ObjectInFolderContainer> list,
1057             boolean foldersOnly, int depth, Set<String> filter, Boolean includeAllowableActions,
1058             Boolean includePathSegments, ObjectInfoHandler objectInfos, boolean requiresObjectInfo) {
1059 
1060         // iterate through children
1061         Iterator<JcrNode> childNodes = jcrFolder.getNodes();
1062         while (childNodes.hasNext()) {
1063             JcrNode child = childNodes.next();
1064 
1065             // folders only?
1066             if (foldersOnly && !child.isFolder()) {
1067                 continue;
1068             }
1069 
1070             // add to list
1071             ObjectInFolderDataImpl objectInFolder = new ObjectInFolderDataImpl();
1072             objectInFolder.setObject(child.compileObjectType(filter, includeAllowableActions, objectInfos,
1073                     requiresObjectInfo));
1074 
1075             if (Boolean.TRUE.equals(includePathSegments)) {
1076                 objectInFolder.setPathSegment(child.getName());
1077             }
1078 
1079             ObjectInFolderContainerImpl container = new ObjectInFolderContainerImpl();
1080             container.setObject(objectInFolder);
1081 
1082             list.add(container);
1083 
1084             // move to next level
1085             if (depth != 1 && child.isFolder()) {
1086                 container.setChildren(new ArrayList<ObjectInFolderContainer>());
1087                 gatherDescendants(child.asFolder(), container.getChildren(), foldersOnly, depth - 1, filter,
1088                         includeAllowableActions, includePathSegments, objectInfos, requiresObjectInfo);
1089             }
1090         }

1093     /**
1094      * Splits a filter statement into a collection of properties.
1095      */
1096     private static Set<String> splitFilter(String filter) {
1097         if (filter == null) {
1098             return null;
1099         }
1100 
1101         if (filter.trim().length() == 0) {
1102             return null;
1103         }
1104 
1105         Set<String> result = new HashSet<String>();
1106         for (String s : filter.split(",")) {
1107             s = s.trim();
1108             if (s.equals("*")) {
1109                 return null;
1110             } else if (s.length() > 0) {
1111                 result.add(s);
1112             }
1113         }
1114 
1115         // set a few base properties
1116         // query name == id (for base type properties)
1117         result.add(PropertyIds.OBJECT_ID);
1118         result.add(PropertyIds.OBJECT_TYPE_ID);
1119         result.add(PropertyIds.BASE_TYPE_ID);
1120 
1121         return result;
1122     }
