455

457             List<ObjectData> rels = relEntry.getValue();
458             String srcId = relEntry.getKey();
459             if (srcId.equals(docId1)) {
460                 assertEquals(1, rels.size());
461                 assertEquals(id1, rels.get(0).getId());
462                 found1 = true;
463             } else if (srcId.equals(docId2)) {
464                 assertEquals(1, rels.size());
465                 assertEquals(id2, rels.get(0).getId());
466                 found2 = true;
467             } else if (srcId.equals(docId3)) {
468                 assertEquals(1, rels.size());
469                 assertEquals(id3, rels.get(0).getId());
470                 found3 = true;
471             } else if (srcId.equals(docId4)) {
472                 assertEquals(1, rels.size());
473                 assertEquals(id4, rels.get(0).getId());
474                 found4 = true;
475             } else if (srcId.equals(targetId)) {
476                 assertEquals(4, rels.size());
477                 found5 = true;
478             }
479         }
480         assertTrue(found1 && found2 && found3 && found4 && found5);
