124     private void logFolderContainer(ObjectInFolderContainer folder, int depth) {
125         StringBuilder prefix = new StringBuilder();
126         for (int i = 0; i < depth; i++) {
127             prefix.append("   ");
128         }
129 
130         log.info(prefix + "name: " + folder.getObject().getPathSegment());
131         List<ObjectInFolderContainer> children = folder.getChildren();
132         if (null != children) {
133             for (ObjectInFolderContainer child : children) {
134                 logFolderContainer(child, depth + 1);
135             }
136         }
137     }

261     private int getSizeOfDescendants(List<ObjectInFolderContainer> objs) {
262         int sum = 0;
263         if (null != objs) {
264             sum = objs.size();
265             for (ObjectInFolderContainer obj : objs) {
266                 if (null != obj.getChildren()) {
267                     sum += getSizeOfDescendants(obj.getChildren());
268                 }
269             }
270         }
271         return sum;
272     }

274     private void createLevel1Folders() {
275         for (int i = 0; i < NUM_ROOT_FOLDERS; i++) {
276             List<PropertyData<?>> properties = new ArrayList<PropertyData<?>>();
277             properties.add(fFactory.createPropertyIdData(PropertyIds.NAME, "Folder " + i));
278             properties.add(fFactory.createPropertyIdData(PropertyIds.OBJECT_TYPE_ID, InMemoryFolderTypeDefinition
279                     .getRootFolderType().getId()));
280             Properties props = fFactory.createPropertiesData(properties);
281             String id = fObjSvc.createFolder(fRepositoryId, props, fRootFolderId, null, null, null, null);
282             if (i == 3) {
283                 fLevel1FolderId = id;
284             }
285         }
286     }
