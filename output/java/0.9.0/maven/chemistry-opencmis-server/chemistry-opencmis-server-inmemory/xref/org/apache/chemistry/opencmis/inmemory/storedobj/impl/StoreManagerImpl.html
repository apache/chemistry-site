160     public TypeDefinitionContainer getTypeById(String repositoryId, String typeId) {
161         TypeManager typeManager = fMapRepositoryToTypeManager.get(repositoryId);
162         if (null == typeManager) {
163             throw new RuntimeException("Unknown repository " + repositoryId);
164         }
165 
166         boolean cmis11 = InMemoryServiceContext.getCallContext().getCmisVersion() != CmisVersion.CMIS_1_0;
167         TypeDefinitionContainer tdc = typeManager.getTypeById(typeId);
168         if (null != tdc && !cmis11) {
169             TypeDefinition td = tdc.getTypeDefinition();
170             if (td.getBaseTypeId() == BaseTypeId.CMIS_ITEM || td.getBaseTypeId() == BaseTypeId.CMIS_SECONDARY
171                     || td.getId().equals(BaseTypeId.CMIS_ITEM.value())
172                     || td.getId().equals(BaseTypeId.CMIS_SECONDARY.value())) {
173                 tdc = null; // filter new types for CMIS 1.0
174             }
175         }
176         return tdc;
177     }

179     public TypeDefinitionContainer getTypeById(String repositoryId, String typeId, boolean includePropertyDefinitions,
180             int depth) {
181         TypeManager typeManager = fMapRepositoryToTypeManager.get(repositoryId);
182         if (null == typeManager) {
183             throw new CmisInvalidArgumentException("Unknown repository " + repositoryId);
184         }
185 
186         TypeDefinitionContainer tc = typeManager.getTypeById(typeId);
187 
188         if (tc != null) {
189             if (depth == -1) {
190                 if (includePropertyDefinitions)
191                     return tc;
192                 else
193                     depth = Integer.MAX_VALUE;
194             } else if (depth == 0 || depth < -1)
195                 throw new CmisInvalidArgumentException("illegal depth value: " + depth);
196 
197             return cloneTypeList(depth, includePropertyDefinitions, tc, null);
198         } else
199             return null;
200     }

227     public List<TypeDefinitionContainer> getRootTypes(String repositoryId, boolean includePropertyDefinitions) {
228         List<TypeDefinitionContainer> result;
229         TypeManager typeManager = fMapRepositoryToTypeManager.get(repositoryId);
230         if (null == typeManager) {
231             throw new CmisInvalidArgumentException("Unknown repository " + repositoryId);
232         }
233         List<TypeDefinitionContainer> rootTypes = typeManager.getRootTypes();
234         
235         // remove cmis:item and cmis:secondary for CMIS 1.0
236         boolean cmis11 = InMemoryServiceContext.getCallContext().getCmisVersion() != CmisVersion.CMIS_1_0;
237         if (!cmis11) {
238             rootTypes = new ArrayList<TypeDefinitionContainer>(rootTypes);
239             TypeDefinitionContainer tcItem = null, tcSecondary = null;
240             for(TypeDefinitionContainer tc : rootTypes) {
241                 if (tc.getTypeDefinition().getId().equals(BaseTypeId.CMIS_ITEM.value()))
242                     tcItem = tc;
243                 if (tc.getTypeDefinition().getId().equals(BaseTypeId.CMIS_SECONDARY.value()))
244                     tcSecondary = tc;
245             }
246             if (tcItem != null)
247                 rootTypes.remove(tcItem);
248             if (tcSecondary != null)
249                 rootTypes.remove(tcSecondary);
250         }
251         
252         if (includePropertyDefinitions) {
253             result = rootTypes;
254         } else {
255             result = new ArrayList<TypeDefinitionContainer>(rootTypes.size());
256             // copy list and omit properties
257             for (TypeDefinitionContainer c : rootTypes) {
258                 AbstractTypeDefinition td = ((AbstractTypeDefinition) c.getTypeDefinition()).clone();
259                 TypeDefinitionContainerImpl tdc = new TypeDefinitionContainerImpl(td);
260                 tdc.setChildren(c.getChildren());
261                 td.setPropertyDefinitions(null);
262                 result.add(tdc);
263             }
264         }
265         return result;
266     }

288     public void initRepositoryInfo(String repositoryId, String repoInfoCreatorClassName) {
289         RepositoryInfoCreator repoCreator = null;
290 
291         if (repoInfoCreatorClassName != null) {
292             Object obj = null;
293             try {
294                 obj = Class.forName(repoInfoCreatorClassName).newInstance();
295             } catch (InstantiationException e) {
296                 throw new RuntimeException(
297                         "Illegal class to create type system, must implement RepositoryInfoCreator interface.", e);
298             } catch (IllegalAccessException e) {
299                 throw new RuntimeException(
300                         "Illegal class to create type system, must implement RepositoryInfoCreator interface.", e);
301             } catch (ClassNotFoundException e) {
302                 throw new RuntimeException(
303                         "Illegal class to create type system, must implement RepositoryInfoCreator interface.", e);
304             }
305 
306             if (obj instanceof RepositoryInfoCreator) {
307                 repoCreator = (RepositoryInfoCreator) obj;
308                 fRepositoryInfo = repoCreator.createRepositoryInfo();
309             } else {
310                 throw new RuntimeException(
311                         "Illegal class to create repository info, must implement RepositoryInfoCreator interface.");
312             }
313         } else {
314             // create a default repository info
315             createRepositoryInfo(repositoryId);

543     /**
544      * traverse tree and replace each need node with a clone. remove properties
545      * on clone if requested, cut children of clone if depth is exceeded.
546      * 
547      * @param depth
548      *            levels of children to copy
549      * @param includePropertyDefinitions
550      *            indicates with or without property definitions
551      * @param tdc
552      *            type definition to clone
553      * @param parent
554      *            parent container where to add clone as child
555      * @return cloned type definition
556      */
557     private static TypeDefinitionContainer cloneTypeList(int depth, boolean includePropertyDefinitions,
558             TypeDefinitionContainer tdc, TypeDefinitionContainer parent) {
559 
560         AbstractTypeDefinition tdClone = ((AbstractTypeDefinition) tdc.getTypeDefinition()).clone();
561         if (!includePropertyDefinitions) {
562             tdClone.setPropertyDefinitions(null);
563         }
564
