383     /*
384      * (non-Javadoc)
385      * 
386      * @see
387      * org.apache.chemistry.opencmis.inmemory.server.BaseServiceValidatorImpl
388      * #createPolicy(org.apache.chemistry.opencmis.commons.server.CallContext,
389      * java.lang.String, java.lang.String,
390      * org.apache.chemistry.opencmis.commons.data.ExtensionsData)
391      */
392     @Override
393     public StoredObject createPolicy(CallContext context, String repositoryId, String folderId, Acl addAces,
394             Acl removeAces, List<String> policyIds, ExtensionsData extension) {
395 
396         if (policyIds != null && policyIds.size() > 0)
397             throw new CmisConstraintException("Applying policies to policies is not supported.");
398         if (folderId != null && folderId.length() > 0)
399             throw new CmisConstraintException("Policies cannot be created in folders.");
400         if (addAces != null || removeAces != null)
401             throw new CmisConstraintException("ACLs on policies are not suported.");
402 
403         StoredObject so = super
404                 .createPolicy(context, repositoryId, folderId, addAces, removeAces, policyIds, extension);
405         checkAllAccess(repositoryId, context.getUsername(), so);
406         return so;
407     }
