179     /**
180      * Converts a repository info object.
181      */
182     public static JSONObject convert(final RepositoryInfo repositoryInfo, final String repositoryUrl,
183             final String rootUrl) {
184         if (repositoryInfo == null) {
185             return null;
186         }
187 
188         JSONObject result = new JSONObject();
189 
190         result.put(JSON_REPINFO_ID, repositoryInfo.getId());
191         result.put(JSON_REPINFO_NAME, repositoryInfo.getName());
192         result.put(JSON_REPINFO_DESCRIPTION, repositoryInfo.getDescription());
193         result.put(JSON_REPINFO_VENDOR, repositoryInfo.getVendorName());
194         result.put(JSON_REPINFO_PRODUCT, repositoryInfo.getProductName());
195         result.put(JSON_REPINFO_PRODUCT_VERSION, repositoryInfo.getProductVersion());
196         result.put(JSON_REPINFO_ROOT_FOLDER_ID, repositoryInfo.getRootFolderId());
197         result.put(JSON_REPINFO_CAPABILITIES, convert(repositoryInfo.getCapabilities()));
198         setIfNotNull(JSON_REPINFO_ACL_CAPABILITIES, convert(repositoryInfo.getAclCapabilities()), result);
199         result.put(JSON_REPINFO_CHANGE_LOG_TOKEN, repositoryInfo.getLatestChangeLogToken());
200         result.put(JSON_REPINFO_CMIS_VERSION_SUPPORTED, repositoryInfo.getCmisVersionSupported());
201         setIfNotNull(JSON_REPINFO_THIN_CLIENT_URI, repositoryInfo.getThinClientUri(), result);
202         setIfNotNull(JSON_REPINFO_CHANGES_INCOMPLETE, repositoryInfo.getChangesIncomplete(), result);
203 
204         JSONArray changesOnType = new JSONArray();
205         if (repositoryInfo.getChangesOnType() != null) {
206             for (BaseTypeId type : repositoryInfo.getChangesOnType()) {
207                 if (type != null) {
208                     changesOnType.add(getJSONStringValue(type.value()));
209                 }
210             }
211         }
212         result.put(JSON_REPINFO_CHANGES_ON_TYPE, changesOnType);
213 
214         setIfNotNull(JSON_REPINFO_PRINCIPAL_ID_ANONYMOUS, repositoryInfo.getPrincipalIdAnonymous(), result);
215         setIfNotNull(JSON_REPINFO_PRINCIPAL_ID_ANYONE, repositoryInfo.getPrincipalIdAnyone(), result);
216 
217         if (repositoryInfo.getExtensionFeatures() != null && !repositoryInfo.getExtensionFeatures().isEmpty()) {
218             JSONArray extendedFeatures = new JSONArray();
219 
220             for (ExtensionFeature feature : repositoryInfo.getExtensionFeatures()) {
221                 JSONObject jsonFeature = new JSONObject();
222 
223                 setIfNotNull(JSON_FEATURE_ID, feature.getId(), jsonFeature);
224                 setIfNotNull(JSON_FEATURE_URL, feature.getUrl(), jsonFeature);
225                 setIfNotNull(JSON_FEATURE_COMMON_NAME, feature.getCommonName(), jsonFeature);
226                 setIfNotNull(JSON_FEATURE_VERSION_LABEL, feature.getVersionLabel(), jsonFeature);
227                 setIfNotNull(JSON_FEATURE_DESCRIPTION, feature.getDescription(), jsonFeature);
228 
229                 if (feature.getFeatureData() != null && !feature.getFeatureData().isEmpty()) {
230                     JSONObject data = new JSONObject();
231                     data.putAll(feature.getFeatureData());
232                     jsonFeature.put(JSON_FEATURE_DATA, data);
233                 }
234 
235                 convertExtension(feature, jsonFeature);
236 
237                 extendedFeatures.add(jsonFeature);
238             }
239 
240             result.put(JSON_REPINFO_EXTENDED_FEATURES, extendedFeatures);
241         }
242 
243         result.put(JSON_REPINFO_REPOSITORY_URL, repositoryUrl);
244         result.put(JSON_REPINFO_ROOT_FOLDER_URL, rootUrl);
245 
246         convertExtension(repositoryInfo, result);
247 
248         return result;
249     }

254     public static JSONObject convert(final RepositoryCapabilities capabilities) {
255         if (capabilities == null) {
256             return null;
257         }
258 
259         JSONObject result = new JSONObject();
260 
261         result.put(JSON_CAP_CONTENT_STREAM_UPDATABILITY,
262                 getJSONEnumValue((capabilities.getContentStreamUpdatesCapability())));
263         result.put(JSON_CAP_CHANGES, getJSONEnumValue(capabilities.getChangesCapability()));
264         result.put(JSON_CAP_RENDITIONS, getJSONEnumValue(capabilities.getRenditionsCapability()));
265         result.put(JSON_CAP_GET_DESCENDANTS, capabilities.isGetDescendantsSupported());
266         result.put(JSON_CAP_GET_FOLDER_TREE, capabilities.isGetFolderTreeSupported());
267         result.put(JSON_CAP_MULTIFILING, capabilities.isMultifilingSupported());
268         result.put(JSON_CAP_UNFILING, capabilities.isUnfilingSupported());
269         result.put(JSON_CAP_VERSION_SPECIFIC_FILING, capabilities.isVersionSpecificFilingSupported());
270         result.put(JSON_CAP_PWC_SEARCHABLE, capabilities.isPwcSearchableSupported());
271         result.put(JSON_CAP_PWC_UPDATABLE, capabilities.isPwcUpdatableSupported());
272         result.put(JSON_CAP_ALL_VERSIONS_SEARCHABLE, capabilities.isAllVersionsSearchableSupported());
273         result.put(JSON_CAP_ORDER_BY, getJSONEnumValue(capabilities.getOrderByCapability()));
274         result.put(JSON_CAP_QUERY, getJSONEnumValue(capabilities.getQueryCapability()));
275         result.put(JSON_CAP_JOIN, getJSONEnumValue(capabilities.getJoinCapability()));
276         result.put(JSON_CAP_ACL, getJSONEnumValue(capabilities.getAclCapability()));
277 
278         if (capabilities.getCreatablePropertyTypes() != null) {
279             CreatablePropertyTypes creatablePropertyTypes = capabilities.getCreatablePropertyTypes();
280 
281             JSONObject creatablePropertyTypesJson = new JSONObject();
282 
283             if (creatablePropertyTypes.canCreate() != null) {
284                 JSONArray canCreate = new JSONArray();
285                 for (PropertyType propType : creatablePropertyTypes.canCreate()) {
286                     if (propType != null) {
287                         canCreate.add(propType.value());
288                     }
289                 }
290                 creatablePropertyTypesJson.put(JSON_CAP_CREATABLE_PROPERTY_TYPES_CANCREATE, canCreate);
291             }
292 
293             convertExtension(creatablePropertyTypes, creatablePropertyTypesJson);
294 
295             result.put(JSON_CAP_CREATABLE_PROPERTY_TYPES, creatablePropertyTypesJson);
296         }
297 
298         if (capabilities.getNewTypeSettableAttributes() != null) {
299             NewTypeSettableAttributes newTypeSettableAttributes = capabilities.getNewTypeSettableAttributes();
300 
301             JSONObject newTypeSettableAttributesJson = new JSONObject();
302             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_ID,
303                     newTypeSettableAttributes.canSetId());
304             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_LOCALNAME,
305                     newTypeSettableAttributes.canSetLocalName());
306             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_LOCALNAMESPACE,
307                     newTypeSettableAttributes.canSetLocalNamespace());
308             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_DISPLAYNAME,
309                     newTypeSettableAttributes.canSetDisplayName());
310             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_QUERYNAME,
311                     newTypeSettableAttributes.canSetQueryName());
312             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_DESCRIPTION,
313                     newTypeSettableAttributes.canSetDescription());
314             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_CREATEABLE,
315                     newTypeSettableAttributes.canSetCreatable());
316             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_FILEABLE,
317                     newTypeSettableAttributes.canSetFileable());
318             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_QUERYABLE,
319                     newTypeSettableAttributes.canSetQueryable());
320             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_FULLTEXTINDEXED,
321                     newTypeSettableAttributes.canSetFulltextIndexed());
322             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_INCLUDEDINSUPERTYTPEQUERY,
323                     newTypeSettableAttributes.canSetIncludedInSupertypeQuery());
324             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_CONTROLABLEPOLICY,
325                     newTypeSettableAttributes.canSetControllablePolicy());
326             newTypeSettableAttributesJson.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES_CONTROLABLEACL,
327                     newTypeSettableAttributes.canSetControllableAcl());
328 
329             convertExtension(newTypeSettableAttributes, newTypeSettableAttributesJson);
330 
331             result.put(JSON_CAP_NEW_TYPE_SETTABLE_ATTRIBUTES, newTypeSettableAttributesJson);
332         }
333 
334         convertExtension(capabilities, result);
335 
336         return result;
337     }

339     /**
340      * Converts an ACL capabilities object.
341      */
342     public static JSONObject convert(final AclCapabilities capabilities) {
343         if (capabilities == null) {
344             return null;
345         }
346 
347         JSONObject result = new JSONObject();
348 
349         result.put(JSON_ACLCAP_SUPPORTED_PERMISSIONS, getJSONEnumValue(capabilities.getSupportedPermissions()));
350         result.put(JSON_ACLCAP_ACL_PROPAGATION, getJSONEnumValue(capabilities.getAclPropagation()));
351 
352         // permissions
353         if (capabilities.getPermissions() != null) {
354             JSONArray permissions = new JSONArray();
355 
356             for (PermissionDefinition permDef : capabilities.getPermissions()) {
357                 JSONObject permission = new JSONObject();
358                 permission.put(JSON_ACLCAP_PERMISSION_PERMISSION, permDef.getId());
359                 permission.put(JSON_ACLCAP_PERMISSION_DESCRIPTION, permDef.getDescription());
360 
361                 permissions.add(permission);
362             }
363 
364             result.put(JSON_ACLCAP_PERMISSIONS, permissions);
365         }
366 
367         // permission mapping
368         if (capabilities.getPermissionMapping() != null) {
369             JSONArray permissionMapping = new JSONArray();
370 
371             for (PermissionMapping permMap : capabilities.getPermissionMapping().values()) {
372                 JSONArray mappingPermissions = new JSONArray();
373                 if (permMap.getPermissions() != null) {
374                     for (String p : permMap.getPermissions()) {
375                         mappingPermissions.add(p);
376                     }
377                 }
378 
379                 JSONObject mapping = new JSONObject();
380                 mapping.put(JSON_ACLCAP_MAPPING_KEY, permMap.getKey());
381                 mapping.put(JSON_ACLCAP_MAPPING_PERMISSION, mappingPermissions);
382 
383                 permissionMapping.add(mapping);
384             }
385 
386             result.put(JSON_ACLCAP_PERMISSION_MAPPING, permissionMapping);
387         }
388 
389         convertExtension(capabilities, result);
390 
391         return result;
392     }

394     public static RepositoryInfo convertRepositoryInfo(final Map<String, Object> json) {
395         if (json == null) {
396             return null;
397         }
398 
399         RepositoryInfoBrowserBindingImpl result = new RepositoryInfoBrowserBindingImpl();
400 
401         result.setId(getString(json, JSON_REPINFO_ID));
402         result.setName(getString(json, JSON_REPINFO_NAME));
403         result.setDescription(getString(json, JSON_REPINFO_DESCRIPTION));
404         result.setVendorName(getString(json, JSON_REPINFO_VENDOR));
405         result.setProductName(getString(json, JSON_REPINFO_PRODUCT));
406         result.setProductVersion(getString(json, JSON_REPINFO_PRODUCT_VERSION));
407         result.setRootFolder(getString(json, JSON_REPINFO_ROOT_FOLDER_ID));
408         result.setRepositoryUrl(getString(json, JSON_REPINFO_REPOSITORY_URL));
409         result.setRootUrl(getString(json, JSON_REPINFO_ROOT_FOLDER_URL));
410         result.setCapabilities(convertRepositoryCapabilities(getMap(json.get(JSON_REPINFO_CAPABILITIES))));
411         result.setAclCapabilities(convertAclCapabilities(getMap(json.get(JSON_REPINFO_ACL_CAPABILITIES))));
412         result.setLatestChangeLogToken(getString(json, JSON_REPINFO_CHANGE_LOG_TOKEN));
413         result.setCmisVersionSupported(getString(json, JSON_REPINFO_CMIS_VERSION_SUPPORTED));
414         result.setThinClientUri(getString(json, JSON_REPINFO_THIN_CLIENT_URI));
415         result.setChangesIncomplete(getBoolean(json, JSON_REPINFO_CHANGES_INCOMPLETE));
416 
417         List<Object> changesOnType = getList(json.get(JSON_REPINFO_CHANGES_ON_TYPE));
418         if (changesOnType != null) {
419             List<BaseTypeId> types = new ArrayList<BaseTypeId>();
420             for (Object type : changesOnType) {
421                 if (type != null) {
422                     types.add(BaseTypeId.fromValue(type.toString()));
423                 }
424             }
425             result.setChangesOnType(types);
426         }
427 
428         result.setPrincipalAnonymous(getString(json, JSON_REPINFO_PRINCIPAL_ID_ANONYMOUS));
429         result.setPrincipalAnyone(getString(json, JSON_REPINFO_PRINCIPAL_ID_ANYONE));
430 
431         List<Object> extendedFeatures = getList(json.get(JSON_REPINFO_EXTENDED_FEATURES));
432         if (extendedFeatures != null) {
433             List<ExtensionFeature> features = new ArrayList<ExtensionFeature>();
434 
435             for (Object extendedFeature : extendedFeatures) {
436                 Map<String, Object> jsonFeature = getMap(extendedFeature);
437 
438                 ExtensionFeatureImpl feature = new ExtensionFeatureImpl();
439                 feature.setId(getString(jsonFeature, JSON_FEATURE_ID));
440                 feature.setUrl(getString(jsonFeature, JSON_FEATURE_URL));
441                 feature.setCommonName(getString(jsonFeature, JSON_FEATURE_COMMON_NAME));
442                 feature.setVersionLabel(getString(jsonFeature, JSON_FEATURE_VERSION_LABEL));
443                 feature.setDescription(getString(jsonFeature, JSON_FEATURE_DESCRIPTION));
444 
445                 Map<String, Object> data = getMap(jsonFeature.get(JSON_FEATURE_DATA));
446                 if (data != null) {
447                     LinkedHashMap<String, String> dataMap = new LinkedHashMap<String, String>();
448                     for (Map.Entry<String, Object> e : data.entrySet()) {
449                         dataMap.put(e.getKey(), (e.getValue() == null ? null : e.getValue().toString()));
450                     }
451 
452                     if (!dataMap.isEmpty()) {
453                         feature.setFeatureData(dataMap);
454                     }
455                 }
456 
457                 convertExtension(jsonFeature, feature, FEATURE_KEYS);
458 
459                 features.add(feature);
460             }
461 
462             if (!features.isEmpty()) {
