112     public static Map<String, String> createSessionParameters(String url, BindingType binding, String username,
113             String password, Authentication authentication, boolean compression, boolean clientCompression,
114             boolean cookies) {
115         Map<String, String> parameters = new LinkedHashMap<String, String>();
116 
117         switch (binding) {
118         case WEBSERVICES:
119             parameters.put(SessionParameter.BINDING_TYPE, BindingType.WEBSERVICES.value());
120             parameters.put(SessionParameter.WEBSERVICES_REPOSITORY_SERVICE, url);
121             parameters.put(SessionParameter.WEBSERVICES_NAVIGATION_SERVICE, url);
122             parameters.put(SessionParameter.WEBSERVICES_OBJECT_SERVICE, url);
123             parameters.put(SessionParameter.WEBSERVICES_VERSIONING_SERVICE, url);
124             parameters.put(SessionParameter.WEBSERVICES_DISCOVERY_SERVICE, url);
125             parameters.put(SessionParameter.WEBSERVICES_MULTIFILING_SERVICE, url);
126             parameters.put(SessionParameter.WEBSERVICES_RELATIONSHIP_SERVICE, url);
127             parameters.put(SessionParameter.WEBSERVICES_ACL_SERVICE, url);
128             parameters.put(SessionParameter.WEBSERVICES_POLICY_SERVICE, url);
129             break;
130         case ATOMPUB:
131             parameters.put(SessionParameter.BINDING_TYPE, BindingType.ATOMPUB.value());
132             parameters.put(SessionParameter.ATOMPUB_URL, url);
133             break;
134         case BROWSER:
135             parameters.put(SessionParameter.BINDING_TYPE, BindingType.BROWSER.value());
136             parameters.put(SessionParameter.BROWSER_URL, url);
137             break;
138         default:
139             parameters.put(SessionParameter.BINDING_TYPE, BindingType.CUSTOM.value());
140         }
141 
142         switch (authentication) {
143         case STANDARD:
144             parameters.put(SessionParameter.USER, username);
145             parameters.put(SessionParameter.PASSWORD, password);
146             break;
147         case NTLM:
148             parameters.put(SessionParameter.USER, username);
149             parameters.put(SessionParameter.PASSWORD, password);
150             parameters.put(SessionParameter.AUTHENTICATION_PROVIDER_CLASS,
151                     CmisBindingFactory.NTLM_AUTHENTICATION_PROVIDER);
152             break;
153         default:
154         }
155 
156         if (compression) {
157             parameters.put(SessionParameter.COMPRESSION, "true");
158         }
159 
160         if (clientCompression) {
161             parameters.put(SessionParameter.CLIENT_COMPRESSION, "true");
162         }
163 
164         if (cookies) {
165             parameters.put(SessionParameter.COOKIES, "true");
166         }
167 
168         // get additional workbench properties from system properties
169         Properties sysProps = System.getProperties();
170         for (String key : sysProps.stringPropertyNames()) {
171             if (key.startsWith(WORKBENCH_PREFIX)) {
172                 parameters.put(key, sysProps.getProperty(key));
173             }
174         }
175 
176         return parameters;
177     }

179     private void connect(Map<String, String> sessionParameters, ObjectFactory objectFactory,
180             AuthenticationProvider authenticationProvider, Cache cache) {
181         this.sessionParameters = sessionParameters;
182 
183         // set a new dummy authenticator
184         // don't send previous credentials to another server
185         Authenticator.setDefault(new Authenticator() {
186         });
187 
188         if (Boolean.parseBoolean(sessionParameters.get(ACCEPT_SELF_SIGNED_CERTIFICATES))) {
189             acceptSelfSignedCertificates();
190         }
191 
192         repositories = SessionFactoryImpl.newInstance().getRepositories(sessionParameters, objectFactory,
193                 authenticationProvider, cache);
194     }

237     private void createOperationContexts() {
238 
239         RepositoryInfo repositoryInfo = getSession().getRepositoryInfo();
240 
241         String supportsAcl = "true";
242         if (repositoryInfo != null && repositoryInfo.getCapabilities() != null
243                 && repositoryInfo.getCapabilities().getAclCapability() == CapabilityAcl.NONE) {
244             supportsAcl = "false";
245         }
246 
247         // object operation context
248         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.FILTER, "*");
249         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ACLS, supportsAcl);
250         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ALLOWABLE_ACTIONS, "true");
251         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_POLICIES, "true");
252         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_RELATIONSHIPS,
253                 IncludeRelationships.BOTH.value());
254         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.RENDITION_FILTER, "*");
255         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.ORDER_BY, null);
256         setDefault(OBJECT_PREFIX, sessionParameters, ClientOperationContext.MAX_ITEMS_PER_PAGE, "1000");
257 
258         objectOperationContext = new ClientOperationContext(OBJECT_PREFIX, sessionParameters);
259 
260         // folder operation context
261         if (!sessionParameters.containsKey(FOLDER_PREFIX + ClientOperationContext.FILTER)) {
262             ObjectType type = session.getTypeDefinition(BaseTypeId.CMIS_DOCUMENT.value());
263 
264             StringBuilder filter = new StringBuilder();
265             for (String propId : FOLDER_PROPERTY_SET) {
266                 PropertyDefinition<?> propDef = type.getPropertyDefinitions().get(propId);
267                 if (propDef != null) {
268                     if (filter.length() > 0) {
269                         filter.append(",");
270                     }
271                     filter.append(propDef.getQueryName());
272                 }
273             }
274 
275             sessionParameters.put(FOLDER_PREFIX + ClientOperationContext.FILTER, filter.toString());
276         }
277 
278         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ACLS, "false");
279         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ALLOWABLE_ACTIONS, "false");
280         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_POLICIES, "false");
281         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_RELATIONSHIPS,
282                 IncludeRelationships.NONE.value());
283         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.RENDITION_FILTER, "cmis:none");
284         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.ORDER_BY, null);
285         setDefault(FOLDER_PREFIX, sessionParameters, ClientOperationContext.MAX_ITEMS_PER_PAGE, "10000");
286 
287         folderOperationContext = new ClientOperationContext(FOLDER_PREFIX, sessionParameters);
288 
289         if (!sessionParameters.containsKey(VERSION_PREFIX + ClientOperationContext.FILTER)) {
290             ObjectType type = session.getTypeDefinition(BaseTypeId.CMIS_DOCUMENT.value());
291 
292             StringBuilder filter = new StringBuilder();
293             for (String propId : VERSION_PROPERTY_SET) {
294                 PropertyDefinition<?> propDef = type.getPropertyDefinitions().get(propId);
295                 if (propDef != null) {
296                     if (filter.length() > 0) {
297                         filter.append(",");
298                     }
299                     filter.append(propDef.getQueryName());
300                 }
301             }
302 
303             sessionParameters.put(VERSION_PREFIX + ClientOperationContext.FILTER, filter.toString());
304         }
305 
306         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ACLS, "false");
307         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_ALLOWABLE_ACTIONS, "false");
308         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_POLICIES, "false");
309         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.INCLUDE_RELATIONSHIPS,
310                 IncludeRelationships.NONE.value());
311         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.RENDITION_FILTER, "cmis:none");
312         setDefault(VERSION_PREFIX, sessionParameters, ClientOperationContext.MAX_ITEMS_PER_PAGE, "10000");
313 
314         versionOperationContext = new ClientOperationContext(VERSION_PREFIX, sessionParameters);
315     }

323     private void acceptSelfSignedCertificates() {
324         TrustManager[] trustAllCerts = new TrustManager[] { new X509TrustManager() {
325             public X509Certificate[] getAcceptedIssuers() {
326                 return null;
327             }
328 
329             public void checkClientTrusted(X509Certificate[] certs, String authType) {
330             }
331 
332             public void checkServerTrusted(X509Certificate[] certs, String authType) {
333             }
334         } };
335 
336         try {
337             SSLContext sc = SSLContext.getInstance("SSL");
338             sc.init(null, trustAllCerts, new java.security.SecureRandom());
339             HttpsURLConnection.setDefaultSSLSocketFactory(sc.getSocketFactory());
340         } catch (Exception e) {
341         }
342     }
