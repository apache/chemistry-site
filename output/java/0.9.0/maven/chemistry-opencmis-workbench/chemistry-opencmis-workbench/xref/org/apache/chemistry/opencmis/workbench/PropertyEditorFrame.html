96      private void createGUI() {
97          setTitle(WINDOW_TITLE);
98          setPreferredSize(new Dimension(800, 600));
99          setMinimumSize(new Dimension(300, 120));
100 
101         setLayout(new BorderLayout());
102 
103         final JPanel panel = new JPanel();
104         panel.setLayout(new BoxLayout(panel, BoxLayout.PAGE_AXIS));
105 
106         final Font labelFont = UIManager.getFont("Label.font");
107         final Font boldFont = labelFont.deriveFont(Font.BOLD, labelFont.getSize2D() * 1.2f);
108 
109         final JPanel topPanel = new JPanel();
110         topPanel.setLayout(new BoxLayout(topPanel, BoxLayout.Y_AXIS));
111         topPanel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
112         final JLabel nameLabel = new JLabel(object.getName());
113         nameLabel.setFont(boldFont);
114         topPanel.add(nameLabel);
115         topPanel.add(new JLabel(object.getId()));
116         add(topPanel, BorderLayout.PAGE_START);
117 
118         JScrollPane scrollPane = new JScrollPane(panel);
119         add(scrollPane, BorderLayout.CENTER);
120 
121         propertyPanels = new ArrayList<PropertyEditorFrame.PropertyInputPanel>();
122 
123         int position = 0;
124 
125         // primary type
126         for (PropertyDefinition<?> propDef : object.getType().getPropertyDefinitions().values()) {
127             boolean isUpdatable = (propDef.getUpdatability() == Updatability.READWRITE)
128                     || (propDef.getUpdatability() == Updatability.WHENCHECKEDOUT && object.getAllowableActions()
129                             .getAllowableActions().contains(Action.CAN_CHECK_IN));
130 
131             if (isUpdatable) {
132                 PropertyInputPanel propertyPanel = new PropertyInputPanel(propDef, object.getPropertyValue(propDef
133                         .getId()), position++);
134 
135                 propertyPanels.add(propertyPanel);
136                 panel.add(propertyPanel);
137             }
138         }
139 
140         // secondary types
141         if (object.getSecondaryTypes() != null) {
142             for (ObjectType secType : object.getSecondaryTypes()) {
143                 if (secType.getPropertyDefinitions() != null) {
144                     for (PropertyDefinition<?> propDef : secType.getPropertyDefinitions().values()) {
145                         boolean isUpdatable = (propDef.getUpdatability() == Updatability.READWRITE)
146                                 || (propDef.getUpdatability() == Updatability.WHENCHECKEDOUT && object
147                                         .getAllowableActions().getAllowableActions().contains(Action.CAN_CHECK_IN));
148 
149                         if (isUpdatable) {
150                             PropertyInputPanel propertyPanel = new PropertyInputPanel(propDef,
151                                     object.getPropertyValue(propDef.getId()), position++);
152 
153                             propertyPanels.add(propertyPanel);
154                             panel.add(propertyPanel);
155                         }
156                     }
157                 }
158             }
159         }
160 
161         JButton updateButton = new JButton("Update");
162         updateButton.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
163         updateButton.setDefaultCapable(true);
164         updateButton.addActionListener(new ActionListener() {
165             public void actionPerformed(ActionEvent event) {
166                 if (doUpdate()) {
167                     dispose();
168                 }
169             }
170         });
171 
172         add(updateButton, BorderLayout.PAGE_END);
173 
174         setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
175         pack();
176         setLocationRelativeTo(null);
177         setVisible(true);
178     }

180     /**
181      * Performs the update.
182      */
183     private boolean doUpdate() {
184         try {
185             setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
186 
187             Map<String, Object> properties = new HashMap<String, Object>();
188             for (PropertyInputPanel propertyPanel : propertyPanels) {
189                 if (propertyPanel.includeInUpdate()) {
190                     properties.put(propertyPanel.getId(), propertyPanel.getValue());
191                 }
192             }
193 
194             if (properties.isEmpty()) {
195                 return false;
196             }
197 
198             ObjectId newId = object.updateProperties(properties, false);
199 
200             if ((newId != null) && newId.getId().equals(model.getCurrentObject().getId())) {
201                 try {
202                     model.reloadObject();
203                     model.reloadFolder();
204                 } catch (Exception ex) {
205                     ClientHelper.showError(null, ex);
206                 }
207             }
208 
209             return true;
210         } catch (Exception ex) {
211             ClientHelper.showError(this, ex);
212             return false;
213         } finally {
214             setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
215         }
216     }

290 
291             changeBox = new JComboBox(new Object[] { "Don't change     ", "Update    ", "Unset     " });
292             titlePanel.add(changeBox, BorderLayout.LINE_END);
293 
294             valueComponents = new LinkedList<JComponent>();
295             if (propDef.getCardinality() == Cardinality.SINGLE) {
296                 JComponent valueField = createInputField(value);
297                 valueComponents.add(valueField);
298                 add(valueField);
299             } else {
300                 if (value instanceof List<?>) {
301                     for (Object v : (List<?>) value) {
302                         JComponent valueField = new MultiValuePropertyInputField(createInputField(v), this, bgColor);
303                         valueComponents.add(valueField);
304                         add(valueField);
305                     }
306                 }
307 
308                 JPanel addPanel = new JPanel(new BorderLayout());
309                 addPanel.setBackground(bgColor);
310                 JButton addButton = new JButton(ICON_ADD);
311                 addButton.addActionListener(new ActionListener() {
312                     @Override
313                     public void actionPerformed(ActionEvent e) {
314                         addNewValue();
315                         setStatus(StatusFlag.Update);
316                     }
317                 });
318                 addPanel.add(addButton, BorderLayout.LINE_END);
319                 add(addPanel);
320 
321                 updatePositions();
322             }
323 
324             setMaximumSize(new Dimension(Short.MAX_VALUE, getPreferredSize().height));
325         }
326 
327         protected JComponent createInputField(Object value) {
328             switch (propDef.getPropertyType()) {
329             case INTEGER:
330                 return new IntegerPropertyInputField(value, this, bgColor);
331             case DECIMAL:
332                 return new DecimalPropertyInputField(value, this, bgColor);
333             case DATETIME:
334                 return new DateTimePropertyInputField(value, this, bgColor);
335             case BOOLEAN:
336                 return new BooleanPropertyInputField(value, this, bgColor);
337             default:
338                 return new StringPropertyInputField(value, this, bgColor);
339             }
340         }
341 
342         private void updatePositions() {
343             int n = valueComponents.size();
344             for (int i = 0; i < n; i++) {
345                 MultiValuePropertyInputField comp = (MultiValuePropertyInputField) valueComponents.get(i);
346                 comp.updatePosition(i, i + 1 == n);
347             }
348         }
349 
350         public void addNewValue() {
351             JComponent valueField = new MultiValuePropertyInputField(createInputField(null), this, bgColor);
352             valueComponents.add(valueField);
353             add(valueField, getComponentCount() - 1);
354 
355             updatePositions();
356             setStatus(StatusFlag.Update);
357 
358             revalidate();
359         }
360 
361         public void removeValue(int pos) {
362             remove(valueComponents.remove(pos));
363 
364             updatePositions();
365             setStatus(StatusFlag.Update);
366 
367             revalidate();
368         }
369 
370         public void moveUp(int pos) {
371             JComponent comp = valueComponents.get(pos);
372             Collections.swap(valueComponents, pos, pos - 1);
373 
374             remove(comp);
375             add(comp, pos);
376 
377             updatePositions();
378             setStatus(StatusFlag.Update);
379 
380             revalidate();
381         }
382 
383         public void moveDown(int pos) {
384             JComponent comp = valueComponents.get(pos);
385             Collections.swap(valueComponents, pos, pos + 1);
386 
387             remove(comp);
388             add(comp, pos + 2);
389 
390             updatePositions();
391             setStatus(StatusFlag.Update);
392 
393             revalidate();
394         }
395 
396         public String getId() {
397             return propDef.getId();
398         }
399 
400         public Object getValue() {
401             Object result = null;
402 
403             if (getStatus() == StatusFlag.Update) {
404                 try {
405                     if (propDef.getCardinality() == Cardinality.SINGLE) {
406                         result = ((PropertyValue) valueComponents.get(0)).getPropertyValue();
407                     } else {
408                         List<Object> list = new ArrayList<Object>();
409                         for (JComponent comp : valueComponents) {
410                             list.add(((PropertyValue) comp).getPropertyValue());
411                         }
412                         result = list;
413                     }
414                 } catch (Exception ex) {
415                     ClientHelper.showError(this, ex);
416                 }
417             }
418 
419             return result;
420         }
421 
422         public boolean includeInUpdate() {
423             return getStatus() != StatusFlag.DontChange;
424         }
425 
426         public void setStatus(StatusFlag status) {
427             switch (status) {
428             case Update:
429                 changeBox.setSelectedIndex(1);
430                 break;
431             case Unset:
432                 changeBox.setSelectedIndex(2);
433                 break;
434             default:
435                 changeBox.setSelectedIndex(0);
436             }
437         }
438 
439         public StatusFlag getStatus() {
440             switch (changeBox.getSelectedIndex()) {
441             case 1:
442                 return StatusFlag.Update;
443             case 2:
444                 return StatusFlag.Unset;
445             default:
446                 return StatusFlag.DontChange;
447             }
448         }
449 
450         public Dimension getMaximumSize() {
451             Dimension size = getPreferredSize();
452             size.width = Short.MAX_VALUE;
453             return size;
454         }
455     }

593     /**
594      * DateTime property.
595      */
596     public static class DateTimePropertyInputField extends JPanel implements PropertyValue {
597         private static final long serialVersionUID = 1L;
598 
599         private static final String[] MONTH_STRINGS;
600 
601         static {
602             String[] months = new java.text.DateFormatSymbols().getMonths();
603             int lastIndex = months.length - 1;
604 
605             if (months[lastIndex] == null || months[lastIndex].length() <= 0) {
606                 String[] monthStrings = new String[lastIndex];
607                 System.arraycopy(months, 0, monthStrings, 0, lastIndex);
608                 MONTH_STRINGS = monthStrings;
609             } else {
610                 MONTH_STRINGS = months;
611             }
612         }
613 
614         private final SpinnerNumberModel day;
615         private final SpinnerListModel month;
616         private final SpinnerNumberModel year;
617         private final SpinnerNumberModel hour;
618         private final SpinnerNumberModel min;
619         private final SpinnerNumberModel sec;
620         private final TimeZone timezone;
621 
622         public DateTimePropertyInputField(final Object value, final UpdateStatus status, final Color bgColor) {
623             setLayout(new FlowLayout(FlowLayout.LEFT, 0, 0));
624             setBackground(bgColor);
625 
626             GregorianCalendar cal = (value == null ? new GregorianCalendar() : (GregorianCalendar) value);
627             timezone = cal.getTimeZone();
628 
629             day = new SpinnerNumberModel(cal.get(Calendar.DATE), 1, 31, 1);
630             addSpinner(new JSpinner(day), status);
631 
632             month = new SpinnerListModel(MONTH_STRINGS);
633             month.setValue(MONTH_STRINGS[cal.get(Calendar.MONTH)]);
634             JSpinner monthSpinner = new JSpinner(month);
635             JComponent editor = monthSpinner.getEditor();
636             if (editor instanceof JSpinner.DefaultEditor) {
637                 JFormattedTextField tf = ((JSpinner.DefaultEditor) editor).getTextField();
638                 tf.setColumns(6);
639                 tf.setHorizontalAlignment(JTextField.RIGHT);
640             }
641             addSpinner(monthSpinner, status);
642 
643             year = new SpinnerNumberModel(cal.get(Calendar.YEAR), 0, 9999, 1);
644             JSpinner yearSpinner = new JSpinner(year);
645             yearSpinner.setEditor(new JSpinner.NumberEditor(yearSpinner, "#"));
646             yearSpinner.getEditor().setBackground(bgColor);
647             addSpinner(yearSpinner, status);
648 
649             add(new JLabel("  "));
650 
651             hour = new SpinnerNumberModel(cal.get(Calendar.HOUR_OF_DAY), 0, 23, 1);
652             JSpinner hourSpinner = new JSpinner(hour);
653             addSpinner(hourSpinner, status);
654 
655             add(new JLabel(":"));
656 
657             min = new SpinnerNumberModel(cal.get(Calendar.MINUTE), 0, 59, 1);
658             JSpinner minSpinner = new JSpinner(min);
659             addSpinner(minSpinner, status);
660 
661             add(new JLabel(":"));
662 
663             sec = new SpinnerNumberModel(cal.get(Calendar.SECOND), 0, 59, 1);
664             JSpinner secSpinner = new JSpinner(sec);
665             addSpinner(secSpinner, status);
666 
667             add(new JLabel(" " + timezone.getDisplayName(true, TimeZone.SHORT)));
668         }
669 
670         private void addSpinner(final JSpinner spinner, final UpdateStatus status) {
671             spinner.addChangeListener(new ChangeListener() {
672                 @Override
673                 public void stateChanged(ChangeEvent e) {
674                     status.setStatus(StatusFlag.Update);
675                 }
676             });
677 
678             add(spinner);
679         }
680 
681         public Object getPropertyValue() {
682             GregorianCalendar result = new GregorianCalendar();
683 
684             result.setTimeZone(timezone);
685 
686             result.set(Calendar.YEAR, year.getNumber().intValue());
687             int mi = 0;
688             String ms = month.getValue().toString();
689             for (int i = 0; i < MONTH_STRINGS.length; i++) {
690                 if (MONTH_STRINGS[i].equals(ms)) {
691                     mi = i;
692                     break;
693                 }
694             }
695             result.set(Calendar.MONTH, mi);
696             result.set(Calendar.DATE, day.getNumber().intValue());
697             result.set(Calendar.HOUR_OF_DAY, hour.getNumber().intValue());
698             result.set(Calendar.MINUTE, min.getNumber().intValue());
699             result.set(Calendar.SECOND, sec.getNumber().intValue());
700 
701             return result;
702         }
